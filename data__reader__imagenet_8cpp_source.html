51 bool imagenet_reader::fetch_datum(Mat& X, int data_id, int mb_idx, int tid) {
52  int num_channel_values = m_image_width * m_image_height * m_image_num_channels;
53  std::string imagepath = get_file_dir() + image_list[data_id].first;
54 
55  int width, height;
56  unsigned char *pixels = m_pixel_bufs[tid].data();
57  bool ret = lbann::image_utils::loadJPG(imagepath.c_str(), width, height, false, pixels);
58  if(!ret) {
59  throw lbann_exception("ImageNet: image_utils::loadJPG failed to load");
60  }
61  if(width != m_image_width || height != m_image_height) {
62  throw lbann_exception("ImageNet: mismatch data size -- either width or height");
63  }
64 
65  for (int p = 0; p < num_channel_values; p++) {
66  X(p, mb_idx) = pixels[p];
67  }
68 
69  auto pixel_col = X(El::IR(0, X.Height()), El::IR(mb_idx, mb_idx + 1));
70  augment(pixel_col, m_image_height, m_image_width, m_image_num_channels);
71  normalize(pixel_col, m_image_num_channels);
72 
73  return true;
74 }

82 void imagenet_reader::load() {
83  std::string imageDir = get_file_dir();
84  std::string imageListFile = get_data_filename();
85 
86  image_list.clear();
87 
88  // load image list
89  FILE *fplist = fopen(imageListFile.c_str(), "rt");
90  if (!fplist) {
91  throw lbann_exception(
92  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
93  " :: failed to open: " + imageListFile);
94  }
95 
96  while (!feof(fplist)) {
97  char imagepath[512];
98  int imagelabel;
99  if (fscanf(fplist, "%s%d", imagepath, &imagelabel) <= 1) {
100  break;
101  }
102  image_list.push_back(std::make_pair(imagepath, imagelabel));
103  }
104  fclose(fplist);
105 
106  // reset indices
107  m_shuffled_indices.clear();
108  m_shuffled_indices.resize(image_list.size());
109  std::iota(m_shuffled_indices.begin(), m_shuffled_indices.end(), 0);
110 
111  select_subset_of_data();
112 }
