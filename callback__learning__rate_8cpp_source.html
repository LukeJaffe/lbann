39 void lbann_callback_learning_rate::setup(model *m) {
40  std::vector<Layer *>& layers = m->get_layers();
41  std::unordered_set<uint> to_update = m_layer_indices;
42  for (size_t l = 0; l < layers.size(); ++l) {
43  Layer *layer = layers[l];
44  uint idx = layer->get_index();
45  // Skip layers without optimizers.
46  optimizable_layer *opt_layer = dynamic_cast<optimizable_layer*>(layer);
47  if (!opt_layer) {
48  continue;
49  }
50  bool has_optimizer = opt_layer->get_optimizer() != nullptr;
51  if (m_layer_indices.empty() && has_optimizer) {
52  to_update.insert(idx);
53  m_last_idx = idx;
54  } else if (m_layer_indices.find(idx) != m_layer_indices.end()) {
55  if (!has_optimizer) {
56  throw lbann_exception(
57  "callback_learning_rate: requested layer " + std::to_string(idx) +
58  " which has NULL optimizer");
59  }
60  m_last_idx = idx;
61  }
62  }
63  m_layer_indices = to_update;
64 }

66 void lbann_callback_learning_rate::on_epoch_end(model *m) {
67  std::vector<Layer *>& layers = m->get_layers();
68  for (size_t l = 0; l < layers.size(); ++l) {
69  Layer *layer = layers[l];
70  uint idx = layer->get_index();
71  // Skip layers without optimizers.
72  optimizable_layer *opt_layer = dynamic_cast<optimizable_layer*>(layer);
73  if (opt_layer == NULL) {
74  continue;
75  }
76  if (m_layer_indices.find(idx) != m_layer_indices.end()) {
77  float old_lr = opt_layer->get_optimizer()->get_learning_rate();
78  float new_lr = schedule(m, layer);
79  if (old_lr != new_lr) {
80  opt_layer->get_optimizer()->set_learning_rate(new_lr);
81  lbann_comm *comm = m->get_comm();
82  if (comm->am_model_master()) {
83  std::cout << "Model " << comm->get_model_rank() <<
84  ": changing layer " << idx << " learning rate to " << new_lr <<
85  " at epoch " << m->get_cur_epoch() << std::endl;
86  }
87  }
88  }
89  }
90 }

119 float lbann_callback_adaptive_learning_rate::schedule(model *m, Layer *l) {
120  optimizable_layer *opt_layer = dynamic_cast<optimizable_layer*>(l);
121  float cur_lr = opt_layer->get_optimizer()->get_learning_rate();
122  double score = m->m_obj_fn->get_mean_value();
123  if (score < m_last_score) {
124  m_last_score = score;
125  m_wait = 0;
126  } else {
127  if (m_wait >= m_patience) {
128  if (is_last_layer(l)) {
129  m_wait = 0;
130  m_last_score = score;
131  }
132  return cur_lr * m_amt;
133  } else {
134  if (is_last_layer(l)) {
135  ++m_wait;
136  }
137  }
138  }
139  return cur_lr;
140 }

154 float lbann_callback_drop_fixed_learning_rate::schedule(model* m, Layer *l) {
155  optimizable_layer *opt_layer = dynamic_cast<optimizable_layer*>(l);
156  float cur_lr = opt_layer->get_optimizer()->get_learning_rate();
157  if (!m_drop_epochs.empty() && m->get_cur_epoch() == m_drop_epochs.back()) {
158  if (l->get_index() == m_last_idx) {

161  return cur_lr * m_amt;

183 void lbann_callback_linear_growth_learning_rate::setup(model *m) {
184  lbann_callback_learning_rate::setup(m);
185  // Compute the learning rate increase.
186  if (!m_layer_indices.empty()) {
187  std::vector<Layer *>& layers = m->get_layers();
188  // Assumes every layer has the same learning rate.
189  uint idx = *m_layer_indices.begin();
190  optimizable_layer *l = dynamic_cast<optimizable_layer *>(layers[idx]);
191  float base_lr = l->get_optimizer()->get_learning_rate();
192  m_inc = (m_target - base_lr) / m_num_epochs;
193  }
194 }

196 float lbann_callback_linear_growth_learning_rate::schedule(model *m,
197  Layer *l) {
198  optimizable_layer *opt_layer = dynamic_cast<optimizable_layer*>(l);
199  float cur_lr = opt_layer->get_optimizer()->get_learning_rate();
200  if (m->get_cur_epoch() < m_delay) {
201  return cur_lr;
202  } else if (m->get_cur_epoch() <= m_num_epochs + m_delay) {
203  return cur_lr + m_inc;
204  } else {
205  return cur_lr;
206  }
207 }
