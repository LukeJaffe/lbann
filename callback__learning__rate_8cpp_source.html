39 void lbann_callback_learning_rate::setup(model *m) {
40  std::vector<Layer *>& layers = m->get_layers();
41  for (size_t l = 0; l < layers.size(); ++l) {
42  Layer *layer = layers[l];
43  uint idx = layer->get_index();
44  // Skip non-learning layers.
45  learning *learning_layer = (learning *) dynamic_cast<learning *> (layer);
46  if(learning_layer == NULL) {
47  continue;
48  }
49  if (m_layer_indices.size() == 0 ||
50  m_layer_indices.find(idx) != m_layer_indices.end()) {
51  if (learning_layer->get_optimizer() != NULL) {
52  m_old_lrs[idx] = learning_layer->get_optimizer()->get_learning_rate();
53  m_last_idx = idx;
54  }
55  }
56  }
57 }

59 void lbann_callback_learning_rate::on_epoch_end(model *m) {
60  std::vector<Layer *>& layers = m->get_layers();
61  for (size_t l = 0; l < layers.size(); ++l) {
62  Layer *layer = layers[l];
63  uint idx = layer->get_index();
64  // Skip non-learning layers.
65  learning *learning_layer = (learning *) dynamic_cast<learning *> (layer);
66  if(learning_layer == NULL) {
67  continue;
68  }
69  if (m_old_lrs.find(idx) != m_old_lrs.end()) {
70  float new_lr = schedule(m, layer);
71  if (new_lr != m_old_lrs[idx]) {
72  m_old_lrs[idx] = new_lr;
73  learning_layer->get_optimizer()->set_learning_rate(new_lr);
74  lbann_comm *comm = m->get_comm();
75  if (comm->am_model_master()) {
76  std::cout << "Model " << comm->get_model_rank() <<
77  ": changing layer " << idx << " learning rate to " << new_lr <<
78  " at epoch " << m->get_cur_epoch() << std::endl;
79  }
80  }
81  }
82  }
83 }

118 float lbann_callback_adaptive_learning_rate::schedule(model *m, Layer *l) {
119  // Skip non-learning layers.
120  learning *learning_layer = (learning *) dynamic_cast<learning *> (l);
121  if(learning_layer == NULL) {
122  return 0;
123  }
124  float cur_lr = learning_layer->get_optimizer()->get_learning_rate();
125  double score = m->m_obj_fn->report_aggregate_avg_obj_fn(execution_mode::validation);
126  if (score < m_last_score) {
127  m_last_score = score;
128  m_wait = 0;
129  } else {
130  if (m_wait >= m_patience) {
131  if (is_last_layer(l)) {
132  m_wait = 0;
133  m_last_score = score;
134  }
135  return cur_lr * m_amt;
136  } else {
137  if (is_last_layer(l)) {
138  ++m_wait;
139  }
140  }
141  }
142  return cur_lr;
143 }
