39 void lbann_callback_learning_rate::setup(model *m) {
40  std::vector<Layer *>& layers = m->get_layers();
41  for (size_t l = 0; l < layers.size(); ++l) {
42  Layer *layer = layers[l];
43  uint idx = layer->get_index();
44  if (m_layer_indices.size() == 0 ||
45  m_layer_indices.find(idx) != m_layer_indices.end()) {
46  if (layer->get_optimizer() != NULL) {
47  m_old_lrs[idx] = layer->get_optimizer()->get_learning_rate();
48  m_last_idx = idx;
49  }
50  }
51  }
52 }

54 void lbann_callback_learning_rate::on_epoch_end(model *m) {
55  std::vector<Layer *>& layers = m->get_layers();
56  for (size_t l = 0; l < layers.size(); ++l) {
57  Layer *layer = layers[l];
58  uint idx = layer->get_index();
59  if (m_old_lrs.find(idx) != m_old_lrs.end()) {
60  float new_lr = schedule(m, layer);
61  if (new_lr != m_old_lrs[idx]) {
62  m_old_lrs[idx] = new_lr;
63  layer->get_optimizer()->set_learning_rate(new_lr);
64  lbann_comm *comm = m->get_comm();
65  if (comm->am_model_master()) {
66  std::cout << "Model " << comm->get_model_rank() <<
67  ": changing layer " << idx << " learning rate to " << new_lr <<
68  " at epoch " << m->get_cur_epoch() << std::endl;
69  }
70  }
71  }
72  }

103 float lbann_callback_adaptive_learning_rate::schedule(model *m, Layer *l) {
104  float cur_lr = l->get_optimizer()->get_learning_rate();
105  double score = m->obj_fn->report_aggregate_avg_obj_fn(execution_mode::validation);
106  if (score < m_last_score) {
107  m_last_score = score;
108  m_wait = 0;
109  } else {
110  if (m_wait >= m_patience) {
111  if (is_last_layer(l)) {
112  m_wait = 0;
113  m_last_score = score;
114  }
115  return cur_lr * m_amt;
116  } else {
117  if (is_last_layer(l)) {
118  ++m_wait;
119  }
120  }
121  }
122  return cur_lr;
123 }
