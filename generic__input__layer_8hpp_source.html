169  void fp_setup_data(int mini_batch_size) override {
170 
171  // Use the predetermined size of the mini-batch to set the current
172  // batch size for the neural network
173  mini_batch_size = get_current_mini_batch_size();
174  this->m_model->set_current_mini_batch_size(mini_batch_size);
175 
176  // Use the precomputed size of the global mini-batch to set the
177  // current effective batch size across all models
178  int total_mini_batch_size = get_current_global_mini_batch_size();
179  this->m_model->set_effective_mini_batch_size(total_mini_batch_size);
180 
181  // Initialize matrices
182  io_layer::fp_setup_data(mini_batch_size);
183 
184  // Once the current mini-batch size is defined, set the standard view for activations only
185  io_buffer->set_local_matrix_bypass(&get_local_activations());
186  io_buffer->set_std_matrix_view(mini_batch_size);

190  void fp_compute() override {
191  execution_mode mode = this->m_model->get_execution_mode();
192  int num_samples_in_batch = io_buffer->fetch_to_local_matrix(get_data_reader(), mode);
193 
194  if(dynamic_cast<partitioned_io_buffer*>(io_buffer) != nullptr) {
195  // Use the predetermined size of the mini-batch to set the current
196  // batch size for the neural network
197  num_samples_in_batch = get_current_mini_batch_size();
198 
199  update_num_samples_processed(num_samples_in_batch);
200  }else if(dynamic_cast<distributed_io_buffer*>(io_buffer) != nullptr) {
201  if(((distributed_io_buffer*) io_buffer)->is_current_root(mode)) {
203  update_num_samples_processed(num_samples_in_batch);
204  }
205 
206  int expected_num_samples_in_batch = this->m_model->get_current_mini_batch_size();
207 
210  num_samples_in_batch = Layer::m_comm->model_broadcast(((distributed_io_buffer*) io_buffer)->current_root_rank(mode), num_samples_in_batch);
211  this->m_model->set_current_mini_batch_size(num_samples_in_batch
212  + get_current_world_master_mini_batch_adjustment(m_comm->get_model_rank()));
213 
214  io_buffer->distribute_from_local_matrix(get_activations(), get_data_reader(), mode);
215 
216  if(num_samples_in_batch !=
217  (expected_num_samples_in_batch - get_current_world_master_mini_batch_adjustment(m_comm->get_model_rank()))) {
218  std::stringstream err;
219  err << __FILE__ << " " << __LINE__ << " :: "
220  << "I/O layers number of samples processed ("<< num_samples_in_batch
221  <<") does not match the mini-batch size ("
222  << (expected_num_samples_in_batch -
223  get_current_world_master_mini_batch_adjustment(m_comm->get_model_rank()))
224  << ")";
225  throw lbann_exception(err.str());
226  }
227  }else {
228  std::stringstream err;
229  err << __FILE__ << " " << __LINE__ << " :: "
230  << "could not fp_compute for I/O layers : encoutered generic_io_buffer type";
231  throw lbann_exception(err.str());
232  }
233  }
