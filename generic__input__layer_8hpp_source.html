176  void fp_set_std_matrix_view() override {
177  // Use the predetermined size of the mini-batch to set the current
178  // batch size for the neural network
179  El::Int cur_mini_batch_size = get_current_mini_batch_size();
180  this->m_model->set_current_mini_batch_size(cur_mini_batch_size);
181 
182  // Use the precomputed size of the global mini-batch to set the
183  // current effective batch size across all models
184  int total_mini_batch_size = get_current_global_mini_batch_size();
185  this->m_model->set_effective_mini_batch_size(total_mini_batch_size);
186 
187  // Once the current mini-batch size is defined, set the standard view for activations only
188  El::View(*m_activations_v, *m_activations, El::ALL, El::IR(0, cur_mini_batch_size));
189  io_buffer->set_local_matrix_bypass(&this->m_activations_v->Matrix());
190  io_buffer->set_std_matrix_view(cur_mini_batch_size);
191  }

196  void fp_compute() override {
197  execution_mode mode = this->m_model->get_execution_mode();
198  int num_samples_in_batch = io_buffer->fetch_to_local_matrix(get_data_reader(), mode);
199 
200  if(dynamic_cast<partitioned_io_buffer*>(io_buffer) != nullptr) {
201  // Use the predetermined size of the mini-batch to set the current
202  // batch size for the neural network
203  num_samples_in_batch = get_current_mini_batch_size();
204 
205  update_num_samples_processed(num_samples_in_batch);
206  }else if(dynamic_cast<distributed_io_buffer*>(io_buffer) != nullptr) {
207  if(((distributed_io_buffer*) io_buffer)->is_current_root(mode)) {
209  update_num_samples_processed(num_samples_in_batch);
210  }
211 
214  this->m_model->set_current_mini_batch_size(Layer::m_comm->model_broadcast(((distributed_io_buffer*) io_buffer)->current_root_rank(mode), num_samples_in_batch));
215 
216  io_buffer->distribute_from_local_matrix(*this->m_activations, get_data_reader(), mode);
217  }else {
218  std::stringstream err;
219  err << __FILE__ << " " << __LINE__ << " :: "
220  << "could not fp_compute for I/O layers : encoutered generic_io_buffer type";
221  throw lbann_exception(err.str());
222  }
223  }

632  }

634  // save our own state
635  // rank 0 reads the file
636  dataset_header header;
637  if (p.get_rank() == 0) {
638  p.read_uint64(persist_type::train, "reader_train_processed", &header.train_proc);
639  p.read_uint64(persist_type::train, "reader_train_total", &header.train_total);

642  p.read_uint64(persist_type::train, "reader_validate_processed", &header.validate_proc);
643  p.read_uint64(persist_type::train, "reader_validate_total", &header.validate_total);
644  }
645 
646  // TODO: assumes homogeneous hardware
647  // broadcast data from rank 0
648  MPI_Bcast(&header, sizeof(header), MPI_BYTE, 0, MPI_COMM_WORLD);
649 
650  // set our fields
651  m_training_dataset.num_samples_processed() = (long) header.train_proc;
652  m_training_dataset.total_samples() = (long) header.train_total;
653  m_testing_dataset.num_samples_processed() = (long) header.test_proc;
654  m_testing_dataset.total_samples() = (long) header.test_total;
655  m_validation_dataset.num_samples_processed() = (long) header.validate_proc;
656  m_validation_dataset.total_samples() = (long) header.validate_total;
657 
658  return true;
659  }
