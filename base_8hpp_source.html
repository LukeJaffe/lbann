65 enum class data_layout {MODEL_PARALLEL, DATA_PARALLEL, invalid};
66 static matrix_format __attribute__((used)) data_layout_to_matrix_format(data_layout layout) {
67  matrix_format format;
68  switch(layout) {
69  case data_layout::MODEL_PARALLEL:
70  format = matrix_format::MC_MR;
71  break;
72  case data_layout::DATA_PARALLEL:
74  format = matrix_format::STAR_STAR;
75  break;
76  default:
77  throw(std::string{} + __FILE__ + " " + std::to_string(__LINE__) + " Invalid data layout selected");
78  }
79  return format;
80 }
