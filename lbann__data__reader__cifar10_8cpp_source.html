66 void cifar10_reader::load() {
67  stringstream err;
68 
69  //open data file
70  string image_dir = get_file_dir();
71  string filename = get_data_filename();
72  stringstream b;
73  b << image_dir << "/" << filename;
74  if (is_master()) {
75  cout << "opening: " << b.str() << endl;
76  }
77  ifstream in(b.str().c_str(), ios::binary);
78  if (not in.good()) {
79  err << __FILE__ << " " << __LINE__
80  << " :: failed to open " << b.str() << " for reading";
81  throw lbann_exception(err.str());
82  }
83 
84  //get number of images, with error checking
85  int n = get_linearized_data_size() + 1; //should be 3073
86  in.seekg(0, in.end);
87  streampos fs = in.tellg();
88  in.seekg(0, in.beg);
89  if (fs % n != 0) {
90  err << __FILE__ << " " << __LINE__
91  << " :: fs % n != 0; fs: " << fs << " n: " << n;
92  throw lbann_exception(err.str());
93  }
94 
95  //reserve space for string images
96  int num_images = fs / n;
97  m_data.resize(num_images);
98  for (size_t h=0; h<m_data.size(); h++) {
99  m_data[h].resize(n);
100  }
101 
102  //read in the images; each image is 1 byte, which is the
103  //label (0-9), and 2072 pixels
104  for (size_t h=0; h<m_data.size(); h++) {
105  in.read((char *)&(m_data[h][0]), n);
106  }
107  in.close();
108 
109  m_shuffled_indices.resize(m_data.size());
110  for (size_t n = 0; n < m_data.size(); n++) {
111  m_shuffled_indices[n] = n;
112  }
113 
114  select_subset_of_data();
115 }

118 int lbann::cifar10_reader::fetch_data(Mat& X) {
119  stringstream err;
120 
121  if(!generic_data_reader::position_valid()) {
122  err << __FILE__ << " " << __LINE__ << " :: lbann::imagenet_reader::fetch_data() - !generic_data_reader::position_valid()";
123  throw lbann_exception(err.str());
124  }
125 
126  int current_batch_size = getm_batch_size();
127  const int end_pos = Min(m_current_pos+current_batch_size, m_shuffled_indices.size());
128  for (int n = m_current_pos; n < end_pos; ++n) {
129  int k = n - m_current_pos;
130  int idx = m_shuffled_indices[n];
131  for (size_t p = 1; p<m_data[idx].size(); p++) {
132  X.Set(p-1, k, m_data[idx][p]);
133  }
134 
135  auto pixel_col = X(IR(0, X.Height()), IR(k, k + 1));
136  augment(pixel_col, m_image_height, m_image_width, m_image_num_channels);
137  normalize(pixel_col, m_image_num_channels);
138  }
139 
140  return end_pos - m_current_pos;
141 }

143 int lbann::cifar10_reader::fetch_label(Mat& Y) {
144  if(!position_valid()) {
145  stringstream err;
146  err << __FILE__<<" "<<__LINE__<< " :: Imagenet data reader error: !position_valid";
147  throw lbann_exception(err.str());
148  }
149 
150  int current_batch_size = getm_batch_size();
151  int n = 0;
152  for (n = m_current_pos; n < m_current_pos + current_batch_size; n++) {
153  if (n >= (int)m_shuffled_indices.size()) {
154  break;
155  }
156  int k = n - m_current_pos;
157  int index = m_shuffled_indices[n];
158  int label = (int)m_data[index][0];
159  Y.Set(label, k, 1);
160  }
161  return (n - m_current_pos);
162 }
