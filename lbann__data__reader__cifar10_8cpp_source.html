68 void DataReader_CIFAR10::load()
69 {
70  stringstream err;
71 
72  //open data file
73  string image_dir = get_file_dir();
74  string filename = get_data_filename();
75  stringstream b;
76  b << image_dir << "/" << filename;
77  if (is_master()) cout << "opening: " << b.str() << endl;
78  ifstream in(b.str().c_str(), ios::binary);
79  if (not in.good()) {
80  err << __FILE__ << " " << __LINE__
81  << " :: failed to open " << b.str() << " for reading";
82  throw lbann_exception(err.str());
83  }
84 
85  //get number of images, with error checking
86  int n = get_linearized_data_size() + 1; //should be 3073
87  in.seekg(0, in.end);
88  streampos fs = in.tellg();
89  in.seekg(0, in.beg);
90  if (fs % n != 0) {
91  err << __FILE__ << " " << __LINE__
92  << " :: fs % n != 0; fs: " << fs << " n: " << n;
93  throw lbann_exception(err.str());
94  }
95 
96  //reserve space for string images
97  int num_images = fs / n;
98  m_data.resize(num_images);
99  for (size_t h=0; h<m_data.size(); h++) {
100  m_data[h].resize(n);
101  }
102 
103  //read in the images; each image is 1 byte, which is the
104  //label (0-9), and 2072 pixels
105  for (size_t h=0; h<m_data.size(); h++) {
106  in.read((char*)&(m_data[h][0]), n);
107  }
108  in.close();
109 
110  ShuffledIndices.resize(m_data.size());
111  for (size_t n = 0; n < m_data.size(); n++) {
112  ShuffledIndices[n] = n;
113  }
114 
115  select_subset_of_data();
116 }

119 int lbann::DataReader_CIFAR10::fetch_data(Mat &X)
120 {
121  stringstream err;
122 
123  if(!DataReader::position_valid()) {
124  err << __FILE__ << " " << __LINE__ << " :: lbann::DataReader_ImageNet::fetch_data() - !DataReader::position_valid()";
125  throw lbann_exception(err.str());
126  }
127 
128  int current_batch_size = getBatchSize();
129  const int end_pos = Min(CurrentPos+current_batch_size, ShuffledIndices.size());
130  for (int n = CurrentPos; n < end_pos; ++n) {
131  int k = n - CurrentPos;
132  int idx = ShuffledIndices[n];
133  for (size_t p = 1; p<m_data[idx].size(); p++) {
134  X.Set(p-1, k, m_data[idx][p]);
135  }
136 
137  auto pixel_col = X(IR(0, X.Height()), IR(k, k + 1));
138  augment(pixel_col, m_image_height, m_image_width, m_image_num_channels);
139  normalize(pixel_col, m_image_num_channels);
140  }
141 
142  return end_pos - CurrentPos;
143 }

145 int lbann::DataReader_CIFAR10::fetch_label(Mat& Y) {
146  if(!position_valid()) {
147  stringstream err;
148  err << __FILE__<<" "<<__LINE__<< " :: Imagenet data reader error: !position_valid";
149  throw lbann_exception(err.str());
150  }
151 
152  int current_batch_size = getBatchSize();
153  int n = 0;
154  for (n = CurrentPos; n < CurrentPos + current_batch_size; n++) {
155  if (n >= (int)ShuffledIndices.size()) {
156  break;
157  }
158  int k = n - CurrentPos;
159  int index = ShuffledIndices[n];
160  int label = (int)m_data[index][0];
161  Y.Set(label, k, 1);
162  }
163  return (n - CurrentPos);
164 }
