71 bool generic_data_reader::update(void) {
73  if(m_use_alt_last_mini_batch_size && ((m_current_mini_batch_idx+1) >= (m_num_mini_batches_per_reader-1))) {
74  // std::cout << "Data reader last update update the current position is " << m_current_pos << " and the next postion is going to be " << (m_current_pos + m_last_mini_batch_stride) << " and the number of samples total is " << m_shuffled_indices.size() << std::endl;
75  m_current_pos += m_last_mini_batch_stride;
76  } else {
77  // std::cout << "Data reader update the current position is " << m_current_pos << " and the next postion is going to be " << (m_current_pos + m_batch_stride) << " and the number of samples total is " << m_shuffled_indices.size() << std::endl;
78  m_current_pos += m_batch_stride;
79  }
80 
82  Zeros(m_indices_fetched_per_mb, m_indices_fetched_per_mb.Width(), 1);
83 
84  if (m_current_pos < (int)m_shuffled_indices.size()) {
85  m_current_mini_batch_idx++;
86  return true;
87  } else {
88  if (not m_first_n) {
89  std::shuffle(m_shuffled_indices.begin(), m_shuffled_indices.end(),
90  get_data_seq_generator());
91  }
92  m_current_mini_batch_idx = 0;
93  m_current_pos = m_base_offset + m_model_offset;
94  return false;
95  }

117 void generic_data_reader::select_subset_of_data(void) {
118  if(!get_firstN()) {
119  std::shuffle(m_shuffled_indices.begin(), m_shuffled_indices.end(), get_data_seq_generator());
120  }
121 
122  if (not (has_max_sample_count() or has_use_percent() or has_validation_percent())) {
123  return;
124  }
125 
126  if (has_max_sample_count()) {
127  size_t count = get_max_sample_count();
128  if(count > getNumData()) {
129  stringstream err;
130  err << __FILE__ << " " << __LINE__
131  << " :: generic_data_reader::select_subset_of_data() - max_sample_count=" << count
132  << " is > getNumData=" << getNumData();
133  throw lbann_exception(err.str());
134  }
135  m_shuffled_indices.resize(get_max_sample_count());
136  } else if (has_use_percent()) {
137  m_shuffled_indices.resize(get_use_percent()*getNumData());
138  }
139 
140  if (has_validation_percent()) {
141  long unused = get_validation_percent()*getNumData(); //getNumData() = m_shuffled_indices.size()
142  long use_me = getNumData() - unused;
143  if (unused > 0) {
144  m_unused_indices=std::vector<int>(m_shuffled_indices.begin() + use_me, m_shuffled_indices.end());
145  m_shuffled_indices.resize(use_me);
146  }
147  }
148 
149  if(!get_firstN()) {
150  std::sort(m_shuffled_indices.begin(), m_shuffled_indices.end());
151  std::sort(m_unused_indices.begin(), m_unused_indices.end());
152  }
153 }

211 bool lbann::generic_data_reader::loadFromCheckpointShared(persist& p, const char *name) {
212  // rank 0 reads the training state file
213  if (p.get_rank() == 0) {
214  char fieldname[1024];
215 
216  // record minibatch index
217  uint64_t val;
218  snprintf(fieldname, sizeof(fieldname), "%s_current_mini_batch_idx", name);
219  p.read_uint64(persist_type::train, fieldname, &val);
220  m_current_mini_batch_idx = (int) val;
221 
222  // get size of ShuffleIndices
223  snprintf(fieldname, sizeof(fieldname), "%s_data_size", name);
224  p.read_uint64(persist_type::train, fieldname, &val);
225  int size = (int) val;
226 
227  // get current position within data
228  snprintf(fieldname, sizeof(fieldname), "%s_data_position", name);
229  p.read_uint64(persist_type::train, fieldname, &val);
230  m_current_pos = (int) val;
231 
232  // resize shuffled index array to hold values
233  m_shuffled_indices.resize(size);
234 
235  // read list of indices
236  snprintf(fieldname, sizeof(fieldname), "%s_data_indices", name);
237  p.read_int32_contig(persist_type::train, fieldname, &m_shuffled_indices[0], (uint64_t) size);
238  }
239 
240  // broadcast minibatch index
241  MPI_Bcast(&m_current_mini_batch_idx, 1, MPI_INT, 0, MPI_COMM_WORLD);
242 
243  // TODO: with multiple readers, make this a scatter
244  // broadcast current position
245  MPI_Bcast(&m_current_pos, 1, MPI_INT, 0, MPI_COMM_WORLD);
246 
247  // broadcast values from rank 0
248  int size = m_shuffled_indices.size();
249  MPI_Bcast(&size, 1, MPI_INT, 0, MPI_COMM_WORLD);
250 
251  // resize shuffled index array to hold values
252  if (p.get_rank() != 0) {
253  m_shuffled_indices.resize(size);
254  }
255 
256  // broadcast index array
257  MPI_Bcast(&m_shuffled_indices[0], size, MPI_INT, 0, MPI_COMM_WORLD);
258 
259  return true;
260 }
