59 bool DataReader::update() {
61  if(m_use_alt_last_mini_batch_size && ((m_current_mini_batch_idx+1) >= (m_num_mini_batches_per_reader-1))) {
62  CurrentPos += m_last_mini_batch_stride;
63  }else {
64  CurrentPos += m_stride;
65  }
66  if (CurrentPos < (int)ShuffledIndices.size()) {
67  m_current_mini_batch_idx++;
68  return true;
69  } else {
70  if (not m_first_n) {
71  std::shuffle(ShuffledIndices.begin(), ShuffledIndices.end(),
72  get_data_seq_generator());
73  }
74  m_current_mini_batch_idx = 0;
75  CurrentPos = m_base_offset + m_model_offset;
76  return false;
77  }
78 }

99 void DataReader::select_subset_of_data() {
100  if(!get_firstN()) {
101  std::shuffle(ShuffledIndices.begin(), ShuffledIndices.end(), get_data_seq_generator());
102  }
103 
104  if (not (has_max_sample_count() or has_use_percent() or has_validation_percent())) {
105  return;
106  }
107 
108  if (has_max_sample_count()) {
109  size_t count = get_max_sample_count();
110  if(count > getNumData()) {
111  stringstream err;
112  err << __FILE__ << " " << __LINE__
113  << " :: DataReader::select_subset_of_data() - max_sample_count=" << count
114  << " is > getNumData=" << getNumData();
115  throw lbann_exception(err.str());
116  }
117  ShuffledIndices.resize(get_max_sample_count());
118  } else if (has_use_percent()) {
119  ShuffledIndices.resize(get_use_percent()*getNumData());
120  }
121 
122  if (has_validation_percent()) {
123  long unused = get_validation_percent()*getNumData(); //getNumData() = ShuffledIndices.size()
124  long use_me = getNumData() - unused;
125  if (unused > 0) {
126  m_unused_indices=std::vector<int>(ShuffledIndices.begin() + use_me, ShuffledIndices.end());
127  ShuffledIndices.resize(use_me);
128  }
129  }
130 
131  if(!get_firstN()) {
132  std::sort(ShuffledIndices.begin(), ShuffledIndices.end());
133  std::sort(m_unused_indices.begin(), m_unused_indices.end());
134  }
135 }

162 void DataReader::calculate_multi_model_data_distribution(lbann_comm *comm) {
163  int max_mini_batch_size = BatchSize;
164  int num_parallel_readers_per_model = (m_stride / comm->get_num_models()) / max_mini_batch_size;
165  int min_stride_across_models = max_mini_batch_size * comm->get_num_models();
166 
167  m_last_mini_batch_size = max_mini_batch_size;
168 
169  int num_whole_mini_batches_per_model = floor(getNumData() / min_stride_across_models);
170  int num_whole_mini_batches_per_reader = floor(num_whole_mini_batches_per_model / num_parallel_readers_per_model);
171  int parallel_readers_with_extra_mini_batch = num_whole_mini_batches_per_model % num_parallel_readers_per_model;
172  int per_model_partial_mini_batch_size = (getNumData() - (num_whole_mini_batches_per_model * min_stride_across_models))/(comm->get_num_models());
173  int world_master_remainder_data = 0;
174 
175  // Compute how many full "parallel" mini-batches are available
176  m_last_mini_batch_threshold = num_whole_mini_batches_per_model * min_stride_across_models;
177 
178  if(comm->get_rank_in_model() < parallel_readers_with_extra_mini_batch) {
179  num_whole_mini_batches_per_reader += 1;
180  m_last_mini_batch_size = max_mini_batch_size;
181  }
182 
183  m_num_mini_batches_per_reader = num_whole_mini_batches_per_reader;
184 
185  int world_master_remainder_adjustment = getNumData()
186  - (num_whole_mini_batches_per_model * min_stride_across_models)
187  - (per_model_partial_mini_batch_size * comm->get_num_models());
188  if(comm->am_world_master()) {
189  world_master_remainder_data = world_master_remainder_adjustment;
190  world_master_remainder_adjustment = 0;
191  }
192  per_model_partial_mini_batch_size += world_master_remainder_data;
193 
195  if(comm->get_rank_in_model() == parallel_readers_with_extra_mini_batch && per_model_partial_mini_batch_size > 0) {
196  m_num_mini_batches_per_reader++;
197  m_last_mini_batch_size = per_model_partial_mini_batch_size;
198  }
199 
200  if(m_last_mini_batch_size > max_mini_batch_size) { throw new lbann_exception("Error in calculating the partial mini-batch size, exceeds the max mini-batch size"); }
201 
204 
206  m_last_mini_batch_stride = m_stride;
207 
208  int last_mini_batch_offset = max(0, num_whole_mini_batches_per_reader - 1) * m_stride;
209 
211  if(comm->get_rank_in_model() == parallel_readers_with_extra_mini_batch && per_model_partial_mini_batch_size > 0) {
212  m_last_mini_batch_stride = (m_last_mini_batch_threshold - m_base_offset - m_model_offset - last_mini_batch_offset)
213  + comm->get_model_rank() * per_model_partial_mini_batch_size + world_master_remainder_adjustment;
214  }
215 
216  // cout << "[" << comm->get_rank_in_world() << "] " << comm->get_model_rank() << " model rank, "<< comm->get_rank_in_model() << " rank in model, num_whole_mini_batches_per_model " << num_whole_mini_batches_per_model << " num_whole_mini_batches_per_reader " << num_whole_mini_batches_per_reader << "(m_num_mini_batches_per_reader=" << m_num_mini_batches_per_reader << ") parallel_readers_with_extra_mini_batch " << parallel_readers_with_extra_mini_batch << " partial_mini_batch_size=" << per_model_partial_mini_batch_size << " last mini bath size=" << m_last_mini_batch_size << " world_master_remainder_data=" << world_master_remainder_data << " threshold " << m_last_mini_batch_threshold << " with a last stride of " << m_last_mini_batch_stride << " and stride of " << m_stride << " and there are " << num_parallel_readers_per_model << " parallel readers per model" << " last mini batch offset = " << last_mini_batch_offset << " parallel reader with extra minibatch = " << parallel_readers_with_extra_mini_batch << " model bracket = " << (parallel_readers_with_extra_mini_batch * max_mini_batch_size + per_model_partial_mini_batch_size + world_master_remainder_data) <<" base ofset "<< m_base_offset << " model offset " << m_model_offset <<endl;
217 
218  return;
219 }

222 bool DataReader::saveToCheckpointShared(persist& p, const char* name)
223 {
224  // rank 0 writes the training state file
225  if (p.m_rank == 0) {
226  char fieldname[1024];
227 
228  // record minibatch index
229  snprintf(fieldname, sizeof(fieldname), "%s_current_mini_batch_idx", name);
230  p.write_uint64(persist_type::train, fieldname, (uint64_t) m_current_mini_batch_idx);
231 
232  // get size of list of training examples
233  int size = ShuffledIndices.size();
234 
235  // record size of ShuffleIndices
236  snprintf(fieldname, sizeof(fieldname), "%s_data_size", name);
237  p.write_uint64(persist_type::train, fieldname, (uint64_t) size);
238 
239  // TODO: each model may have a different position, need to gather and write these
240  // record current position within training data
241  snprintf(fieldname, sizeof(fieldname), "%s_data_position", name);
242  p.write_uint64(persist_type::train, fieldname, (uint64_t) CurrentPos);
243 
244  // write list of indices
245  snprintf(fieldname, sizeof(fieldname), "%s_data_indices", name);
246  p.write_int32_contig(persist_type::train, fieldname, &ShuffledIndices[0], (uint64_t) size);
247  }
248 
249  return true;
250 }

253 bool lbann::DataReader::loadFromCheckpointShared(persist& p, const char* name)
254 {
255  // rank 0 reads the training state file
256  if (p.m_rank == 0) {
257  char fieldname[1024];
258 
259  // record minibatch index
260  uint64_t val;
261  snprintf(fieldname, sizeof(fieldname), "%s_current_mini_batch_idx", name);
262  p.read_uint64(persist_type::train, fieldname, &val);
263  m_current_mini_batch_idx = (int) val;
264 
265  // get size of ShuffleIndices
266  snprintf(fieldname, sizeof(fieldname), "%s_data_size", name);
267  p.read_uint64(persist_type::train, fieldname, &val);
268  int size = (int) val;
269 
270  // get current position within data
271  snprintf(fieldname, sizeof(fieldname), "%s_data_position", name);
272  p.read_uint64(persist_type::train, fieldname, &val);
273  CurrentPos = (int) val;
274 
275  // resize shuffled index array to hold values
276  ShuffledIndices.resize(size);
277 
278  // read list of indices
279  snprintf(fieldname, sizeof(fieldname), "%s_data_indices", name);
280  p.read_int32_contig(persist_type::train, fieldname, &ShuffledIndices[0], (uint64_t) size);
281  }
282 
283  // broadcast minibatch index
284  MPI_Bcast(&m_current_mini_batch_idx, 1, MPI_INT, 0, MPI_COMM_WORLD);
285 
286  // TODO: with multiple readers, make this a scatter
287  // broadcast current position
288  MPI_Bcast(&CurrentPos, 1, MPI_INT, 0, MPI_COMM_WORLD);
289 
290  // broadcast values from rank 0
291  int size = ShuffledIndices.size();
292  MPI_Bcast(&size, 1, MPI_INT, 0, MPI_COMM_WORLD);
293 
294  // resize shuffled index array to hold values
295  if (p.m_rank != 0) {
296  ShuffledIndices.resize(size);
297  }
298 
299  // broadcast index array
300  MPI_Bcast(&ShuffledIndices[0], size, MPI_INT, 0, MPI_COMM_WORLD);
301 
302  return true;
303 }
