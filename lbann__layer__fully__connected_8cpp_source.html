41 // FullyConnectedLayer : single network layer class
43 // WB structure: (num units "neurons / filters" x (num features + 1))
44 // Each row represents a neuron / filter
45 // There is a column for each feature coming in from the previous layer plus 1 for the bias
46 // [W00 ... B0]
47 // [| |]
48 // [Wn0 Bn]
49 //
50 // WB_D structure:
51 // [dW dB]
52 // D structure:
53 // [D ]
54 // Z, Zs, Act, Acts structure:
55 // [Acts ]

130 void lbann::FullyConnectedLayer::setup(int numPrevNeurons) {
131  Layer::setup(numPrevNeurons);
132  if(optimizer != NULL) {
133  optimizer->setup(numPrevNeurons+1, NumNeurons);
134  }
135 
136  // Initialize matrices
137  // Note: the weights-bias matrix has an extra column so it includes bias term
138  Zeros(*m_weights, NumNeurons, numPrevNeurons+1);
139  Zeros(*m_weights_gradient, NumNeurons, numPrevNeurons + 1);
140  Zeros(*m_prev_activations, numPrevNeurons, m_mini_batch_size);
141  Zeros(*m_weighted_sum, NumNeurons, m_mini_batch_size);
142  Zeros(*m_activations, NumNeurons, m_mini_batch_size);
143  Zeros(*m_prev_error_signal, NumNeurons, m_mini_batch_size);
144  Zeros(*m_error_signal, numPrevNeurons, m_mini_batch_size); // m_error_signal holds the product of m_weights^T * m_prev_error_signal
145 
147  View(*m_activation_weights_v, *m_weights, ALL, IR(0, numPrevNeurons));
148  View(*m_bias_weights_v, *m_weights, ALL, IR(numPrevNeurons));
149 
151  View(*m_activation_weights_gradient_v, *m_weights_gradient, ALL, IR(0, numPrevNeurons));
152  View(*m_bias_weights_gradient_v, *m_weights_gradient, ALL, IR(numPrevNeurons));
153 
155  initialize_matrix(*m_activation_weights_v, m_weight_initialization, numPrevNeurons, NumNeurons);
156 
158  Ones(*m_bias_bp_t, 1, m_mini_batch_size);
159 }

171 void lbann::FullyConnectedLayer::fp_linearity()
172 {
173  // Apply forward prop linearity
174 
175  // Apply bias
176  Copy(*m_bias_weights_v, *m_bias_weights_repl);
177  const Mat& local_bias_weights = m_bias_weights_repl->Matrix();
178  IndexDependentFill(m_weighted_sum_v->Matrix(), (std::function<DataType(El::Int,El::Int)>)
179  ([&local_bias_weights](El::Int r, El::Int c)->DataType {
180  return local_bias_weights.Get(r);
181  }));
182  Scale(m_bias_term, *m_weighted_sum_v);
183 
184  // Apply weight matrix
185  switch(m_data_layout) {
186  case data_layout::MODEL_PARALLEL:
187  Gemm(NORMAL, NORMAL, DataType(1),
188  *m_activation_weights_v,
189  *m_prev_activations_v,
190  DataType(1),
191  *m_weighted_sum_v);
192  break;
193  case data_layout::DATA_PARALLEL:
194  Gemm(NORMAL, NORMAL, DataType(1),
195  m_activation_weights_v->LockedMatrix(),
196  m_prev_activations_v->LockedMatrix(),
197  DataType(1),
198  m_weighted_sum_v->Matrix());
199  break;
200  }
201 
202  // Copy result to output matrix
203  Copy(*m_weighted_sum_v, *m_activations_v);
204 
205 }

207 void lbann::FullyConnectedLayer::bp_linearity()
208 {
209 
210  switch(m_data_layout) {
211  case data_layout::MODEL_PARALLEL:
212  // Compute the partial delta update for the next lower layer
213  Gemm(TRANSPOSE, NORMAL, DataType(1),
214  *m_activation_weights_v,
215  *m_prev_error_signal_v,
216  DataType(0),
217  *m_error_signal_v);
218  // Compute update for activation weights
219  Gemm(NORMAL, TRANSPOSE, DataType(1)/get_effective_minibatch_size(),
220  *m_prev_error_signal_v,
221  *m_prev_activations_v,
222  DataType(0),
223  *m_activation_weights_gradient_v);
224  // Compute update for bias terms
225  Gemv(NORMAL, DataType(1)/get_effective_minibatch_size(),
226  *m_prev_error_signal_v,
227  *m_bias_bp_t_v,
228  DataType(0),
229  *m_bias_weights_gradient_v);
230  break;
231  case data_layout::DATA_PARALLEL:
232  // Compute the partial delta update for the next lower layer
233  Gemm(TRANSPOSE, NORMAL, DataType(1),
234  m_activation_weights_v->LockedMatrix(),
235  m_prev_error_signal_v->LockedMatrix(),
236  DataType(0),
237  m_error_signal_v->Matrix());
238  // Compute update for activation weights
239  Gemm(NORMAL, TRANSPOSE, DataType(1)/get_effective_minibatch_size(),
240  m_prev_error_signal_v->LockedMatrix(),
241  m_prev_activations_v->LockedMatrix(),
242  DataType(0),
243  m_activation_weights_gradient_v->Matrix());
244  // Compute update for bias terms
245  Gemv(NORMAL, DataType(1)/get_effective_minibatch_size(),
246  m_prev_error_signal_v->LockedMatrix(),
247  m_bias_bp_t_v->LockedMatrix(),
248  DataType(0),
249  m_bias_weights_gradient_v->Matrix());
250  // Add gradients from all processes
251  AllReduce(*m_weights_gradient,
252  m_weights_gradient->RedundantComm());
253  break;
254  }

258 DataType lbann::FullyConnectedLayer::computeCost(DistMat &deltas) {
259  DataType avg_error = 0.0, total_error = 0.0;
260  // Compute the L2 norm on the deltas (activation - y)
261  ColSumMat norms;
262  ColumnTwoNorms(deltas, norms);
263  int c = 0;
264  // Sum the local, total error
265  for(int r = 0; r < norms.LocalHeight(); r++) {
266  total_error += norms.GetLocal(r,c);
267  }
268  mpi::AllReduce(total_error, norms.DistComm());
269  avg_error = total_error / norms.Height();
270  return avg_error;
271 }

289 DataType lbann::FullyConnectedLayer::checkGradient(Layer& PrevLayer, const DataType Epsilon)
290 {
291  DistMat WB_E1(m_weights->Grid());
292  DistMat WB_E2(m_weights->Grid());
293  DistMat Zs_E1(m_weighted_sum->Grid());
294  DistMat Zs_E2(m_weighted_sum->Grid());
295  DistMat Acts_E1(m_activations->Grid());
296  DistMat Acts_E2(m_activations->Grid());
297  DataType grad_diff = 0;
298  DataType grad_sum = 0;
299 
300  Zeros(WB_E1, m_weights->Height(), m_weights->Width());
301  Zeros(WB_E2, m_weights->Height(), m_weights->Width());
302  Zeros(Zs_E1, m_weighted_sum->Height(), m_weighted_sum->Width());
303  Zeros(Zs_E2, m_weighted_sum->Height(), m_weighted_sum->Width());
304  Zeros(Acts_E1, m_activations->Height(), m_activations->Width());
305  Zeros(Acts_E2, m_activations->Height(), m_activations->Width());
306 
307  Copy(*m_weights, WB_E1);
308  Copy(*m_weights, WB_E2);
309 
310  DataType sum_error = 0;
311  Int prow = 0;
312  Int pcol = 0;
313 
314  if(WB_E1.IsLocal(prow, pcol)) {
315  Int _prow = WB_E1.LocalRow(prow);
316  Int _pcol = WB_E1.LocalCol(pcol);
317  WB_E1.SetLocal(_prow, _pcol, WB_E1.GetLocal(_prow, _pcol) + Epsilon);
318  }
319  if(WB_E2.IsLocal(prow, pcol)) {
320  Int _prow = WB_E2.LocalRow(prow);
321  Int _pcol = WB_E2.LocalCol(pcol);
322  WB_E2.SetLocal(_prow, _pcol, WB_E2.GetLocal(_prow, _pcol) - Epsilon);
323  }
324  for (Int row = 0; row < m_weights->Height(); row++) {
325  for (Int col = 0; col < m_weights->Width(); col++) {
326  // printf("Updating %d: %d x %d\n", Index, row, col);
327  if(WB_E1.IsLocal(prow, pcol)) {
328  Int _prow = WB_E1.LocalRow(prow);
329  Int _pcol = WB_E1.LocalCol(pcol);
330  WB_E1.SetLocal(_prow, _pcol, WB_E1.GetLocal(_prow, _pcol) - Epsilon);
331  }
332  if(WB_E2.IsLocal(prow, pcol)) {
333  Int _prow = WB_E2.LocalRow(prow);
334  Int _pcol = WB_E2.LocalCol(pcol);
335  WB_E2.SetLocal(_prow, _pcol, WB_E2.GetLocal(_prow, _pcol) + Epsilon);
336  }
337  if(WB_E1.IsLocal(row, col)) {
338  Int _row = WB_E1.LocalRow(row);
339  Int _col = WB_E1.LocalCol(col);
340  WB_E1.SetLocal(_row, _col, WB_E1.GetLocal(_row, _col) + Epsilon);
341  }
342  if(WB_E2.IsLocal(row, col)) {
343  Int _row = WB_E2.LocalRow(row);
344  Int _col = WB_E2.LocalCol(col);
345  WB_E2.SetLocal(_row, _col, WB_E2.GetLocal(_row, _col) - Epsilon);
346  }
347 
348  // J(theta)
349  // this->fp_linearity(*WB, *(PrevLayer.Acts), *Zs, *Acts);
350  //this->fp_nonlinearity(*Acts);
351 
352  // J(thetaPlus(i))
353  // this->fp_linearity(WB_E1, *(PrevLayer.Acts), Zs_E1, Acts_E1);
354  //this->fp_nonlinearity(Acts_E1);
355 
356  // J(thetaMinus(i))
357  // this->fp_linearity(WB_E2, *(PrevLayer.Acts), Zs_E2, Acts_E2);
358  //this->fp_nonlinearity(Acts_E2);
359 
360  // this->getCost();
361 
362  // gradApprox(i) = J(thetaPlus(i)) - J(thetaMinus(i)) / (2*Epsilon)
363  Axpy(-1.0, *m_activations, Acts_E1);
364  Axpy(-1.0, *m_activations, Acts_E2);
365 
366  bool bad_E1 = false, bad_E2 = false;
367  for(Int r = 0; r < m_activations->Height(); r++) {
368  for(Int c = 0; c < m_activations->Width(); c++) {
369  if(Acts_E1.IsLocal(r,c)) {
370  Int _r = Acts_E1.LocalRow(r);
371  Int _c = Acts_E1.LocalCol(c);
372  if((Acts_E1.GetLocal(_r,_c) > 1e-12 || Acts_E1.GetLocal(_r,_c) < -1e-12) && r != row) {
373  bad_E1 = true;
374  // cout << "Acts_E1=["<< r << "," << c << "]="<<Acts_E1.GetLocal(_r,_c)<<endl;
375  }
376  }
377  if(Acts_E2.IsLocal(r,c)) {
378  Int _r = Acts_E2.LocalRow(r);
379  Int _c = Acts_E2.LocalCol(c);
380  if((Acts_E2.GetLocal(_r,_c) > 1e-12 || Acts_E2.GetLocal(_r,_c) < -1e-12) && r != row) {
381  bad_E2 = true;
382  // cout << "Acts_E2=["<< r << "," << c << "]="<<Acts_E2.GetLocal(_r,_c)<<endl;
383  }
384  }
385  }
386  }
387 
388  if(bad_E1) {
389  if(Acts_E1.Grid().Rank() == 0) {
390  printf("BAD ENTRY Acts_E1 %d x %d\n", row, col);
391  }
392  cout.precision(20);
393  Print(Acts_E1);
394  if(Acts_E1.Grid().Rank() == 0) {
395  printf("Acts\n");
396  }
397  Print(*m_activations);
398  }
399  if(bad_E2) {
400  if(Acts_E2.Grid().Rank() == 0) {
401  printf("BAD ENTRY Acts_E2 %d x %d\n", row, col);
402  }
403  Print(Acts_E2);
404  }
405 
406  prow = row;
407  pcol = col;
408  }
409  }
410  DataType grad_error = sqrt(grad_diff / grad_sum);
411  return grad_error;
