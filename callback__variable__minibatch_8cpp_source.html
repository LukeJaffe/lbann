37 void lbann_callback_variable_minibatch::on_train_begin(model *m) {
38  // Avoid issues with the train method being called multiple times.
39  if (m->get_cur_epoch() != 0) {
40  return;
41  }
42  input_layer* input = dynamic_cast<input_layer*>(m->get_layers()[0]);
43  if (!input) {
44  throw lbann_exception("variable_minibatch: could not get input layer");
45  }
46  if (m_starting_mbsize > m->get_max_mini_batch_size()) {
47  throw lbann_exception(
48  "variable_minibatch: starting mini-batch size is larger than max");
49  }
50  input->calculate_num_iterations_per_epoch_training_spans_models(
51  m_starting_mbsize);
52 }

54 void lbann_callback_variable_minibatch::on_epoch_end(model *m) {
55  input_layer* input = dynamic_cast<input_layer*>(m->get_layers()[0]);
56  int new_mbsize = change_minibatch_size(m);
57  if (new_mbsize != m_current_mini_batch_size &&
58  new_mbsize <= m->get_max_mini_batch_size()) {
59  input->calculate_num_iterations_per_epoch_training_spans_models(new_mbsize);
60  std::vector<Layer*>& layers = m->get_layers();
61  // Determine the current learning rate.
62  float cur_lr = 0;
63  for (size_t l = 0; l < layers.size(); ++l) {
64  // Skip non-learning layers.
65  learning *learning_layer = dynamic_cast<learning*>(layers[l]);
66  if (learning_layer == NULL) {
67  continue;
68  }
69  optimizer *opt = learning_layer->get_optimizer();
70  if (opt) {
71  cur_lr = opt->get_learning_rate();
72  break;
73  }
74  }
75  // Potentially update the layers' learning rate.
76  float new_lr = change_learning_rate(m, cur_lr, m_current_mini_batch_size,
77  new_mbsize);
78  for (size_t l = 0; l < layers.size(); ++l) {
79  // Skip non-learning layers.
80  learning *learning_layer = dynamic_cast<learning*>(layers[l]);
81  if (learning_layer == NULL) {
82  continue;
83  }
84  optimizer *opt = learning_layer->get_optimizer();
85  if (opt) {
86  opt->set_learning_rate(new_lr);
87  }
88  }
89  lbann_comm *comm = m->get_comm();
90  if (comm->am_model_master()) {
91  std::cout << "Model " << comm->get_model_rank() <<
92  ": changing mini-batch size to " << new_mbsize <<
93  " and learning rate to " << new_lr << " at epoch " <<
94  m->get_cur_epoch() << std::endl;
95  }
96  // Update our tracking of the mini-batch size.
97  m_current_mini_batch_size = new_mbsize;
98  }
99 }
