60 void imagenet_readerSingle::load() {
61  string image_dir = get_file_dir();
62  string base_filename = get_data_filename();
63 
64  //open offsets file, with error checking
65  stringstream b;
66  b << image_dir << "/" << base_filename << "_offsets.txt";
67  if (is_master()) {
68  cout << "opening: " << b.str() << " " << endl;
69  }
70  ifstream in(b.str().c_str());
71  if (not in.is_open() and in.good()) {
72  stringstream err;
73  err << __FILE__ << " " << __LINE__
74  << " :: failed to open " << b.str() << " for reading";
75  throw lbann_exception(err.str());
76  }
77 
78  //read the offsets file
79  size_t num_images;
80  in >> num_images;
81  if (is_master()) {
82  cout << "num images: " << num_images << endl;
83  }
84  m_offsets.reserve(num_images);
85  m_offsets.push_back(make_pair(0,0));
86  size_t last_offset = 0;
87  size_t offset;
88  int label;
89  while (in >> offset >> label) {
90  m_offsets.push_back(make_pair(offset + last_offset, label));
91  last_offset = m_offsets.back().first;
92  }
93 
94  if (num_images+1 != m_offsets.size()) {
95  stringstream err;
96  err << __FILE__ << " " << __LINE__
97  << " :: we read " << m_offsets.size() << " offsets, but should have read " << num_images;
98  throw lbann_exception(err.str());
99  }
100  in.close();
101 
102  open_data_stream();
103 
104  m_shuffled_indices.resize(m_offsets.size());
105  for (size_t n = 0; n < m_offsets.size()-1; n++) {
106  m_shuffled_indices[n] = n;
107  }
108 
109  select_subset_of_data();
110 }

113 bool imagenet_readerSingle::fetch_datum(Mat& X, int data_id, int mb_idx, int tid) {
114  int num_channel_values = m_image_width * m_image_height * m_image_num_channels;
115  stringstream err;
116  int width, height;
117 
118  if (static_cast<size_t>(data_id+1) >= m_offsets.size()) {
119  err << __FILE__ << " " << __LINE__ << " :: data_id= " << data_id << " is larger than m_offsets.size()= " << m_offsets.size() << " -2";
120  throw lbann_exception(err.str());
121  }
122  size_t start = m_offsets[data_id].first;
123  size_t end = m_offsets[data_id+1].first;
124 
125  if (end > m_file_size) {
126  err << __FILE__ << " " << __LINE__ << " :: end= " << end << " is larger than m_file_size= " << m_file_size << " for P_" << get_rank() << " with role: " << get_role() << " m_offsets.size(): " << m_offsets.size() << " mb_idx: " << mb_idx << " data_id: " << data_id;
127  throw lbann_exception(err.str());
128  }
129 
130  int ssz = end - start;
131 
132  if (ssz <= 0) {
133  err << "P_" << get_rank() << " start: " << start << " end: " << end << " ssz= " << ssz << " is <= 0";
134  throw lbann_exception(err.str());
135  }
136 
137  m_work_buffer.resize(ssz);
138  m_data_filestream.seekg(start);
139  m_data_filestream.read((char *)&m_work_buffer[0], ssz);
140 
141  unsigned char *pixels = m_pixel_bufs[tid].data();
142  bool ret = lbann::image_utils::loadJPG(m_work_buffer, width, height, false, pixels);
143 
144  if(!ret) {
145  err << __FILE__ << " " << __LINE__ << " :: ImageNetSingle: image_utils::loadJPG failed to load index: " << data_id;
146  throw lbann_exception(err.str());
147  }
148  if(width != m_image_width || height != m_image_height) {
149  err << __FILE__ << " " << __LINE__ << " :: ImageNetSingle: mismatch data size -- either width or height";
150  throw lbann_exception(err.str());
151  }
152 
153  for (El::Int p = 0; p < num_channel_values; p++) {
154  X.Set(p, mb_idx, pixels[p]);
155  }
156 
157  auto pixel_col = X(El::IR(0, X.Height()), El::IR(mb_idx, mb_idx + 1));
158  augment(pixel_col, m_image_height, m_image_width, m_image_num_channels);
159  normalize(pixel_col, m_image_num_channels);
160 
161  return true;
162 }

183  stringstream b;
184  b << image_dir << "/" << base_filename << "_data.bin";
185  if (is_master()) {

187  }
188  m_data_filestream.open(b.str().c_str(), ios::in | ios::binary);
189  if (not m_data_filestream.is_open() and m_data_filestream.good()) {
190  stringstream err;

193  throw lbann_exception(err.str());
194  }
195  m_data_filestream.seekg(0, m_data_filestream.end);
