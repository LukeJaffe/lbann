64  double compute_metric(ElMat& predictions_v, ElMat& ground_truth_v) {
65  // Simple approach: Gather all predictions to the model master.
66  // TODO: Better to gather only the top k predictions from each model.
67  int num_errors = 0;
68  if (this->m_comm->am_model_master()) {
69  m_gathered_predictions = predictions_v;
70  m_gathered_ground_truth = ground_truth_v;
71  const El::Int mbsize = m_gathered_predictions.Width();
72  const El::Int num_classes = m_gathered_predictions.Height();
73  // Compute the top k predictions in each entry of the minibatch.
74  Mat top_predictions(mbsize, m_top_k);
75  std::vector<El::Int> indices(num_classes);
76  std::iota(indices.begin(), indices.end(), 0);
77  for (El::Int mb_idx = 0; mb_idx < mbsize; ++mb_idx) {
78  // We want the indices of the top k predictions for this entry.
79  // This partially sorts an auxiliary array of indices based on the
80  // predictions.
81  std::partial_sort(indices.begin(), indices.begin() + m_top_k,
82  indices.end(),
83  [mb_idx, this] (El::Int a, El::Int b) -> bool {
84  return m_gathered_predictions.GetLocal(a, mb_idx) >
85  m_gathered_predictions.GetLocal(b, mb_idx); });
86  // Compute the index of the ground truth value.
87  El::Int ground_truth_idx = -1;
88  for (El::Int class_idx = 0; class_idx < num_classes; ++class_idx) {
89  if (m_gathered_ground_truth.GetLocal(class_idx, mb_idx) ==
90  DataType(1)) {
91  ground_truth_idx = class_idx;
92  }
93  }
94  // Now check if those top indices contain the true value.
95  bool found = false;
96  for (auto pred = indices.begin(); pred != indices.begin() + m_top_k;
97  ++pred) {
98  if (*pred == ground_truth_idx) {
99  found = true;
100  break;
101  }
102  }
103  if (!found) {
104  ++num_errors;
105  }
106  }
107  // Broadcast the number of errors to the rest of the models.
108  this->m_comm->model_broadcast(0, num_errors);
109  } else {
110  num_errors = this->m_comm->template model_broadcast<int>(0);
111  }
112  return num_errors;
113  }

115  double report_metric(execution_mode mode) {
116  statistics *stats = get_statistics(mode);
117  double errors_per_epoch = stats->m_error_per_epoch;
118  long samples_per_epoch = stats->m_samples_per_epoch;
119 
120  double accuracy = (double)(samples_per_epoch - errors_per_epoch) /
121  samples_per_epoch * 100;
122  string score = std::to_string(accuracy);

124  return accuracy;
