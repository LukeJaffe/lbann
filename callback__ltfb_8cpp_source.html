74 void lbann_callback_ltfb::on_batch_end(model *m) {
75  if (m->get_cur_step() % m_round_size != 0) {
76  return; // Not the end of a round.
77  }
78 
79  int partner = select_partner();
80  if (partner == m_comm->get_model_rank()) {
81  // No partner this round, skip.
82  return;
83  }
84  // Transfer data to secondary model.
85  exchange(m, partner);
86  // Evaluate on tournament data.
87  // Have to cast, assumes deep_neural_network.
88  deep_neural_network *dnn = dynamic_cast<deep_neural_network *>(m);
89  deep_neural_network *remote_dnn =
90  dynamic_cast<deep_neural_network *>(m_remote_model);
91  double local_acc = evaluate(dnn);
92  double remote_acc = evaluate(remote_dnn);
93  // If the remote is better, keep it.
94  if (remote_acc > local_acc) {
95  if (m_comm->am_model_master()) {
96  std::cout << m_comm->get_model_rank() << ": (step " << m->get_cur_step() << ") Replacing local model (" << local_acc << ") with " <<
97  "model " << partner << " (" << remote_acc << ")" << std::endl;
98  }
99  replace_with_remote(m);
100  }
101 }

103 int lbann_callback_ltfb::select_partner() {
104  int my_partner = 0;
105  // Master generates partners for everyone.
106  if (m_comm->am_world_master()) {
107  std::vector<int> ranks(m_comm->get_num_models());
108  std::iota(ranks.begin(), ranks.end(), 0);
109  std::shuffle(ranks.begin(), ranks.end(), get_fast_generator());
110  // Adjacent pairs become partners.
111  // Dilate so that we can do one scatter to every process.
112  std::vector<int> partners(
113  m_comm->get_num_models() * m_comm->get_procs_per_model());
114  for (size_t i = 0; i < (ranks.size() & ~1); i += 2) {
115  int rank1 = ranks[i];
116  int rank2 = ranks[i+1];
117  std::fill_n(partners.begin() + rank1*m_comm->get_procs_per_model(),
118  m_comm->get_procs_per_model(), rank2);
119  std::fill_n(partners.begin() + rank2*m_comm->get_procs_per_model(),
120  m_comm->get_procs_per_model(), rank1);
121  }
122  // Handle the last rank if needed.
123  if (partners.size() % 2 != 0) {
124  int last_rank = ranks[ranks.size() - 1];
125  std::fill_n(partners.begin() + last_rank*m_comm->get_procs_per_model(),
126  m_comm->get_procs_per_model(), last_rank);
127  }
128  El::mpi::Scatter(partners.data(), 1, &my_partner, 1, 0,
129  El::mpi::COMM_WORLD);
130  } else {
131  El::mpi::Scatter((int *) nullptr, 0, &my_partner, 1, 0, El::mpi::COMM_WORLD);
132  }
133  return my_partner;
134 }

136 void lbann_callback_ltfb::exchange(model *m, int partner) {
137  std::vector<Layer *>& layers = m->get_layers();
138  std::vector<Layer *>& remote_layers = m_remote_model->get_layers();
139  const std::type_info& learning_type = typeid(learning);
140  for (size_t i = 0; i < layers.size(); ++i) {
141  // Only exchange learning layers.
142  if (std::type_index(learning_type) ==
143  std::type_index(typeid(*layers[i]))) {
144  learning *layer = dynamic_cast<learning*>(layers[i]);
145  learning *remote_layer = dynamic_cast<learning*>(remote_layers[i]);
146  // TODO: Support sending optimizer state.
147  ElMat& weights = layer->get_weights();
148  ElMat& remote_weights = remote_layer->get_weights();
149  if (weights.Height() > 0) {
150  m_comm->sendrecv(weights.LockedBuffer(),
151  weights.LocalHeight()*weights.LocalWidth(),
152  partner,
153  remote_weights.Buffer(),
154  weights.LocalHeight()*weights.LocalWidth(),
155  partner);
156  }
157  }
158  }

161 double lbann_callback_ltfb::evaluate(deep_neural_network *m) {
162  m->evaluate(execution_mode::validation);
163  m->set_execution_mode(execution_mode::training); // Reset execution mode.
164  double acc = 0;
165  const std::type_info& ca_model_type = typeid(
166  metrics::categorical_accuracy<data_layout::MODEL_PARALLEL>);
167  const std::type_info& ca_data_type = typeid(
168  metrics::categorical_accuracy<data_layout::DATA_PARALLEL>);
169  for (auto&& metric : m->get_metrics()) {
170  const std::type_info& m_type = typeid(*metric);
171  if (std::type_index(ca_model_type) == std::type_index(m_type) ||
172  std::type_index(ca_data_type) == std::type_index(m_type)) {
173  acc = metric->report_metric(execution_mode::validation);
174  break;
175  }
176  }
177  return acc;
178 }
