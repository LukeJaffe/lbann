45 void data_store_pilot2_molecular::setup() {
46  double tm1 = get_time();
47  std::stringstream err;
48  m_owner = (int)m_reader->get_compound_rank() == (int)m_rank;
49  m_owner_rank = m_reader->get_compound_rank();
50 
51  if (m_owner) std::cerr << "starting data_store_pilot2_molecular::setup() for role: "
52  << m_reader->get_role() << "; owning processor: " << m_owner_rank << std::endl;
53  if (m_owner) std::cerr << "calling generic_data_store::setup()\n";
54  generic_data_store::setup();
55 
56  if (! m_in_memory) {
57  err << __FILE__ << " " << __LINE__ << " :: "
58  << "not yet implemented";
59  throw lbann_exception(err.str());
60  }
61 
62  else {
63  //sanity check
64  pilot2_molecular_reader *reader = dynamic_cast<pilot2_molecular_reader*>(m_reader);
65  if (reader == nullptr) {
66  err << __FILE__ << " " << __LINE__ << " :: "
67  << "dynamic_cast<data_reader_pilot2_molecular*>(m_reader) failed";
68  throw lbann_exception(err.str());
69  }
70  m_pilot2_reader = reader;
71 
72  // get list of indices used in calls to generic_data_reader::fetch_data
73  size_t s2 = 0;
74  for (auto t1 : (*m_minibatch_indices)) {
75  s2 += t1.size();
76  }
77  m_my_minibatch_indices.reserve(s2);
78  for (auto t1 : (*m_minibatch_indices)) {
79  for (auto t2 : t1) {
80  m_my_minibatch_indices.push_back(t2);
81  }
82  }
83 
84  // allocate storage for the data that will be passed to the data reader's
85  // fetch_datum method.
86  m_data_buffer.resize(omp_get_max_threads());
87  int num_features = m_pilot2_reader->get_num_features();
88  int num_neighbors = m_pilot2_reader->get_num_neighbors();
89  for (size_t j=0; j<m_data_buffer.size(); j++) {
90  m_data_buffer[j].resize(num_features * (num_neighbors+1));
91  }
92 
93  if (m_owner) {
94  std::cerr << "calling construct_data_store()\n";
95  construct_data_store();
96  }
97 
98  if (m_owner) std::cerr << "calling bcast_offsets()\n";
99  bcast_offsets();
100 
101  if (m_owner) std::cerr << "calling build_nabor_map()\n";
102  build_nabor_map();
103 
104 #if 0
105  if (m_extended_testing) {
106  if (m_owner) std::cerr << "calling setup_extended_testing\n";
107  setup_extended_testing();
108  }
109 #endif
110 
111  MPI_Win_create(m_data.data(), m_data.size(), sizeof(double), MPI_INFO_NULL, m_comm->get_model_comm().comm, &m_win);
112 
113  if (m_owner) std::cerr << "calling exchange_data()\n";
114  exchange_data();
115  }
116 
117  if (m_owner) {
118  std::cerr << "data_store_pilot2_molecular::setup time: " << get_time() - tm1 << "\n";
119  }

122 void data_store_pilot2_molecular::construct_data_store() {
123  std::stringstream err;
124 
125  // allocate memory for the data store
126  int num_features = m_pilot2_reader->get_num_features();
127  m_data.resize(m_num_global_indices*num_features, 0.0);
128 
129  // get the feature and neighbor data from the pilot2_molecular data reader
130  if (m_pilot2_reader->get_word_size() == 4) {
131  err << __FILE__ << " " << __LINE__ << " :: "
132  << "not implemented for word_size = 4; please ask Dave Hysom to fix";
133  throw lbann_exception(err.str());
134  }
135  double *features_8 = m_pilot2_reader->get_features_8();
136 
137  int num_samples_per_frame = m_pilot2_reader->get_num_samples_per_frame();
138 
139  size_t offset = 0;
140  size_t jj = 0;
141  for (size_t j=0; j<m_num_global_indices; j++) {
142  int data_id = (*m_shuffled_indices)[j];
143  m_offsets[data_id] = offset;
144  offset += num_features;
145  fill_in_data(data_id, jj, num_samples_per_frame, num_features, features_8);
146  }
147 }

151 void data_store_pilot2_molecular::fill_in_data(
152  const int data_id, size_t &jj, const int num_samples_per_frame,
153  const int num_features, double *features) {
154  const int frame = m_pilot2_reader->get_frame(data_id);
155  const int frame_offset = frame * num_features * num_samples_per_frame;
156  const int intra_frame_data_id = data_id - frame * num_samples_per_frame;
157  double *data = features + frame_offset + intra_frame_data_id * num_features;
158  for (int i=0; i<num_features; i++) {
159  m_data[jj++] = m_pilot2_reader->scale_data<double>(i, data[i]);
160  }
161 }

163 void data_store_pilot2_molecular:: bcast_offsets() {
164  int n = m_offsets.size();
165  MPI_Bcast(&n, 1, MPI_INT, m_owner_rank, m_comm->get_model_comm().comm);
166  std::vector<int> w(n*2);
167 
168  size_t jj = 0;
169  if (m_owner) {
170  for (auto t : m_offsets) {
171  w[jj++] = t.first;
172  w[jj++] = t.second;
173  }
174  }
175  MPI_Bcast(w.data(), n*2, MPI_INT, m_owner_rank, m_comm->get_model_comm().comm);
176 
177  if (! m_owner) {
178  for (size_t j=0; j<w.size(); j+= 2) {
179  m_offsets[w[j]] = w[j+1];
180  }
181  }

184 void data_store_pilot2_molecular::exchange_data() {
185  double tm1 = get_time();
186  std::stringstream err;
187 
188  //get set of molecules required for the next epoch
189  std::unordered_set<int> required_molecules;
190  for (auto t : m_my_minibatch_indices) {
191  int data_id = (*m_shuffled_indices)[t];
192  required_molecules.insert(data_id);
193  if (m_neighbors.find(data_id) == m_neighbors.end()) {
194  err << __FILE__ << " :: " << __LINE__ << " :: "
195  << " m_neighbors.find(" << data_id << " failed";
196  throw lbann_exception(err.str());
197  }
198  for (auto t2 : m_neighbors[data_id]) {
199  required_molecules.insert(t2);
200  }
201  }
202 
203  //allocate storage
204  m_my_molecules.resize(required_molecules.size());
205  int num_features = m_pilot2_reader->get_num_features();
206  for (size_t j=0; j< m_my_molecules.size(); j++) {
207  m_my_molecules[j].resize(num_features);
208  }
209 
210  MPI_Win_fence(MPI_MODE_NOSTORE|MPI_MODE_NOPUT, m_win);
211  m_molecule_hash.clear();
212 
213  size_t jj = 0;
214  for (auto idx : required_molecules) {
215  if (idx != -1) {
216  if (m_offsets.find(idx) == m_offsets.end()) {
217  err << __FILE__ << " :: " << __LINE__ << " :: "
218  << " m_offsets.find(" << idx << ") failed";
219  throw lbann_exception(err.str());
220  }
221  int offset = m_offsets[idx];
222  m_molecule_hash[idx] = jj;
223  if (jj >= m_my_molecules.size()) throw lbann_exception("ERROR 1");
224 
225  MPI_Get(m_my_molecules[jj].data(), num_features, MPI_DOUBLE,
226  m_owner_rank, offset, num_features, MPI_DOUBLE, m_win);
227  ++jj;
228  }
229  }
230  MPI_Win_fence(MPI_MODE_NOSTORE|MPI_MODE_NOPUT, m_win);
231  if (m_owner) {
232  std::cout << "data_store_pilot2_molecular::exchange_data() time: " << get_time() - tm1 << std::endl;
233  }
234 }

237 void data_store_pilot2_molecular::build_nabor_map() {
238  //bcast neighbor data
239  int sz;
240  if (m_owner) sz= m_pilot2_reader->get_neighbors_data_size();
241  MPI_Bcast(&sz, 1, MPI_INT, m_owner_rank, m_comm->get_model_comm().comm);
242  double *neighbors_8;
243  std::vector<double> work;
244  if (m_owner) {
245  neighbors_8 = m_pilot2_reader->get_neighbors_8();
246  } else {
247  work.resize(sz);
248  neighbors_8 = work.data();
249  }
250  MPI_Bcast(neighbors_8, sz, MPI_DOUBLE, m_owner_rank, m_comm->get_model_comm().comm);
251 
252  //fill in the nabors map
253  for (auto data_id : (*m_shuffled_indices)) {
254  int frame = m_pilot2_reader->get_frame(data_id);
255  int max_neighborhood = m_pilot2_reader->get_max_neighborhood();
256  int num_samples_per_frame = m_pilot2_reader->get_num_samples_per_frame();
257  const int neighbor_frame_offset = frame * num_samples_per_frame * (2 * max_neighborhood);
258  const int intra_frame_data_id = data_id - frame * num_samples_per_frame;
259  int num_neighbors = m_pilot2_reader->get_num_neighbors();
260  m_neighbors[data_id].reserve(num_neighbors);
261  double *neighbor_data = neighbors_8 + neighbor_frame_offset + intra_frame_data_id * (2 * max_neighborhood);
262  for (int i=1; i<num_neighbors + 1; ++i) {
263  int neighbor_id = neighbor_data[i];
264  m_neighbors[data_id].push_back(neighbor_id);
265  }
266  }
267 }

269 void data_store_pilot2_molecular::get_data_buf(int data_id, int tid, std::vector<double> *&buf) {
270  std::stringstream err;
271  std::vector<double> &v = m_data_buffer[tid];
272  std::fill(v.begin(), v.end(), 0.0);
273  if (m_neighbors.find(data_id) == m_neighbors.end()) {
274  err << __FILE__ << " " << __LINE__ << " :: "
275  << data_id << " not found in m_neighbors (primary molecule)";
276  throw lbann_exception(err.str());
277  }
278  if (m_molecule_hash.find(data_id) == m_molecule_hash.end()) {
279  err << __FILE__ << " " << __LINE__ << " :: "
280  << data_id << " not found in m_molecule_hash";
281  throw lbann_exception(err.str());
282  }
283 
284  //fill in data for the primary molecule
285  size_t jj = 0;
286  std::vector<double> &d1 = m_my_molecules[m_molecule_hash[data_id]];
287  for (size_t j=0; j<d1.size(); j++) {
288  v[jj++] = d1[j];
289  }
290 
291  //fill in data for the primary molecule's neighbor
292  std::vector<int> &nabors = m_neighbors[data_id];
293  int num_features = m_pilot2_reader->get_num_features();
294  for (size_t h=0; h<nabors.size(); h++) {
295  if (nabors[h] != -1) {
296  if (m_molecule_hash.find(data_id) == m_molecule_hash.end()) {
297  err << __FILE__ << " " << __LINE__ << " :: "
298  << nabors[h] << " not found in m_molecule_hash (neighbor)";
299  throw lbann_exception(err.str());
300  }
301  std::vector<double> &d2 = m_my_molecules[m_molecule_hash[nabors[h]]];
302  for (size_t i=0; i<d2.size(); i++) {
303  v[jj++] = d2[i];
304  }
305  } else {
306  jj += num_features;
307  }
308  }
309 
310  buf = &v;
311 }
