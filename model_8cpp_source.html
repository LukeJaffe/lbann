322 bool model::need_checkpoint() {
323  /* TODO: since we're using clocks, this requires a bcast for each call,
324  * we could use number of samples processed to make a local decision */
325 
326  // if none of our checkpoint conditions are set, assume we're not checkpointing
327  if (m_checkpoint_epochs == 0 &&
328  m_checkpoint_steps == 0 &&
329  m_checkpoint_secs == 0.0) {
330  return false;
331  }
332 
333  // assume that we won't checkpoint
334  int flag = 0;
335 
336  // if at start of epoch and evenly divide
337  if (flag == 0 && m_checkpoint_epochs > 0) {
338  if (at_epoch_start()) {
339  flag = (int) (m_current_epoch % m_checkpoint_epochs == 0);
340  }
341  }
342 
343  // if our current step is evenly divisable by checkpoint steps,
344  // take a checkpoint
345  if (flag == 0 && m_checkpoint_steps > 0) {
346  flag = (int) (m_current_step % m_checkpoint_steps == 0);
347  }
348 
349  // check the clock if time-based checkpoint is enabled
350  if (flag == 0 && m_checkpoint_secs != 0.0) {
351  // have rank 0 determine whether we should checkpoint
352  // to avoid issues with clock skew, we rely on rank 0 to make decision
353  if (m_comm->am_world_master()) {
354  // get the current time
355  double current = MPI_Wtime();
356 
357  // compute time next checkpoint is due
358  double next = m_checkpoint_last + m_checkpoint_secs;
359 
360  // determine whether it's time for a checkpoint
361  flag = (current >= next);
362  }
363 
364  // get flag from rank 0
365  MPI_Bcast(&flag, 1, MPI_INT, 0, MPI_COMM_WORLD);
366  }
367 
368  return (bool)flag;
369 }

372 static bool write_latest(const char *dir, const char *name, int epoch, int train) {
373  // define filename
374  char filename[1024];
375  sprintf(filename, "%s/%s", dir, name);
376 
377  // open the file for writing
378  int fd = openwrite(filename);
379  if (fd != -1) {
380  write_uint32(fd, "epoch", (uint32_t)epoch);
381  write_uint32(fd, "train", (uint32_t)train);
382 
383  // close our file
384  closewrite(fd, filename);
385  }
386 
387  return true;

391 static bool read_latest(const char *dir, const char *name, int *epochLast, int *trainLast) {
392  // assume we don't have a file, we'll return -1 in that case
393  *epochLast = -1;
394  *trainLast = -1;
395 
396  // define filename
397  char filename[1024];
398  sprintf(filename, "%s/%s", dir, name);
399 
400  // open the file for reading
401  int fd = openread(filename);
402  if (fd != -1) {
403  // read epoch from file
404  uint32_t epoch;
405  read_uint32(fd, "epoch", &epoch);
406  *epochLast = (int) epoch;
407 
408  // read epoch from file
409  uint32_t train;
410  read_uint32(fd, "train", &train);
411  *trainLast = train;
412 
413  // close our file
414  closeread(fd, filename);
415  }
416 
417  return true;
418 }

427 bool model::checkpointShared() {
428  // if the checkpoint directory is not defined, bail
429  if (m_checkpoint_dir.length() == 0) {
430  return false;
431  }
432 
433  // time how long this takes
434  Timer timer;
435 
436  // get checkpoint directory
437  const char *dir = m_checkpoint_dir.c_str();
438 
439  // read current epoch and step counters from model
440  int epoch = m_current_epoch;
441  int step = m_current_step;
442 
443  // let user know we're saving a checkpoint
444  MPI_Barrier(MPI_COMM_WORLD);
445  if (m_comm->am_world_master()) {
446  timer.Start();
447  printf("Checkpoint: epoch %d step %d ...\n", epoch, step);
448  fflush(stdout);
449  }
450 
451  // create top level directory
452  //const char* dir = trainParams.ParameterDir.c_str();
453  makedir(dir);
454 
455  // create subdirectory for this epoch
456  char epochdir[1024];
457  snprintf(epochdir, sizeof(epochdir), "%s/shared.epoch.%d.step.%d", dir, epoch, step);
458 
459  // start our checkpoint
460  persist p;
461  p.open_checkpoint(epochdir);
462 
463  // call virtual function to checkpoint model state
464  this->save_to_checkpoint_shared(p);
465 
466  // close our checkpoint
467  p.close_checkpoint();
468 
469  uint64_t bytes_count = p.get_bytes();
470 
471  // write epoch number to current file, we do this at the end so as to only update
472  // this file when we know we have a new valid checkpoint
473  if (m_comm->am_world_master()) {
474  write_latest(dir, "shared.last", epoch, step);
475  }
476 
477  // stop timer and report cost
478  MPI_Barrier(MPI_COMM_WORLD);
479  if (m_comm->am_world_master()) {
480  double secs = timer.Stop();
481  double bw = 0.0;
482  if (secs > 0.0) {
483  bw = ((double) bytes_count) / (secs * 1024.0 * 1024.0);
484  }
485  printf("Checkpoint complete: Epoch=%d Step=%d (%f secs, %llu bytes, %f MB/sec)\n",
486  epoch, step, secs, (unsigned long long) bytes_count, bw
487  );
488  fflush(stdout);
489  }
490 
491  // saved a checkpoint, update our last checkpoint time
492  m_checkpoint_last = MPI_Wtime();
493 
494  return true;
495 }

499  if (m_checkpoint_dir.length() == 0) {
500  return false;
501  }
502 
503  // get top level directory
504  const char *dir = m_checkpoint_dir.c_str();
505 
506  // read epoch number from current file
507  int epoch, step;
508  if (m_comm->am_world_master()) {
