113 void lbann::NetworkParams::parse_network_string(void) {
114  Network.clear();
115  const std::string delim = ",";
116  size_t start = 0U;
117  size_t end = NetworkStr.find(delim);
118  while(end != std::string::npos) {
119  Network.push_back(std::stoi(NetworkStr.substr(start, end - start), nullptr, 0));
120  start = end + delim.length();
121  end = NetworkStr.find(delim, start);
122  }
123  Network.push_back(std::stoi(NetworkStr.substr(start, end), nullptr, 0));
124 }
