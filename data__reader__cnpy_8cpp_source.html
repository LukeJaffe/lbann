44 bool cnpy_reader::fetch_datum(Mat& X, int data_id, int mb_idx, int tid) {
45  if (m_data.word_size == 4) {
46  float *tmp = (float *)m_data.data;
47  float *data = tmp + data_id;
48  for (int j=0; j<m_num_features; j++) {
49  X.Set(j, mb_idx, data[j]);
50  }
51  } else if (m_data.word_size == 8) {
52  double *tmp = (double *)m_data.data;
53  double *data = tmp + data_id;
54  for (int j=0; j<m_num_features; j++) {
55  X.Set(j, mb_idx, data[j]);
56  }
57  } else {
58  throw lbann_exception(
59  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
60  " unknown word size: " + std::to_string(m_data.word_size) +
61  " we only support 4 (float) or 8 (double)");
62  }
63 
64  return true;
65 }
