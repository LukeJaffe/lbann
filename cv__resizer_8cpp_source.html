108 bool cv_resizer::determine_initial_roi(const cv::Mat& image) {
109  if ((m_roi_sz.first > 0) && (m_roi_sz.second > 0)) {
110  // do initial crop
111  const int cx_roi = static_cast<int>(image.cols * m_center.first);
112  const int cy_roi = static_cast<int>(image.rows * m_center.second);
113 
114  int roi_width = m_roi_sz.first;
115  int roi_height = m_roi_sz.second;
116 
117  int x0 = cx_roi - roi_width/2;
118  int y0 = cy_roi - roi_height/2;
119 
120  if (x0 < 0) {
121  roi_width += x0;
122  x0 = 0;
123  }
124 
125  if (y0 < 0) {
126  roi_height += y0;
127  y0 = 0;
128  }
129 
130  if (x0+roi_width > image.cols) {
131  roi_width = image.cols - x0;
132  }
133 
134  if (y0+roi_height > image.rows) {
135  roi_height = image.rows - y0;
136  }
137 
138  m_roi = cv::Rect(x0, y0, roi_width, roi_height);
139 
140  m_center_roi.x = static_cast<float>(cx_roi - x0)/roi_width;
141  m_center_roi.y = static_cast<float>(cy_roi - y0)/roi_height;
142  } else if ((m_roi_sz.first == 0) && (m_roi_sz.second == 0)) {
143  m_roi = cv::Rect(0, 0, image.cols, image.rows);
144  m_center_roi.x = m_center.first;
145  m_center_roi.y = m_center.second;
146  }
147  return true;
148 }

151 bool cv_resizer::determine_zoom_ratio(const cv::Mat& image) {
152  const unsigned int x_center = static_cast<unsigned int>(m_roi.width * m_center_roi.x);
153  const unsigned int half_width_short =
154  std::min(m_roi.width - x_center, x_center);
155 
156  const unsigned int y_center = static_cast<unsigned int>(m_roi.height * m_center_roi.y);
157  const unsigned int half_height_short =
158  std::min(m_roi.height - y_center, y_center);
159 
160  if ((half_width_short == 0u) || (half_height_short == 0u)) {
161  return false;
162  }
163 
164  m_hzoom = m_width / (2.0f * half_width_short);
165  m_vzoom = m_height / (2.0f * half_height_short);
166 
167  if (m_uniform_zoom) {
168  m_hzoom = m_vzoom = std::max(m_hzoom, m_vzoom);
169  }
170 
171  m_interpolation = cv::INTER_LINEAR;
172 
173  if ((m_hzoom < 1.0f) && (m_vzoom < 1.0f)) {
174  m_interpolation = cv::INTER_AREA; // (better for shrinking)
175  } else if ((m_hzoom > 1.0f) && (m_vzoom > 1.0f)) {
176 #if 0
177  m_interpolation = cv::INTER_CUBIC; // (slow but better)
178 #else
179  m_interpolation = cv::INTER_LINEAR; // (faster but ok)
180 #endif
181  }
182 
183  return true;
184 }

188  m_enabled = false; // unless this method is successful, stays disabled
189 
190  _LBANN_SILENT_EXCEPTION(image.empty(), "", false)
191 
192  bool ok = ((m_width > 0u) && (m_height > 0u));
193  ok = ok && check_center(image);
194  ok = ok && determine_initial_roi(image);
195  ok = ok && determine_zoom_ratio(image);

206 bool cv_resizer::apply(cv::Mat& image) {
207  m_enabled = false; // turn off as it is applied
208 
209  _LBANN_SILENT_EXCEPTION(image.empty(), "", false)
210 
211  //std::cout << "initial crop; at (" << m_roi.x << ", " << m_roi.y
212  // << ") for " << m_roi.width << " x " << m_roi.height << std::endl;
213 
214  cv::Mat image_roi = image(m_roi);
215 
216  const unsigned int zoomed_width = static_cast<unsigned int>(image_roi.cols * m_hzoom + 0.5f);
217  const unsigned int zoomed_height = static_cast<unsigned int>(image_roi.rows * m_vzoom + 0.5f);
218 
219  _LBANN_MILD_EXCEPTION((zoomed_width < m_width) || (zoomed_height < m_height),
220  "Zoom ratio is not large enough.", false);
221 
222  //std::cout << "zoom to " << zoomed_width << " x " << zoomed_height << std::endl;
223 
224  cv::Mat zoomed_image;
225  cv::resize(image_roi, zoomed_image, cv::Size(zoomed_width, zoomed_height), 0, 0, m_interpolation);
226 
227  int cx = static_cast<int>(zoomed_width * m_center_roi.x);
228  int cy = static_cast<int>(zoomed_height * m_center_roi.y);
229 
230  int px0 = cx - m_width/2;
231  int py0 = cy - m_height/2;
232  if (px0 < 0) {
233  px0 = 0;
234  }
235  if (py0 < 0) {
236  py0 = 0;
237  }
238 
239  if (static_cast<unsigned int>(px0 + m_width) > zoomed_width) {
240  px0 = zoomed_width - m_width;
241  }
242 
243  if (static_cast<unsigned int>(py0 + m_height) > zoomed_height) {
244  py0 = zoomed_height - m_height;
245  }
246 
247  _LBANN_MILD_EXCEPTION((px0 < 0) || (py0 < 0), "Failed to resize", false);
248 
249  //std::cout << "taking at (" << px0 << ", " << py0
250  // << ") for " << m_width << " x " << m_height << std::endl;
251 
252  image = zoomed_image(cv::Rect(px0, py0, m_width, m_height));
253 
254  return true;
255 }
