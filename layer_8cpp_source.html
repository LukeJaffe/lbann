244 void Layer::forward_prop() {
245  const auto fp_start = get_time();
246 
247  // Setup matrix data, e.g. input matrices
248  fp_setup_data(m_model->get_current_mini_batch_size());
249 
250  #if defined(LBANN_HAS_CUDNN) && defined(LBANN_DEBUG)
251  // Synchronize GPUs and check for errors
252  if (m_using_gpus) { this->m_cudnn->check_error(); }
253  #endif // defined(LBANN_HAS_CUDNN) && defined(LBANN_DEBUG)
254 
255  // Apply layer's compute function
256  const auto fp_compute_start = get_time();
257  fp_compute();
258  m_fp_compute_time += get_time() - fp_compute_start;
259 
260  // Add this layer as a gradient source for weight optimizers
261  for (auto&& w : m_weights) {
262  optimizer* opt = w->get_optimizer();
263  if (opt != nullptr) { opt->add_gradient_source(this); }
264  }
265 
266  #if defined(LBANN_HAS_CUDNN) && defined(LBANN_DEBUG)
267  // Synchronize GPUs and check for errors
268  if (m_using_gpus) { this->m_cudnn->check_error(); }
269  #endif // defined(LBANN_HAS_CUDNN) && defined(LBANN_DEBUG)
270 
271  m_fp_time += get_time() - fp_start;
272 }

320 void Layer::summarize_stats(lbann_summary& summarizer, int step) {
321  std::string prefix = m_name + "/";
322  summarizer.reduce_scalar(prefix + "fp_time", m_fp_time, step);
323  summarizer.reduce_scalar(prefix + "bp_time", m_bp_time, step);
324  summarizer.reduce_scalar(prefix + "update_time", m_update_time, step);
325  reset_counters();
326  // Combine the optimizer step time from all the weights.
327  double step_time = 0.0;
328  for (weights *w : get_weights()) {
329  optimizer *opt = w->get_optimizer();
330  if (opt) {
331  step_time += opt->get_step_time();
332  opt->reset_counters();
333  }
334  }
335  summarizer.reduce_scalar(prefix + "opt_time", step_time, step);
336 }

508 void Layer::setup_dims() {
509 
510  // Get dimensions of previous neuron tensor
511  if(m_parent_layers.empty()) {
512  m_prev_neuron_dims.assign(1, 0);
513  } else {
514  m_prev_neuron_dims = m_parent_layers.front()->fp_output_dims(this);
515  }
516  m_num_prev_neuron_dims = m_prev_neuron_dims.size();
517  m_num_prev_neurons = std::accumulate(m_prev_neuron_dims.begin(),
