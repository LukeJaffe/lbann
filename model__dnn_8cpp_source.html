67 void deep_neural_network::train(int num_epochs) {
68  do_train_begin_cbs();
69 
70  // Epoch main loop
71  for (int epoch = 0; epoch < num_epochs; ++epoch) {
72 
73  // Check if training has been terminated
74  if (get_terminate_training()) {
75  break;
76  }
77 
78  // due to restart, may not always be at start of epoch
79  // use mini batch index in data reader to signify start of epoch
80  if (at_epoch_start()) {
81  ++m_current_epoch;
82  do_epoch_begin_cbs();
83  }
84 
85  // Set the execution mode to training
86  m_execution_mode = execution_mode::training;
87  for (size_t l = 0; l < m_layers.size(); ++l) {
88  m_layers[l]->set_execution_mode(execution_mode::training);
89  }
90 
91  // Train on mini-batches until data set is traversed
92  // Note: The data reader shuffles the data after each epoch
93  m_obj_fn->reset_statistics();
94  for (auto&& m : m_metrics) {
95  m->reset_metric();
96  }
97  bool finished_epoch = false;
98  while (!finished_epoch) {
99  finished_epoch = train_mini_batch();
100 
101  // save a checkpoint if needed
102  if (need_checkpoint()) {
103  checkpointShared();
104  }
105  }
106 
107  if (is_execution_mode_valid(execution_mode::validation)) {
108  // Evaluate model on validation set
109  // TODO: do we need validation callbacks here?
110  // do_validation_begin_cbs();
111  evaluate(execution_mode::validation);
112  // do_validation_end_cbs();
113  }
114 
115  do_epoch_end_cbs();
116 
117  // save checkpoint after epoch
118  if (need_checkpoint()) {
119  checkpointShared();
120  }
121  }
122 
123  do_train_end_cbs();
124 }

126 bool deep_neural_network::train_mini_batch() {
127  do_batch_begin_cbs();
128 
129  // Forward propagation
130  do_model_forward_prop_begin_cbs();
131  for (size_t l = 0u; l < m_layers.size(); ++l) {
132  do_layer_forward_prop_begin_cbs(m_layers[l]);
133  m_layers[l]->forward_prop();
134  do_layer_forward_prop_end_cbs(m_layers[l]);
135  }
136  do_model_forward_prop_end_cbs();
137 
138  // Record and reset objective function value
139  m_obj_fn->record_and_reset_value();
140 
141  // Backward propagation
142  do_model_backward_prop_begin_cbs();
143  for (size_t l = m_layers.size(); l-- > 0u;) {
144  do_layer_backward_prop_begin_cbs(m_layers[l]);
145  m_layers[l]->back_prop();
146  do_layer_backward_prop_end_cbs(m_layers[l]);
147  }
148  do_model_backward_prop_end_cbs();
149 
151  for (size_t l = m_layers.size() - 1; l > 0u; --l) {
152  m_layers[l]->update();
153  }
154  const bool data_set_processed = m_layers[0]->update();
155 
156  do_batch_end_cbs();
157  ++m_current_step; // Update the current step once the entire mini-batch is complete
158  return data_set_processed;
159 }

161 void deep_neural_network::evaluate(execution_mode mode) {
162  switch(mode) {
163  case execution_mode::validation:
164  do_validation_begin_cbs();
165  break;
166  case execution_mode::testing:
167  do_test_begin_cbs();
168  break;
169  default:
170  throw lbann_exception("Illegal execution mode in evaluate function");
171  }
172 
173  // Set the execution mode
174  m_execution_mode = mode;
175  for (size_t l = 0u; l < m_layers.size(); ++l) {
176  m_layers[l]->set_execution_mode(mode);
177  }
178 
179  // Evaluate on mini-batches until data set is traversed
180  // Note: The data reader shuffles the data after each epoch
181  m_obj_fn->reset_statistics();
182  for (auto&& m : m_metrics) {
183  m->reset_metric();
184  }
185  bool finished_epoch = false;
186  while (!finished_epoch) {
187  finished_epoch = evaluate_mini_batch();
188  }
189 
190  switch(mode) {
191  case execution_mode::validation:
192  do_validation_end_cbs();
193  break;
194  case execution_mode::testing:
195  do_test_end_cbs();
196  break;
197  default:
198  throw lbann_exception("Illegal execution mode in evaluate function");
199  }
200 
201  return;
202 }

204 bool deep_neural_network::evaluate_mini_batch() {
205  do_batch_evaluate_begin_cbs();
206 
207  // forward propagation (mini-batch)
208  do_model_evaluate_forward_prop_begin_cbs();
209  for (size_t l = 0u; l < m_layers.size(); l++) {
210  do_layer_evaluate_forward_prop_begin_cbs(m_layers[l]);
211  m_layers[l]->forward_prop();
212  do_layer_evaluate_forward_prop_end_cbs(m_layers[l]);
213  }
214  do_model_evaluate_forward_prop_end_cbs();
215 
216  // Record and reset objective function value
217  m_obj_fn->record_and_reset_value();
218 
219  // Update layers
220  // Note: should only affect the input and target layers
221  for (size_t l = m_layers.size() - 1; l > 0u; --l) {
222  m_layers[l]->update();
223  }
224  const bool data_set_processed = m_layers[0]->update();
225  do_batch_evaluate_end_cbs();
226  switch(m_execution_mode) {
227  case execution_mode::validation:
228  ++m_current_validation_step;
229  break;
230  case execution_mode::testing:
231  ++m_current_testing_step;
232  break;
233  default:
234  throw lbann_exception("Illegal execution mode in evaluate mini-batch function");
235  }
236  return data_set_processed;
