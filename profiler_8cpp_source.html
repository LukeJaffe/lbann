106 int lbann_callback_profiler::get_color(Layer *l) {
107  const std::string &lname = l->get_name();
108  int idx = 4;
109  if (lname == "fully_connected") {
110  idx = 5;
111  } else if (lname == "convolution") {
112  idx = 6;
113  } else if (lname == "pooling_layer") {
114  idx = 7;
115  } else if (lname == "input_layer_distributed_minibatch_parallel_io") {
116  idx = 8;
117  }
118  return colors[idx % num_colors];
119 }
