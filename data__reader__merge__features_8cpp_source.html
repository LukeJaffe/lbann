81 void data_reader_merge_features::load() {
82  // Load each data reader separately.
83  for (auto&& reader : m_data_readers) {
84  reader->load();
85  m_data_size += reader->get_linearized_data_size();
86  }
87  // Verify the readers have the same number of samples.
88  int num_samples = m_data_readers[0]->get_num_data();
89  for (auto&& reader : m_data_readers) {
90  if (num_samples != reader->get_num_data()) {
91  throw lbann_exception(
92  "data_reader_merge_features: data readers do not have the same amount of data");
93  }
94  }
95  m_label_reader->load();
96  // Reset indices.
97  m_shuffled_indices.resize(num_samples);
98  std::iota(m_shuffled_indices.begin(), m_shuffled_indices.end(), 0);
99  select_subset_of_data();
100 }
