84  void set_original_layer(Layer *original_layer) {

88  void setup_dims() override {
89  generic_target_layer::setup_dims();
90  this->m_neuron_dims = m_original_layer->get_neuron_dims();
91  this->m_num_neuron_dims = m_original_layer->get_num_neuron_dims();
92  this->m_num_neurons = m_original_layer->get_num_neurons();
93  if(this->m_num_neurons != this->m_num_prev_neurons) {
94  throw lbann_exception("reconstruction_layer: original layer ("
95  + std::to_string(this->m_num_neurons)

97  + std::to_string(this->m_num_prev_neurons)
98  +") do not have the same number of neurons");
99  }
100  }

102  protected:
103  void fp_set_std_matrix_view() override {
104  int64_t cur_mini_batch_size = this->m_model->get_current_mini_batch_size();
105 
106  generic_target_layer::fp_set_std_matrix_view();
107 
108  //view of original layer
109  AbsDistMat& orig_acts = m_original_layer->get_activations();
110  original_layer_act_v = orig_acts.Construct(orig_acts.Grid(),orig_acts.Root());
111  El::View(*original_layer_act_v, orig_acts, El::ALL, El::IR(0, cur_mini_batch_size));
112  }

114  void fp_compute() override {
115 
116  //Copy prev (decoder) activations for greedy layer wise training
117  El::Copy(*original_layer_act_v,*this->m_activations_v);

119  }

132  void summarize_stats(lbann_summary& summarizer, int step) override {
133  std::string tag = this->m_name + "/ReconstructionCost";
134  execution_mode mode = this->m_model->get_execution_mode();
135  summarizer.reduce_scalar(tag, this->m_model->get_objective_function()->get_mean_value(mode), step);
136  // Skip target layer (for now).

138  }
