91  void fp_compute() {
92  //Copy prev (decoder) activations for greedy layer wise training
93  El::Copy(*this->m_prev_activations_v,*this->m_activations);
94  // Compute cost will be sum of squared error of fp_input (linearly transformed to m_activations)
95  // and original layer fp_input/original input
