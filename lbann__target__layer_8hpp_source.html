103  void epoch_print() const {
104  double obj_cost = this->m_neural_network_model->m_obj_fn->report_aggregate_avg_obj_fn(execution_mode::training);
105  if (this->m_comm->am_world_master()) {
106  std::vector<double> avg_obj_fn_costs(this->m_comm->get_num_models());
107  this->m_comm->intermodel_gather(obj_cost, avg_obj_fn_costs);
108  for (size_t i = 0; i < avg_obj_fn_costs.size(); ++i) {
109  std::cout << "Model " << i << " average " << _to_string(this->m_neural_network_model->m_obj_fn->type) << ": " << avg_obj_fn_costs[i] <<
110  std::endl;
111  }
112  } else {
113  this->m_comm->intermodel_gather(obj_cost, this->m_comm->get_world_master());
114  }
