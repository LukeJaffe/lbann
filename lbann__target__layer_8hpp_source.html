105  void epoch_print() const {

107  if (this->m_comm->am_world_master()) {
108  std::vector<double> avg_obj_fn_costs(this->m_comm->get_num_models());
109  this->m_comm->intermodel_gather(obj_cost, avg_obj_fn_costs);

111  std::cout << "Model " << i << " average " << _to_string(this->m_neural_network_model->m_obj_fn->type) << ": " << avg_obj_fn_costs[i] <<
112  std::endl;
113  }
114  } else {
115  this->m_comm->intermodel_gather(obj_cost, this->m_comm->get_world_master());
116  }
