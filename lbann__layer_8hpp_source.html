100 static layer_category __attribute__((used)) _layer_type_to_category(layer_type l) {
101  switch(l) {
102  case layer_type::fully_connected:
103  case layer_type::softmax:
104  case layer_type::convolution:
105  case layer_type::pooling:
106  case layer_type::local_response_normalization:
107  return layer_category::compute;
108  case layer_type::input_distributed_minibatch:
109  case layer_type::input_distributed_minibatch_parallel_io:
110  case layer_type::input_partitioned_minibatch_parallel_io:
111  case layer_type::target_distributed_minibatch:
112  case layer_type::target_distributed_minibatch_parallel_io:
113  case layer_type::target_partitioned_minibatch_parallel_io:
114  return layer_category::io;
115  case layer_type::reconstruction:
116  return layer_category::SPECIAL;
117  case layer_type::INVALID:
118  return layer_category::INVALID;
119  default:
120  throw(std::string{} + __FILE__ + " " + std::to_string(__LINE__) + " Invalid layer_type specified");
121  }
122  return layer_category::INVALID;
