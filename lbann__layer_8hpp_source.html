68  case layer_type::convolution:
69  return "convolution";
70  case layer_type::pooling:
71  return "pooling";
72  case layer_type::local_response_normalization:
73  return "local_response_normalization";

75  return "input_distributed_minibatch";
76  case layer_type::input_distributed_minibatch_parallel_io:
77  return "input_distributed_minibatch_parallel_io";

79  return "target_distributed_minibatch";

81  return "target_distributed_minibatch_parallel_io";

85  return "INVALID";
