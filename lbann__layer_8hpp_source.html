89 static layer_category __attribute__((used)) _layer_type_to_category(layer_type l) {
90  switch(l) {
91  case layer_type::fully_connected:
92  case layer_type::convolution:
93  return layer_category::learning;
94  case layer_type::softmax:
95  case layer_type::activation:
96  return layer_category::activation;
97  case layer_type::pooling:
98  return layer_category::transform;
99  case layer_type::local_response_normalization:
100  case layer_type::dropout:
101  case layer_type::batch_normalization:
102  return layer_category::regularizer;
103  case layer_type::input_distributed_minibatch:
104  case layer_type::input_distributed_minibatch_parallel_io:
105  case layer_type::input_partitioned_minibatch_parallel_io:
106  case layer_type::target_distributed_minibatch:
107  case layer_type::target_distributed_minibatch_parallel_io:
108  case layer_type::target_partitioned_minibatch_parallel_io:
109  return layer_category::io;
110  case layer_type::reconstruction:
111  return layer_category::special;
112  case layer_type::INVALID:
113  return layer_category::invalid;
114  default:
115  throw(std::string{} + __FILE__ + " " + std::to_string(__LINE__) + " Invalid layer_type specified");
116  }
117  return layer_category::invalid;
118 }
