77 bool data_reader_nci::fetch_datum(Mat& X, int data_id, int mb_idx, int tid) {
78  string line;
79  std::getline(m_ifs[tid]->seekg(m_index_map[data_id]),line);
80  std::istringstream lstream(line);
81  std::string field;
82  int col = 0, f=0;
83 
84  while(getline(lstream, field, ' ')) {
85  col++;
86  //if(col == 4) m_labels[data_id] = this->map_label_2int(field);
87  if (col == 3) {
88  if (field.empty()) {
89  break;
90  } else {
91  m_responses[data_id] = static_cast<DataType>(stof(field));
92  }
93  }else if(col == 4) {
94  m_labels[data_id] = this->map_label_2int(field);
95  }else if (col > 5) {
96  if(field.empty()) {
97  field = "0"; //set empty feature field (unit) to zero
98  }
99  X.Set(f,mb_idx,stof(field));
100  f++;
101  }//end if col > 5
102  }// end while loop
103  auto data_col = X(El::ALL, El::IR(mb_idx, mb_idx+1));
104  normalize(data_col, 1);
105  return true;
106 }

128 void data_reader_nci::load() {
129  std::string infile = get_data_filename();
130  std::ifstream ifs(infile.c_str());
131  if (!ifs) {
132  throw lbann_exception(
133  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
134  " :: data_reader_nci::load(): can't open file: " + infile);
135  }
136  m_infile = infile;
137  std::string line;
138  int i;
139 
140  m_index_map.push_back(ifs.tellg());
141 #ifdef NCI_HAS_HEADER
142  // Skip the first line of the file, which is a header
143  std::getline (ifs, line); // to skip the header
144  m_index_map.pop_back();
145  m_index_map.push_back(ifs.tellg());
146 #endif
147 
148  while(std::getline (ifs, line) ) {
149 #if 0 // enable to skip empty or comment lines
150  static const std::string whitespaces(" \t\f\v\n\r");
151  std::size_t pos = line.find_first_not_of(whitespaces);
152  if ((pos == std::string::npos) || (line[pos] == '#')) {
153  m_index_map.pop_back();
154  m_index_map.push_back(ifs.tellg());
155  continue;
156  }
157 #endif
158 
159  m_index_map.push_back(ifs.tellg()); // The beginning of the next data to read
160  m_num_samples++;
161 
162  std::string field;
163  std::istringstream lstream(line);
164  i=0;
165  m_num_features = 0;
166  while(std::getline(lstream, field, ' ')) {
167  i++;
168  if (i > 5) {
169  m_num_features++;
170  }
171  }
172  }
173  ifs.close();
174  m_index_map.pop_back();
175  m_index_map.shrink_to_fit();
176  m_responses.resize(m_num_samples);
177  m_labels.resize(m_num_samples);
178  // reset indices
179  m_shuffled_indices.clear();
180  m_shuffled_indices.resize(m_num_samples);
181  for (size_t n = 0; n < m_shuffled_indices.size(); ++n) {
182  m_shuffled_indices[n] = n;
183  }
184 
185  select_subset_of_data();
186 }
