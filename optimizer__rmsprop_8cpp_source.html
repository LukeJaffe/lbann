93 void rmsprop::update(const AbsDistMat *gradient) {
94  // Get local matrix data
95  const int local_height = m_parameters->LocalHeight();
96  const int local_width = m_parameters->LocalWidth();
97  DataType *parameters_buffer = m_parameters->Buffer();
98  const int parameters_ldim = m_parameters->LDim();
99  const DataType *gradient_buffer = gradient->LockedBuffer();
100  const int gradient_ldim = gradient->LDim();
101  DataType *cache_buffer = m_cache->Buffer();
102  const int cache_ldim = m_cache->LDim();
103 
104  // Check if matrix data is contiguous
105  if(parameters_ldim != local_height
106  || gradient_ldim != local_height
107  || cache_ldim != local_height) {
108  // Update with non-contiguous data
109  #pragma omp parallel for collapse(2)
110  for(int j=0; j<local_width; ++j) {
111  for(int i=0; i<local_height; ++i) {
112  DataType& x = parameters_buffer[i+j*parameters_ldim];
113  const DataType g = gradient_buffer[i+j*gradient_ldim];
114  DataType& c = cache_buffer[i+j*cache_ldim];
115  c = m_decay_rate * c + (DataType(1) - m_decay_rate) * g * g;
116  x -= m_learning_rate * g / (Sqrt(c) + m_eps);
117  }
118  }
119  } else {
120  // Update with contiguous data
121  #pragma omp parallel for
122  for(int i=0; i<local_height*local_width; ++i) {
123  DataType& x = parameters_buffer[i];
124  const DataType g = gradient_buffer[i];
125  DataType& c = cache_buffer[i];
126  c = m_decay_rate * c + (DataType(1) - m_decay_rate) * g * g;
127  x -= m_learning_rate * g / (Sqrt(c) + m_eps);
128  }
129  }
130 }
