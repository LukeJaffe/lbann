33 void im2col(const Mat& im,
34  Mat& col,
35  const std::vector<Int>& im_dims,
36  const std::vector<Int>& im_pads,
37  Int num_im_channels,
38  const std::vector<Int>& window_dims,
39  const std::vector<Int>& window_strides) {
40 
41  // Input and output parameters
42  const int num_channels = num_im_channels;
43  const int num_dims = im_dims.size();
44  const int output_height = col.Height();
45  const int output_width = col.Width();
46  const DataType *__restrict__ input_buffer = im.LockedBuffer();
47  DataType *__restrict__ output_buffer = col.Buffer();
48 
49  // Call optimized routine if data is 2D
50  if(num_dims == 2) {
51  im2col_2d(input_buffer, output_buffer,
52  im_dims[1], im_dims[0], im_pads[1], im_pads[0], num_channels,
53  window_dims[1], window_dims[0],
54  window_strides[1], window_strides[0]);
55  return;
56  }
57 
58  // im2col parameters
59  std::vector<int> input_dim(num_dims);
60  std::vector<int> window_dim(num_dims);
61  std::vector<int> offset_start(num_dims);
62  std::vector<int> offset_end(num_dims);
63  std::vector<int> offset_stride(num_dims);
64  std::vector<int> offset_num(num_dims);
65  for(int d = 0; d < num_dims; ++d) {
66  input_dim[d] = im_dims[d];
67  window_dim[d] = window_dims[d];
68  offset_start[d] = -im_pads[d];
69  offset_end[d] = im_dims[d] + im_pads[d] - window_dims[d] + 1;
70  offset_stride[d] = window_strides[d];
71  offset_num[d] = (offset_end[d] - offset_start[d] + offset_stride[d] - 1) / offset_stride[d];
72  }
73 
74  // Iterate through output matrix columns
75  #pragma omp parallel for
76  for(int output_col = 0; output_col < output_width; ++output_col) {
77 
78  // Initialize arrays
79  std::vector<int> offset_pos(num_dims);
80  std::vector<int> window_pos(num_dims);
81 
82  // Get position of current offset
83  int output_col_remainder = output_col;
84  for(int d = num_dims-1; d >= 0; --d) {
85  const int offset = output_col_remainder % offset_num[d];
86  offset_pos[d] = offset_start[d] + offset * offset_stride[d];
87  output_col_remainder /= offset_num[d];
88  }
89 
90  // Iterate through output matrix entries
91  for(int output_row = 0; output_row < output_height; ++output_row) {
92  const int output_index = output_row + output_col * output_height;
93 
94  // Get position in window and channel
95  int output_row_remainder = output_row;
96  for(int d = num_dims-1; d >= 0; --d) {
97  window_pos[d] = output_row_remainder % window_dim[d];
98  output_row_remainder /= window_dim[d];
99  }
100  const int channel = output_row_remainder;
101 
102  // Get input matrix entry
103  bool input_pos_valid = true;
104  int input_index = channel;
105  for(int d = 0; d < num_dims; ++d) {
106  const int input_pos = offset_pos[d] + window_pos[d];
107  input_pos_valid = input_pos_valid && 0 <= input_pos && input_pos < input_dim[d];
108  input_index = input_pos + input_index * input_dim[d];
109  }
110 
111  // Copy input matrix entry to output matrix if valid
112  output_buffer[output_index]
113  = input_pos_valid ? input_buffer[input_index] : DataType(0);
114 
115  }
116  }

120 void col2im(const Mat& col,
121  Mat& im,
122  const std::vector<Int>& im_dims,
123  const std::vector<Int>& im_pads,
124  Int num_im_channels,
125  const std::vector<Int>& window_dims,
126  const std::vector<Int>& window_strides) {
127 
128  // Input and output parameters
129  const int num_channels = num_im_channels;
130  const int num_dims = im_dims.size();
131  const int input_height = col.Height();
132  const int input_width = col.Width();
133  const int output_size = im.Height() * im.Width();
134  const DataType *__restrict__ input_buffer = col.LockedBuffer();
135  DataType *__restrict__ output_buffer = im.Buffer();
136 
137  // Call optimized routine if data is 2D
138  if(num_dims == 2) {
139  col2im_2d(input_buffer, output_buffer,
140  im_dims[1], im_dims[0], im_pads[1], im_pads[0], num_channels,
141  window_dims[1], window_dims[0],
142  window_strides[1], window_strides[0]);
143  return;
144  }
145 
146  // col2im parameters
147  std::vector<int> output_dim(num_dims);
148  std::vector<int> window_dim(num_dims);
149  std::vector<int> offset_start(num_dims);
150  std::vector<int> offset_end(num_dims);
151  std::vector<int> offset_stride(num_dims);
152  std::vector<int> offset_num(num_dims);
153  for(int d = 0; d < num_dims; ++d) {
154  output_dim[d] = im_dims[d];
155  window_dim[d] = window_dims[d];
156  offset_start[d] = -im_pads[d];
157  offset_end[d] = im_dims[d] + im_pads[d] - window_dims[d] + 1;
158  offset_stride[d] = window_strides[d];
159  offset_num[d] = (offset_end[d] - offset_start[d] + offset_stride[d] - 1) / offset_stride[d];
160  }
161 
162  // Iterate through output entries
163  #pragma omp parallel for
164  for(int output_index = 0; output_index < output_size; ++output_index) {
165 
166  // Initialize arrays
167  std::vector<int> output_pos(num_dims);
168  std::vector<int> first_offset(num_dims);
169  std::vector<int> last_offset(num_dims);
170  std::vector<int> offset(num_dims);
171 
172  // Get position of output entry
173  int output_index_remainder = output_index;
174  for(int d = num_dims-1; d >= 0; --d) {
175  output_pos[d] = output_index_remainder % output_dim[d];
176  output_index_remainder /= output_dim[d];
177  }
178  const int channel = output_index_remainder;
179 
180  // Initialize output entry
181  DataType output_entry = 0;
182 
183  // Get window offsets containing output entry
184  for(int d = 0; d < num_dims; ++d) {
185  first_offset[d] = (output_pos[d] - offset_start[d] - window_dim[d]) / offset_stride[d] + 1;
186  first_offset[d] = Max(first_offset[d], 0);
187  last_offset[d] = (output_pos[d] - offset_start[d]) / offset_stride[d];
188  last_offset[d] = Min(last_offset[d], offset_num[d] - 1);
189  offset[d] = first_offset[d];
190  }
191 
192  // Iterate through window offsets containing output entry
193  while(offset[0] <= last_offset[0]) {
194 
195  // Get input entry corresponding to input entry
196  int input_col = 0;
197  int input_row = channel;
198  for(int d = 0; d < num_dims; ++d) {
199  const int window_pos = output_pos[d] - (offset_start[d] + offset[d] * offset_stride[d]);
200  input_col = offset[d] + input_col * offset_num[d];
201  input_row = window_pos + input_row * window_dim[d];
202  }
203  const int input_index = input_row + input_col * input_height;
204 
205  // Add input entry to output entry
206  output_entry += input_buffer[input_index];
207 
208  // Move to next window offset
209  ++offset[num_dims-1];
210  for(int d = num_dims-1; d >= 1; --d) {
211  if(offset[d] > last_offset[d]) {
212  offset[d] = first_offset[d];
213  ++offset[d-1];
214  }
215  }
216 
217  }
218 
219  // Update output entry
220  output_buffer[output_index] = output_entry;
221 
222  }

226 void im2col_2d(const DataType *__restrict__ input_buffer,
227  DataType *__restrict__ output_buffer,
228  int input_dim_x,
229  int input_dim_y,
230  int input_pad_x,
231  int input_pad_y,
232  int num_channels,
233  int window_dim_x,
234  int window_dim_y,
235  int offset_stride_x,
236  int offset_stride_y) {
237 
238  // im2col parameters
239  const int offset_start_x = -input_pad_x;
240  const int offset_start_y = -input_pad_y;
241  const int offset_end_x = input_dim_x + input_pad_x - window_dim_x + 1;
242  const int offset_end_y = input_dim_y + input_pad_y - window_dim_y + 1;
243  const int offset_num_x = (offset_end_x - offset_start_x + offset_stride_x - 1) / offset_stride_x;
244  const int offset_num_y = (offset_end_y - offset_start_y + offset_stride_y - 1) / offset_stride_y;
245  const int output_height = num_channels * window_dim_x * window_dim_y;
246  const int output_width = offset_num_x * offset_num_y;
247 
248  // Iterate through output matrix entries
249  #pragma omp parallel for collapse(5)
250  for(int offset_y = 0; offset_y < offset_num_y; ++offset_y) {
251  for(int offset_x = 0; offset_x < offset_num_x; ++offset_x) {
252  for(int channel = 0; channel < num_channels; ++channel) {
253  for(int window_pos_y = 0;
254  window_pos_y < window_dim_y;
255  ++window_pos_y) {
256  for(int window_pos_x = 0;
257  window_pos_x < window_dim_x;
258  ++window_pos_x) {
259 
260  // Get input entry
261  const int offset_pos_y = offset_start_y + offset_y * offset_stride_y;
262  const int offset_pos_x = offset_start_x + offset_x * offset_stride_x;
263  const int input_pos_y = offset_pos_y + window_pos_y;
264  const int input_pos_x = offset_pos_x + window_pos_x;
265  const int input_index = (input_pos_x
266  + input_pos_y * input_dim_x
267  + channel * input_dim_x * input_dim_y);
268  const bool input_pos_valid = (0 <= input_pos_y
269  && input_pos_y < input_dim_y
270  && 0 <= input_pos_x
271  && input_pos_x < input_dim_x);
272 
273  // Get output entry
274  const int output_row = (window_pos_x
275  + window_pos_y * window_dim_x
276  + channel * window_dim_x * window_dim_y);
277  const int output_col = offset_x + offset_y * offset_num_x;
278  const int output_index = output_row + output_col * output_height;
279 
280  // Copy input entry to output entry if valid
281  output_buffer[output_index]
282  = input_pos_valid ? input_buffer[input_index] : DataType(0);

292 void col2im_2d(const DataType *__restrict__ input_buffer,
293  DataType *__restrict__ output_buffer,
294  int output_dim_x,
295  int output_dim_y,
296  int output_pad_x,
297  int output_pad_y,
298  int num_channels,
299  int window_dim_x,
300  int window_dim_y,
301  int offset_stride_x,
302  int offset_stride_y) {
303 
304  // col2im parameters
305  const int offset_start_x = -output_pad_x;
306  const int offset_start_y = -output_pad_y;
307  const int offset_end_x = output_dim_x + output_pad_x - window_dim_x + 1;
308  const int offset_end_y = output_dim_y + output_pad_y - window_dim_y + 1;
309  const int offset_num_x = (offset_end_x - offset_start_x + offset_stride_x - 1) / offset_stride_x;
310  const int offset_num_y = (offset_end_y - offset_start_y + offset_stride_y - 1) / offset_stride_y;
311  const int input_height = num_channels * window_dim_x * window_dim_y;
312  const int input_width = offset_num_x * offset_num_y;
313  const int output_size = num_channels * output_dim_x * output_dim_y;
314 
315  // Iterate through output entries
316  #pragma omp parallel for collapse(3)
317  for(int channel = 0; channel < num_channels; ++channel) {
318  for(int output_pos_y = 0;
319  output_pos_y < output_dim_y;
320  ++output_pos_y) {
321  for(int output_pos_x = 0;
322  output_pos_x < output_dim_x;
323  ++output_pos_x) {
324 
325  // Get output entry
326  const int output_index = (output_pos_x
327  + output_pos_y * output_dim_x
328  + channel * output_dim_x * output_dim_y);
329  DataType output_entry = 0;
330 
331  // Get window offsets containing output entry
332  const int offset_x_lower = (output_pos_x - offset_start_x - window_dim_x) / offset_stride_x + 1;
333  const int offset_y_lower = (output_pos_y - offset_start_y - window_dim_y) / offset_stride_y + 1;
334  const int offset_x_upper = (output_pos_x - offset_start_x) / offset_stride_x;
335  const int offset_y_upper = (output_pos_y - offset_start_y) / offset_stride_y;
336  const int first_offset_x = Max(offset_x_lower, 0);
337  const int first_offset_y = Max(offset_y_lower, 0);
338  const int last_offset_x = Min(offset_x_upper, offset_num_x - 1);
339  const int last_offset_y = Min(offset_y_upper, offset_num_y - 1);
340 
341  // Iterate through window offsets
342  for(int offset_y = first_offset_y;
343  offset_y <= last_offset_y;
344  ++offset_y) {
345  const int window_pos_y = output_pos_y - (offset_start_y + offset_y * offset_stride_y);
346  for(int offset_x = first_offset_x;
347  offset_x <= last_offset_x;
348  ++offset_x) {
349  const int window_pos_x = output_pos_x - (offset_start_x + offset_x * offset_stride_x);
350 
351  // Get input entry
352  const int input_row = (window_pos_x
353  + window_pos_y * window_dim_x
354  + channel * window_dim_x * window_dim_y);
355  const int input_col = offset_x + offset_y * offset_num_x;
356  const int input_index = input_row + input_col * input_height;
357 
358  // Add input entry to output entry
359  output_entry += input_buffer[input_index];
360 
361  }
362  }
363 
364  // Update output entry
365  output_buffer[output_index] = output_entry;
366 
367  }
368  }
369  }
