72 void lbann_callback_dump_minibatch_sample_indices::on_evaluate_forward_prop_end(model *m, Layer *l) {
73  switch(m->get_execution_mode()) {
74  case execution_mode::validation:
75  dump_to_file(m, l, m->get_cur_validation_step());
76  break;
77  case execution_mode::testing:
78  dump_to_file(m, l, m->get_cur_testing_step());
79  break;
80  default:
81  throw lbann_exception("lbann_callback_dump_minibatch_sample_indices: invalid execution phase");
82  }
