143 void lbann::local_response_normalization_layer::setup_gpu() {
144 #ifndef __LIB_CUDNN
145  throw lbann_exception("lbann_layer_local_response_normalization: cuDNN not detected");
146 #else
147 
148  // Initialize descriptors
149  checkCUDNN(cudnnCreateTensorDescriptor(&m_tensor_desc));
150  checkCUDNN(cudnnCreateLRNDescriptor(&m_lrn_desc));
151 
152  // Set input tensor descriptor
153  std::vector<int> dims(m_num_dims+2);
154  dims[0] = m_mini_batch_size_per_gpu;
155  dims[1] = m_num_channels;
156  for(Int i=0; i<m_num_dims; ++i) {
157  dims[i+2] = m_dims[i];
158  }
159  std::vector<int> strides(m_num_dims+2);
160  strides[m_num_dims + 1] = 1;
161  for(Int i=m_num_dims; i>=0; --i) {
162  strides[i] = strides[i+1] * dims[i+1];
163  }
164  checkCUDNN(cudnnSetTensorNdDescriptor(m_tensor_desc,
165  m_cudnn->get_cudnn_data_type(),
166  m_num_dims+2,
167  dims.data(),
168  strides.data()));
169 
170  // Set local response normalization descriptor
171  checkCUDNN(cudnnSetLRNDescriptor(m_lrn_desc,
172  (unsigned int) m_window_width,
173  (double) m_lrn_alpha,
174  (double) m_lrn_beta,
175  (double) m_lrn_k));
176 
177  // Allocate GPU memory
178  m_cudnn->allocate_on_gpus(m_weighted_sum_d,
179  NumNeurons,
180  m_mini_batch_size_per_gpu);
181  m_cudnn->allocate_on_gpus(m_activations_d,
182  NumNeurons,
183  m_mini_batch_size_per_gpu);
184  m_cudnn->allocate_on_gpus(m_error_signal_d,
185  m_num_prev_neurons,

217 void lbann::local_response_normalization_layer::fp_linearity_gpu() {
218 #ifndef __LIB_CUDNN
219  throw lbann_exception("lbann_layer_local_response_normalization: cuDNN not detected");
220 #else
221 
222  // Useful constants
223  const DataType one = 1;
224  const DataType zero = 0;
225 
226  // Perform local response normalization with each GPU
227  const Int num_gpus = m_cudnn->get_num_gpus();
228  for(Int i=0; i<num_gpus; ++i) {
229  checkCUDA(cudaSetDevice(m_cudnn->get_gpu(i)));

252 void lbann::local_response_normalization_layer::fp_linearity_cpu() {
253 
254  // Get local matrices
255  const Mat& prev_activations_local = m_prev_activations_v->LockedMatrix();
256  Mat& weighted_sum_local = m_weighted_sum_v->Matrix();
257  Mat& activations_local = m_activations_v->Matrix();
258 
259  // Input and output entries are divided amongst channels
260  const Int num_per_channel = NumNeurons / m_num_channels;
261 
263  // activations(i) = prev_activations(i) / scale_factor(i) ^ beta
264  // scale_factor(i)
265  // = k + alpha / window_width * sum( prev_activations(j) ^ 2 )
266  // Note: The sum is over entries in the normalization window.
268 
269  // Iterate through data samples in mini-batch
270  #pragma omp parallel for collapse(2)
271  for(Int sample = 0; sample < prev_activations_local.Width(); ++sample) {
272  // Iterate through positions in sample
273  for(Int pos = 0; pos < num_per_channel; ++pos) {
274 
275  // Initialize normalization window
276  Int window_start = - m_window_width / 2;
277  Int window_end = m_window_width / 2;
278  DataType window_sum = 0;
279  for(Int c = Max(window_start, 0);
280  c <= Min(window_end, m_num_channels-1);
281  ++c) {
282  const DataType x
283  = prev_activations_local.Get(pos + num_per_channel*c, sample);
284  window_sum += x * x;
285  }
286 
287  // Iterate through channels at current position
288  for(Int channel = 0; channel < m_num_channels; ++channel) {
289  const Int index = pos + num_per_channel * channel;
290 
291  // Apply local response normalization to current entry
292  const DataType input_entry = prev_activations_local.Get(index, sample);
293  const DataType scale_factor = m_lrn_k + m_lrn_alpha / m_window_width * window_sum;
294  const DataType output_entry = input_entry * Pow(scale_factor, -m_lrn_beta);
295  weighted_sum_local.Set(index, sample, output_entry);
296 
297  // Shift normalization window by one entry
298  if(window_start >= 0) {
299  const Int i = pos + num_per_channel*window_start;
300  const DataType x = prev_activations_local.Get(i, sample);
301  window_sum -= x * x;
302  }
303  ++window_start;
304  ++window_end;
305  if(window_end < m_num_channels) {
306  const Int i = pos + num_per_channel*window_end;
307  const DataType x = prev_activations_local.Get(i, sample);
308  window_sum += x * x;
309  }
310 
311  }
312 
313  }
314 
315  }
316 
317  // weighted_sum and output are identical after fp linearity step
318  Copy(weighted_sum_local, activations_local);
319 
320 }

357 void lbann::local_response_normalization_layer::bp_linearity_cpu() {
358 
359  // Get local matrices
360  const Mat& prev_activations_local = m_prev_activations_v->LockedMatrix();
361  const Mat& activations_local = m_activations_v->LockedMatrix();
362  const Mat& prev_error_signal_local = m_prev_error_signal_v->LockedMatrix();
363  Mat& error_signal_local = m_error_signal_v->Matrix();
364 
365  // Initialize error signal to zero
366  Zero(error_signal_local);
367 
368  // Input and output entries are divided amongst channels
369  const Int num_per_channel = NumNeurons / m_num_channels;
370 
372  // error_signal(i)
373  // = prev_error_signal(i) / scale_factor(i) ^ beta
374  // - 2 * alpha * beta / window_width * prev_activations(i)
375  // * sum( prev_error_signal(j) * activations(j)
376  // / scale_factor(j) )
377  // Note: See comments in fp_linearity_cpu for a definition of
378  // scale_factor. The sum is over entries in the normalization
379  // window.
381 
382  // Iterate through data samples in mini-batch
383  #pragma omp parallel for collapse(2)
384  for(Int sample = 0; sample < prev_activations_local.Width(); ++sample) {
385  // Iterate through positions in sample
386  for(Int pos = 0; pos < num_per_channel; ++pos) {
387 
388  // Initialize normalization window
389  Int window_start = - m_window_width / 2;
390  Int window_end = m_window_width / 2;
391  DataType window_sum = 0;
392  for(Int c = Max(window_start, 0);
393  c <= Min(window_end, m_num_channels-1);
394  ++c) {
395  const DataType x
396  = prev_activations_local.Get(pos + num_per_channel*c, sample);
397  window_sum += x * x;
398  }
399 
400  // Iterate through channels at current position
401  DataType error_signal_update;
402  for(Int channel = 0; channel < m_num_channels; ++channel) {
403  const Int index = pos + num_per_channel * channel;
404 
405  // Get data for current entry
406  const DataType activations_entry = activations_local.Get(index, sample);
407  const DataType prev_error_signal_entry = prev_error_signal_local.Get(index, sample);
408  const DataType scale_factor = m_lrn_k + m_lrn_alpha / m_window_width * window_sum;
409 
410  // Update current error signal entry
411  error_signal_update = prev_error_signal_entry * Pow(scale_factor, -m_lrn_beta);
412  error_signal_local.Update(index, sample, error_signal_update);
413 
414  // Update error signal entries in normalization window
415  for(Int c = Max(window_start, 0);
416  c <= Min(window_end, m_num_channels-1);
417  ++c) {
418  const Int i = pos + num_per_channel * c;
419  const DataType prev_activations_entry = prev_activations_local.Get(i, sample);
420  error_signal_update
421  = (-2 * m_lrn_alpha * m_lrn_beta / m_window_width * prev_activations_entry
422  * prev_error_signal_entry * activations_entry / scale_factor);
423  error_signal_local.Update(i, sample, error_signal_update);
424  }
425 
426  // Shift normalization window by one entry
427  if(window_start >= 0) {
428  const Int i = pos + num_per_channel*window_start;
429  const DataType x = prev_activations_local.Get(i, sample);
430  window_sum -= x * x;
431  }
432  ++window_start;
433  ++window_end;
434  if(window_end < m_num_channels) {
435  const Int i = pos + num_per_channel*window_end;
436  const DataType x = prev_activations_local.Get(i, sample);
437  window_sum += x * x;
438  }
439 
440  }
441 
442  }
443 
444  }
