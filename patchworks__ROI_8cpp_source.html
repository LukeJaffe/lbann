70 bool ROI::set_overlapping_region(const cv::Mat& img)
71 {
72  if (!is_valid() || (img.data == NULL)) return false;
73  if (m_left < 0) m_left = 0;
74  if (m_top < 0) m_top = 0;
75  if (m_right > img.cols) m_right = img.cols;
76  if (m_bottom > img.rows) m_bottom = img.rows;
77  if (m_right == undefined_coordinate) m_right = img.cols;
78  if (m_bottom == undefined_coordinate) m_bottom = img.rows;
79 
80  return true;
81 }
