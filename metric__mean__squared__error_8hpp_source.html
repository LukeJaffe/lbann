56  double compute_metric(AbsDistMat& predictions_v, AbsDistMat& groundtruth_v) {
57 
58  // Get local matrices and matrix parameters
59  const Mat& predictions_local = predictions_v.LockedMatrix();
60  const Mat& groundtruth_local = groundtruth_v.LockedMatrix();
61  const El::Int height = predictions_v.Height();
62  const El::Int width = predictions_v.Width();
63  const El::Int local_height = predictions_local.Height();
64  const El::Int local_width = predictions_local.Width();
65 
66  // Compute mean squared error
67  double mse = 0.0;
68  for(El::Int col = 0; col < local_width; ++col) {
69  for(El::Int row = 0; row < local_height; ++row) {
70  const double pred_val = predictions_local(row, col);
71  const double true_val = groundtruth_local(row, col);
72  const double error = pred_val - true_val;
73  mse += error * error;
74  }
75  }
76  mse /= height * width;
77  mse = El::mpi::AllReduce(mse, predictions_v.DistComm());
78  return mse;
79 
80  }
