64 int lbann::imagenet_reader_cv::fetch_data(Mat& X) {
65  if(!generic_data_reader::position_valid()) {
66  stringstream err;
67  err << __FILE__<<" "<<__LINE__<< " :: Imagenet data reader load error: !position_valid";
68  throw lbann_exception(err.str());
69  }
70 
71  const int num_channel_values = m_image_width * m_image_height * m_image_num_channels;
72  const int current_batch_size = getm_batch_size();
73  const int end_pos = Min(m_current_pos+current_batch_size, m_shuffled_indices.size());
74 
75  #pragma omp parallel for
76  for (int n = m_current_pos; n < end_pos; ++n) {
77 
78  int k = n - m_current_pos;
79  int index = m_shuffled_indices[n];
80  string imagepath = m_image_dir + image_list[index].first;
81 
82  int width=0, height=0, img_type=0;
83  ::Mat X_v;
84  View(X_v, X, IR(0, X.Height()), IR(k, k + 1));
85  cv_process pp(*m_pp);
86  bool ret = lbann::image_utils::load_image(imagepath, width, height, img_type, pp, X_v);
87  if(!ret) {
88  throw lbann_exception("ImageNet: image_utils::loadJPG failed to load");
89  }
90  if (_BUILTIN_FALSE((width * height * CV_MAT_CN(img_type)) != num_channel_values)) {
91  throw lbann_exception("ImageNet: mismatch data size -- either width or height");
92  }
93  }
94 
95  return end_pos - m_current_pos;
96 }

98 int lbann::imagenet_reader_cv::fetch_label(Mat& Y) {
99  if(!position_valid()) {
100  stringstream err;
101  err << __FILE__<<" "<<__LINE__<< " :: Imagenet data reader error: !position_valid";
102  throw lbann_exception(err.str());
103  }
104 
105  int current_batch_size = getm_batch_size();
106  int n = 0;
107  for (n = m_current_pos; n < m_current_pos + current_batch_size; n++) {
108  if (n >= (int)m_shuffled_indices.size()) {
109  break;
110  }
111 
112  int k = n - m_current_pos;
113  int index = m_shuffled_indices[n];
114  int label = image_list[index].second;
115 
116  Y.Set(label, k, 1);
117  }
118  return (n - m_current_pos);
119 }

121 void lbann::imagenet_reader_cv::load() {
122  string imageDir = get_file_dir();
123  string imageListFile = get_data_filename();
124 
125  m_image_dir = imageDir;
126  image_list.clear();
127 
128  // load image list
129  FILE *fplist = fopen(imageListFile.c_str(), "rt");
130  if (!fplist) {
131  stringstream err;
132  err << __FILE__ << " " << __LINE__ << "failed to open: " << imageListFile << endl;
133  throw lbann_exception(err.str());
134  }
135 
136  while (!feof(fplist)) {
137  char imagepath[512];
138  int imagelabel;
139  if (fscanf(fplist, "%s%d", imagepath, &imagelabel) <= 1) {
140  break;
141  }
142  image_list.push_back(make_pair(imagepath, imagelabel));
143  }
144  fclose(fplist);
145 
146  // reset indices
147  m_shuffled_indices.clear();
148  m_shuffled_indices.resize(image_list.size());
149  for (size_t n = 0; n < image_list.size(); n++) {
150  m_shuffled_indices[n] = n;
151  }
152 
153  select_subset_of_data();
154 }
