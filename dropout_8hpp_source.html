98  void fp_compute() {
99  if (this->get_execution_mode() != execution_mode::training ||
100  m_keep_prob < 0.0f) {
101  // Copy previous activations over.
102  El::Copy(*(this->m_prev_activations), *(this->m_activations));
103  return;
104  }
105  ElMat *input_acts = this->m_prev_activations;
106  const El::Int local_height = input_acts->LocalHeight();
107  const El::Int local_width = input_acts->LocalWidth();
108 
109 #ifdef LBANN_PROCDET_DROPOUT
110  const El::Int global_height = input_acts->Height();
111  bernoulli_fill_procdet(*m_cur_mask, input_acts->Height(),
112  input_acts->Width(), m_keep_prob);
113  *m_cur_mask *= DataType(1.0) / m_keep_prob;
114  if (input_acts->GlobalRow(local_height - 1) == global_height - 1) {
115  for (El::Int col = 0; col < local_width; ++col) {
116  m_cur_mask->SetLocal(local_height - 1, col, DataType(1.0));
117  }
118  }
119  El::Hadamard(*input_acts, *m_cur_mask, *(this->m_activations));
120 #else
121  Mat& local_input_acts = input_acts->Matrix();
122  Mat& local_output_acts = this->m_activations->Matrix();
123 
124  // Construct dropout mask
125  // Note: Construct Bernoulli matrix and scale by 1/m_keep_prob.
126  m_cur_mask->Resize(local_height, local_width);
127  El::EntrywiseMap(*m_cur_mask,
128  (std::function<DataType(const DataType&)>)
129  ([this](const DataType& z)->DataType {
130  auto& gen = get_fast_generator();
131  std::bernoulli_distribution dist(m_keep_prob);
132  return dist(gen) ? DataType(1) / m_keep_prob : DataType(0);
133  }));
134  // Apply dropout mask to local activations
135  El::Hadamard(local_input_acts, *m_cur_mask, local_output_acts);
136 #endif // LBANN_PROCDET_DROPOUT
137  }

140  void bp_compute() {
141  // Terminate early when not training.
142  if (this->get_execution_mode() != execution_mode::training) {
143  return;
144  }
145  if (m_keep_prob < 0.0f) {
146  // Copy error signal through.
147  El::Copy(*(this->m_prev_error_signal), *(this->m_error_signal));
148  return;
149  }
150 
151 #ifdef LBANN_PROCDET_DROPOUT
152  El::Hadamard(*(this->m_prev_error_signal), *m_cur_mask, *(this->m_error_signal));
153 #else
154  // Re-weight the incoming loss using dropout mask
155  Mat& local_prev_error_signal = this->m_prev_error_signal->Matrix();
156  Mat& local_error_signal = this->m_error_signal->Matrix();
157  El::Hadamard(local_prev_error_signal, *m_cur_mask, local_error_signal);
158 #endif // LBANN_PROCDET_DROPOUT
159  }
