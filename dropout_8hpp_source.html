96 
97  if (this->m_model->get_execution_mode() != execution_mode::training

100  El::LockedView(output, input);

102

104  const DataType scale = DataType(1) / m_keep_prob;
105  const int height = input.Height();

107  m_mask->Resize(height, width);
108 #ifdef LBANN_SEQUENTIAL_CONSISTENCY
109  bernoulli_fill_procdet(*m_mask, height, width, m_keep_prob);
110  *m_mask *= scale;
111 #else
112  El::EntrywiseMap(*m_mask,

114  ([this,scale](const DataType& z)->DataType {

118  }));
