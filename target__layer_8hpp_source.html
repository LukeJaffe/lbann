100  virtual void setup_dims() {

103  this->m_neuron_dims = io_layer::get_data_dims();
104  this->m_num_neuron_dims = this->m_neuron_dims.size();

106  this->m_neuron_dims.end(),

116  virtual void setup_data() {
117  io_layer::setup_data();
118  std::stringstream err;
119 
120  // The setup should be taken out here. Instead it should be done at the parent scope
121  if(!this->m_shared_data_reader) {

126  m_validation_dataset.m_data_reader->setup();

133  if(this->m_num_prev_neurons != this->m_num_neurons) {
134  err << __FILE__ << " " << __LINE__
135  << " :: " << get_name() << " this->m_num_prev_neurons != this->m_num_neurons; this->m_num_prev_neurons= " << this->m_num_prev_neurons << " this->m_num_neurons= " << this->m_num_neurons << endl;
136  throw lbann_exception(err.str());
137  }
138 
139  if(this->m_neural_network_model->m_obj_fn == NULL) {
140  err << __FILE__ << " " << __LINE__
141  << " :: lbann_target_layer: target layer has invalid objective function pointer";
142  throw lbann_exception(err.str());
143  }
144  for (auto&& m : this->m_neural_network_model->get_metrics()) {
145  m->setup(this->m_num_neurons,
146  this->m_neural_network_model->get_max_mini_batch_size());
147  m->m_neural_network_model = this->m_neural_network_model;
148  }

150  // Initialize objective function
151  this->m_neural_network_model->m_obj_fn->setup(*this->m_prev_layer);

173  void summarize_stats(lbann_summary& summarizer, int step) {
174  std::string obj_name = this->m_neural_network_model->m_obj_fn->name();
175  // Replace spaces with _ for consistency.
176  std::transform(obj_name.begin(), obj_name.end(), obj_name.begin(),
177  [] (char c) { return c == ' ' ? '_' : c; });
178  std::string tag = "layer" + std::to_string(this->m_index) + "/objective_" +
179  obj_name;
180  summarizer.reduce_scalar(
181  tag,
182  this->m_neural_network_model->m_obj_fn->get_mean_value(),
183  step);
184  io_layer::summarize_stats(summarizer, step);
185  }

187  void epoch_print() const {
188  double obj_cost = this->m_neural_network_model->m_obj_fn->get_mean_value();
189  if (this->m_comm->am_world_master()) {
190  std::vector<double> avg_obj_fn_costs(this->m_comm->get_num_models());
191  this->m_comm->intermodel_gather(obj_cost, avg_obj_fn_costs);
192  for (size_t i = 0; i < avg_obj_fn_costs.size(); ++i) {
193  std::cout << "Model " << i << " average " <<
194  this->m_neural_network_model->m_obj_fn->name() << ": " <<
195  avg_obj_fn_costs[i] << std::endl;
196  }
197  } else {
198  this->m_comm->intermodel_gather(obj_cost, this->m_comm->get_world_master());
199  }
200  }
