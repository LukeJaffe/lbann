53  virtual void setup_dims() {

113  void summarize_stats(lbann_summary& summarizer, int step) {
114  std::string obj_name = this->m_neural_network_model->m_obj_fn->name();
115  // Replace spaces with _ for consistency.
116  std::transform(obj_name.begin(), obj_name.end(), obj_name.begin(),
117  [] (char c) { return c == ' ' ? '_' : c; });
118  std::string tag = "layer" + std::to_string(this->m_index) + "/objective_" +
119  obj_name;
120  summarizer.reduce_scalar(
121  tag,
122  this->m_neural_network_model->m_obj_fn->report_aggregate_avg_obj_fn(
123  execution_mode::training), step);
124  io_layer::summarize_stats(summarizer, step);
125  }

127  void epoch_print() const {
128  double obj_cost = this->m_neural_network_model->m_obj_fn->report_aggregate_avg_obj_fn(execution_mode::training);
129  if (this->m_comm->am_world_master()) {
130  std::vector<double> avg_obj_fn_costs(this->m_comm->get_num_models());
131  this->m_comm->intermodel_gather(obj_cost, avg_obj_fn_costs);
132  for (size_t i = 0; i < avg_obj_fn_costs.size(); ++i) {
133  std::cout << "Model " << i << " average " <<
134  this->m_neural_network_model->m_obj_fn->name() << ": " <<
135  avg_obj_fn_costs[i] << std::endl;
136  }
137  } else {
138  this->m_comm->intermodel_gather(obj_cost, this->m_comm->get_world_master());
139  }
140  }
