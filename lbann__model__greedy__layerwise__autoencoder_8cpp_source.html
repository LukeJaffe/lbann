164 bool lbann::greedy_layerwise_autoencoder::train_mini_batch(size_t phase_index)
165 {
166  size_t phase_end = phase_index+2;
167  do_batch_begin_cbs();
168 
169  // Forward propagation
170  do_model_forward_prop_begin_cbs();
171  //@todo; optimize this? change start index from 0 to phase_index
172  for (size_t l = 0; l <= phase_end; ++l) {
173  do_layer_forward_prop_begin_cbs(m_layers[l]);
174  m_layers[l]->forwardProp();
175  do_layer_forward_prop_end_cbs(m_layers[l]);
176  }
177  do_model_forward_prop_end_cbs();
178 
179  ++m_current_step;
180 
181  // Backward propagation
182  do_model_backward_prop_begin_cbs();
183  //@todo; optimize to backprop up to phase_index and not 0
184  for (size_t l = phase_end+1; l-- > 0;) {
185  do_layer_backward_prop_begin_cbs(m_layers[l]);
186  m_layers[l]->backProp();
187  do_layer_backward_prop_end_cbs(m_layers[l]);
188  }
189  do_model_backward_prop_end_cbs();
190 
193  for (size_t l = phase_end; l > phase_index; --l) {
194  m_layers[l]->update();
195  }
196  const bool data_set_processed = m_layers[0]->update();
197 
198  do_batch_end_cbs();
199  return data_set_processed;
200 }
