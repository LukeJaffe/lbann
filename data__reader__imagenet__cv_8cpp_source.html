53 bool imagenet_reader_cv::replicate_preprocessor(const std::shared_ptr<cv_process>& pp) {
54  if (!pp) return false;
55 
56  const int nthreads = omp_get_max_threads();
57 
58  // Construct thread private preprocessing objects out of a shared pointer
59  if (std::dynamic_pointer_cast<cv_process_patches>(pp)) {
60  m_pps.clear();
61  m_ppps.resize(nthreads);
62  const std::shared_ptr<const cv_process_patches> ppp = std::dynamic_pointer_cast<cv_process_patches>(pp);
63 
64  #pragma omp parallel for schedule(static, 1)
65  for (int i = 0; i < nthreads; ++i) {
66  std::unique_ptr<cv_process_patches> ppu(new cv_process_patches(*ppp));
67  m_ppps[i] = std::move(ppu);
68  }
69 
70  for (int i = 0; i < nthreads; ++i)
71  if (!m_ppps[i]) return false;
72  } else {
73  m_pps.resize(nthreads);
74  m_ppps.clear();
75 
76  #pragma omp parallel for schedule(static, 1)
77  for (int i = 0; i < nthreads; ++i) {
78  std::unique_ptr<cv_process> ppu(new cv_process(*pp));
79  m_pps[i] = std::move(ppu);
80  }
81 
82  for (int i = 0; i < nthreads; ++i)
83  if (!m_pps[i]) return false;
84  }
85 
86  if (nthreads <= 0) {
87  std::stringstream err;
88  err << __FILE__<<" "<<__LINE__<< " :: Imagenet data reader construction error: invalid nthreads";
89  throw lbann_exception(err.str());
90  }
91 
92  return true;
93 }

196 int imagenet_reader_cv::fetch_data(std::vector<::Mat>& X) {
197  if (m_ppps.size() == 0u) {
198  // TODO: If the patch handling is separated into a child class, this check is not necessary
199  throw lbann_exception("ImageNet: imagenet_reader_cv::fetch_data invalid patch processor");
200  }
201  const int nthreads = omp_get_max_threads();
202  if(!position_valid()) {
203  throw lbann_exception(
204  std::string{} + __FILE__ + " " + std::to_string(__LINE__)
205  + " :: generic data reader load error: !position_valid"
206  + " -- current pos = " + std::to_string(m_current_pos)
207  + " and there are " + std::to_string(m_shuffled_indices.size()) + " indices");
208  }
209  if (X.size() == 0u) {
210  throw lbann_exception(
211  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
212  " :: imagenet data reader fetch error: X.size()==0");
213  }
214  const El::Int x_height = X[0].Height();
215  const El::Int x_width = X[0].Width();
216 
219  #pragma omp parallel for schedule(static, 1)
220  for (int t = 0; t < nthreads; t++) {
221  preprocess_data_source(omp_get_thread_num());
222  }
223 
224  const int current_batch_size = get_current_mini_batch_size();
225  const int end_pos = std::min(static_cast<size_t>(m_current_pos+current_batch_size),
226  m_shuffled_indices.size());
227  const int mb_size = std::min(
228  El::Int{((end_pos - m_current_pos) + m_sample_stride - 1) / m_sample_stride},
229  x_width);
230 
231  for (auto & x : X) {
232  El::Zeros(x, x_height, x_width);
233  }
234  El::Zeros(m_indices_fetched_per_mb, mb_size, 1);
235  #pragma omp parallel for schedule(static, 1)
236  for (int s = 0; s < mb_size; s++) {
237  // Catch exceptions within the OpenMP thread.
238  try {
239  const int n = m_current_pos + (s * m_sample_stride);
240  const int index = m_shuffled_indices[n];
241  fetch_datum(X, index, s, omp_get_thread_num());
242  m_indices_fetched_per_mb.Set(s, 0, index);
243  } catch (lbann_exception& e) {
244  lbann_report_exception(e);
245  } catch (exception& e) {
246  El::ReportException(e);
247  }
248  }
249 
252  #pragma omp parallel for schedule(static, 1)
253  for (int t = 0; t < nthreads; t++) {
254  postprocess_data_source(omp_get_thread_num());
255  }
256 
257  return mb_size;
258 }

266 void imagenet_reader_cv::load() {
267  //const std::string imageDir = get_file_dir();
268  const std::string imageListFile = get_data_filename();
269 
270  m_image_list.clear();
271 
272  // load image list
273  FILE *fplist = fopen(imageListFile.c_str(), "rt");
274  if (!fplist) {
275  throw lbann_exception(
276  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
277  " :: failed to open: " + imageListFile);
278  }
279 
280  while (!feof(fplist)) {
281  char imagepath[512];
282  int imagelabel;
283  if (fscanf(fplist, "%s%d", imagepath, &imagelabel) <= 1) {
284  break;
285  }
286  m_image_list.push_back(std::make_pair(imagepath, imagelabel));
287  }
288  fclose(fplist);
289 
290  // reset indices
291  m_shuffled_indices.clear();
292  m_shuffled_indices.resize(m_image_list.size());
293  std::iota(m_shuffled_indices.begin(), m_shuffled_indices.end(), 0);
294 
295  select_subset_of_data();
296 }
