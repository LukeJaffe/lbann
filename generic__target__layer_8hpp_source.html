87  void setup_dims() override {

91  this->m_num_neuron_dims = this->m_neuron_dims.size();
92  this->m_num_neurons = std::accumulate(this->m_neuron_dims.begin(),

132  void fp_compute() override {
133  execution_mode mode = this->m_model->get_execution_mode();
134  int num_samples_in_batch = io_buffer->fetch_to_local_matrix(m_paired_input_layer->get_data_reader(), mode);
135 
136  if(dynamic_cast<partitioned_io_buffer*>(io_buffer) != nullptr) {
137  update_num_samples_processed(num_samples_in_batch);
138  }else if(dynamic_cast<distributed_io_buffer*>(io_buffer) != nullptr) {
139  if(((distributed_io_buffer*) io_buffer)->is_current_root(mode)) {
141  update_num_samples_processed(num_samples_in_batch);
142  }
143 
144  int curr_mini_batch_size = this->m_model->get_current_mini_batch_size();
145  if(((distributed_io_buffer*) io_buffer)->is_current_root(mode) && num_samples_in_batch != curr_mini_batch_size) {
146  throw lbann_exception("lbann_target_layer_distributed_minibatch: number of labels ("
147  + std::to_string(num_samples_in_batch) + ") does not match the current mini-batch size ("
148  + std::to_string(curr_mini_batch_size) + ")."
149  );
150  }
152  io_buffer->distribute_from_local_matrix(*this->m_activations, m_paired_input_layer->get_data_reader(), mode);
153  }else {
154  std::stringstream err;
155  err << __FILE__ << " " << __LINE__ << " :: "
156  << "could not fp_compute for I/O layers : encoutered generic_io_buffer type";
157  throw lbann_exception(err.str());
158  }
159  }
