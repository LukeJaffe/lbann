34 void lbann_callback_checknan::on_forward_prop_end(model *m, Layer *l) {
35  // Skip output layer.
36  if (l->get_index() == (int) m->get_layers().size() - 1) {
37  return;
38  }
39  DistMat& acts = (DistMat&) l->get_activations();
40  if (!is_good(acts)) {
41  lbann_comm *comm = m->get_comm();
42  dump_network(m);
43  throw lbann_exception(
44  "checknan: [" + std::to_string(comm->get_rank_in_world()) +
45  "]: error in layer " + std::to_string(l->get_index()) +
46  " gradients (step=" + std::to_string(m->get_cur_step()) + ")");
47  }
48 }

50 void lbann_callback_checknan::on_backward_prop_end(model *m, Layer *l) {
51  // Skip non-learning layers.
52  learning *learning_layer = (learning *) dynamic_cast<learning *> (l);
53  if(learning_layer == NULL) {
54  return;
55  }
56  // Skip input/output layers.
57  if (l->get_index() == 0 || l->get_index() == (int) m->get_layers().size() - 1) {
58  return;
59  }
60  DistMat& grad = (DistMat&) learning_layer->get_weights_gradient();
61  if (!is_good(grad)) {
62  lbann_comm *comm = m->get_comm();
63  dump_network(m);
64  throw lbann_exception(
65  "checknan: [" + std::to_string(comm->get_rank_in_world()) +
66  "]: error in layer " + std::to_string(l->get_index()) +
67  " gradients (step=" + std::to_string(m->get_cur_step()) + ")");
68  }

71 void lbann_callback_checknan::on_batch_end(model *m) {
72  std::vector<Layer *>& layers = m->get_layers();
73  // Skip input/output layers-- they don't have weights.
74  for (size_t i = 1; i < layers.size() - 1; ++i) {
75  Layer *l = layers[i];
76  // Skip non-learning layers.
77  learning *learning_layer = (learning *) dynamic_cast<learning *> (l);
78  if(learning_layer == NULL) {
79  continue;
80  }
81  DistMat& weights = (DistMat&) learning_layer->get_weights();
82  if (!is_good(weights)) {
83  lbann_comm *comm = m->get_comm();
84  dump_network(m);

87  "]: error in layer " + std::to_string(l->get_index()) +

93 bool lbann_callback_checknan::is_good(const DistMat& m) {
94  const Mat& lm = m.LockedMatrix();
95  const Int height = lm.Height();
96  const Int width = lm.Width();
97  for (Int col = 0; col < width; ++col) {
98  for (Int row = 0; row < height; ++row) {
99  const DataType val = lm(row, col);
100  if (std::isnan(val)) {
101  std::cout << "Found NaN at (" << row << ", " << col << ")!" << std::endl;
102  return false;
103  } else if (std::isinf(val)) {
104  std::cout << "Found inf (" << row << ", " << col << ")!" << std::endl;
105  return false;
106  }
107  }
108  }
109  return true;
110 }

112 void lbann_callback_checknan::dump_network(model *m) {
113  std::vector<Layer*>& layers = m->get_layers();
114  // Dump only the local matrices because not every rank will necessarily
115  // have bad data, and the check is purely local.
116  const std::string prefix =
117  "model" + std::to_string(m->get_comm()->get_model_rank()) +
118  "-rank" + std::to_string(m->get_comm()->get_rank_in_model()) +
119  "-epoch" + std::to_string(m->get_cur_epoch()) + "-step" +
120  std::to_string(m->get_cur_step()) + "-layer";
121  for (size_t idx = 0; idx < layers.size(); ++idx) {
122  Layer *l = layers[idx];
123  // Dump activations.
124  El::Write(l->get_activations().Matrix(),
125  prefix + std::to_string(l->get_index()) + "-Activations",
126  El::ASCII);
127  learning *learning_layer = dynamic_cast<learning*>(l);
128  if (learning_layer != NULL) {
129  El::Write(learning_layer->get_weights().Matrix(),
130  prefix + std::to_string(l->get_index()) + "-WeightsBiases",
131  El::ASCII);
132  El::Write(learning_layer->get_weights_gradient().Matrix(),
133  prefix + std::to_string(l->get_index()) + "-Gradients",
134  El::ASCII);
135  }
136  }
137 }
