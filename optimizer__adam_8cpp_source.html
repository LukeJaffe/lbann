118 void adam::setup(AbsDistMat *parameters) {
119  optimizer::setup(parameters);
120 
121  // Initialize Adam cache
122  switch(m_matrix_format) {
123  case matrix_format::MC_MR:
124  m_moment1 = new DistMat(m_comm->get_model_grid());
125  m_moment2 = new DistMat(m_comm->get_model_grid());
126  break;
127  case matrix_format::STAR_STAR:
128  m_moment1 = new StarMat(m_comm->get_model_grid());
129  m_moment2 = new StarMat(m_comm->get_model_grid());
130  break;
131  case matrix_format::MC_STAR:
132  m_moment1 = new RowSumMat(m_comm->get_model_grid());
133  m_moment2 = new RowSumMat(m_comm->get_model_grid());
134  break;
135  case matrix_format::STAR_VC:
136  m_moment1 = new StarVCMat(m_comm->get_model_grid());
137  m_moment2 = new StarVCMat(m_comm->get_model_grid());
138  break;
139  default:
140  throw lbann_exception("lbann_optimizer_adam: invalid data layout");
141  }
142  El::Zeros(*m_moment1, m_height, m_width);
143  El::Zeros(*m_moment2, m_height, m_width);
144 }

159 void adam::update(const AbsDistMat *gradient) {
160  m_current_beta1 *= m_beta1;
161  m_current_beta2 *= m_beta2;
162  // Precompute the bias correction and learning rate.
163  const DataType correction = m_learning_rate *
164  (std::sqrt(DataType(1) - m_current_beta2)
165  / (DataType(1) - m_current_beta1));
166 
167  // Get local matrix data
168  const int local_height = m_parameters->LocalHeight();
169  const int local_width = m_parameters->LocalWidth();
170  DataType *parameters_buffer = m_parameters->Buffer();
171  const int parameters_ldim = m_parameters->LDim();
172  const DataType *gradient_buffer = gradient->LockedBuffer();
173  const int gradient_ldim = gradient->LDim();
174  DataType *moment1_buffer = m_moment1->Buffer();
175  const int moment1_ldim = m_moment1->LDim();
176  DataType *moment2_buffer = m_moment2->Buffer();
177  const int moment2_ldim = m_moment2->LDim();
178 
179  // Check if matrix data is contiguous
180  if(parameters_ldim != local_height
181  || gradient_ldim != local_height
182  || moment1_ldim != local_height
183  || moment2_ldim != local_height) {
184  // Update with non-contiguous data
185  #pragma omp parallel for collapse(2)
186  for(int j=0; j<local_width; ++j) {
187  for(int i=0; i<local_height; ++i) {
188  DataType& x = parameters_buffer[i+j*parameters_ldim];
189  // See below; avoid denormalization.
190  const DataType g = gradient_buffer[i+j*gradient_ldim] + m_eps;
191  DataType& m1 = moment1_buffer[i+j*moment1_ldim];
192  DataType& m2 = moment2_buffer[i+j*moment2_ldim];
193  m1 = m_beta1 * m1 + (DataType(1) - m_beta1) * g;
194  m2 = m_beta2 * m2 + (DataType(1) - m_beta2) * g * g;
195  x -= correction * m1 / (std::sqrt(m2) + m_eps);
196  }
197  }
198  } else {
199  // Update with contiguous data
200  #pragma omp parallel for
201  for(int i=0; i<local_height*local_width; ++i) {
202  DataType& x = parameters_buffer[i];
203  // We add eps here because sometimes the gradient is small enough that
204  // g*g can become denormalized, which can significantly impact the
205  // performance.
206  const DataType g = gradient_buffer[i] + m_eps;
207  DataType& m1 = moment1_buffer[i];
208  DataType& m2 = moment2_buffer[i];
209  m1 = m_beta1 * m1 + (DataType(1) - m_beta1) * g;
210  m2 = m_beta2 * m2 + (DataType(1) - m_beta2) * g * g;
211  x -= correction * m1 / (std::sqrt(m2) + m_eps);
212  }
213  }
214 }
