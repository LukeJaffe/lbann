74 class sigmoid_layer : public Activation {
75 protected:
76  DataType act(const DataType& z) {
77  return (DataType(1) / (DataType(1) + std::exp(-z)));
78  }
79  DataType act_prime(const DataType& z) {
80  const DataType sigz = act(z);
81  return sigz * (DataType(1) - sigz);
82  }
83 };

86 class tanh_layer : public Activation {
87 protected:
88  DataType act(const DataType& z) {
89  return std::tanh(z);
90  }
91  DataType act_prime(const DataType& z) {
92  const DataType e = std::exp(DataType(2)*z);
93  return (e - DataType(1)) / (e + DataType(1));
94  }

128 class leaky_reLU_layer : public Activation {
129 public:
131  leaky_reLU_layer(DataType leak = 0.01f) : m_leak(leak) {}
132 protected:
133  DataType act(const DataType& z) {
134  return std::max(m_leak * z, z);
135  }
136  DataType act_prime(const DataType& z) {

148 class softplus_layer : public Activation {
149 protected:
150  DataType act(const DataType& z) {
151  // Warning: Not numerically stable.
152  // Better approach is to determine a threshold so that for large z,
153  // softplus(z) ~= z and for small z, softplus(z) ~= exp(z).
154  return std::log1p(std::exp(z));
155  }
156  DataType act_prime(const DataType& z) {
157  return DataType(1.0) / (DataType(1.0) + std::exp(-z));
158  }
159 };

165 class smooth_reLU_layer : public Activation {
166 protected:
167  DataType act(const DataType& z) {
168  return z / (DataType(1) + std::exp(-z));
169  }
170  DataType act_prime(const DataType& z) {
171  const DataType sigz = DataType(1) / (DataType(1) + std::exp(-z));
172  return sigz + z*sigz - z*sigz*sigz;
173  }
174 };

186 class ELU_layer : public Activation {
187 public:
194  ELU_layer(DataType alpha = 1.0f) : m_alpha(alpha) {}
195 protected:
196  DataType act(const DataType& z) {
197  return (z > DataType(0)) ? z : (m_alpha*std::expm1(z));
198  }
199  DataType act_prime(const DataType& z) {
200  return (z > DataType(0)) ? DataType(1) : (m_alpha*std::expm1(z) + m_alpha);
201  }
202 private:
203  DataType m_alpha;
204 };

207 template<typename... Args>
208 Activation* new_activation(activation_type act_fn, Args... params) {
209  switch (act_fn) {
210  case activation_type::SIGMOID:
211  return new sigmoid_layer();
212  case activation_type::TANH:
213  return new tanh_layer();
214  case activation_type::RELU:
215  return new reLU_layer();
216  case activation_type::ID:
217  return new id_layer();
218  case activation_type::LEAKY_RELU:
219  return new leaky_reLU_layer(params...);
220  case activation_type::SOFTPLUS:
221  return new softplus_layer();
222  case activation_type::SMOOTH_RELU:
223  return new smooth_reLU_layer();
224  case activation_type::ELU:
225  return new ELU_layer(params...);
226  default:
227  throw lbann_exception("Unsupported activation type.");
228  }
229  return nullptr; // Never reached.
230 }
