42 template <data_layout T_layout>
43 class selu_dropout : public regularizer_layer {
44  public:
46  selu_dropout(int index,
47  lbann_comm *comm,
48  float keep_prob=0.95f,
49  DataType alpha = DataType(1.6732632423543772848170429916717),
50  DataType scale = DataType(1.0507009873554804934193349852946)) :
51  regularizer_layer(index, comm),
52  m_keep_prob(keep_prob) {
53 #ifdef LBANN_PROCDET_DROPOUT
54  throw lbann_exception("selu_dropout: deterministic dropout not supported");
55 #endif
56  // Setup the data distribution
57  initialize_distributed_matrices();
58  // Compute alpha' and the affine transform.
59  m_alpha_prime = -scale*alpha;
60  m_a = keep_prob +
61  m_alpha_prime*m_alpha_prime*keep_prob*(DataType(1) - keep_prob);
62  m_a = DataType(1) / std::sqrt(m_a);
63  m_b = -m_a * m_alpha_prime*(DataType(1) - keep_prob);
64  }
65 
66  selu_dropout(const selu_dropout& other) :
67  regularizer_layer(other),
68  m_alpha_prime(other.m_alpha_prime),
69  m_a(other.m_a),
70  m_b(other.m_b),
71  m_keep_prob(other.m_keep_prob) {
72  m_cur_mask = new Mat(*other.m_cur_mask);
73  }

101  void fp_compute() {
102  if (this->get_execution_mode() != execution_mode::training ||
103  m_keep_prob < 0.0f) {
104  // Copy previous activations over.
105  El::Copy(*(this->m_prev_activations), *(this->m_activations_v));
106  return;
107  }
108  ElMat *input_acts = this->m_prev_activations;
109  const El::Int local_height = input_acts->LocalHeight();
110  const El::Int local_width = input_acts->LocalWidth();
111 
112  Mat& local_input_acts = input_acts->Matrix();
113  Mat& local_output_acts = this->m_activations_v->Matrix();
114 
115  // Construct and apply mask and the affine transform.
116  // TODO: Optimize.
117  El::Bernoulli(*m_cur_mask, local_height, local_width, m_keep_prob);
118  for (El::Int col = 0; col < local_width; ++col) {
119  for (El::Int row = 0; row < local_height; ++row) {
120  local_output_acts(row, col) = m_a *
121  (local_input_acts(row, col)*(*m_cur_mask)(row, col) +
122  m_alpha_prime*(1 - (*m_cur_mask)(row, col))) + m_b;
123  }
124  }
125  }
126 
128  void bp_compute() {
129  // Terminate early when not training.
130  if (this->get_execution_mode() != execution_mode::training) {
131  return;
132  }
133  if (m_keep_prob < 0.0f) {
134  // Copy error signal through.
135  El::Copy(*(this->m_prev_error_signal), *(this->m_error_signal_v));
136  return;
137  }
138 
139  Mat& local_prev_error_signal = this->m_prev_error_signal->Matrix();
140  Mat& local_error_signal = this->m_error_signal_v->Matrix();
141  const El::Int local_height = local_prev_error_signal.Height();
142  const El::Int local_width = local_prev_error_signal.Width();
143  // Reweight with the affine scale factor and the dropout mask.
144  for (El::Int col = 0; col < local_width; ++col) {
145  for (El::Int row = 0; row < local_height; ++row) {
146  local_error_signal(row, col) =
147  m_a * local_prev_error_signal(row, col) * (*m_cur_mask)(row, col);
148  }
149  }
150  }
