43 class selu_dropout : public regularizer_layer {
44  public:
46  selu_dropout(int index,
47  lbann_comm *comm,
48  int mini_batch_size,
49  float keep_prob=0.95f,
50  DataType alpha = DataType(1.6732632423543772848170429916717),
51  DataType scale = DataType(1.0507009873554804934193349852946)) :
52  regularizer_layer(index, comm, mini_batch_size),
53  m_keep_prob(keep_prob) {
54 #ifdef LBANN_PROCDET_DROPOUT
55  throw lbann_exception("selu_dropout: deterministic dropout not supported");
56 #endif
57  // Setup the data distribution
58  initialize_distributed_matrices();
59  // Compute alpha' and the affine transform.
60  m_alpha_prime = -scale*alpha;
61  m_a = keep_prob +
62  m_alpha_prime*m_alpha_prime*keep_prob*(DataType(1) - keep_prob);
63  m_a = DataType(1) / std::sqrt(m_a);
64  m_b = -m_a * m_alpha_prime*(DataType(1) - keep_prob);
65  }
66

68  regularizer_layer(other),
69  m_alpha_prime(other.m_alpha_prime),
70  m_a(other.m_a),

72  m_keep_prob(other.m_keep_prob) {
73  m_cur_mask = new Mat(*other.m_cur_mask);
74  }

102  void fp_compute() {
103  if (this->get_execution_mode() != execution_mode::training ||
104  m_keep_prob < 0.0f) {
105  // Copy previous activations over.
106  El::Copy(*(this->m_prev_activations), *(this->m_activations));
107  return;
108  }
109  ElMat *input_acts = this->m_prev_activations;
110  const El::Int local_height = input_acts->LocalHeight();
111  const El::Int local_width = input_acts->LocalWidth();
112 
113  Mat& local_input_acts = input_acts->Matrix();
114  Mat& local_output_acts = this->m_activations->Matrix();
115 
116  // Construct and apply mask and the affine transform.
117  // TODO: Optimize.
118  El::Bernoulli(*m_cur_mask, local_height, local_width, m_keep_prob);
119  for (El::Int col = 0; col < local_width; ++col) {
120  for (El::Int row = 0; row < local_height; ++row) {
121  local_output_acts(row, col) = m_a *
122  (local_input_acts(row, col)*(*m_cur_mask)(row, col) +
123  m_alpha_prime*(1 - (*m_cur_mask)(row, col))) + m_b;
124  }
125  }
126  }
127 
129  void bp_compute() {
130  // Terminate early when not training.
131  if (this->get_execution_mode() != execution_mode::training) {
132  return;
133  }
134  if (m_keep_prob < 0.0f) {
135  // Copy error signal through.
136  El::Copy(*(this->m_prev_error_signal), *(this->m_error_signal));
137  return;
138  }
139 
140  Mat& local_prev_error_signal = this->m_prev_error_signal->Matrix();
141  Mat& local_error_signal = this->m_error_signal->Matrix();
142  const El::Int local_height = local_prev_error_signal.Height();
143  const El::Int local_width = local_prev_error_signal.Width();
144  // Reweight with the affine scale factor and the dropout mask.
145  for (El::Int col = 0; col < local_width; ++col) {
146  for (El::Int row = 0; row < local_height; ++row) {
147  local_error_signal(row, col) =
148  m_a * local_prev_error_signal(row, col) * (*m_cur_mask)(row, col);
149  }
150  }
151  }
