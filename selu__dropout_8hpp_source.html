42 template <data_layout T_layout>
43 class selu_dropout : public regularizer_layer {
44  public:
46  selu_dropout(int index,
47  int num_neurons,
48  lbann_comm *comm,
49  int mini_batch_size,
50  float keep_prob=0.95f,
51  DataType alpha = DataType(1.6732632423543772848170429916717),
52  DataType scale = DataType(1.0507009873554804934193349852946)) :
53  regularizer_layer(index, comm, mini_batch_size),
54  m_keep_prob(keep_prob) {
55 #ifdef LBANN_PROCDET_DROPOUT
56  throw lbann_exception("selu_dropout: deterministic dropout not supported");
57 #endif
58  // Setup the data distribution
59  initialize_distributed_matrices();
60  this->m_type = layer_type::dropout;
61  this->m_num_neurons = num_neurons;
62  this->m_num_neuron_dims = 1;
63  this->m_neuron_dims.assign(1, num_neurons);
64  // Compute alpha' and the affine transform.
65  m_alpha_prime = -scale*alpha;
66  m_a = keep_prob +
67  m_alpha_prime*m_alpha_prime*keep_prob*(DataType(1) - keep_prob);
68  m_a = DataType(1) / std::sqrt(m_a);
69  m_b = -m_a * m_alpha_prime*(DataType(1) - keep_prob);
70  }
71 
72  ~selu_dropout() {
73  delete m_cur_mask;
74  }
75 
76  virtual inline void initialize_distributed_matrices();
77  virtual inline data_layout get_data_layout() { return T_layout; }

79  void setup(int num_prev_neurons) {
80  regularizer_layer::setup(num_prev_neurons);
81  this->m_num_prev_neuron_dims = 1;
82  this->m_prev_neuron_dims.assign(1, num_prev_neurons);
83  this->m_num_neurons = num_prev_neurons;
84  Zeros(*(this->m_activations), this->m_num_neurons, this->m_mini_batch_size);
85  Zeros(*(this->m_error_signal), num_prev_neurons, this->m_mini_batch_size);
86  }
87 
88  protected:
90  void fp_compute() {
91  if (this->get_execution_mode() != execution_mode::training ||
92  m_keep_prob < 0.0f) {
93  // Copy previous activations over.
94  El::Copy(*(this->m_prev_activations), *(this->m_activations));
95  return;
96  }
97  ElMat *input_acts = this->m_prev_activations;
98  const El::Int local_height = input_acts->LocalHeight();
99  const El::Int local_width = input_acts->LocalWidth();
100 
101  Mat& local_input_acts = input_acts->Matrix();
102  Mat& local_output_acts = this->m_activations->Matrix();
103 
104  // Construct and apply mask and the affine transform.
105  // TODO: Optimize.
106  El::Bernoulli(*m_cur_mask, local_height, local_width, m_keep_prob);
107  for (El::Int col = 0; col < local_width; ++col) {
108  for (El::Int row = 0; row < local_height; ++row) {
109  local_output_acts(row, col) = m_a *
110  (local_input_acts(row, col)*(*m_cur_mask)(row, col) +
111  m_alpha_prime*(1 - (*m_cur_mask)(row, col))) + m_b;
112  }
113  }
114  }
115 
117  void bp_compute() {
118  // Terminate early when not training.
119  if (this->get_execution_mode() != execution_mode::training) {
120  return;
121  }
122  if (m_keep_prob < 0.0f) {
123  // Copy error signal through.
124  El::Copy(*(this->m_prev_error_signal), *(this->m_error_signal));
125  return;
126  }
127 
128  Mat& local_prev_error_signal = this->m_prev_error_signal->Matrix();
129  Mat& local_error_signal = this->m_error_signal->Matrix();
130  const El::Int local_height = local_prev_error_signal.Height();
131  const El::Int local_width = local_prev_error_signal.Width();
132  // Reweight with the affine scale factor and the dropout mask.
133  for (El::Int col = 0; col < local_width; ++col) {
134  for (El::Int row = 0; row < local_height; ++row) {
135  local_error_signal(row, col) =
136  m_a * local_prev_error_signal(row, col) * (*m_cur_mask)(row, col);
137  }
138  }
139  }
140 
141  private:
143  DataType m_alpha_prime;
145  DataType m_a;
147  DataType m_b;
149  float m_keep_prob;
151  Mat *m_cur_mask;
152 };
