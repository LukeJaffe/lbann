49 sequential_model::sequential_model(const sequential_model& other) :
50  model(other) {
51  // First copy over the layers.
52  for (const auto& l : other.m_layers) {
53  m_layers.push_back(l->copy());
54  }
55  // Update pointers for each layer.
56  for (size_t l = 0; l < m_layers.size(); ++l) {
57  m_layers[l]->set_neural_network_model(this);
58  Layer* prev_layer = l > 0 ? m_layers[l-1] : nullptr;
59  Layer* next_layer = l < m_layers.size() - 1 ? m_layers[l+1] : nullptr;
60  m_layers[l]->setup_pointers(prev_layer, next_layer);
61  }
62 }

295 void sequential_model::setup(int start_index, int end_index) {
296  if(end_index <= 0) {
297  end_index = m_layers.size();
298  }
299 
300  // Setup each layer
301  for (int l=start_index; l<end_index; ++l) {
302  m_layers[l]->set_neural_network_model(this);
303  Layer* prev_layer = l > 0 ? m_layers[l-1] : nullptr;
304  Layer* next_layer = l < end_index-1 ? m_layers[l+1] : nullptr;
305  m_layers[l]->setup(prev_layer, next_layer);
306  m_layers[l]->check_setup();
307  m_layers[l]->set_index(l);

309  cout << std::setw(3) << l << ":[" << std::setw(15) << m_layers[l]->get_name() << "] Set up a layer with input " << std::setw(7) << m_layers[l]->get_num_prev_neurons() << " and " << std::setw(7) << m_layers[l]->get_num_neurons() << " neurons." << endl;
310  }
311  }
