82 bool lbann::image_utils::loadBMP(const char *Imagefile, int& Width, int& Height, int& BPP, bool Flip, unsigned char *&Pixels) {
83  FILE *infile = fopen(Imagefile, "rb");
84  if (infile == NULL) {
85  fprintf(stderr, "can't open %s\n", Imagefile);
86  return false;
87  }
88 
89  // Read Bitmap File Header
90  BMP_FILEHEADER header;
91  fread(&header, sizeof(BMP_FILEHEADER), 1, infile);
92  if (header.bfType != BMP_HEADER_MARKER) {
93  fclose(infile);
94  return false;
95  }
96 
97  // Read Bitmap Info
98  int bisize = header.bfOffBits - sizeof(BMP_FILEHEADER);
99  BMP_INFO *info = (BMP_INFO *)malloc(bisize);
100  fread(info, bisize, 1, infile);
101 
102  // Check Palette Count
103  if (info->bmiHeader.biClrUsed != 0 || info->bmiHeader.biBitCount != 24) {
104  free(info);
105  fclose(infile);
106  return false;
107  }
108 
109  // Read DIB Bits
110  int bitrowsize = ((info->bmiHeader.biWidth * info->bmiHeader.biBitCount + 31) / 32) * 4;
111  int bitsize = bitrowsize * info->bmiHeader.biHeight;
112  unsigned char *bits = (unsigned char *)malloc(bitsize);
113  fread(bits, bitsize, 1, infile);
114 
115  // Set Pixels
116  Width = info->bmiHeader.biWidth;
117  Height = info->bmiHeader.biHeight;
118  BPP = 3;
119  Pixels = new unsigned char[Width * Height * BPP];
120  for (int y = 0; y < Height; y++) {
121  for (int x = 0; x < Width; x++) {
122  int offset = (Flip) ? ((Height - 1 - y) * Width + x) : (y * Width + x);
123  Pixels[offset] = bits[y * bitrowsize + x * 3 + 2];
124  Pixels[offset + Height*Width] = bits[y * bitrowsize + x * 3 + 1];
125  Pixels[offset + 2*Height*Width] = bits[y * bitrowsize + x * 3 + 0];
126  }
127  }
128 
129  free(info);
130  free(bits);
131  fclose(infile);
132 
133  return true;
134 }

146 bool lbann::image_utils::loadPGM(const char *Imagefile, int& Width, int& Height, int& BPP, bool Flip, unsigned char *&Pixels) {
147  FILE *infile = fopen(Imagefile, "rb");
148  if (infile == NULL) {
149  fprintf(stderr, "can't open %s\n", Imagefile);
150  return false;
151  }
152 
153  char format[5];
154  fscanf(infile, "%s", format);
155  int width, height;
156  fscanf(infile, "%d%d", &width, &height);
157  int maxpixel;
158  fscanf(infile, "%d", &maxpixel);
159 
160  Width = width;
161  Height = height;
162  BPP = 1;
163  Pixels = new unsigned char[Width * Height * BPP];
164 
165  for (int y = 0; y < Height; y++) {
166  for (int x = 0; x < Width; x++) {
167  int offset = (Flip) ? ((Height - 1 - y) * Width + x) : (y * Width + x);
168  int pixel = fgetc(infile);
169  Pixels[offset] = (unsigned char)((double)pixel / maxpixel * 255.0);
170  }
171  }
172 
173  fclose(infile);
174  return true;
175 }
