128  void fp_compute_gpu() override {
129  #ifndef LBANN_HAS_CUDNN
130  throw lbann_exception("relu_layer: cuDNN not detected");
131  #else
132  // Useful constants
133  const DataType one = 1;
134  const DataType zero = 0;
135 #ifndef LBANN_HAS_DISTCONV
136  // Apply activation on each GPU
137  const int num_gpus = this->m_cudnn->get_num_gpus();
138  for(int i = 0; i < num_gpus; ++i) {
139  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
140  CHECK_CUDNN(cudnnSetStream(this->m_cudnn->get_handle(i),
141  this->m_cudnn->get_stream(i)));
142  CHECK_CUDNN(cudnnActivationForward(this->m_cudnn->get_handle(i),
143  m_activation_cudnn_desc,
144  &one,
145  this->m_prev_activations_cudnn_desc,
146  this->m_prev_activations_d[0].get_locked_data(i),
147  &zero,
148  this->m_activations_cudnn_desc,
149  this->m_activations_d[0].get_data(i)));
150  }
151 #else
152  assert0(dc::tensor::View(
153  m_prev_activations_e,
154  m_prev_activations_d[0].get_locked_data(0)));
155  assert0(dc::tensor::View(
156  m_activations_e, m_activations_d[0].get_data(0)));
157  m_relu->set_num_samples(this->m_model->get_current_mini_batch_size());
158  m_relu->forward(one, m_prev_activations_e,
159  zero, m_activations_e);
160 #endif // LBANN_HAS_DISTCONV
161  #endif // LBANN_HAS_CUDNN
162  }

164  void bp_compute_gpu() override {
165  #ifndef LBANN_HAS_CUDNN
166  throw lbann_exception("relu_layer: cuDNN not detected");
167  #else
168  // Useful constants
169  const DataType one = 1;
170 #ifndef LBANN_HAS_DISTCONV
171  // Apply activation derivative on each GPU
172  const int num_gpus = this->m_cudnn->get_num_gpus();
173  for(int i = 0; i < num_gpus; ++i) {

177  CHECK_CUDNN(cudnnActivationBackward(this->m_cudnn->get_handle(i),
178  m_activation_cudnn_desc,
179  &one,
180  this->m_prev_activations_cudnn_desc,
181  this->m_prev_activations_d[0].get_locked_data(i),
182  this->m_prev_error_signals_cudnn_desc,
183  this->m_prev_error_signals_d[0].get_locked_data(i),
184  this->m_activations_cudnn_desc,

186  &one,
187  this->m_error_signals_cudnn_desc,
188  this->m_error_signals_d[0].get_data(i)));
189  }
190 #else
191  assert0(dc::tensor::View(
192  m_activations_e, m_activations_d[0].get_data(0)));
193  assert0(dc::tensor::View(
194  m_prev_activations_e,

209  void setup_tensors() {
210 #ifndef LBANN_HAS_DISTCONV
211  throw lbann_exception(
212  std::string {} + __FILE__ + " " + std::to_string(__LINE__) + " :: " +
213  "Layer: DISTCONV not detected");
214 #else
215  MPIPrintStreamDebug()
216  << "relu: setup_tensors."
217  << "\n";
218 
219  m_distconv_enabled = true;
220 
221  // REFACTORING: duplicated at convolution::setup_tensors
222  Array4 input_tensor_shape =
223  {m_prev_neuron_dims[2], m_prev_neuron_dims[1],
224  m_prev_neuron_dims[0],
225  this->m_model->get_max_mini_batch_size()};
226 
227  Array4 input_local_shape = input_tensor_shape;
228  // Assuming single GPU per rank
229  input_local_shape[3] = m_max_mini_batch_size_per_gpu;
230  Array4 division_block_size = {1, 1, 1, 1};
231 
232  LocaleMPI loc(m_comm->get_model_comm().comm);
233  // Sample distribution
234  Array4 sample_decomposition = {1, 1, 1, m_comm->get_procs_per_model()};
235 
236  // prev_activations
237  m_prev_activations_e = ConstTensorDev(input_tensor_shape, loc,
238  Dist(sample_decomposition),
239  input_local_shape,
240  division_block_size);
241  assert0(dc::tensor::View(
242  m_prev_activations_e,
243  m_prev_activations_d[0].get_locked_data(0)));
244 
245  Array4 output_tensor_shape = {m_neuron_dims[2], m_neuron_dims[1],
246  m_neuron_dims[0],
247  this->m_model->get_max_mini_batch_size()};
248  Array4 output_local_shape = output_tensor_shape;
249  output_local_shape[3] = m_max_mini_batch_size_per_gpu;
250 
251  // activations
252  m_activations_e = TensorDev(output_tensor_shape, loc,
253  Dist(sample_decomposition),
254  output_local_shape,
255  division_block_size);
256  assert0(dc::tensor::View(
257  m_activations_e, m_activations_d[0].get_data(0)));
258 
259 
260  // prev_error_signals
261  m_prev_error_signals_e = ConstTensorDev(output_tensor_shape, loc,
262  Dist(sample_decomposition),
263  output_local_shape,
264  division_block_size);
265  assert0(dc::tensor::View(
266  m_prev_error_signals_e,
267  m_prev_error_signals_d[0].get_locked_data(0)));
268 
269  // error_signals
270  m_error_signals_e = TensorDev(input_tensor_shape, loc,
271  Dist(sample_decomposition),
272  input_local_shape, division_block_size);
273  assert0(dc::tensor::View(
274  m_error_signals_e,
275  m_error_signals_d[0].get_data(0)));
276 
277  // Init the dc::Pooling layer
278  m_relu = new dc::ReLU<dc::cudnn::BackendCUDNN>(
279  *this->m_cudnn->get_distconv_backend());
280 
281  m_relu->setup(m_prev_activations_e,
282  m_activations_e,
283  m_error_signals_e,
284  m_prev_error_signals_e);
285 #endif
286  }
