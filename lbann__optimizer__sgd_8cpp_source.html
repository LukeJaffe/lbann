78 void lbann::sgd::update(const AbsDistMat* gradient)
79 {
80 
81  // Update learning rate and iteration count
82  m_learning_rate *= DataType(1) / ( 1 + m_decay * m_iterations );
83  ++m_iterations;
84 
85  if(m_momentum == DataType(0)) {
86  // Vanilla SGD
87  Axpy(-m_learning_rate, *gradient, *m_parameters);
88  }
89  else {
90 
91  // Get local matrix data
92  const Int local_height = m_parameters->LocalHeight();
93  const Int local_width = m_parameters->LocalWidth();
94  DataType* parameters_buffer = m_parameters->Buffer();
95  const Int parameters_ldim = m_parameters->LDim();
96  const DataType* gradient_buffer = gradient->LockedBuffer();
97  const Int gradient_ldim = gradient->LDim();
98  DataType* velocity_buffer = m_velocity->Buffer();
99  const Int velocity_ldim = m_velocity->LDim();
100 
101  // Check if matrix data is contiguous
102  if(parameters_ldim != local_height
103  || gradient_ldim != local_height
104  || velocity_ldim != local_height) {
105  // (Nesterov) momentum SGD for non-contiguous data
106 #pragma omp parallel for collapse(2)
107  for(Int j=0; j<local_width; ++j) {
108  for(Int i=0; i<local_height; ++i) {
109  const DataType g = gradient_buffer[i+j*gradient_ldim];
110  DataType& v = velocity_buffer[i+j*velocity_ldim];
111  DataType& x = parameters_buffer[i+j*parameters_ldim];
112  v = m_momentum * v - m_learning_rate * g;
113  x += m_nesterov ? m_momentum * v - m_learning_rate * g : v;
114  }
115  }
116  }
117  else {
118  if(m_nesterov) {
119  // Nesterov's accelerated gradient descent for contiguous data
120 #pragma omp parallel for
121  for(Int i=0; i<local_height*local_width; ++i) {
122  DataType& x = parameters_buffer[i];
123  const DataType g = gradient_buffer[i];
124  DataType& v = velocity_buffer[i];
125  v = m_momentum * v - m_learning_rate * g;
126  x += m_momentum * v - m_learning_rate * g;
127  }
128  }
129  else {
130  // Momentum SGD for contiguous data
131 #pragma omp parallel for
132  for(Int i=0; i<local_height*local_width; ++i) {
133  DataType& x = parameters_buffer[i];
134  const DataType g = gradient_buffer[i];
135  DataType& v = velocity_buffer[i];
136  v = m_momentum * v - m_learning_rate * g;
137  x += v;
138  }
139  }
140  }
141 
142  }
143 
144 }
