71  if(!generic_data_reader::position_valid()) {
72  stringstream err;
73  err << __FILE__<<" "<<__LINE__<< " :: MNIST data reader load error: !position_valid";

77  int pixelcount = m_image_width * m_image_height;
78  int current_batch_size = getm_batch_size();

81  std::vector<float> pixels(pixelcount);

85

89  // }
90 
91  if (n >= (int)m_shuffled_indices.size()) {
92  break;

97

141 void lbann::mnist_reader::load(void) {
142  if (is_master()) {
143  cerr << "starting lbann::mnist_reader::load\n";
144  }
145  m_image_data.clear();
146 
147  string FileDir = get_file_dir();
148  string ImageFile = get_data_filename();
149  string LabelFile = get_label_filename();
150 
151  // set filepath
152  string imagepath = FileDir + __DIR_DELIMITER + ImageFile;
153  string labelpath = FileDir + __DIR_DELIMITER + LabelFile;
154 
155  // read labels
156  FILE *fplbl = fopen(labelpath.c_str(), "rb");
157  if (!fplbl) {
158  stringstream err;
159  err << __FILE__<<" "<<__LINE__<< " :: MNIST data reader: failed to open file: " << labelpath;
160  throw lbann_exception(err.str());
161  }
162 
163  int magicnum1, numitems1;
164  fread(&magicnum1, 4, 1, fplbl);
165  fread(&numitems1, 4, 1, fplbl);
166  __swapEndianInt((unsigned int&)magicnum1);
167  __swapEndianInt((unsigned int&)numitems1);
168 
169  // read images
170  FILE *fpimg = fopen(imagepath.c_str(), "rb");
171  if (!fpimg) {
172  stringstream err;
173  err << __FILE__<<" "<<__LINE__<< " :: MNIST data reader: failed to open file: " << imagepath;
174  throw lbann_exception(err.str());
175  }
176 
177  int magicnum2, numitems2, imgwidth, imgheight;
178  fread(&magicnum2, 4, 1, fpimg);
179  fread(&numitems2, 4, 1, fpimg);
180  fread(&imgwidth, 4, 1, fpimg);
181  fread(&imgheight, 4, 1, fpimg);
182  __swapEndianInt((unsigned int&)magicnum2);
183  __swapEndianInt((unsigned int&)numitems2);
184  __swapEndianInt((unsigned int&)imgwidth);
185  __swapEndianInt((unsigned int&)imgheight);
186 
187  if (numitems1 != numitems2) {
188  fclose(fplbl);
189  fclose(fpimg);
190  stringstream err;
191  err << __FILE__<<" "<<__LINE__<< " :: MNIST data reader: numitems1 != numitems2";
192  throw lbann_exception(err.str());
193  }
194 
195  // set to array
196  m_image_data.resize(numitems1);
197  for (int n = 0; n < numitems1; n++) {
198  m_image_data[n].resize(1+(imgwidth * imgheight));
199  fread(&m_image_data[n][0], 1, 1, fplbl);
200  fread(&m_image_data[n][1], imgwidth * imgheight, 1, fpimg);
201  }
202  fclose(fpimg);
203  fclose(fplbl);
204 
205  // reset indices
206  m_shuffled_indices.clear();
207  m_shuffled_indices.resize(m_image_data.size());
208  for (size_t n = 0; n < m_shuffled_indices.size(); n++) {
209  m_shuffled_indices[n] = n;
210  }
211  if (is_master()) {
212  cerr << "calling select_subset_of_data; m_shuffled_indices.size: " << m_shuffled_indices.size() << endl;
213  }
214  select_subset_of_data();
215 }
