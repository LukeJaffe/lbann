29 double digamma(double x) {
30  double result = 0.0;
31  double z = x;
32  while (z < 8.0) {
33  result -= 1.0/z;
34  z += 1.0;
35  }
36  double s = 1.0/z;
37  double s2 = s*s;
38  double s4 = s2*s2;
39  double s6 = s2*s4;
40  double s8 = s4*s4;
41  result += std::log(z) - s/2.0 - s2/12.0 + s4/120.0 - s6/252.0 + s8/240.0 - (5.0/660.0)*s4*s6 +(691.0/32760.0)*s6*s6 - s6*s8/12.0;
42  return result;
43 }

129 EvalType polya_negloglike::evaluate_compute(const AbsDistMat& predictions,
130  const AbsDistMat& ground_truth) {
131 
132  // Initialize workspace
133  m_counts->Resize(1, predictions.Width());
134  m_alpha_sums->Resize(1, predictions.Width());
135  m_lgamma_alpha_sums->Resize(1, predictions.Width());
136  m_lgamma_alpha_level_count_sums->Resize(1, predictions.Width());
137  Mat& counts_local = m_counts->Matrix();
138  Mat& alpha_sums_local = m_alpha_sums->Matrix();
139  Mat& lgamma_alpha_sums_local = m_lgamma_alpha_sums->Matrix();
140  Mat& lgamma_alpha_level_count_sums_local = m_lgamma_alpha_level_count_sums->Matrix();
141 
142  // Local matrices
143  const Mat& predictions_local = predictions.LockedMatrix();
144  const Mat& ground_truth_local = ground_truth.LockedMatrix();
145 
146  // Matrix parameters
147  const int width = predictions.Width();
148  const int local_height = predictions_local.Height();
149  const int local_width = predictions_local.Width();
150 
151  #pragma omp parallel for
152  for (int col = 0; col < local_width; ++col) {
153  DataType count = DataType(0);
154  DataType alpha_sum = DataType(0);
155  DataType lgamma_alpha_sum = DataType(0);
156  DataType lgamma_alpha_level_count_sum = DataType(0);
157  for (int row = 0; row < local_height; ++row) {
158  const DataType true_val = ground_truth_local(row, col);
159  const DataType pred_val = predictions_local(row, col);
160  count += true_val;
161  alpha_sum += pred_val;
162  lgamma_alpha_sum += std::lgamma(pred_val);
163  lgamma_alpha_level_count_sum += std::lgamma(pred_val + true_val);
164  }
165  counts_local(0, col) = count;
166  alpha_sums_local(0, col) = alpha_sum;
167  lgamma_alpha_sums_local(0, col) = lgamma_alpha_sum;
168  lgamma_alpha_level_count_sums_local(0, col) = lgamma_alpha_level_count_sum;
169  }
170  get_comm().allreduce(*m_counts, m_counts->RedundantComm());
171  get_comm().allreduce(*m_alpha_sums, m_alpha_sums->RedundantComm());
172  get_comm().allreduce(*m_lgamma_alpha_sums, m_lgamma_alpha_sums->RedundantComm());
173  get_comm().allreduce(*m_lgamma_alpha_level_count_sums,
174  m_lgamma_alpha_level_count_sums->RedundantComm());
175 
176  // Compute mean objective function value across mini-batch
177  EvalType local_sum = EvalType(0);
178  for (int col = 0; col < local_width; ++col) {
179  local_sum += (- std::lgamma(alpha_sums_local(0, col))
180  + std::lgamma(counts_local(0, col) + alpha_sums_local(0, col))
181  - lgamma_alpha_level_count_sums_local(0, col)
182  + lgamma_alpha_sums_local(0, col));
183  }
184  return get_comm().allreduce(local_sum / width, m_counts->DistComm());
185 
186 }

188 void polya_negloglike::differentiate_compute(const AbsDistMat& predictions,
189  const AbsDistMat& ground_truth,
190  AbsDistMat& gradient) {
191 
192  // Local matrices
193  const Mat& predictions_local = predictions.LockedMatrix();
194  const Mat& ground_truth_local = ground_truth.LockedMatrix();
195  const Mat& counts_local = m_counts->LockedMatrix();
196  const Mat& alpha_sums_local = m_alpha_sums->LockedMatrix();
197  Mat& gradient_local = gradient.Matrix();
198 
199  // Matrix parameters
200  const El::Int local_height = gradient_local.Height();
201  const El::Int local_width = gradient_local.Width();
202 
203  // Compute gradient
204  #pragma omp parallel for collapse(2)
205  for (El::Int col = 0; col < local_width; ++col) {
206  for (El::Int row = 0; row < local_height; ++row) {
207  const DataType true_val = ground_truth_local(row, col);
208  const DataType pred_val = predictions_local(row, col);
209  const DataType alpha_sum = alpha_sums_local(0, col);
210  const DataType count = counts_local(0, col);
211  gradient_local(row, col) = (- digamma(alpha_sum)
212  + digamma(count + alpha_sum)
213  - digamma(true_val + pred_val)
214  + digamma(pred_val));
215  }
216  }
