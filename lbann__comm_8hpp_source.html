497  static inline bool is_sendable(const Mat& mat) {
498  // This assumes we do not transmit mat with a datatype smaller than
499  // DataType.

501  size_t count = (size_t) mat.Height() * (size_t) mat.Width();
502  return count <= (size_t) std::numeric_limits<int>::max();
503  }
