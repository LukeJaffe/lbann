504  static inline bool is_sendable(const Mat& mat) {
505  // This assumes we do not transmit mat with a datatype smaller than
506  // DataType.
507  // MPI uses "int" as its count type; do calculations with larger ints.
508  size_t count = (size_t) mat.Height() * (size_t) mat.Width();
509  return count <= (size_t) std::numeric_limits<int>::max();
510  }
