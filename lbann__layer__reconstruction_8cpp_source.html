68 void lbann::reconstruction_layer::setup(int num_prev_neurons) {
69  target_layer::setup(num_prev_neurons);
70  Layer::setup(num_prev_neurons);
71 
72  // Initialize weight-bias matrix
73  Zeros(*m_weights, m_num_neurons, num_prev_neurons);
74 
75  // Initialize weights
76  initialize_matrix(*m_weights, m_weight_initialization, num_prev_neurons, m_num_neurons);
77 
78  // Initialize other matrices
79  Zeros(*m_error_signal, num_prev_neurons, m_mini_batch_size); // m_error_signal holds the product of m_weights^T * m_prev_error_signal
80  Zeros(*m_activations, m_num_neurons, m_mini_batch_size); //clear up m_activations before copying fp_input to it
81  Zeros(*m_weights_gradient, m_num_neurons,num_prev_neurons); //clear up before filling with new results
82  Zeros(*m_prev_error_signal, m_num_neurons, m_mini_batch_size); //clear up before filling with new results
83  Zeros(*m_prev_activations, num_prev_neurons, m_mini_batch_size);
84 
85  // Initialize optimizer
86  if(m_optimizer != NULL) {
87  m_optimizer->setup(m_weights);
88  }

90 }

92 void lbann::reconstruction_layer::fp_linearity() {
93  //m_activations is linear transformation of m_weights * m_prev_activations^T
94  Gemm(NORMAL, NORMAL, (DataType) 1., *m_weights, *m_prev_activations_v, (DataType) 0.0, *m_activations_v);
95 
96  int64_t curr_mini_batch_size = neural_network_model->get_current_mini_batch_size();
97  DistMat original_layer_act_v;
98  //view of original layer
99  View(original_layer_act_v,*(m_original_layer->m_activations),IR(0,m_original_layer->m_activations->Height()),IR(0,curr_mini_batch_size));
100  // Compute cost will be sum of squared error of fp_input (linearly transformed to m_activations)
101  // and original layer fp_input/original input
102  DataType avg_error = neural_network_model->obj_fn->compute_obj_fn(*m_activations_v, original_layer_act_v);
103  aggregate_cost += avg_error;
104  num_forwardprop_steps++;
105 }

107 void lbann::reconstruction_layer::bp_linearity() {
108 
109  // delta = (activation - y)
110  // delta_w = delta * activation_prev^T
111 
112  int64_t curr_mini_batch_size = neural_network_model->get_current_mini_batch_size();
113  DistMat original_layer_act_v;
114 
115  //view of original layer
116  View(original_layer_act_v,*(m_original_layer->m_activations),IR(0,m_original_layer->m_activations->Height()),IR(0,curr_mini_batch_size));
117 
118  // Compute error signal
119  neural_network_model->obj_fn->compute_obj_fn_derivative(m_prev_layer_type, *m_activations_v, original_layer_act_v,*m_prev_error_signal_v);
120 
121  //m_prev_error_signal_v is the error computed by objective function
122  //is really not previous, but computed in this layer
123  //@todo: rename as obj_error_signal
124 
125  // Compute the partial delta update for the next lower layer
126  Gemm(TRANSPOSE, NORMAL, DataType(1), *m_weights, *m_prev_error_signal_v, DataType(0), *m_error_signal_v);
127 
128  // Compute update for activation weights
129  Gemm(NORMAL, TRANSPOSE, DataType(1)/get_effective_minibatch_size(), *m_prev_error_signal_v,
130  *m_prev_activations_v,DataType(0), *m_weights_gradient);
131 }
