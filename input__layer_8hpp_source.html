117  virtual void fp_set_std_matrix_view() {
118  // Use the predetermined size of the mini-batch to set the current
119  // batch size for the neural network
120  El::Int cur_mini_batch_size = generic_data_distribution::get_current_mini_batch_size();
121  this->m_neural_network_model->set_current_mini_batch_size(cur_mini_batch_size);
122 
123  // Use the precomputed size of the global mini-batch to set the
124  // current effective batch size across all models
125  int total_mini_batch_size = generic_data_distribution::get_current_global_mini_batch_size();
126  this->m_neural_network_model->set_effective_mini_batch_size(total_mini_batch_size);
127 
128  // Once the current mini-batch size is defined, set the standard view for activations only
129  El::View(*m_activations_v, *m_activations, El::ALL, El::IR(0, cur_mini_batch_size));
130  }
