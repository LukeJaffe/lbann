54 lbann::lbann_comm::lbann_comm(int _procs_per_model) :
55  procs_per_model(_procs_per_model), num_model_barriers(0),
56  num_intermodel_barriers(0), num_global_barriers(0), bytes_sent(0),
57  bytes_received(0) {
58 
59  // Initialize parameters
60  int world_size = mpi::Size(mpi::COMM_WORLD);
61  if (procs_per_model == 0) {
62  procs_per_model = world_size;
63  }
64  num_models = world_size / procs_per_model;
65  model_rank = mpi::Rank(mpi::COMM_WORLD) / procs_per_model;
66  rank_in_model = mpi::Rank(mpi::COMM_WORLD) % procs_per_model;
67 
68  // Check if parameters are valid
69  if (procs_per_model > world_size) {
70  stringstream err;
71  err << __FILE__ << " " << __LINE__
72  << " :: Not enough processes to create one model; procs_per_model: "
73  << procs_per_model << " is larger than world_size: " << world_size;
74  throw lbann_exception(err.str());
75  }
76  if (world_size % procs_per_model != 0) {
77  stringstream err;
78  err << __FILE__ << " " << __LINE__
79  << " :: Procs per model does not divide total number of procs; procs_per_model: "
80  << procs_per_model << " total number of procs (world size): " << world_size;
81  throw lbann_exception(err.str());
82  }
83 
84  // Initialize model and intermodel communicators
85  mpi::Split(mpi::COMM_WORLD, model_rank, rank_in_model, model_comm);
86  mpi::Split(mpi::COMM_WORLD, rank_in_model, model_rank, intermodel_comm);
87 
88  // Initialize Elemental grid
89  grid = new Grid(model_comm);
90 
91  // Initialize node communicators
92  setup_node_comm();
93  procs_per_node = mpi::Size(node_comm);
94  rank_in_node = mpi::Rank(node_comm);
95 
96 }

235 void lbann::lbann_comm::intermodel_allreduce(
236  Mat& mat, int max_recv_count,
237  std::function<uint8_t*(Mat&, IR, IR, int&, bool)> send_transform,
238  std::function<int(uint8_t*, Mat&)> recv_transform,
239  std::function<int(uint8_t*, Mat&)> recv_apply_transform) {
240  // If not a power-of-2, we can't use the recursive doubling.
241  const int nprocs = get_num_models();
242  if (nprocs & (nprocs - 1)) {
243  pe_ring_allreduce(intermodel_comm, mat, max_recv_count,
244  send_transform, recv_transform,
245  recv_apply_transform);
246  } else {
247  // TODO: Don't hardcode this.
248  if (mat.Height() <= 64 && mat.Width() <= 64) {
249  recursive_doubling_allreduce_pow2(
250  intermodel_comm, mat, max_recv_count,
251  send_transform, recv_apply_transform);
252  } else {
253  pe_ring_allreduce(intermodel_comm, mat, max_recv_count,
254  send_transform, recv_transform,
255  recv_apply_transform);
256  }
257  }

260 void lbann::lbann_comm::recursive_doubling_allreduce_pow2(
261  mpi::Comm comm, Mat& mat, int max_recv_count,
262  std::function<uint8_t*(Mat&, IR, IR, int&, bool)> send_transform,
263  std::function<int(uint8_t*, Mat&)> recv_apply_transform) {
264  const int rank = mpi::Rank(comm);
265  const int nprocs = mpi::Size(comm);
266  // This implementation requires a power-of-2 number of processes.
267  if (nprocs & (nprocs - 1)) {
268  return;
269  }
270  uint8_t* recv_buf = get_collective_buffer(max_recv_count);
271  unsigned int mask = 1;
272  while (mask < nprocs) {
273  int partner = rank ^ mask; // The rank we exchange with this step.
274  // Transform the data we want to send.
275  int send_size;
276  uint8_t* send_buf = send_transform(mat, ALL, ALL, send_size, false);
277  bytes_sent += send_size;
278  mpi::SendRecv(send_buf, send_size, partner,
279  recv_buf, max_recv_count, partner, comm);
280  // Transform and reduce the received data.
281  int recv_size = recv_apply_transform(recv_buf, mat);
282  bytes_received += recv_size;
283  mask <<= 1;
284  }
285 }

287 void lbann::lbann_comm::pe_ring_allreduce(
288  mpi::Comm comm, Mat& mat, int max_recv_count,
289  std::function<uint8_t*(Mat&, IR, IR, int&, bool)> send_transform,
290  std::function<int(uint8_t*, Mat&)> recv_transform,
291  std::function<int(uint8_t*, Mat&)> recv_apply_transform) {
292  const int rank = mpi::Rank(comm);
293  const int nprocs = mpi::Size(comm);
294  // Compute the number of columns each processor sends.
295  // If it doesn't divide evenly, give one extra to the earlier ranks.
296  const Int cols_per_proc = mat.Width() / nprocs;
297  const Int cols_remainder = mat.Width() % nprocs;
298  // Compute the lengths/ends of the slices.
299  std::vector<Int> slice_lengths(nprocs, cols_per_proc);
300  for (int i = 0; i < cols_remainder; ++i) {
301  slice_lengths[i] += 1;
302  }
303  std::vector<Int> slice_ends(nprocs);
304  std::partial_sum(slice_lengths.begin(), slice_lengths.end(),
305  slice_ends.begin());
306  uint8_t* recv_buf = get_collective_buffer(max_recv_count);
307  // Local slice of our accumulated data.
308  auto accum_view = mat(ALL, IR(slice_ends[rank] - slice_lengths[rank],
309  slice_ends[rank]));
310  // Do a pairwise-exchange reduce-scatter.
311  for (int step = 1; step < nprocs; ++step) {
312  // Compute where we send to/receive from.
313  const int dst = (rank + step) % nprocs;
314  const int src = (rank - step + nprocs) % nprocs;
315  // Transform the data we send. We do not look at the same chunk of data
316  // twice.
317  int send_size;
318  uint8_t* send_buf = send_transform(
319  mat, ALL, IR(slice_ends[dst] - slice_lengths[dst], slice_ends[dst]),
320  send_size, true);
321  bytes_sent += send_size;
322  mpi::SendRecv(send_buf, send_size, dst,
323  recv_buf, max_recv_count, src, comm);
324  int recv_size = recv_apply_transform(recv_buf, accum_view);
325  bytes_received += recv_size;
326  }
327  // Do a ring allgather.
328  const int src = (rank - 1 + nprocs) % nprocs;
329  const int dst = (rank + 1) % nprocs;
330  // Apply the transform to our locally-accumulated slice of the data.
331  int send_size;
332  // Do the first step where we forward our local data.
333  {
334  uint8_t* send_buf = send_transform(
335  mat, ALL, IR(slice_ends[rank] - slice_lengths[rank], slice_ends[rank]),
336  send_size, false);
337  const int data_src = (rank - 1 + nprocs) % nprocs;
338  bytes_sent += send_size;
339  mpi::SendRecv(send_buf, send_size, dst,
340  recv_buf, max_recv_count, src, comm);
341  auto recv_view = mat(ALL,
342  IR(slice_ends[data_src] - slice_lengths[data_src],
343  slice_ends[data_src]));
344  int recv_size = recv_transform(recv_buf, recv_view);
345  bytes_received += recv_size;
346  send_size = recv_size;
347  }
348  // Now do the remaining nprocs - 2 steps.
349  // We always send from recv_buf and receive to recv_buf2, swapping
350  // pointers to avoid copying.
351  uint8_t* recv_buf2 = get_collective_buffer(max_recv_count, 1);
352  for (int step = 1; step < nprocs - 1; ++step) {
353  // Compute where the data we get is coming from.
354  const int data_src = (rank - step - 1 + nprocs) % nprocs;
355  auto recv_view = mat(ALL,
356  IR(slice_ends[data_src] - slice_lengths[data_src],
357  slice_ends[data_src]));
358  bytes_sent += send_size;
359  mpi::SendRecv(recv_buf, send_size, dst,
360  recv_buf2, max_recv_count, src, comm);
361  int recv_size = recv_transform(recv_buf2, recv_view);
362  bytes_received += recv_size;
363  // Swap the send and receive buffers.
364  std::swap(recv_buf, recv_buf2);
365  send_size = recv_size;
366  }
367 }

369 void lbann::lbann_comm::ring_allreduce(
370  mpi::Comm comm, Mat& mat, int max_recv_count,
371  std::function<uint8_t*(Mat&, IR, IR, int&, bool)> send_transform,
372  std::function<int(uint8_t*, Mat&)> recv_transform,
373  std::function<int(uint8_t*, Mat&)> recv_apply_transform) {
374  const int rank = mpi::Rank(comm);
375  const int nprocs = mpi::Size(comm);
376  // Compute the number of columns each processor sends.
377  const Int cols_per_proc = mat.Width() / nprocs;
378  const Int cols_remainder = mat.Width() % nprocs;
379  // Compute the lengths/ends of the slices.
380  std::vector<Int> slice_lengths(nprocs, cols_per_proc);
381  for (int i = 0; i < cols_remainder; ++i) {
382  slice_lengths[i] += 1;
383  }
384  std::vector<Int> slice_ends(nprocs);
385  std::partial_sum(slice_lengths.begin(), slice_lengths.end(),
386  slice_ends.begin());
387  uint8_t* recv_buf = get_collective_buffer(max_recv_count);
388  // Compute source/destination in the ring.
389  const int src = (rank - 1 + nprocs) % nprocs;
390  const int dst = (rank + 1) % nprocs;
391  // Do a ring-based reduce-scatter.
392  // This is like the pairwise-exchange reduce-scatter except instead of
393  // rank i accumulating only slice i, the slices are cycled around and
394  // each node accumulates its portion into the slice when it passes
395  // through. After the nprocs-1 steps slice k will be on rank
396  // (k + nprocs - 1) % nprocs.
397  for (int step = 0; step < nprocs - 1; ++step) {
398  // Compute the slices to send/recv.
399  const int send_slice = (rank - step + nprocs) % nprocs;
400  const int recv_slice = (rank - step - 1 + nprocs) % nprocs;
401  // Transform the data to send.
402  int send_size;
403  uint8_t* send_buf = send_transform(
404  mat, ALL, IR(slice_ends[send_slice] - slice_lengths[send_slice],
405  slice_ends[send_slice]), send_size, false);
406  mpi::SendRecv(send_buf, send_size, dst,
407  recv_buf, max_recv_count, src, comm);
408  auto recv_view = mat(
409  ALL, IR(slice_ends[recv_slice] - slice_lengths[recv_slice],
410  slice_ends[recv_slice]));
411  int recv_size = recv_apply_transform(recv_buf, recv_view);
412  }
413  // Do a ring allgather, first applying the transform to local data.
414  int send_size;
415  {
416  const int send_slice = (rank + 1) % nprocs;
417  const int recv_slice = rank;
418  uint8_t* send_buf = send_transform(
419  mat, ALL, IR(slice_ends[send_slice] - slice_lengths[send_slice],
420  slice_ends[send_slice]), send_size, false);
421  mpi::SendRecv(send_buf, send_size, dst,
422  recv_buf, max_recv_count, src, comm);
423  auto recv_view = mat(ALL,
424  IR(slice_ends[recv_slice] - slice_lengths[recv_slice],
425  slice_ends[recv_slice]));
426  int recv_size = recv_transform(recv_buf, recv_view);
427  send_size = recv_size;
428  }
429  uint8_t* recv_buf2 = get_collective_buffer(max_recv_count, 1);
430  for (int step = 1; step < nprocs - 1; ++step) {
431  const int send_slice = (rank - step + 1 + nprocs) % nprocs;
432  const int recv_slice = (rank - step + nprocs) % nprocs;
433  auto recv_view = mat(ALL,
434  IR(slice_ends[recv_slice] - slice_lengths[recv_slice],
435  slice_ends[recv_slice]));
436  mpi::SendRecv(recv_buf, send_size, dst,
437  recv_buf2, max_recv_count, src, comm);
438  int recv_size = recv_transform(recv_buf2, recv_view);
439  // Swap the send and receive buffers.
440  std::swap(recv_buf, recv_buf2);
441  send_size = recv_size;
442  }
443 }

445 void lbann::lbann_comm::setup_node_comm() {
446 
447  // Get string specifying compute node
448  char node_name[MPI_MAX_PROCESSOR_NAME];
449  int node_name_len;
450  checkMPI(MPI_Get_processor_name(node_name, &node_name_len));
451  const std::string node_string(node_name);
452 
453  // Hash node names and split MPI processes
454  int hash = std::hash<std::string>()(node_string);
455  hash = hash >= 0 ? hash : -hash; // Make sure hash is non-negative
456  mpi::Comm hash_comm;
457  mpi::Split(mpi::COMM_WORLD, hash, mpi::Rank(mpi::COMM_WORLD), hash_comm);
458  const int hash_comm_size = mpi::Size(hash_comm);
459 
460  // Compare node names and split MPI processes
461  char* node_name_list = new char[hash_comm_size*MPI_MAX_PROCESSOR_NAME];
462  checkMPI(MPI_Allgather(node_name, MPI_MAX_PROCESSOR_NAME, MPI_CHAR,
463  node_name_list, MPI_MAX_PROCESSOR_NAME, MPI_CHAR,
464  hash_comm.comm));
465  int node_num = mpi::Rank(hash_comm);
466  for(int i=0; i<hash_comm_size; ++i) {
467  const std::string other_node_string(node_name_list + i*MPI_MAX_PROCESSOR_NAME);
468  if(node_string == other_node_string) {
469  node_num = i;
470  break;
471  }
472  }
473  delete[] node_name_list;
474  mpi::Split(hash_comm, node_num, mpi::Rank(mpi::COMM_WORLD), node_comm);
475  mpi::Free(hash_comm);
476 
477  // Set up the list of model ranks on this node.
478  int node_comm_size = mpi::Size(node_comm);
479  for (int i = 0; i < node_comm_size; ++i) {
480  model_ranks_on_node.push_back(mpi::Translate(node_comm, i, model_comm));
481  }
482 }
