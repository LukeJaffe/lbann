58 EvalType l2_weight_regularization::local_squared_l2_norm(const Mat& mat) const {
59  const El::Int height = mat.Height();
60  const El::Int width = mat.Width();
61  const El::Int ldim = mat.LDim();
62  const DataType* __restrict__ buf = mat.LockedBuffer();
63  EvalType sqsum = EvalType(0);
64  // Check if data is contiguous.
65  if (ldim == height) {
66  const El::Int size = height*width;
67  #pragma omp parallel for reduction(+:sqsum)
68  for (El::Int i = 0; i < size; ++i) {
69  const EvalType val = buf[i];
70  sqsum += val * val;
71  }
72  } else {
73  #pragma omp parallel for reduction(+:sqsum) collapse(2)
74  for (El::Int j = 0; j < width; ++j) {
75  for (El::Int i = 0; i < height; ++i) {
76  const EvalType val = buf[i + j*ldim];
77  sqsum += val * val;
78  }
79  }
80  }
81  return sqsum;
82 }

84 EvalType l2_weight_regularization::evaluate() {
85  if (m_scale_factor == EvalType(0)) { return EvalType(0); }
86  auto value = EvalType(0);
87  for (weights* w : m_weights) {
88  cudnn::cudnn_manager* cudnn = w->get_cudnn_manager();
89  if (cudnn != nullptr) {
90 #ifdef LBANN_HAS_CUDNN
91  CHECK_CUDA(cudaSetDevice(cudnn->get_gpu(0)));
92  EvalType norm = cublas::nrm2(cudnn->get_cublas_handle(0),
93  w->get_size(),
94  w->get_values_gpu()[0], 1);
95  value += norm * norm;
96 #endif // LBANN_HAS_CUDNN
97  } else {
98  // Further optimization: Can batch allreduces on the same communicator.
99  const AbsDistMat& values = w->get_values();
100  EvalType local_norm = local_squared_l2_norm(values.LockedMatrix());
101  value += get_comm().allreduce(local_norm, values.DistComm());
102  }
103  }
104  return m_scale_factor * value;
105 }

107 void l2_weight_regularization::compute_weight_regularization() {
108  if (m_scale_factor == EvalType(0)) { return; }
109  for (weights* w : m_weights) {
110  optimizer* opt = w->get_optimizer();
111  if (w->get_cudnn_manager() != nullptr) {
112 #ifdef LBANN_HAS_CUDNN
113  std::vector<DataType*> values_d = w->get_values_gpu();
114  opt->add_to_gradient_gpu(values_d, 2 * m_scale_factor);
115 #endif // LBANN_HAS_CUDNN
116  } else {
117  opt->add_to_gradient(w->get_values(), 2 * m_scale_factor);
118  }
119  }
120 }
