71 void lbann_callback_imcomm::setup(model* m) {
72  std::vector<Layer*>& layers = m->get_layers();
73  layer_params.resize(layers.size());
74  for (size_t layer = 0; layer < layers.size(); ++layer) {
75  imcomm_params& params = layer_params[layer];
76  if (param_choices.find(layer) != param_choices.end()) {
77  params = param_choices[layer];
78  } else if (layer != 0 && layer != layers.size() - 1) {
79  // Don't do communication for input/output layers unless explicitly told.
80  // Also don't do communication for layers with no gradients.
81  if (layers[layer]->get_weights_biases_gradient().Height() == 0) {
82  params.ct = NONE;
83  } else {
84  params.ct = default_ct;
85  }
86  }
87  if (params.ct != NONE) {
88  // Update the effective mini-batch size so averaging is done properly.
89  layers[layer]->set_effective_minibatch_size(
90  layers[layer]->get_minibatch_size() * m->get_comm()->get_num_models());
91  // Check if reshaping is needed.
92  // Currently only automatically reshapes conv layers. (But ignores bias.)
93  if (layers[layer]->m_type == layer_type::convolution) {
94  convolutional_layer* conv_layer = (convolutional_layer*) layers[layer];
95  params.reshape_height = conv_layer->m_num_input_channels *
96  std::accumulate(conv_layer->m_filter_dims.begin(),
97  conv_layer->m_filter_dims.end(),
98  Int(1), std::multiplies<Int>());
99  params.reshape_width = conv_layer->m_num_output_channels;
100  }
101  if (ct_does_quantization(params.ct)) {
102  if (params.reshape_height) {
103  Zeros(params.error, params.reshape_height, params.reshape_width);
104  } else {
105  const ElMat& gradients = layers[layer]->get_weights_biases_gradient();
106  Zeros(params.error, gradients.LocalHeight(), gradients.LocalWidth());
107  }
108  }
109  }
110  }

113 void lbann_callback_imcomm::on_epoch_end(model* m) {
114  lbann_comm* comm = m->get_comm();
115  if (comm->get_num_models() == 1 ||
116  m->get_execution_mode() != execution_mode::training) {
117  return; // No point with only one model.
118  }
119  std::vector<Layer*>& layers = m->get_layers();
120  for (size_t layer = 0; layer < layers.size(); ++layer) {
121  imcomm_params& params = layer_params[layer];
122  if (ct_does_quantization(params.ct)) {
123  comm->intermodel_sum_matrix(params.error);
124  Mat& local_gradients = layers[layer]->get_weights_biases_gradient().Matrix();
125  if (params.reshape_height > 0) {
126  Mat reshaped;
127  reshape_mat(local_gradients, reshaped, params.reshape_height,
128  params.reshape_width);
129  reshaped = params.error;
130  } else {
131  local_gradients = params.error;
132  }
133  // Apply optimizer update with accumulated gradient error.
134  layers[layer]->update();
135  Zero(params.error);
136  }
137  }

140 void lbann_callback_imcomm::on_backward_prop_end(model* m) {
141  lbann_comm* comm = m->get_comm();
142  if (comm->get_num_models() == 1 ||
143  m->get_execution_mode() != execution_mode::training) {
144  return; // No point with only one model.
145  }
146  std::vector<Layer*>& layers = m->get_layers();
147  for (size_t layer = 0; layer < layers.size(); ++layer) {
148  double start_time = get_time();
149  imcomm_params& params = layer_params[layer];
150  if (params.ct == NONE) continue;
151  Mat& local_gradients =
152  layers[layer]->get_weights_biases_gradient().Matrix();
153  Mat* reshaped = &local_gradients;
154  if (params.reshape_height > 0 && ct_does_quantization(params.ct)) {
155  if (layers[layer]->m_type == layer_type::convolution) {
156  convolutional_layer* conv_layer = (convolutional_layer*) layers[layer];
157  // Currently ignores the bias.
158  Mat grad_view = local_gradients(IR(0, conv_layer->m_filter_size), ALL);
159  reshape_mat(grad_view, *reshaped, params.reshape_height,
160  params.reshape_width);
161  } else {
162  reshape_mat(local_gradients, *reshaped, params.reshape_height,
163  params.reshape_width);
164  }
165  }
166  switch (params.ct) {
167  case NORMAL:
168  comm->intermodel_sum_matrix(*reshaped);
169  break;
170  case ONEBIT_QUANTIZATION:
171  quantizer.intermodel_sum_onebit_quantized(
172  comm, *reshaped, params.error);
173  break;
174  case THRESH_QUANTIZATION:
175  quantizer.intermodel_sum_threshold_quantized(
176  comm, *reshaped, params.error, params.pos_thresh, params.neg_thresh);
177  break;
178  case ADAPTIVE_QUANTIZATION:
179  quantizer.intermodel_sum_adaptive_quantized(
180  comm, *reshaped, params.error, params.proportion);
181  break;
182  default:
183  throw lbann_exception("imcomm: unknown comm type");
184  }
185  double im_time = get_time() - start_time;
186  do_summary(m, layers[layer], im_time);
187  }

190 void lbann_callback_imcomm::do_summary(model* m, Layer* layer,
191  double im_time) {
192  if (summarizer == nullptr) return;
193  uint idx = layer->get_index();
194  lbann_comm* comm = m->get_comm();
195  std::string prefix = "layer" + std::to_string(
196  static_cast<long long>(idx)) + "/imcomm_";
197  summarizer->reduce_scalar(prefix + "time",
198  im_time, m->get_cur_step());
199  size_t bytes_sent = 0;
200  size_t bytes_received = 0;
201  if (ct_does_quantization(layer_params[idx].ct)) {
202  bytes_sent = comm->get_ar_bytes_sent();
203  bytes_received = comm->get_ar_bytes_received();
204  } else {
205  // Use the same approximation the comm layer does.
206  const Mat& local_gradients =
207  layer->get_weights_biases_gradient().LockedMatrix();
208  bytes_sent =
209  sizeof(DataType) * local_gradients.Height() * local_gradients.Width();
210  bytes_received =
211  sizeof(DataType) * local_gradients.Height() * local_gradients.Width();
212  }
213  summarizer->reduce_scalar(prefix + "bytes_sent",
214  bytes_sent, m->get_cur_step());
215  summarizer->reduce_scalar(prefix + "bytes_received",
216  bytes_received, m->get_cur_step());
217  if (ct_does_quantization(layer_params[idx].ct)) {
218  summarizer->reduce_scalar(prefix + "rs_bytes_sent",
219  comm->get_ar_rs_bytes_sent(),
220  m->get_cur_step());
221  summarizer->reduce_scalar(prefix + "ag_bytes_sent",
222  comm->get_ar_ag_bytes_sent(),
223  m->get_cur_step());
224  summarizer->reduce_scalar(prefix + "rs_bytes_received",
225  comm->get_ar_rs_bytes_received(),
226  m->get_cur_step());
227  summarizer->reduce_scalar(prefix + "ag_bytes_received",
228  comm->get_ar_ag_bytes_received(),
229  m->get_cur_step());
230  summarizer->reduce_scalar(prefix + "ar_send_trans_time",
231  comm->get_ar_send_transform_time(),
232  m->get_cur_step());
233  summarizer->reduce_scalar(prefix + "ar_recv_trans_time",
234  comm->get_ar_recv_transform_time(),
235  m->get_cur_step());
236  summarizer->reduce_scalar(prefix + "ar_recv_apply_trans_time",
237  comm->get_ar_recv_apply_transform_time(),
238  m->get_cur_step());
239  if (layer_params[idx].ct == ADAPTIVE_QUANTIZATION) {
240  summarizer->reduce_scalar(prefix + "quantized_count",
241  quantizer.get_quantized_count(),
242  m->get_cur_step());
243  }
244  quantizer.reset_counters();
245  comm->reset_stats_counters();
246  }
