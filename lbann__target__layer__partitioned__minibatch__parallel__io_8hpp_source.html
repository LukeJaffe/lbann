53  if(!m_shared_data_reader) {

59  io_layer::setup_data_readers_for_training(base_offset,
60  batch_stride,
61  m_num_parallel_readers_training,
62  model_offset);
63  partitioned_minibatch_parallel_io::calculate_num_iterations_per_epoch(m_training_dataset.data_reader);
65  io_layer::setup_data_readers_for_evaluation(Layer::comm->get_rank_in_model(),
66  Layer::m_mini_batch_size,

68  } else {
69  io_layer::setup_data_readers_for_training(Layer::comm->get_rank_in_model(),
70  Layer::m_mini_batch_size,

98  target_layer::update_num_samples_processed(num_samples_in_batch);

100  int64_t curr_mini_batch_size = neural_network_model->get_current_mini_batch_size();

109  }

111  return;
112  }
