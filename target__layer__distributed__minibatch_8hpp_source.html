101  void fp_compute() {
102  int num_samples_in_batch = fetch_to_local_matrix(Y_local);
103  if(is_current_root()) {
105  target_layer::update_num_samples_processed(num_samples_in_batch);
106  }
107 
108  int curr_mini_batch_size = this->m_neural_network_model->get_current_mini_batch_size();
109  if(is_current_root() && num_samples_in_batch != curr_mini_batch_size) {
110  throw lbann_exception("lbann_target_layer_distributed_minibatch: number of labels does not match the current mini-batch size.");
111  }
113  distribute_from_local_matrix(Y_local, Ys);
114  Copy(Ys, *this->m_activations);
115 
117  DataType avg_error = this->m_neural_network_model->m_obj_fn->compute_obj_fn(*this->m_prev_activations_v, *this->m_activations_v);
118  this->m_neural_network_model->m_obj_fn->record_obj_fn(this->m_execution_mode, avg_error);
119 
120  for (auto&& m : this->m_neural_network_model->get_metrics()) {
121  double num_errors = m->compute_metric(*this->m_prev_activations_v, *this->m_activations_v);
122  m->record_error(num_errors, curr_mini_batch_size);
123  }
124 
125  return;
126  }
