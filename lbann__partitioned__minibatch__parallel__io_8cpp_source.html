186  int max_mini_batch_size = data_reader->m_batch_size;

188  int min_stride_across_models = max_mini_batch_size * comm->get_num_models();
189 
190  data_reader->m_last_mini_batch_size = max_mini_batch_size;
191 
192  int num_whole_mini_batches_per_model = floor(data_reader->getNumData() / min_stride_across_models);
193  int num_whole_mini_batches_per_reader = floor(num_whole_mini_batches_per_model / num_parallel_readers_per_model);
194  // int parallel_readers_with_extra_mini_batch = num_whole_mini_batches_per_model % num_parallel_readers_per_model;
195  int per_model_partial_mini_batch_size = (data_reader->getNumData() - (num_whole_mini_batches_per_model * min_stride_across_models))/(comm->get_num_models());
196  int world_master_remainder_data = 0;
197 
198  // Compute how many full "parallel" mini-batches are available
199  data_reader->m_last_mini_batch_threshold = num_whole_mini_batches_per_model * min_stride_across_models;
200 
201  data_reader->m_num_mini_batches_per_reader = num_whole_mini_batches_per_reader;
202 
203  int world_master_remainder_adjustment = data_reader->getNumData()
204  - (num_whole_mini_batches_per_model * min_stride_across_models)
205  - (per_model_partial_mini_batch_size * comm->get_num_models());
206  if(comm->am_world_master()) {
207  world_master_remainder_data = world_master_remainder_adjustment;
208  world_master_remainder_adjustment = 0;
209  }
210  per_model_partial_mini_batch_size += world_master_remainder_data;
211 
212  if(per_model_partial_mini_batch_size > 0 || world_master_remainder_adjustment > 0) {
213  data_reader->m_num_mini_batches_per_reader++;
214  data_reader->m_last_mini_batch_size = per_model_partial_mini_batch_size;
215  }
216 
217  data_reader->m_num_iterations_per_epoch = data_reader->m_num_mini_batches_per_reader;
218 
219  if(data_reader->m_last_mini_batch_size > max_mini_batch_size) {
220  throw new lbann_exception("Error in calculating the partial mini-batch size, exceeds the max mini-batch size");
221  }

232  if(/*comm->get_rank_in_model() == parallel_readers_with_extra_mini_batch && */per_model_partial_mini_batch_size > 0 || world_master_remainder_adjustment > 0) {
233  data_reader->m_last_mini_batch_stride = (data_reader->m_last_mini_batch_threshold - data_reader->m_base_offset - data_reader->m_model_offset - last_mini_batch_offset) + comm->get_rank_in_model();
