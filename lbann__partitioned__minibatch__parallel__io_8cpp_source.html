117 bool lbann::partitioned_minibatch_parallel_io::is_data_set_processed() {
118  int num_readers_done = 0;
119  int max_active_parallel_readers = get_num_parallel_readers(); // When calculating if all parallel readers are done, include the maximum number,
120  // not just the ones in the last round. This will ensure that all readers, that had data
121  // will have partitioned it.
122  int num_parallel_readers = m_num_valid_readers;
123 
124  int num_iterations_per_epoch = get_num_iterations_per_epoch();
125 
126  m_local_reader_done = !update_data_reader();
127 
129  if(m_local_reader_done) {
130  num_readers_done = 1;
131  }
132 
134  num_readers_done = m_comm->model_allreduce(num_readers_done);
135  if(m_cur_step_in_epoch == (num_iterations_per_epoch - 1) /*num_readers_done >= max_active_parallel_readers*/) {
136  m_local_reader_done = false;
137  m_root = 0;
138  m_num_data_per_epoch = 0;
141  m_cur_step_in_epoch = 0;
142  return true;
143  } else {
144  m_cur_step_in_epoch++;

185 void lbann::partitioned_minibatch_parallel_io::calculate_num_iterations_per_epoch(generic_data_reader *data_reader) {
186  int max_mini_batch_size = data_reader->getm_batch_max();
187  int num_parallel_readers_per_model = max(1, (data_reader->get_batch_stride() / m_comm->get_num_models()) / max_mini_batch_size);
188  int min_stride_across_models = max_mini_batch_size * m_comm->get_num_models();
189 
190  data_reader->set_last_mini_batch_size(max_mini_batch_size);
191 
192  int num_whole_mini_batches_per_model = floor(data_reader->getNumData() / min_stride_across_models);
193  int num_whole_mini_batches_per_reader = floor(num_whole_mini_batches_per_model / num_parallel_readers_per_model);
194  // int parallel_readers_with_extra_mini_batch = num_whole_mini_batches_per_model % num_parallel_readers_per_model;
195  int per_model_partial_mini_batch_size = (data_reader->getNumData() - (num_whole_mini_batches_per_model * min_stride_across_models))/(m_comm->get_num_models());
196  int world_master_remainder_data = 0;
197 
198  // Compute how many full "parallel" mini-batches are available
199  data_reader->set_last_mini_batch_threshold(num_whole_mini_batches_per_model * min_stride_across_models);
200 
201  data_reader->set_num_mini_batches_per_reader(num_whole_mini_batches_per_reader);
202 
203  int world_master_remainder_adjustment = data_reader->getNumData()
204  - (num_whole_mini_batches_per_model * min_stride_across_models)
205  - (per_model_partial_mini_batch_size * m_comm->get_num_models());
206  if(m_comm->am_world_master()) {
207  world_master_remainder_data = world_master_remainder_adjustment;
208  world_master_remainder_adjustment = 0;
209  }
210  per_model_partial_mini_batch_size += world_master_remainder_data;
211 
212  if(per_model_partial_mini_batch_size > 0 || world_master_remainder_adjustment > 0) {
213  data_reader->set_num_mini_batches_per_reader(data_reader->get_num_mini_batches_per_reader()+1);
214  data_reader->set_last_mini_batch_size(per_model_partial_mini_batch_size);
215  }
216 
217  data_reader->set_num_iterations_per_epoch(data_reader->get_num_mini_batches_per_reader());
218 
219  if(data_reader->get_last_mini_batch_size() > max_mini_batch_size) {
220  throw new lbann_exception("Error in calculating the partial mini-batch size, exceeds the max mini-batch size");
221  }
222 
225 
227  data_reader->set_last_mini_batch_stride(data_reader->get_batch_stride());
228 
229  int last_mini_batch_offset = max(0, num_whole_mini_batches_per_reader - 1) * data_reader->get_batch_stride();
230 
232  if(/*m_comm->get_rank_in_model() == parallel_readers_with_extra_mini_batch && */per_model_partial_mini_batch_size > 0 || world_master_remainder_adjustment > 0) {
233  data_reader->set_last_mini_batch_stride((data_reader->get_last_mini_batch_threshold() - data_reader->get_base_offset() - data_reader->get_model_offset() - last_mini_batch_offset) + m_comm->get_rank_in_model());
234  }
