67 int lbann::generic_data_reader::fetch_data(Mat& X) {
68  int nthreads = omp_get_max_threads();
69  if(!position_valid()) {
70  throw lbann_exception(
71  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
72  " :: generic data reader load error: !position_valid");
73  }
74 
77  #pragma omp parallel for schedule(static, 1)
78  for (int t = 0; t < nthreads; t++) {
79  preprocess_data_source(omp_get_thread_num());
80  }
81 
82  int current_batch_size = getm_batch_size();
83  const int end_pos = std::min(static_cast<size_t>(m_current_pos+current_batch_size),
84  m_shuffled_indices.size());
85  const int mb_size = std::min(
86  El::Int{((end_pos - m_current_pos) + m_sample_stride - 1) / m_sample_stride},
87  X.Width());
88 
89  El::Zeros(X, X.Height(), X.Width());
90  El::Zeros(m_indices_fetched_per_mb, mb_size, 1);
91  #pragma omp parallel for
92  for (int s = 0; s < mb_size; s++) {
93  int n = m_current_pos + (s * m_sample_stride);
94  int index = m_shuffled_indices[n];
95  bool valid = fetch_datum(X, index, s, omp_get_thread_num());
96  if (!valid) {
97  throw lbann_exception(
98  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
99  " :: generic data reader load error: datum not valid");
100  }
101  m_indices_fetched_per_mb.Set(s, 0, index);
102  }
103 
106  #pragma omp parallel for schedule(static, 1)
107  for (int t = 0; t < nthreads; t++) {
108  postprocess_data_source(omp_get_thread_num());
109  }
110 
111  return mb_size;
112 }

222 bool generic_data_reader::update() {
224  if(m_use_alt_last_mini_batch_size && ((m_current_mini_batch_idx+1) >= (m_num_mini_batches_per_reader-1))) {
225  // std::cout << "Data reader last update update the current position is " << m_current_pos << " and the next postion is going to be " << (m_current_pos + m_last_mini_batch_stride) << " and the number of samples total is " << m_shuffled_indices.size() << std::endl;
226  m_current_pos += m_last_mini_batch_stride;
227  } else {
228  // std::cout << "Data reader update the current position is " << m_current_pos << " and the next postion is going to be " << (m_current_pos + m_batch_stride) << " and the number of samples total is " << m_shuffled_indices.size() << std::endl;
229  m_current_pos += m_batch_stride;
230  }
231 
233  El::Zeros(m_indices_fetched_per_mb, m_indices_fetched_per_mb.Width(), 1);
234 
235  if (m_current_pos < (int)m_shuffled_indices.size()) {
236  m_current_mini_batch_idx++;
237  return true;
238  } else {
239  if (not m_first_n) {
240  std::shuffle(m_shuffled_indices.begin(), m_shuffled_indices.end(),
241  get_data_seq_generator());
242  }
243  m_current_mini_batch_idx = 0;
244  m_current_pos = m_base_offset + m_model_offset;
245  return false;
246  }

268 void generic_data_reader::select_subset_of_data() {
269  if(!get_firstN()) {
270  std::shuffle(m_shuffled_indices.begin(), m_shuffled_indices.end(), get_data_seq_generator());
271  }
272 
273  if (not (has_max_sample_count() or has_use_percent() or has_validation_percent())) {
274  return;
275  }
276 
277  if (has_max_sample_count()) {
278  size_t count = get_max_sample_count();
279  if(count > getNumData()) {
280  throw lbann_exception(
281  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
282  " :: generic_data_reader::select_subset_of_data() - max_sample_count=" +
283  std::to_string(count) + " is > getNumData=" +
284  std::to_string(getNumData()));
285  }
286  m_shuffled_indices.resize(get_max_sample_count());
287  } else if (has_use_percent()) {
288  m_shuffled_indices.resize(get_use_percent()*getNumData());
289  }
290 
291  if (has_validation_percent()) {
292  long unused = get_validation_percent()*getNumData(); //getNumData() = m_shuffled_indices.size()
293  long use_me = getNumData() - unused;
294  if (unused > 0) {
295  m_unused_indices=std::vector<int>(m_shuffled_indices.begin() + use_me, m_shuffled_indices.end());
296  m_shuffled_indices.resize(use_me);
297  }
298  }
299 
300  if(!get_firstN()) {
301  std::sort(m_shuffled_indices.begin(), m_shuffled_indices.end());
302  std::sort(m_unused_indices.begin(), m_unused_indices.end());
303  }

313 bool generic_data_reader::saveToCheckpointShared(persist& p, const char *name) {
314  // rank 0 writes the training state file
315  if (p.get_rank() == 0) {
316  char fieldname[1024];
317 
318  // record minibatch index
319  snprintf(fieldname, sizeof(fieldname), "%s_current_mini_batch_idx", name);
320  p.write_uint64(persist_type::train, fieldname, (uint64_t) m_current_mini_batch_idx);
321 
322  // get size of list of training examples

324 
325  // record size of ShuffleIndices
326  snprintf(fieldname, sizeof(fieldname), "%s_data_size", name);
327  p.write_uint64(persist_type::train, fieldname, (uint64_t) size);
328 
329  // TODO: each model may have a different position, need to gather and write these

335  snprintf(fieldname, sizeof(fieldname), "%s_data_indices", name);

337  }

343 bool lbann::generic_data_reader::loadFromCheckpointShared(persist& p, const char *name) {
344  // rank 0 reads the training state file
345  if (p.get_rank() == 0) {
346  char fieldname[1024];
347 
348  // record minibatch index
349  uint64_t val;
350  snprintf(fieldname, sizeof(fieldname), "%s_current_mini_batch_idx", name);
351  p.read_uint64(persist_type::train, fieldname, &val);
352  m_current_mini_batch_idx = (int) val;
353 
354  // get size of ShuffleIndices
355  snprintf(fieldname, sizeof(fieldname), "%s_data_size", name);
356  p.read_uint64(persist_type::train, fieldname, &val);
357  int size = (int) val;
358 
359  // get current position within data
360  snprintf(fieldname, sizeof(fieldname), "%s_data_position", name);
361  p.read_uint64(persist_type::train, fieldname, &val);
362  m_current_pos = (int) val;
363 
364  // resize shuffled index array to hold values
365  m_shuffled_indices.resize(size);
366 
367  // read list of indices
368  snprintf(fieldname, sizeof(fieldname), "%s_data_indices", name);
369  p.read_int32_contig(persist_type::train, fieldname, &m_shuffled_indices[0], (uint64_t) size);
370  }
371 
372  // broadcast minibatch index
373  MPI_Bcast(&m_current_mini_batch_idx, 1, MPI_INT, 0, MPI_COMM_WORLD);
374 
375  // TODO: with multiple readers, make this a scatter
376  // broadcast current position
377  MPI_Bcast(&m_current_pos, 1, MPI_INT, 0, MPI_COMM_WORLD);
378 
379  // broadcast values from rank 0
380  int size = m_shuffled_indices.size();
381  MPI_Bcast(&size, 1, MPI_INT, 0, MPI_COMM_WORLD);
382 
383  // resize shuffled index array to hold values
384  if (p.get_rank() != 0) {
385  m_shuffled_indices.resize(size);
386  }
387 
388  // broadcast index array
389  MPI_Bcast(&m_shuffled_indices[0], size, MPI_INT, 0, MPI_COMM_WORLD);
390 
391  return true;
392 }
