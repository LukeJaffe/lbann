126  void push_back_parent(const Layer *parent) {
127 
128  // Check if parent layer is null pointer
129  if(parent == NULL) {
130  if(m_comm->am_world_master()) {
131  std::cerr << "concatenation_layer: could not add parent layer since pointer is null" << "\n";
132  }
133  return;
134  }
135 
136  // Add parent layer if it isn't in list of parents
137  auto parent_pos = std::find(m_parents.begin(), m_parents.end(), parent);
138  if(parent_pos == m_parents.end()) {
139  m_parents.push_back(parent);
140  }
141  else {
142  throw lbann_exception("concatenation_layer: could not add parent layer since it is already in list of parents");
143  }
144

168  void setup_dims() {
169 
170  // Initialize previous layer dimensions with first parent layer
171  transform::setup_dims();
172 
173  // Check if concatenation axis is valid
174  if(m_concatenation_axis < 0
175  || m_concatenation_axis >= this->m_num_neuron_dims) {
176  throw lbann_exception("concatenation_layer: invalid concatenation axis");
177  }
178 
179  // Get concatenation axis indices corresponding to each parent layer
180  m_concatenation_points.empty();
181  m_concatenation_points.push_back(0);
182  m_concatenation_points.push_back(this->m_neuron_dims[m_concatenation_axis]);
183  for(size_t i=1; i<m_parents.size(); ++i) {
184 
185  // Get parent layer dimensions
186  std::vector<int> parent_dims = m_parents[i]->fp_output_dims(this);
187 
188  // Check if parent layer has valid dimensions
189  if((int) parent_dims.size() != this->m_num_neuron_dims) {
190  throw lbann_exception("concatenation_layer: parent layer has invalid number of dimensions");
191  }
192  for(size_t d=0; d<parent_dims.size(); ++d) {
193  if((int) d != m_concatenation_axis
194  && this->m_neuron_dims[d] != parent_dims[d]) {
195  throw lbann_exception("concatenation_layer: parent layer has invalid dimensions");
196  }
197  }
198 
199  // Get concatentation axis upper bound for parent layer
200  m_concatenation_points.push_back(m_concatenation_points.back()
201  + parent_dims[m_concatenation_axis]);
202 
203  }
204 
205  // Update neuron dimensions
206  this->m_neuron_dims[m_concatenation_axis] = m_concatenation_points.back();
207  this->m_num_neurons = std::accumulate(m_neuron_dims.begin(),
208  m_neuron_dims.end(),
209  1,
210  std::multiplies<int>());
211 
212  }

280  void fp_compute_gpu() {
281  #ifndef __LIB_CUDNN
282  throw lbann_exception("concatenation_layer: cuDNN not detected");
283  #else
284 
285  // Split the activations tensor into slices of width 1 along the
286  // concatenation axis
287  const int num_slices
288  = std::accumulate(this->m_neuron_dims.begin(),
289  this->m_neuron_dims.begin() + m_concatenation_axis,
290  1,
291  std::multiplies<int>());
292  const int slice_unit_size
293  = std::accumulate(this->m_neuron_dims.begin() + m_concatenation_axis + 1,
294  this->m_neuron_dims.end(),
295  1,
296  std::multiplies<int>());
297  const int output_slice_dim = this->m_neuron_dims[m_concatenation_axis];
298  const int output_slice_size = output_slice_dim * slice_unit_size;
299 
300  // Copy entries in each child to error signal tensor
301  const int num_gpus = this->m_cudnn->get_num_gpus();
302  for(size_t parent_index = 0; parent_index < m_parents.size(); ++parent_index) {
303  const Layer* parent = m_parents[parent_index];
304 
305  // Get child error signal on GPUs
306  std::vector<DataType*> input;
307  if(parent_index == 0) {
308  input = this->m_prev_activations_d;
309  }
310  else {
311  std::vector<void*> work_spaces = this->m_cudnn->get_work_spaces();
312  for(int i=0; i<num_gpus; ++i) {
313  input.push_back((DataType*) work_spaces[i]);
314  }
315  if(parent->using_gpus()) {
316  parent->get_gpu_fp_output(input, this);
317  }
318  else {
319  parent->get_fp_output(*this->m_prev_activations, this);
320  this->m_cudnn->scatter_to_gpus(input,
321  this->m_prev_activations->LockedMatrix(),
322  this->m_mini_batch_size_per_gpu);
323  }
324  }
325 
326  // Split previous error signal tensor into slices
327  const int input_slice_dim = m_concatenation_points[parent_index+1] - m_concatenation_points[parent_index];
328  const int input_slice_size = input_slice_dim * slice_unit_size;
329  const int input_size = num_slices * input_slice_size;
330  const int slice_offset = m_concatenation_points[parent_index] * slice_unit_size;
331 
332  // Copy slices from previous error signal tensor into error signal tensor
333  for(int slice = 0; slice < num_slices; ++slice) {
334  std::vector<DataType*> input_slice(num_gpus), output_slice(num_gpus);
335  for(int i = 0; i < num_gpus; ++i) {
336  input_slice[i] = input[i] + slice * input_slice_size;
337  output_slice[i] = this->m_activations_d[i] + slice * output_slice_size + slice_offset;
338  }
339  this->m_cudnn->copy_on_gpus(output_slice,
340  input_slice,
341  input_slice_size,
342  this->m_mini_batch_size_per_gpu,
343  input_size,
344  this->m_num_neurons);
345  }
346 
347  }
348 
349  #endif // #ifndef __LIB_CUDNN

352  void fp_compute_cpu() {
353 
354  // Split the neuron tensor into slices of width 1 along the
355  // concatenation axis
356  const int num_slices
357  = std::accumulate(this->m_neuron_dims.begin(),
358  this->m_neuron_dims.begin() + m_concatenation_axis,
359  1,
360  std::multiplies<int>());
361  const int slice_unit_size
362  = std::accumulate(this->m_neuron_dims.begin() + m_concatenation_axis + 1,
363  this->m_neuron_dims.end(),
364  1,
365  std::multiplies<int>());
366  const int output_slice_dim = this->m_neuron_dims[m_concatenation_axis];
367  const int output_slice_size = output_slice_dim * slice_unit_size;
368 
369  // Copy entries in each parent to neuron tensor
370  for(size_t i = 0; i < m_parents.size(); ++i) {
371 
372  // Split previous neuron tensor into slices
373  m_parents[i]->get_fp_output(*this->m_prev_activations, this);
374  const int input_slice_dim = m_concatenation_points[i+1] - m_concatenation_points[i];
375  const int input_slice_size = input_slice_dim * slice_unit_size;
376  const int slice_offset_start = m_concatenation_points[i] * slice_unit_size;
377  const int slice_offset_end = m_concatenation_points[i+1] * slice_unit_size;
378 
379  // Copy slices from previous neuron tensor into neuron tensor
380  for(int slice = 0; slice < num_slices; ++slice) {
381  El::LockedView(*m_input_slice_v,
382  *this->m_prev_activations,
383  El::IR(slice * input_slice_size,
384  (slice+1) * input_slice_size),
385  El::ALL);
386  El::View(*m_output_slice_v,
387  *this->m_activations_v,
388  El::IR(slice * output_slice_size + slice_offset_start,
389  slice * output_slice_size + slice_offset_end),
390  El::ALL);
391  El::Copy(*m_input_slice_v, *m_output_slice_v);
392  }
393 
394  }

398  void get_bp_output(AbsDistMat& bp_output, const Layer* prev_layer) const {
399 
400  // Check if input is in the list of parent layers
401  const int parent_index = (std::find(m_parents.begin(),
402  m_parents.end(),
403  prev_layer)
404  - m_parents.begin());
405  if(parent_index >= (int) m_parents.size()) {
406  transform::get_bp_output(bp_output, prev_layer);
407  }
408 
409  // Split the error signal tensor into slices of width 1 along the
410  // concatenation axis
411  const int num_slices
412  = std::accumulate(this->m_neuron_dims.begin(),
413  this->m_neuron_dims.begin() + m_concatenation_axis,
414  1,
415  std::multiplies<int>());
416  const int slice_unit_size
417  = std::accumulate(this->m_neuron_dims.begin() + m_concatenation_axis + 1,
418  this->m_neuron_dims.end(),
419  1,
420  std::multiplies<int>());
421  const int input_slice_dim = this->m_neuron_dims[m_concatenation_axis];
422  const int output_slice_dim = m_concatenation_points[parent_index+1] - m_concatenation_points[parent_index];
423  const int input_slice_size = input_slice_dim * slice_unit_size;
424  const int output_slice_size = output_slice_dim * slice_unit_size;
425  const int slice_offset_start = m_concatenation_points[parent_index] * slice_unit_size;
426  const int slice_offset_end = m_concatenation_points[parent_index+1] * slice_unit_size;
427 
428  if(num_slices == 1
429  && m_error_signal_v->DistData() == bp_output.DistData()) {
430  // Return view of error signal slice
431  El::LockedView(bp_output,
432  *this->m_error_signal_v,
433  El::IR(slice_offset_start, slice_offset_end),
434  El::ALL);
435  }
436  else {
437  // Copy slices from error signal tensor into output
438  bp_output.Empty(false);
439  bp_output.Resize(output_slice_size, m_error_signal_v->Width());
440  AbsDistMat* output_slice_v
441  = bp_output.Construct(bp_output.Grid(), bp_output.Root());
442  for(int slice = 0; slice < num_slices; ++slice) {
443  El::LockedView(*m_input_slice_v,
444  *this->m_error_signal_v,
445  El::IR(slice * input_slice_size + slice_offset_start,
446  slice * input_slice_size + slice_offset_end),
447  El::ALL);
448  El::View(*output_slice_v,
449  bp_output,
450  El::IR(slice * output_slice_size,
451  (slice+1) * output_slice_size),
452  El::ALL);
453  El::Copy(*m_input_slice_v, *output_slice_v);
454  }
455  delete output_slice_v;
456  }
457 
458  }
