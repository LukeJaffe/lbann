50 void lbann_callback_ltfb::on_batch_end(model *m) {
51  if (m->get_cur_step() % m_round_size != 0) {
52  return; // Not the end of a round.
53  }
54 
55  int partner = select_partner();
56  if (partner == m_comm->get_model_rank()) {
57  // No partner this round, skip.
58  return;
59  }
60  // Transfer data to secondary model.
61  exchange(m, partner);
62  // Evaluate on tournament data.
63  // Have to cast, assumes deep_neural_network.
64  deep_neural_network *dnn = static_cast<deep_neural_network *>(m);
65  deep_neural_network *remote_dnn =
66  static_cast<deep_neural_network *>(m_remote_model);
67  dnn->evaluate(execution_mode::validation);
68  remote_dnn->evaluate(execution_mode::validation);
69  // Reset the execution mode.
70  m->set_execution_mode(execution_mode::training);
71  m_remote_model->set_execution_mode(execution_mode::training);
72  // Assume there is a categorical accuracy metric.
73  double local_acc = 0;
74  double remote_acc = 0;
75  for (auto&& metric : m->metrics) {
76  if (metric->type == metrics::metric_type::categorical_accuracy) {
77  local_acc = metric->report_metric(execution_mode::validation);
78  break;
79  }
80  }
81  for (auto&& metric : m_remote_model->metrics) {
82  if (metric->type == metrics::metric_type::categorical_accuracy) {
83  remote_acc = metric->report_metric(execution_mode::validation);
84  break;
85  }
86  }
87  // If the remote is better, keep it.
88  if (remote_acc > local_acc) {
89  if (m_comm->am_model_master()) {
90  std::cout << m_comm->get_model_rank() << ": (step " << m->get_cur_step() << ") Replacing local model (" << local_acc << ") with " <<
91  "model " << partner << " (" << remote_acc << ")" << std::endl;
92  }
93  replace_with_remote(m);
94  }
95 }

97 int lbann_callback_ltfb::select_partner() {
98  int my_partner = 0;
99  // Master generates partners for everyone.
100  if (m_comm->am_world_master()) {
101  std::vector<int> ranks(m_comm->get_num_models());
102  std::iota(ranks.begin(), ranks.end(), 0);
103  std::shuffle(ranks.begin(), ranks.end(), get_fast_generator());
104  // Adjacent pairs become partners.
105  // Dilate so that we can do one scatter to every process.
106  std::vector<int> partners(
107  m_comm->get_num_models() * m_comm->get_procs_per_model());
108  for (size_t i = 0; i < (ranks.size() & ~1); i += 2) {
109  int rank1 = ranks[i];
110  int rank2 = ranks[i+1];
111  std::fill_n(partners.begin() + rank1*m_comm->get_procs_per_model(),
112  m_comm->get_procs_per_model(), rank2);
113  std::fill_n(partners.begin() + rank2*m_comm->get_procs_per_model(),
114  m_comm->get_procs_per_model(), rank1);
115  }
116  // Handle the last rank if needed.
117  if (partners.size() % 2 != 0) {
118  int last_rank = ranks[ranks.size() - 1];
119  std::fill_n(partners.begin() + last_rank*m_comm->get_procs_per_model(),
120  m_comm->get_procs_per_model(), last_rank);
121  }
122  El::mpi::Scatter(partners.data(), 1, &my_partner, 1, 0,
123  El::mpi::COMM_WORLD);
124  } else {
125  El::mpi::Scatter((int *) nullptr, 0, &my_partner, 1, 0, El::mpi::COMM_WORLD);
126  }
127  return my_partner;
128 }

130 void lbann_callback_ltfb::exchange(model *m, int partner) {
131  std::vector<Layer *>& layers = m->get_layers();
132  std::vector<Layer *>& remote_layers = m_remote_model->get_layers();
133  // Skip input/target layers.
134  for (size_t i = 1; i < layers.size() - 1; ++i) {
135  Layer *layer = layers[i];
136  Layer *remote_layer = remote_layers[i];
137  // TODO: Support sending optimizer state.
138  ElMat& weights = layer->get_weights_biases();
139  ElMat& remote_weights = remote_layer->get_weights_biases();
140  if (weights.Height() > 0) {
141  m_comm->sendrecv(weights.LockedBuffer(),
142  weights.LocalHeight()*weights.LocalWidth(),
143  partner,
144  remote_weights.Buffer(),
145  weights.LocalHeight()*weights.LocalWidth(),
146  partner);
147  }
148  }
149 }
