50 double categorical_cross_entropy::compute_categorical_cross_entropy(const AbsDistMat& predictions_v,
51  const AbsDistMat& groundtruth_v) {
52 
53  // Compute categorical cross entropy on current process
54  double total_error = 0;
55  for(int c = 0; c < groundtruth_v.LocalWidth(); c++) {
56  for(int r = 0; r < groundtruth_v.LocalHeight(); r++) {
57  const DataType true_val = groundtruth_v.GetLocal(r,c);
58  if(true_val != DataType(0)) {
59  double pred_val = predictions_v.GetLocal(r,c);
60  if(pred_val > DataType(0)) {
61  total_error += - true_val * std::log(pred_val);
62  } else {
63  total_error = DataType(INFINITY);
64  }
65  }
66  }
67  }
68 
69  // Get categorical cross entropy by summing results from all processes
70  total_error = El::mpi::AllReduce(total_error, groundtruth_v.DistComm());
71  return total_error;
72 
73 }

76 double categorical_cross_entropy::compute_obj_fn(const AbsDistMat& predictions_v,
77  const AbsDistMat& groundtruth_v) {
78  int cur_mini_batch_size = groundtruth_v.Width();
79 
80  double total_error = compute_categorical_cross_entropy(predictions_v, groundtruth_v);
81 
82  double avg_error = total_error / cur_mini_batch_size;
83 
84  return avg_error;

93  if (std::type_index(typeid(prev_layer)) ==
94  std::type_index(typeid(softmax_layer<data_layout::MODEL_PARALLEL>)) ||
95  std::type_index(typeid(prev_layer)) ==

97  El::Copy(predictions_v, error_signal_v);
98  El::Axpy(DataType(-1), groundtruth_v, error_signal_v);
99  }
100 
101  // Compute error signal (default case)
102  // Note: error_signal = - groundtruth ./ predictions
103  else {
104  El::IndexDependentFill(error_signal_v.Matrix(),
105  (std::function<DataType(El::Int,El::Int)>)

107  const DataType true_val = groundtruth_v.GetLocal(r,c);
108  if(true_val != DataType(0))
109  return - true_val / predictions_v.GetLocal(r,c);
110  else {
111  return DataType(0);
