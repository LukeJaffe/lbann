38 bool offline_patches_npz::load(const std::string filename, size_t first_n) {
39  m_item_class_list.clear();
40  m_file_root_list.clear();
41  m_file_variant_list.clear();
42 
43  // The list of arrays expected to be packed in the file.
44  // 'max_class' and 'variant_divider' are scalar values, and known in advance.
45  const std::set<std::string> dict =
46  {"item_root_list",
47  "item_variant_list",
48  "item_class_list",
49  "file_root_list",
50  "file_variant_list",
51  "max_class",
52  "variant_divider"};
53 
54  if (!check_if_file_exists(filename)) {
55  return false;
56  }
57  cnpy::npz_t dataset = cnpy::npz_load(filename);
58 
59  // check if all the arrays are included
60  for (const auto& np : dataset) {
61  if (dict.find(np.first) == dict.end()) {
62  return false;
63  }
64  }
65 
66  if (first_n > 0u) { // to use only first_n samples
67  shrink_to_fit(dataset["item_root_list"], first_n);
68  shrink_to_fit(dataset["item_variant_list"], first_n);
69  }
70  // Set the array of index sequences for root type
71  m_item_root_list = dataset["item_root_list"];
72 
73  if (first_n > 0u) { // to use only first_n samples
74  }
75  // Set the array of index sequences for variant type
76  m_item_variant_list = dataset["item_variant_list"];
77 
78  { // load the label array into a vector of label_t (uint8_t)
79  cnpy::NpyArray d_item_class_list = dataset["item_class_list"];
80  m_checked_ok = (d_item_class_list.shape.size() == 1u);
81  if (m_checked_ok) {
82  // In case of shrinking to first_n, make sure the size is consistent
83  const size_t num_samples = m_item_root_list.shape[0];
84  m_item_class_list.resize(num_samples);
85  for (size_t i=0u; i < num_samples; ++i) {
86  std::string digits(data_ptr<char>(d_item_class_list, {i}), d_item_class_list.word_size);
87  m_item_class_list[i] = static_cast<label_t>(atoi(digits.c_str()));
88  }
89  }
90  cnpy::npz_t::iterator it = dataset.find("item_class_list");
91  dataset.erase(it); // to keep memory footprint as low as possible
92  }
93 
94  { // load the array of dictionary substrings of root type
95  cnpy::NpyArray d_file_root_list = dataset["file_root_list"];
96  m_checked_ok = m_checked_ok && (d_file_root_list.shape.size() == 1u);
97  if (m_checked_ok) {
98  const size_t num_roots = d_file_root_list.shape[0];
99  m_file_root_list.resize(num_roots);
100  for (size_t i=0u; i < num_roots; ++i) {
101  std::string file_root(data_ptr<char>(d_file_root_list, {i}), d_file_root_list.word_size);
102  m_file_root_list[i] = std::string(file_root.c_str());
103  }
104  }
105  cnpy::npz_t::iterator it = dataset.find("file_root_list");
106  dataset.erase(it); // to keep memory footprint as low as possible
107  }
108  //for (const auto& fl: m_file_root_list) std::cout << fl << std::endl;
109 
110  { // load the array of dictionary substrings of variant type
111  cnpy::NpyArray d_file_variant_list = dataset["file_variant_list"];
112  m_checked_ok = m_checked_ok && (d_file_variant_list.shape.size() == 1u);
113  if (m_checked_ok) {
114  const size_t num_variants = d_file_variant_list.shape[0];
115  m_file_variant_list.resize(num_variants);
116  for (size_t i=0u; i < num_variants; ++i) {
117  std::string file_variant(data_ptr<char>(d_file_variant_list, {i}), d_file_variant_list.word_size);
118  m_file_variant_list[i] = std::string(file_variant.c_str());
119  }
120  }
121  cnpy::npz_t::iterator it = dataset.find("file_variant_list");
122  dataset.erase(it); // to keep memory footprint as low as possible
123  }
124  //for (const auto& fl: m_file_variant_list) std::cout << fl << std::endl;
125 
126  m_checked_ok = m_checked_ok && check_data();
127 
128  if (!m_checked_ok) {
129  //std::cout << get_description();
130  m_item_class_list.clear();
131  m_file_root_list.clear();
132  m_file_variant_list.clear();
133  throw lbann_exception("offline_patches_npz: loaded data not consistent");
134  }
135 
136  return m_checked_ok;
137 }

140 bool offline_patches_npz::check_data() const {
141  bool ok = (m_item_root_list.shape.size() == 2u) &&
142  (m_item_variant_list.shape.size() == 3u) &&
143  (m_file_root_list.size() > 0u) &&
144  (m_file_variant_list.size() > 0u) &&
145  (m_item_root_list.shape[0] == m_item_class_list.size()) &&
146  (m_item_variant_list.shape[0] == m_item_class_list.size()) &&
147  (m_item_root_list.shape[1] == m_num_patches) &&
148  (m_item_variant_list.shape[1] == m_num_patches) &&
149  (m_item_variant_list.shape[2] > 0u) &&
150  (m_item_root_list.word_size == sizeof(size_t)) &&
151  (m_item_variant_list.word_size == sizeof(size_t));
152  return ok;
153 }

187 offline_patches_npz::sample_t offline_patches_npz::get_sample(const size_t idx) const {
188  if (!m_checked_ok || idx >= get_num_samples()) {
189  throw lbann_exception("offline_patches_npz: invalid sample index");
190  }
191 
192  std::vector<std::string> file_names;
193 
194  for (size_t p = 0u; p < m_num_patches; ++p) {
195  const size_t root = data<size_t>(m_item_root_list, {idx, p});
196  if (root >= m_file_root_list.size()) {
197  using std::to_string;
198  throw lbann_exception("offline_patches_npz: invalid file_root_list index: "
199  + to_string(root) + " >= " + to_string(m_file_root_list.size()));
200  }
201  std::string file_name = m_file_root_list.at(root);
202 
203  const size_t* variant = &(data<size_t>(m_item_variant_list, {idx, p, 0u}));
204  const int ve = m_item_variant_list.shape.back()-1;
205  for (int i = 0; i < ve; ++i) {
206  file_name += m_file_variant_list.at(variant[i]) + m_variant_divider;
207  }
208 
209  file_name += m_file_variant_list.at(variant[ve]);
210  file_names.push_back(file_name);
211  }
212  return std::make_pair(file_names, m_item_class_list[idx]);
213 }

225 size_t offline_patches_npz::compute_cnpy_array_offset(
226  const cnpy::NpyArray& na, const std::vector<size_t> indices) {
227 
228  bool ok = (indices.size() == na.shape.size());
229  size_t unit_stride = 1u;
230  size_t offset = 0u;
231 
232  for (size_t i = indices.size(); ok && (i-- > 0u); ) {
233  ok = (indices[i] < na.shape[i]);
234  offset += indices[i] * unit_stride;
235  unit_stride *= na.shape[i];
236  }
237  if (!ok) {
238  throw lbann_exception("offline_patches_npz: invalid data index");
239  }
240  return offset;
241 }

244 void offline_patches_npz::shrink_to_fit(cnpy::NpyArray& na, size_t sz) {
245  if ((na.shape.size() == 0u) || (na.shape[0] <= sz)) return;
246  size_t new_size = sz;
247  for(size_t i = 1u; i < na.shape.size(); ++i) {
248  new_size *= na.shape[i];
249  }
250  na.data_holder->resize(new_size*na.word_size);
251  na.data_holder->shrink_to_fit();
252  na.num_vals = new_size;
253  na.shape[0] = sz;
254 }
