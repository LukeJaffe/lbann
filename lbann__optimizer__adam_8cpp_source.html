85 void lbann::adam::update(const AbsDistMat* gradient)
86 {
87 
88  m_current_beta1 *= m_beta1;
89  m_current_beta2 *= m_beta2;
90  const DataType correction = ( Sqrt(DataType(1) - m_current_beta2)
91  / (DataType(1) - m_current_beta1) );
92 
93  // Get local matrix data
94  const Int local_height = m_parameters->LocalHeight();
95  const Int local_width = m_parameters->LocalWidth();
96  DataType* parameters_buffer = m_parameters->Buffer();
97  const Int parameters_ldim = m_parameters->LDim();
98  const DataType* gradient_buffer = gradient->LockedBuffer();
99  const Int gradient_ldim = gradient->LDim();
100  DataType* moment1_buffer = m_moment1->Buffer();
101  const Int moment1_ldim = m_moment1->LDim();
102  DataType* moment2_buffer = m_moment2->Buffer();
103  const Int moment2_ldim = m_moment2->LDim();
104 
105  // Check if matrix data is contiguous
106  if(parameters_ldim != local_height
107  || gradient_ldim != local_height
108  || moment1_ldim != local_height
109  || moment2_ldim != local_height) {
110  // Update with non-contiguous data
111 #pragma omp parallel for collapse(2)
112  for(Int j=0; j<local_width; ++j) {
113  for(Int i=0; i<local_height; ++i) {
114  DataType& x = parameters_buffer[i+j*parameters_ldim];
115  const DataType g = gradient_buffer[i+j*gradient_ldim];
116  DataType& m1 = moment1_buffer[i+j*moment1_ldim];
117  DataType& m2 = moment2_buffer[i+j*moment2_ldim];
118  m1 = m_beta1 * m1 + (DataType(1) - m_beta1) * g;
119  m2 = m_beta2 * m2 + (DataType(1) - m_beta2) * g * g;
120  x -= m_learning_rate * correction * m1 / (Sqrt(m2) + m_eps);
121  }
122  }
123  }
124  else {
125  // Update with contiguous data
126 #pragma omp parallel for
127  for(Int i=0; i<local_height*local_width; ++i) {
128  DataType& x = parameters_buffer[i];
129  const DataType g = gradient_buffer[i];
130  DataType& m1 = moment1_buffer[i];
131  DataType& m2 = moment2_buffer[i];
132  m1 = m_beta1 * m1 + (DataType(1) - m_beta1) * g;
133  m2 = m_beta2 * m2 + (DataType(1) - m_beta2) * g * g;
134  x -= m_learning_rate * correction * m1 / (Sqrt(m2) + m_eps);
135  }
136  }
137 
138 }
