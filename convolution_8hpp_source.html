155  void setup_dims() {
156 
157  // Initialize previous neuron tensor dimensions
158  base_convolution_layer::setup_dims();
159 
160  // Initialize convolution kernel dimensions
161  this->m_kernel_dims.insert(this->m_kernel_dims.begin() + 1,
162  this->m_prev_neuron_dims[0]);
163 
164  // Check if previous neuron tensor dimensions are valid
165  #ifdef LBANN_DEBUG
166  if(this->m_num_neuron_dims != (int) this->m_kernel_dims.size() - 1) {
167  throw lbann_exception("convolution_layer: neuron tensor dimensions are unexpected");
168  }
169  #endif
170 
171  // Initialize neuron tensor dimensions
172  this->m_neuron_dims[0] = this->m_kernel_dims[0];
173  for(int i=0; i<this->m_num_neuron_dims-1; ++i) {
174  const int effective_dim = (this->m_prev_neuron_dims[i+1]
175  + 2*this->m_conv_pads[i]
176  - this->m_kernel_dims[i+2] + 1);
177  this->m_neuron_dims[i+1]= ((effective_dim + this->m_conv_strides[i] - 1)
178  / this->m_conv_strides[i]);
179  }
180  this->m_num_neurons = std::accumulate(this->m_neuron_dims.begin(),
181  this->m_neuron_dims.end(),
182  1,
183  std::multiplies<int>());
184 
185  // Get size of convolutional kernel
186  this->m_kernel_size = std::accumulate(m_kernel_dims.begin(),
187  m_kernel_dims.end(),
188  1,
189  std::multiplies<int>());
190 
191  }

193  void setup_data() {
194  if(m_bias_scaling_factor == DataType(0)) {

210  void setup_views() {

226  void setup_gpu() {
227  base_convolution_layer::setup_gpu();
228  #ifndef __LIB_CUDNN
229  throw lbann_exception("convolution_layer: cuDNN not detected");
230  #else
231 
232  // Maximum workspace size
233  cudaDeviceProp device_props;
234  CHECK_CUDA(cudaGetDeviceProperties(&device_props, 0));
235  const size_t work_space_limit = device_props.totalGlobalMem / 2;
236 
237  // Choose algorithms
238  CHECK_CUDNN(cudnnGetConvolutionForwardAlgorithm(this->m_cudnn->get_handle(),
239  this->m_prev_neurons_cudnn_desc,
240  m_kernel_cudnn_desc,
241  m_convolution_cudnn_desc,
242  this->m_neurons_cudnn_desc,
243  CUDNN_CONVOLUTION_FWD_SPECIFY_WORKSPACE_LIMIT,
244  work_space_limit,
245  &this->m_convolution_cudnn_algorithm));
246  CHECK_CUDNN(cudnnGetConvolutionBackwardDataAlgorithm(this->m_cudnn->get_handle(),
247  m_kernel_cudnn_desc,
248  this->m_neurons_cudnn_desc,
249  m_convolution_cudnn_desc,
250  this->m_prev_neurons_cudnn_desc,
251  CUDNN_CONVOLUTION_BWD_DATA_SPECIFY_WORKSPACE_LIMIT,
252  work_space_limit,
253  &this->m_transposed_convolution_cudnn_algorithm));
254  CHECK_CUDNN(cudnnGetConvolutionBackwardFilterAlgorithm(this->m_cudnn->get_handle(),
255  this->m_prev_neurons_cudnn_desc,
256  this->m_neurons_cudnn_desc,
257  m_convolution_cudnn_desc,
258  m_kernel_cudnn_desc,
259  CUDNN_CONVOLUTION_BWD_FILTER_SPECIFY_WORKSPACE_LIMIT,
260  work_space_limit,
261  &this->m_kernel_gradient_cudnn_algorithm));
262 
263  // Initialize work space
264  size_t max_work_space = 0;
265  size_t required_work_space;
266  CHECK_CUDNN(cudnnGetConvolutionForwardWorkspaceSize(this->m_cudnn->get_handle(),
267  this->m_prev_neurons_cudnn_desc,
268  this->m_kernel_cudnn_desc,
269  this->m_convolution_cudnn_desc,
270  this->m_neurons_cudnn_desc,
271  this->m_convolution_cudnn_algorithm,
272  &required_work_space));
273  max_work_space = std::max(max_work_space, required_work_space);
274  CHECK_CUDNN(cudnnGetConvolutionBackwardDataWorkspaceSize(this->m_cudnn->get_handle(),
275  this->m_kernel_cudnn_desc,
276  this->m_neurons_cudnn_desc,
277  this->m_convolution_cudnn_desc,
278  this->m_prev_neurons_cudnn_desc,
279  this->m_transposed_convolution_cudnn_algorithm,
280  &required_work_space));
281  max_work_space = std::max(max_work_space, required_work_space);
282  CHECK_CUDNN(cudnnGetConvolutionBackwardFilterWorkspaceSize(this->m_cudnn->get_handle(),
283  this->m_prev_neurons_cudnn_desc,
284  this->m_neurons_cudnn_desc,
285  this->m_convolution_cudnn_desc,
286  this->m_kernel_cudnn_desc,
287  this->m_kernel_gradient_cudnn_algorithm,
288  &required_work_space));
289  max_work_space = std::max(max_work_space, required_work_space);
290  for(int i=0; i<this->m_cudnn->get_num_gpus(); ++i) {
291  if(max_work_space > this->m_cudnn->get_work_space_size(i)) {
292  this->m_cudnn->set_work_space_size(i, max_work_space);
293  }
294  }
295 
296  #endif // __LIB_CUDNN
