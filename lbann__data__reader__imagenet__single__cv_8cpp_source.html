87 void lbann::DataReader_ImageNetSingle_cv::load()
88 {
89  string image_dir = get_file_dir();
90  string base_filename = get_data_filename();
91 
92  //open offsets file, with error checking
93  stringstream b;
94  b << image_dir << "/" << base_filename << "_offsets.txt";
95  if (is_master()) cout << "opening: " << b.str() << " " << endl;
96  ifstream in(b.str().c_str());
97  if (not in.is_open() and in.good()) {
98  stringstream err;
99  err << __FILE__ << " " << __LINE__
100  << " :: failed to open " << b.str() << " for reading";
101  throw lbann_exception(err.str());
102  }
103 
104  //read the offsets file
105  int n;
106  in >> n;
107  if (is_master()) cout << "num images: " << n << endl;
108  m_offsets.reserve(n);
109  m_offsets.push_back(make_pair(0,0));
110  size_t last_offset = 0;
111  size_t offset;
112  int label;
113  while (in >> offset >> label) {
114  m_offsets.push_back(make_pair(offset + last_offset, label));
115  last_offset = m_offsets.back().first;
116  }
117 
118  if (n+1 != m_offsets.size()) {
119  stringstream err;
120  err << __FILE__ << " " << __LINE__
121  << " :: we read " << m_offsets.size() << " offsets, but should have read " << n;
122  throw lbann_exception(err.str());
123  }
124  in.close();
125 
126  openDataStream();
127 
128  ShuffledIndices.resize(m_offsets.size());
129  for (size_t n = 0; n < m_offsets.size()-1; n++) {
130  ShuffledIndices[n] = n;
131  }
132 
133  select_subset_of_data();
134 }

137 int lbann::DataReader_ImageNetSingle_cv::fetch_data(Mat &X)

139  stringstream err;
140 
141  if(!DataReader::position_valid()) {
142  err << __FILE__ << " " << __LINE__ << " :: lbann::DataReader_ImageNet::fetch_data() - !DataReader::position_valid()";
143  throw lbann_exception(err.str());
144  }
145 
146  const int num_channel_values = m_image_width * m_image_height * m_image_num_channels;
147  int width, height;
148  int current_batch_size = getBatchSize();
149  const int end_pos = Min(CurrentPos+current_batch_size, ShuffledIndices.size());
150 
151  for (int n = CurrentPos; n < end_pos; ++n) {
152  int k = n - CurrentPos;
153  int idx = ShuffledIndices[n];
154  if (idx > m_offsets.size()-1) {
155  err << __FILE__ << " " << __LINE__ << " :: idx= " << idx << " is larger than m_offsets.size()= " << m_offsets.size() << " -2";
156  throw lbann_exception(err.str());

158  size_t start = m_offsets[idx].first;
159  size_t end = m_offsets[idx+1].first;
160 
161  if (end > m_file_size) {
162  err << __FILE__ << " " << __LINE__ << " :: end= " << end << " is larger than m_file_size= " << m_file_size << " for P_" << get_rank() << " with role: " << get_role() << " m_offsets.size(): " << m_offsets.size() << " n: " << n << " idx: " << idx;
163  throw lbann_exception(err.str());
164  }
165 
166  int ssz = end - start;
167 
168  if (ssz <= 0) {
169  err << "P_" << get_rank() << " start: " << start << " end: " << end << " ssz= " << ssz << " is <= 0";
170  throw lbann_exception(err.str());
171  }
172 
173  m_work_buffer.resize(ssz);
174  m_data_filestream.seekg(start);
175  m_data_filestream.read((char*)&m_work_buffer[0], ssz);
176 
177  int img_type=0;
178  ::Mat X_v;
179  View(X_v, X, IR(0, X.Height()), IR(k, k + 1));
180  bool ret = image_utils::import_image(m_work_buffer, width, height, img_type, *m_pp, X_v);
181 /*
182  unsigned char *p = &m_pixels[0];
183  bool ret = lbann::image_utils::loadJPG(m_work_buffer, width, height, false, p);
184 */
185 
186  if (_BUILTIN_FALSE(!ret)) {
187  stringstream err;
188  err << __FILE__ << " " << __LINE__ << " :: ImageNetSingle: image_utils::loadJPG failed to load index: " << idx;
189  throw lbann_exception(err.str());
190  }
191  if (_BUILTIN_FALSE((width * height * CV_MAT_CN(img_type)) != num_channel_values)) {
192  stringstream err;
193  err << __FILE__ << " " << __LINE__ << " :: ImageNetSingle: mismatch data size -- either width or height";
194  throw lbann_exception(err.str());
195  }
196 /*
197  for (size_t p = 0; p < m_pixels.size(); p++) {
198  X.Set(p, k, m_pixels[p]);
199  }
200 
201  auto pixel_col = X(IR(0, X.Height()), IR(k, k + 1));
202  augment(pixel_col, m_image_height, m_image_width, m_image_num_channels);
203  normalize(pixel_col, m_image_num_channels);
204 */
205  }
206 
207  return end_pos - CurrentPos;
