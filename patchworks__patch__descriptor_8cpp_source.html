104 bool patch_descriptor::get_first_patch(ROI& patch)
105 {
106  int x_center = (m_sample_area.width()+1)/2 + m_sample_area.left();
107  int y_center = (m_sample_area.height()+1)/2 + m_sample_area.top();
108  int x_margin = 0;
109  int y_margin = 0;
110 
111  if (m_mode_center == 0u) {
112  // Consider the jitter for a patch at the boundary of an image
113  x_margin = (m_width+1)/2 + m_jitter;
114  y_margin = (m_height+1)/2 + m_jitter;
115  } else if (m_mode_center == 1u) {
116  // The jitter for the center patch is a part of gap.
117  //if (m_jitter > m_gap) return false;
118  x_margin = m_width + (m_width+1)/2 + 2*m_jitter + m_gap;
119  y_margin = m_height + (m_height+1)/2 + 2*m_jitter + m_gap;
120  }
121 
122  if ((m_mode_center == 0u || m_mode_center == 1u)) {
123  // area where the center of a center patch can be in
124  ROI center_patch_area;
125  bool ok = center_patch_area.set_by_corners(x_margin + m_sample_area.left(),
126  y_margin + m_sample_area.top(),
127  m_sample_area.width() - x_margin,
128  m_sample_area.height() - y_margin);
129  if (!ok) {
130  std::cout << "invalid center patch area: " << center_patch_area << std::endl;
131  return false;
132  }
133  if (!center_patch_area.is_valid()) return false;
134 
135  // randomly generate the center coordinate within the center patch area
136  rand_t rg_center;
137  rg_center.init_uniform_int(0, 0, center_patch_area.width()-1);
138  rg_center.init_uniform_int(1, 0, center_patch_area.height()-1);
139 
140  x_center = rg_center.gen_uniform_int(0) + center_patch_area.left();
141  y_center = rg_center.gen_uniform_int(1) + center_patch_area.top();
142  }
143 
144  if (m_jitter > 0u) { // apply position jitter if enabled
145  x_center += m_rg_jitter.gen_uniform_int(0) - m_jitter;
146  y_center += m_rg_jitter.gen_uniform_int(1) - m_jitter;
147  }
148 
149  // set the center patch
150  ROI p;
151  if (!p.set_by_center(x_center, y_center, m_width, m_height) ||
152  !(m_sample_area >= p))
153  return false;
154 
155  m_patch_center = p;
156  patch = p;
157  positions.clear();
158  positions.push_back(patch);
159 
160  return true;
161 }

163 bool patch_descriptor::get_next_patch(ROI& patch)
164 {
165  bool got_one = false;
166 
167  do {
168  ROI p = m_patch_center;
169 
170  if (m_cur_patch_idx >= m_displacements.size()) return false;
171  p.move(m_displacements[m_cur_patch_idx++]);
172 
173  if (m_jitter > 0u) {
174  const int x_jitter = m_rg_jitter.gen_uniform_int(0) - m_jitter;
175  const int y_jitter = m_rg_jitter.gen_uniform_int(1) - m_jitter;
176  p.move(displacement_type(x_jitter, y_jitter));
177  }
178 
179  if (p.is_valid() && (m_sample_area >= p)) {
180  patch = p;
181  got_one = true;
182  }
183  } while (!got_one);
184 
185  positions.push_back(patch);
186  return true;
187 }
