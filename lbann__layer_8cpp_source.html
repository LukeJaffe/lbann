141 void lbann::Layer::forwardProp() {
142  double fp_start = get_time();
143 
144  // Get incoming activations and convert matrix distribution if necessary
145  if(fp_input != NULL) { // Input layers will not have a valid fp_input
146  DistData curr_dist = m_prev_activations->DistData();
147  DistData prev_dist = fp_input->DistData();
148  if(curr_dist.colDist == prev_dist.colDist
149  && curr_dist.rowDist == prev_dist.rowDist) {
150  View(*m_prev_activations, *fp_input);
151  } else {
152  *m_prev_activations = *fp_input;
153  }
154  }
155 
156  // Set matrix views based on current mini-batch size
157  fp_set_std_matrix_view();
158 
159 #ifdef __LIB_CUDNN
160  // Transfer inputs from CPU to GPUs if needed
161  if(m_using_gpus) {
162  if(!m_prev_layer_using_gpus) {
163  m_cudnn->scatter_to_gpus(m_prev_activations_d,
164  m_prev_activations_v->LockedMatrix(),
165  m_mini_batch_size_per_gpu);
166  } else {
167  m_prev_activations_d = *fp_input_d;
168  }
169  }
170 #endif
171 
172  // Apply connection regularization. (e.g. DropConnect).
173  for(size_t i=0; i<regularizers.size(); ++i) {
174  regularizers[i]->fp_connections();
175  }
176 
177  // Layer layer's linearity.
178  double fp_lin_start = get_time();
179  fp_linearity();
180  fp_linearity_time += get_time() - fp_lin_start;
181 
182  // Apply weight regularization (e.g. L2 normalization).
183  for(size_t i=0; i<regularizers.size(); ++i) {
184  regularizers[i]->fp_weights();
185  }
186 
187  // Apply activation function/nonlinearity.
188  double fp_nonlin_start = get_time();
189  fp_nonlinearity();
190  fp_nonlinearity_time += get_time() - fp_nonlin_start;
191 
192  // Apply activation regularization (e.g. Dropout).
193  for(size_t i=0; i<regularizers.size(); ++i) {
194  regularizers[i]->fp_activations();
195  }
196 
197 #ifdef __LIB_CUDNN
198  // Transfer outputs from GPUs to CPU if needed
199  if(m_using_gpus && !m_next_layer_using_gpus) {
200  m_cudnn->gather_from_gpus(m_activations_v->Matrix(),
201  m_activations_d,
202  m_mini_batch_size_per_gpu);
203  m_cudnn->synchronize();
204  }
205 #endif
206 
207  fp_time += get_time() - fp_start;
208 }

210 void lbann::Layer::backProp() {
211  double bp_start = get_time();
212 
213  // Set the view for all of the standard matrices based on the
214  // current mini-batch size
215  // bp_set_std_matrix_view();
216 
217  // Get incoming loss and convert matrix distribution if necessary
218  if(bp_input != NULL) { // Target layers will not have a valid bp_input
219  DistData curr_dist = m_prev_error_signal->DistData();
220  DistData next_dist = bp_input->DistData();
221  if(curr_dist.colDist == next_dist.colDist
222  && curr_dist.rowDist == next_dist.rowDist) {
223  View(*m_prev_error_signal, *bp_input);
224  View(*m_prev_error_signal_v,
225  *m_prev_error_signal,
226  ALL,
227  IR(0, m_prev_error_signal_v->Width()));
228  } else {
229  *m_prev_error_signal = *bp_input;
230  }
231  }
232 
233 #ifdef __LIB_CUDNN
234  // Transfer inputs from CPU to GPUs
235  if(m_using_gpus) {
236  if(!m_next_layer_using_gpus) {
237  m_cudnn->scatter_to_gpus(m_prev_error_signal_d,
238  m_prev_error_signal_v->LockedMatrix(),
239  m_mini_batch_size_per_gpu);
240  } else {
241  m_prev_error_signal_d = *bp_input_d;
242  }
243  }
244 #endif
245 
246  // Backprop activation regularization.
247  for(Int i=regularizers.size()-1; i>=0; --i) {
248  regularizers[i]->bp_activations();
249  }
250 
251  // Backprop the activation function/nonlinearity.
252  double bp_nonlin_start = get_time();
253  bp_nonlinearity();
254  bp_nonlinearity_time += get_time() - bp_nonlin_start;
255 
256  // Backprop weight regularization.
257  for(Int i=regularizers.size()-1; i>=0; --i) {
258  regularizers[i]->bp_weights();
259  }
260 
261  // Backprop the layer's linearity.
262  double bp_lin_start = get_time();
263  bp_linearity();
264  bp_linearity_time += get_time() - bp_lin_start;
265 
266  // Backprop connection regularization.
267  for(Int i=regularizers.size()-1; i>=0; --i) {
268  regularizers[i]->bp_connections();
269  }
270 
271 #ifdef __LIB_CUDNN
272  // Transfer outputs from GPUs to CPU
273  if(m_using_gpus && !m_prev_layer_using_gpus) {
274  m_cudnn->gather_from_gpus(m_error_signal_v->Matrix(),
275  m_error_signal_d,
276  m_mini_batch_size_per_gpu);
277  m_cudnn->synchronize();
278  }
279 #endif
280 
281  bp_time += get_time() - bp_start;
282 }

312  summarizer.reduce_max(prefix + "max", acts, step);
313  summarizer.reduce_stdev(prefix + "stdev", acts, step);
314  prefix = "layer" + std::to_string(static_cast<long long>(m_index)) + "/";
315  summarizer.reduce_scalar(prefix + "fp_time", fp_time, step);
316  summarizer.reduce_scalar(prefix + "fp_linearity_time", fp_linearity_time, step);
317  summarizer.reduce_scalar(prefix + "fp_nonlinearity_time", fp_nonlinearity_time, step);
318  summarizer.reduce_scalar(prefix + "bp_time", bp_time, step);

320  summarizer.reduce_scalar(prefix + "bp_nonlinearity_time", bp_nonlinearity_time, step);

323 }

465 void lbann::Layer::fp_set_std_matrix_view() {
466  Int cur_mini_batch_size = m_neural_network_model->get_current_mini_batch_size();
467 
468  View(*m_prev_activations_v, *m_prev_activations, ALL, IR(0, cur_mini_batch_size));
469  if (m_prev_error_signal->Height() > 0) {
470  // No previous error signal for the final layer.
471  View(*m_prev_error_signal_v, *m_prev_error_signal, ALL,
472  IR(0, cur_mini_batch_size));
473  }
474  View(*m_weighted_sum_v, *m_weighted_sum, ALL, IR(0, cur_mini_batch_size));
475  View(*m_error_signal_v, *m_error_signal, ALL, IR(0, cur_mini_batch_size));
476  View(*m_activations_v, *m_activations, ALL, IR(0, cur_mini_batch_size));
477 
478  // Update the layer's effective mini-batch size so it averages properly.
483  if(cur_mini_batch_size != m_mini_batch_size || 1) {
484  // When the current mini-batch is partial, check with the other
485  // models to figure out the entire size of the complete mini-batch
486  Int total_mini_batch_size = m_comm->intermodel_allreduce((Int) cur_mini_batch_size);
487  set_effective_minibatch_size(total_mini_batch_size);
488  } else {
489  set_effective_minibatch_size(cur_mini_batch_size * m_comm->get_num_models());
490  }
491 }
