132 void lbann::Layer::forwardProp() {
133  double fp_start = get_time();
134 
135 #ifdef __LIB_CUDNN
136  // Pin host memory if needed for GPU memory transfers
137  if(m_using_gpus && !m_prev_layer_using_gpus && !m_fp_input_pinned) {
138  if(fp_input != NULL
139  && m_prev_activations->DistData().colDist == fp_input->DistData().colDist
140  && m_prev_activations->DistData().rowDist == fp_input->DistData().rowDist) {
141  m_cudnn->pin_matrix(*fp_input);
142  }
143  else {
144  m_cudnn->pin_matrix(*m_prev_activations);
145  }
146  m_fp_input_pinned = true;
147  }
148  if(m_using_gpus && !m_next_layer_using_gpus && !m_fp_output_pinned) {
149  m_cudnn->pin_matrix(*m_activations);
150  m_fp_output_pinned = true;
151  }
152 #endif
153 
154  // Get incoming activations and convert matrix distribution if necessary
155  if(fp_input != NULL) { // Input layers will not have a valid fp_input
156  if(m_prev_activations->DistData().colDist == fp_input->DistData().colDist
157  && m_prev_activations->DistData().rowDist == fp_input->DistData().rowDist) {
158  View(*m_prev_activations, *fp_input);
159  } else {
160  Copy(*fp_input, *m_prev_activations);
161  }
162  }
163 
164  // Set matrix views based on current mini-batch size
165  fp_set_std_matrix_view();
166 
167 #ifdef __LIB_CUDNN
168  // Transfer inputs from CPU to GPUs if needed
169  if(m_using_gpus) {
170  if(!m_prev_layer_using_gpus) {
171  m_cudnn->scatter_to_gpus(m_prev_activations_d,
172  m_prev_activations_v->LockedMatrix(),
173  m_mini_batch_size_per_gpu);
174  } else {
175  m_prev_activations_d = *fp_input_d;
176  }
177  }
178 #endif
179 
180  // Apply layer's compute function
181  double fp_compute_start = get_time();
182  //cerr << "CALLING fp_compute()\n";
183  fp_compute();
184  //cerr << "DONE CALLING fp_compute()\n";
185  fp_compute_time += get_time() - fp_compute_start;
186 
187 #ifdef __LIB_CUDNN
188  // Transfer outputs from GPUs to CPU if needed
189  if(m_using_gpus && !m_next_layer_using_gpus) {
190  if(!m_fp_output_pinned) {
191  m_cudnn->pin_matrix(*m_activations);
192  }
193  m_cudnn->gather_from_gpus(m_activations_v->Matrix(),
194  m_activations_d,
195  m_mini_batch_size_per_gpu);
196  m_cudnn->synchronize();
197  }
198 #endif
199 
200  fp_time += get_time() - fp_start;
201 }

203 void lbann::Layer::backProp() {
204  double bp_start = get_time();
205 
206 #ifdef __LIB_CUDNN
207  // Pin host memory if needed for GPU memory transfers
208  if(m_using_gpus && !m_next_layer_using_gpus && !m_bp_input_pinned) {
209  if(bp_input != NULL
210  && m_prev_error_signal->DistData().colDist == bp_input->DistData().colDist
211  && m_prev_error_signal->DistData().rowDist == bp_input->DistData().rowDist) {
212  m_cudnn->pin_matrix(*bp_input);
213  }
214  else {
215  m_cudnn->pin_matrix(*m_prev_error_signal);
216  }
217  m_bp_input_pinned = true;
218  }
219  if(m_using_gpus && !m_prev_layer_using_gpus && !m_bp_output_pinned) {
220  m_cudnn->pin_matrix(*m_error_signal);
221  m_bp_output_pinned = true;
222  }
223 #endif
224 
225  // Get incoming loss and convert matrix distribution if necessary
226  if(bp_input != NULL) { // Target layers will not have a valid bp_input
227  if(m_prev_error_signal->DistData().colDist == bp_input->DistData().colDist
228  && m_prev_error_signal->DistData().rowDist == bp_input->DistData().rowDist) {
229  View(*m_prev_error_signal, *bp_input);
230  } else {
231  Copy(*bp_input, *m_prev_error_signal);
232  }
233  }
234 
235  // Set the view for all of the standard matrices based on the
236  // current mini-batch size
237  bp_set_std_matrix_view();
238 
239 #ifdef __LIB_CUDNN
240  // Transfer inputs from CPU to GPUs
241  if(m_using_gpus) {
242  if(!m_next_layer_using_gpus) {
243  m_cudnn->scatter_to_gpus(m_prev_error_signal_d,
244  m_prev_error_signal_v->LockedMatrix(),
245  m_mini_batch_size_per_gpu);
246  } else {
247  m_prev_error_signal_d = *bp_input_d;
248  }
249  }
250 #endif
251 
252  // Backprop the compute function.
253  double bp_compute_start = get_time();
254  bp_compute();
255  bp_compute_time += get_time() - bp_compute_start;
256 
257 #ifdef __LIB_CUDNN
258  // Transfer outputs from GPUs to CPU
259  if(m_using_gpus && !m_prev_layer_using_gpus) {
260  m_cudnn->gather_from_gpus(m_error_signal_v->Matrix(),
261  m_error_signal_d,
262  m_mini_batch_size_per_gpu);
263  m_cudnn->synchronize();
264  }
265 #endif
266 
267  bp_time += get_time() - bp_start;
268 }

381 void lbann::Layer::fp_set_std_matrix_view() {
382  Int cur_mini_batch_size = m_neural_network_model->get_current_mini_batch_size();
383  View(*m_prev_activations_v, *m_prev_activations, ALL, IR(0, cur_mini_batch_size));
384  View(*m_activations_v, *m_activations, ALL, IR(0, cur_mini_batch_size));
385 
386  // Update the layer's effective mini-batch size so it averages properly.
391  if(cur_mini_batch_size != m_mini_batch_size || 1) {
392  // When the current mini-batch is partial, check with the other
393  // models to figure out the entire size of the complete mini-batch
394  Int total_mini_batch_size = m_comm->intermodel_allreduce((Int) cur_mini_batch_size);
395  set_effective_minibatch_size(total_mini_batch_size);
396  } else {
397  set_effective_minibatch_size(cur_mini_batch_size * m_comm->get_num_models());
398  }
399 }
