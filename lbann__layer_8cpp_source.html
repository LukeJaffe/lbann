143 void lbann::Layer::forwardProp() {
144  double fp_start = get_time();
145 
146  // Get incoming activations and convert matrix distribution if necessary
147  if(fp_input != NULL) { // Input layers will not have a valid fp_input
148  DistData curr_dist = m_prev_activations->DistData();
149  DistData prev_dist = fp_input->DistData();
150  if(curr_dist.colDist == prev_dist.colDist
151  && curr_dist.rowDist == prev_dist.rowDist) {
152  View(*m_prev_activations, *fp_input);
153  }
154  else {
155  *m_prev_activations = *fp_input;
156  }
157  }
158 
159  // Set matrix views based on current mini-batch size
160  fp_set_std_matrix_view();
161 
162 #ifdef __LIB_CUDNN
163  // Transfer inputs from CPU to GPUs if needed
164  if(m_using_gpus) {
165  if(!m_prev_layer_using_gpus) {
166  m_cudnn->scatter_to_gpus(m_prev_activations_d,
167  m_prev_activations_v->LockedMatrix(),
168  m_mini_batch_size_per_gpu);
169  }
170  else {
171  m_prev_activations_d = *fp_input_d;
172  }
173  }
174 #endif
175 
176  // Apply connection regularization. (e.g. DropConnect).
177  for (regularizer* reg : regularizers) reg->fp_connections();
178 
179  // Layer layer's linearity.
180  double fp_lin_start = get_time();
181  fp_linearity();
182  fp_linearity_time += get_time() - fp_lin_start;
183 
184  // Apply weight regularization (e.g. L2 normalization).
185  for (regularizer* reg : regularizers) reg->fp_weights();
186 
187  // Apply activation function/nonlinearity.
188  double fp_nonlin_start = get_time();
189  fp_nonlinearity();
190  fp_nonlinearity_time += get_time() - fp_nonlin_start;
191 
192  // Apply activation regularization (e.g. Dropout).
193  for (regularizer* reg : regularizers) reg->fp_activations();
194 
195 #ifdef __LIB_CUDNN
196  // Transfer outputs from GPUs to CPU if needed
197  if(m_using_gpus && !m_next_layer_using_gpus) {
198  m_cudnn->gather_from_gpus(m_activations_v->Matrix(),
199  m_activations_d,
200  m_mini_batch_size_per_gpu);
201  m_cudnn->synchronize();
202  }
203 #endif
204 
205  fp_time += get_time() - fp_start;
206 }

208 void lbann::Layer::backProp() {
209  double bp_start = get_time();
210 
211  // Set the view for all of the standard matrices based on the
212  // current mini-batch size
213  // bp_set_std_matrix_view();
214 
215  // Get incoming loss and convert matrix distribution if necessary
216  if(bp_input != NULL) { // Target layers will not have a valid bp_input
217  DistData curr_dist = m_prev_error_signal->DistData();
218  DistData next_dist = bp_input->DistData();
219  if(curr_dist.colDist == next_dist.colDist
220  && curr_dist.rowDist == next_dist.rowDist) {
221  View(*m_prev_error_signal, *bp_input);
222  View(*m_prev_error_signal_v,
223  *m_prev_error_signal,
224  ALL,
225  IR(0, m_prev_error_signal_v->Width()));
226  }
227  else {
228  *m_prev_error_signal = *bp_input;
229  }
230  }
231 
232 #ifdef __LIB_CUDNN
233  // Transfer inputs from CPU to GPUs
234  if(m_using_gpus) {
235  if(!m_next_layer_using_gpus) {
236  m_cudnn->scatter_to_gpus(m_prev_error_signal_d,
237  m_prev_error_signal_v->LockedMatrix(),
238  m_mini_batch_size_per_gpu);
239  }
240  else {
241  m_prev_error_signal_d = *bp_input_d;
242  }
243  }
244 #endif
245 
246  // Backprop activation regularization.
247  for (regularizer* reg : regularizers) reg->bp_activations();
248 
249  // Backprop the activation function/nonlinearity.
250  double bp_nonlin_start = get_time();
251  bp_nonlinearity();
252  bp_nonlinearity_time += get_time() - bp_nonlin_start;
253 
254  // Backprop weight regularization.
255  for (regularizer* reg : regularizers) reg->bp_weights();
256 
257  // Backprop the layer's linearity.
258  double bp_lin_start = get_time();
259  bp_linearity();
260  bp_linearity_time += get_time() - bp_lin_start;
261 
262  // Backprop connection regularization.
263  for (regularizer* reg : regularizers) reg->bp_connections();
264 
265 #ifdef __LIB_CUDNN
266  // Transfer outputs from GPUs to CPU
267  if(m_using_gpus && !m_prev_layer_using_gpus) {
268  m_cudnn->gather_from_gpus(m_error_signal_v->Matrix(),
269  m_error_signal_d,
270  m_mini_batch_size_per_gpu);
271  m_cudnn->synchronize();
272  }
273 #endif
274 
275  bp_time += get_time() - bp_start;
276 }

466 void lbann::Layer::fp_set_std_matrix_view() {
467  Int cur_mini_batch_size = neural_network_model->get_current_mini_batch_size();
468 
469  if(m_prev_activations->Width() != 0)
470  View(*m_prev_activations_v, *m_prev_activations, ALL, IR(0, cur_mini_batch_size));
471  if(m_prev_error_signal->Width() != 0)
472  View(*m_prev_error_signal_v, *m_prev_error_signal, ALL, IR(0, cur_mini_batch_size));
473  if(m_weighted_sum->Width() != 0)
474  View(*m_weighted_sum_v, *m_weighted_sum, ALL, IR(0, cur_mini_batch_size));
475  if(m_error_signal->Width() != 0)
476  View(*m_error_signal_v, *m_error_signal, ALL, IR(0, cur_mini_batch_size));
477  if(m_activations->Width() != 0)
478  View(*m_activations_v, *m_activations, ALL, IR(0, cur_mini_batch_size));
479 
480  // Update the layer's effective mini-batch size so it averages properly.
481  if(cur_mini_batch_size != m_mini_batch_size) {
482  // When the current mini-batch is partial, check with the other
483  // models to figure out the entire size of the complete mini-batch
484  Int total_mini_batch_size = comm->intermodel_allreduce((Int) cur_mini_batch_size);
485  set_effective_minibatch_size(total_mini_batch_size);
486  }
487  else {
488  set_effective_minibatch_size(cur_mini_batch_size * comm->get_num_models());
489  }
490 }
