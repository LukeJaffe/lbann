124  virtual void fp_compute_cpu() {
125 
126  // Get local matrices
127  const Mat& prev_activations_local = this->m_prev_activations->LockedMatrix();
128  Mat& activations_local = this->m_activations->Matrix();
129 
130  // Local matrix parameters
131  const int local_height = prev_activations_local.Height();
132  const int local_width = prev_activations_local.Width();
133  const int prev_activations_ldim = prev_activations_local.LDim();
134  const int activations_ldim = activations_local.LDim();
135  const DataType* __restrict__ prev_activations_buffer
136  = prev_activations_local.LockedBuffer();
137  DataType* __restrict__ activations_buffer = activations_local.Buffer();
138 
139  // Apply activation function
140  if(prev_activations_ldim == local_height
141  && activations_ldim == local_height) {
142  // Contiguous data
143  #pragma omp parallel for
144  for(int i = 0; i < local_height * local_width; ++i) {
145  const DataType prev_activations_entry = prev_activations_buffer[i];
146  DataType& activations_entry = activations_buffer[i];
147  activations_entry = activation_function(prev_activations_entry);
148  }
149  } else {
150  // Non-contiguous data
151  #pragma omp parallel for collapse(2)
152  for(int col = 0; col < local_width; ++col) {
153  for(int row = 0; row < local_height; ++row) {
154  const DataType prev_activations_entry
155  = prev_activations_buffer[row + col * prev_activations_ldim];
156  DataType& activations_entry
157  = activations_buffer[row + col * activations_ldim];
158  activations_entry = activation_function(prev_activations_entry);
159  }
160  }
161  }
162 
163  }

165  virtual void bp_compute_cpu() {
166 
167  // Get local matrices
168  const Mat& prev_activations_local = this->m_prev_activations->LockedMatrix();
169  const Mat& prev_error_signal_local = this->m_prev_error_signal->LockedMatrix();
170  Mat& error_signal_local = this->m_error_signal->Matrix();
171 
172  // Local matrix parameters
173  const int local_height = prev_activations_local.Height();
174  const int local_width = prev_activations_local.Width();
175  const int prev_activations_ldim = prev_activations_local.LDim();
176  const int prev_error_signal_ldim = prev_error_signal_local.LDim();
177  const int error_signal_ldim = error_signal_local.LDim();
178  const DataType* __restrict__ prev_activations_buffer
179  = prev_activations_local.LockedBuffer();
180  const DataType* __restrict__ prev_error_signal_buffer
181  = prev_error_signal_local.LockedBuffer();
182  DataType* __restrict__ error_signal_buffer = error_signal_local.Buffer();
183 
184  // Apply activation function back propagation
185  if(prev_activations_ldim == local_height
186  && prev_error_signal_ldim == local_height
187  && error_signal_ldim == local_height) {
188  // Contiguous data
189  #pragma omp parallel for
190  for(int i = 0; i < local_height * local_width; ++i) {
191  const DataType prev_activations_entry = prev_activations_buffer[i];
192  const DataType prev_error_signal_entry = prev_error_signal_buffer[i];
193  DataType& error_signal_entry = error_signal_buffer[i];
194  error_signal_entry
195  = activation_function_gradient(prev_activations_entry) * prev_error_signal_entry;
196  }
197  } else {
198  // Non-contiguous data
199  #pragma omp parallel for collapse(2)
200  for(int col = 0; col < local_width; ++col) {
201  for(int row = 0; row < local_height; ++row) {
202  const DataType prev_activations_entry
203  = prev_activations_buffer[row + col * prev_activations_ldim];
204  const DataType prev_error_signal_entry
205  = prev_error_signal_buffer[row + col * prev_error_signal_ldim];
206  DataType& error_signal_entry

208  error_signal_entry
