36 void lbann_callback_dump_minibatch_sample_indices::dump_to_file(model *m, Layer *l, int64_t step) {
37  const std::string prefix = m_basename + _to_string(m->get_execution_mode()) + "-model" +
38  std::to_string(m->get_comm()->get_model_rank()) +
39  "-rank" + std::to_string(m->get_comm()->get_rank_in_model()) +
40  "-epoch" + std::to_string(m->get_cur_epoch()) + "-step" +
41  std::to_string(step) + "-layer";
42  if (!dynamic_cast<io_layer*>(l) || l->get_index() != 0) {
43  return;
44  }
45 
46  El::Matrix<El::Int>* indices = l->get_sample_indices_per_mb();
47 
48  if(indices->Height() != 0 && indices->Width() != 0) {
49  El::Write(*indices,
50  prefix + std::to_string(l->get_index()) +
51  "-MB_Sample_Indices",
52  El::ASCII);
53  }
54 }

60 void lbann_callback_dump_minibatch_sample_indices::on_evaluate_forward_prop_end(model *m, Layer *l) {
61  switch(m->get_execution_mode()) {
62  case execution_mode::validation:
63  dump_to_file(m, l, m->get_cur_validation_step());
64  break;
65  case execution_mode::testing:
66  dump_to_file(m, l, m->get_cur_testing_step());
67  break;
68  default:
69  throw lbann_exception("lbann_callback_dump_minibatch_sample_indices: invalid execution phase");
70  }
