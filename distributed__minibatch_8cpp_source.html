120 bool lbann::distributed_minibatch::is_data_set_processed() {
121  int num_readers_done = 0;
122  int max_active_parallel_readers = get_num_parallel_readers(); // When calculating if all parallel readers are done, include the maximum number,
123  // not just the ones in the last round. This will ensure that all readers, that had data
124  // will have distributed it.
125  int num_parallel_readers = m_num_valid_readers;
126 
127  if(m_comm->get_rank_in_model() < num_parallel_readers) {
128  if((m_comm->get_rank_in_model()+1)%num_parallel_readers == m_root) {
129  if(m_local_data_valid) {
130  stringstream err;
131  err << __FILE__ << " "<< __LINE__
132  << " :: lbann_input_layer_distributed_minibatch: all valid data was not processed.";
133  throw lbann_exception(err.str());
134  }
135  m_local_reader_done = !update_data_reader();
136  }
137  }
138 
140  if(m_local_reader_done) {
141  num_readers_done = 1;
142  }
143 
145  num_readers_done = m_comm->model_allreduce(num_readers_done);
146  if(num_readers_done == max_active_parallel_readers) {
147  m_local_reader_done = false;
148  m_root = 0;
149  m_num_data_per_epoch = 0;
152  return true;
153  } else {
154  return false;
155  }
156 }

158 int lbann::distributed_minibatch::compute_max_num_parallel_readers(long data_set_size, int mini_batch_size, int num_parallel_readers) {
160  if(data_set_size != 0) {
161  int max_num_parallel_readers = num_parallel_readers;
162  while(ceil((float)data_set_size / (float)(mini_batch_size * m_comm->get_num_models())) < max_num_parallel_readers) {
163  max_num_parallel_readers--;
164  }
165  if(max_num_parallel_readers != num_parallel_readers) {
166  std::cout << "Warning the training data set size " << data_set_size
167  << " is too small for the number of requested parallel readers "
168  << num_parallel_readers << ", using " << max_num_parallel_readers << "."
169  << std::endl;
170  }
171  return max_num_parallel_readers;
172  } else {
173  return 0;
174  }

177 void lbann::distributed_minibatch::calculate_num_iterations_per_epoch(generic_data_reader *data_reader) {
178  int max_mini_batch_size = data_reader->getm_batch_max();
179  int num_parallel_readers_per_model = max(1, (data_reader->get_batch_stride() / m_comm->get_num_models()) / max_mini_batch_size);
180  int min_stride_across_models = max_mini_batch_size * m_comm->get_num_models();
181 
182  data_reader->set_last_mini_batch_size(max_mini_batch_size);
183 
184  int num_whole_mini_batches_per_model = floor(data_reader->getNumData() / min_stride_across_models);
185  int num_whole_mini_batches_per_reader = floor(num_whole_mini_batches_per_model / num_parallel_readers_per_model);
186  int parallel_readers_with_extra_mini_batch = num_whole_mini_batches_per_model % num_parallel_readers_per_model;
187  int per_model_partial_mini_batch_size = (data_reader->getNumData() - (num_whole_mini_batches_per_model * min_stride_across_models))/(m_comm->get_num_models());
188  int world_master_remainder_data = 0;
189 
190  // Compute how many full "parallel" mini-batches are available
191  data_reader->set_last_mini_batch_threshold(num_whole_mini_batches_per_model * min_stride_across_models);
192 
193  if(m_comm->get_rank_in_model() < parallel_readers_with_extra_mini_batch) {
194  num_whole_mini_batches_per_reader += 1;
195  data_reader->set_last_mini_batch_size(max_mini_batch_size);
196  }
197 
198  data_reader->set_num_mini_batches_per_reader(num_whole_mini_batches_per_reader);
199 
200  int world_master_remainder_adjustment = data_reader->getNumData()
201  - (num_whole_mini_batches_per_model * min_stride_across_models)
202  - (per_model_partial_mini_batch_size * m_comm->get_num_models());
203  if(m_comm->am_world_master()) {
204  world_master_remainder_data = world_master_remainder_adjustment;
205  world_master_remainder_adjustment = 0;
206  }
207  per_model_partial_mini_batch_size += world_master_remainder_data;
208 
210  if(m_comm->get_rank_in_model() == parallel_readers_with_extra_mini_batch && per_model_partial_mini_batch_size > 0) {
211  data_reader->set_num_mini_batches_per_reader(data_reader->get_num_mini_batches_per_reader()+1);
212  data_reader->set_last_mini_batch_size(per_model_partial_mini_batch_size);
213  }
214 
215  if(data_reader->get_last_mini_batch_size() > max_mini_batch_size) {
216  throw new lbann_exception("Error in calculating the partial mini-batch size, exceeds the max mini-batch size");
217  }
218 
221 
223  data_reader->set_last_mini_batch_stride(data_reader->get_batch_stride());
224 
225  int last_mini_batch_offset = max(0, num_whole_mini_batches_per_reader - 1) * data_reader->get_batch_stride();
226 
228  if(m_comm->get_rank_in_model() == parallel_readers_with_extra_mini_batch && per_model_partial_mini_batch_size > 0) {
229  data_reader->set_last_mini_batch_stride((data_reader->get_last_mini_batch_threshold() - data_reader->get_base_offset() - data_reader->get_model_offset() - last_mini_batch_offset)
230  + m_comm->get_model_rank() * per_model_partial_mini_batch_size + world_master_remainder_adjustment);
231  }
232
