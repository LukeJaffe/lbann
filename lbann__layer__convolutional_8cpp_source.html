51 convolutional_layer::convolutional_layer(const uint index,
52  const Int num_dims,
53  const Int num_input_channels,
54  const Int* input_dims,
55  const Int num_output_channels,
56  const Int* filter_dims,
57  const Int* conv_pads,
58  const Int* conv_strides,
59  const Int mini_batch_size,
60  const activation_type activation,
61  const weight_initialization init,
62  lbann_comm* comm,
63  optimizer* opt,
64  std::vector<regularizer*> regs,
65  cudnn::cudnn_manager* cudnn)
66  : Layer(data_layout::DATA_PARALLEL, index, comm, opt, mini_batch_size, activation, regs),
67  m_weight_initialization(init),
68  m_num_dims(num_dims),
69  m_num_input_channels(num_input_channels),
70  m_num_output_channels(num_output_channels)
71 {
72  m_type = layer_type::convolution;
73 
74  // Initialize input dimensions and convolution parameters
75  m_input_dims.resize(m_num_dims);
76  m_filter_dims.resize(m_num_dims);
77  m_filter_size = m_num_input_channels*m_num_output_channels;
78  m_conv_pads.resize(num_dims);
79  m_conv_strides.resize(num_dims);
80  for(Int i=0; i<num_dims; ++i) {
81  m_input_dims[i] = input_dims[i];
82  m_filter_dims[i] = filter_dims[i];
83  m_filter_size *= filter_dims[i];
84  m_conv_pads[i] = conv_pads[i];
85  m_conv_strides[i] = conv_strides[i];
86  }
87 
88  // Calculate output dimensions
89  m_output_dims.resize(num_dims);
90  NumNeurons = num_output_channels;
91  for(Int i=0; i<num_dims; ++i) {
92  m_output_dims[i] = input_dims[i]+2*conv_pads[i]-filter_dims[i]+1;
93  m_output_dims[i] = (m_output_dims[i]+conv_strides[i]-1)/conv_strides[i];
94  NumNeurons *= m_output_dims[i];
95  }
96 
97 #ifdef __LIB_CUDNN
98  m_weights_gradient_per_gpu = StarMat(comm->get_model_grid());
99 #endif // #ifdef __LIB_CUDNN
100 
101 #ifdef __LIB_CUDNN
102 
103  to_pin_fwd = false;
104  to_pin_bwd = false;
105  is_pinned_fwd = false;
106  is_pinned_bwd = false;
107 
108  // Initialize cuDNN objects
109  m_input_desc = NULL;
110  m_output_desc = NULL;
111  m_bias_desc = NULL;
112  m_filter_desc = NULL;
113  m_convolution_desc = NULL;
114  m_activation_desc = NULL;
115  m_forward_algo = CUDNN_CONVOLUTION_FWD_ALGO_IMPLICIT_GEMM;
116  m_backward_filter_algo = CUDNN_CONVOLUTION_BWD_FILTER_ALGO_0;
117  m_backward_data_algo = CUDNN_CONVOLUTION_BWD_DATA_ALGO_0;
118 
119  // Set parameters for GPU implementation
120  if(cudnn) {
121  m_using_gpus = true;
122  m_cudnn = cudnn;
123  const Int num_gpus = m_cudnn->get_num_gpus();
124  const Int num_processes = comm->get_procs_per_model();
125  const Int local_mini_batch_size = (mini_batch_size + num_processes - 1) / num_processes;
126  m_mini_batch_size_per_gpu = (local_mini_batch_size + num_gpus - 1) / num_gpus;
127 
128  // Default behavior set to pin memory blocks used by cuDNN
129  pin_mem();
130  }
131 
132 #endif // #ifdef __LIB_CUDNN
133 
134 }

173 void convolutional_layer::setup(const int num_prev_neurons)
174 {
175  Layer::setup(num_prev_neurons);
176 
177 #ifdef __LIB_CUDNN
178  // Setup cuDNN objects
179  if(m_using_gpus) {
180  setup_gpu();
181  }
182 #endif // #ifdef __LIB_CUDNN
183 
184 #ifdef LBANN_DEBUG
185  // Check if input dimensions are valid
186  Int num_inputs = m_num_input_channels;
187  for(Int i=0; i<m_num_dims; ++i)
188  num_inputs *= m_input_dims[i];
189  if(num_inputs != m_num_prev_neurons) {
190  throw lbann_exception("lbann_layer_convolutional: unexpected number of input neurons");
191  }
192 #endif // #ifdef LBANN_DEBUG
193 
194  // Initialize matrices
195  Zeros(*m_weights, m_filter_size+m_num_output_channels, 1);
196  Zeros(*m_weights_gradient, m_filter_size+m_num_output_channels, 1);
197 #ifdef __LIB_CUDNN
198  if(m_using_gpus) {
199  Zeros(m_weights_gradient_per_gpu,
200  m_filter_size+m_num_output_channels,
201  m_cudnn->get_num_gpus());
202  }
203 #endif // #ifdef __LIB_CUDNN
204  Zeros(*m_prev_activations, m_num_prev_neurons, m_mini_batch_size);
205  Zeros(*m_error_signal, m_num_prev_neurons, m_mini_batch_size);
206  Zeros(*m_activations, NumNeurons, m_mini_batch_size);
207  Zeros(*m_prev_error_signal, NumNeurons, m_mini_batch_size);
208  Zeros(*m_weighted_sum, NumNeurons, m_mini_batch_size);
209 
210  // Initialize filters
211  StarMat filter;
212  View(filter, *m_weights, IR(0,m_filter_size), ALL);
213  const Int fan_in = m_filter_size / m_num_output_channels;
214  const Int fan_out = m_filter_size / m_num_input_channels;
215  initialize_matrix(filter, m_weight_initialization, fan_in, fan_out);
216 
217  // Initialize optimizer
218  if(m_optimizer != NULL)
219  m_optimizer->setup(m_weights);
220

223 void lbann::convolutional_layer::setup_gpu()
224 {
225 #ifndef __LIB_CUDNN
226  throw lbann_exception("lbann_layer_convolutional: cuDNN not detected");
227 #else
228 
229  // Get device properties
230  cudaDeviceProp device_props;
231  checkCUDA(cudaGetDeviceProperties(&device_props, 0));
232 
233  // Initialize descriptors
234  checkCUDNN(cudnnCreateTensorDescriptor(&m_input_desc));
235  checkCUDNN(cudnnCreateTensorDescriptor(&m_output_desc));
236  checkCUDNN(cudnnCreateTensorDescriptor(&m_bias_desc));
237  checkCUDNN(cudnnCreateFilterDescriptor(&m_filter_desc));
238  checkCUDNN(cudnnCreateConvolutionDescriptor(&m_convolution_desc));
239  checkCUDNN(cudnnCreateActivationDescriptor(&m_activation_desc));
240 
241  // Set input tensor descriptor
242  std::vector<int> input_dims(m_num_dims+2);
243  input_dims[0] = m_mini_batch_size_per_gpu;
244  input_dims[1] = m_num_input_channels;
245  for(Int i=0; i<m_num_dims; ++i)
246  input_dims[i+2] = m_input_dims[i];
247  std::vector<int> input_strides(m_num_dims+2);
248  input_strides[m_num_dims + 1] = 1;
249  for(Int i=m_num_dims; i>=0; --i)
250  input_strides[i] = input_strides[i+1] * input_dims[i+1];
251  checkCUDNN(cudnnSetTensorNdDescriptor(m_input_desc,
252  m_cudnn->get_cudnn_data_type(),
253  m_num_dims+2,
254  input_dims.data(),
255  input_strides.data()));
256 
257  // Set filter descriptor
258  std::vector<int> filter_dims(m_num_dims+2);
259  filter_dims[0] = m_num_output_channels;
260  filter_dims[1] = m_num_input_channels;
261  for(Int i=0; i<m_num_dims; ++i)
262  filter_dims[i+2] = m_filter_dims[i];
263  checkCUDNN(cudnnSetFilterNdDescriptor(m_filter_desc,
264  m_cudnn->get_cudnn_data_type(),
265  CUDNN_TENSOR_NCHW,
266  m_num_dims+2,
267  filter_dims.data()));
268 
269  // Set convolution descriptor
270  // Note: upscales are not supported as of cuDNN v5.1
271  std::vector<int> conv_upscales(m_num_dims, 1);
272  std::vector<int> conv_pads(m_num_dims);
273  for(Int i=0; i<m_num_dims; ++i)
274  conv_pads[i] = m_conv_pads[i];
275  std::vector<int> conv_strides(m_num_dims);
276  for(Int i=0; i<m_num_dims; ++i)
277  conv_strides[i] = m_conv_strides[i];
278  checkCUDNN(cudnnSetConvolutionNdDescriptor(m_convolution_desc,
279  m_num_dims,
280  conv_pads.data(),
281  conv_strides.data(),
282  conv_upscales.data(),
283  CUDNN_CONVOLUTION,
284  m_cudnn->get_cudnn_data_type()));
285 
286  // Set output tensor descriptor
287  std::vector<int> output_dims(m_num_dims+2);
288 #ifdef LBANN_DEBUG
289  checkCUDNN(cudnnGetConvolutionNdForwardOutputDim(m_convolution_desc,
290  m_input_desc,
291  m_filter_desc,
292  m_num_dims+2,
293  output_dims.data()));
294  if(output_dims[0] != m_mini_batch_size_per_gpu)
295  throw lbann_exception("lbann_layer_convolutional: invalid output dimensions");
296  if(output_dims[1] != m_num_output_channels)
297  throw lbann_exception("lbann_layer_convolutional: invalid output dimensions");
298  for(Int i=0; i<m_num_dims; ++i) {
299  if(output_dims[i+2] != m_output_dims[i]) {
300  throw lbann_exception("lbann_layer_convolutional: invalid output dimensions");
301  }
302  }
303 #else
304  output_dims[0] = m_mini_batch_size_per_gpu;
305  output_dims[1] = m_num_output_channels;
306  for(Int i=0; i<m_num_dims; ++i)
307  output_dims[i+2] = m_output_dims[i];
308 #endif // #ifdef LBANN_DEBUG
309  std::vector<int> output_strides(m_num_dims+2);
310  output_strides[m_num_dims + 1] = 1;
311  for(Int i=m_num_dims; i>=0; --i)
312  output_strides[i] = output_strides[i+1] * output_dims[i+1];
313  checkCUDNN(cudnnSetTensorNdDescriptor(m_output_desc,
314  m_cudnn->get_cudnn_data_type(),
315  m_num_dims+2,
316  output_dims.data(),
317  output_strides.data()));
318 
319  // Set output tensor descriptor
320  std::vector<int> bias_dims(m_num_dims+2, 1);
321  bias_dims[1] = m_num_output_channels;
322  std::vector<int> bias_strides(m_num_dims+2, 1);
323  bias_strides[0] = bias_dims[1];
324  checkCUDNN(cudnnSetTensorNdDescriptor(m_bias_desc,
325  m_cudnn->get_cudnn_data_type(),
326  m_num_dims+2,
327  bias_dims.data(),
328  bias_strides.data()));
329 
330  // Choose algorithms
331  checkCUDNN(cudnnGetConvolutionForwardAlgorithm(m_cudnn->get_handle(),
332  m_input_desc,
333  m_filter_desc,
334  m_convolution_desc,
335  m_output_desc,
336  CUDNN_CONVOLUTION_FWD_SPECIFY_WORKSPACE_LIMIT,
337  device_props.totalGlobalMem/2,
338  &m_forward_algo));
339  checkCUDNN(cudnnGetConvolutionBackwardFilterAlgorithm(m_cudnn->get_handle(),
340  m_input_desc,
341  m_output_desc,
342  m_convolution_desc,
343  m_filter_desc,
344  CUDNN_CONVOLUTION_BWD_FILTER_SPECIFY_WORKSPACE_LIMIT,
345  device_props.totalGlobalMem/2,
346  &m_backward_filter_algo));
347  checkCUDNN(cudnnGetConvolutionBackwardDataAlgorithm(m_cudnn->get_handle(),
348  m_filter_desc,
349  m_output_desc,
350  m_convolution_desc,
351  m_input_desc,
352  CUDNN_CONVOLUTION_BWD_DATA_SPECIFY_WORKSPACE_LIMIT,
353  device_props.totalGlobalMem/2,
354  &m_backward_data_algo));
355 
356  // Initialize work space
357  size_t max_work_space = 0;
358  size_t required_work_space;
359  checkCUDNN(cudnnGetConvolutionForwardWorkspaceSize(m_cudnn->get_handle(),
360  m_input_desc,
361  m_filter_desc,
362  m_convolution_desc,
363  m_output_desc,
364  m_forward_algo,
365  &required_work_space));
366  max_work_space = Max(max_work_space, required_work_space);
367  checkCUDNN(cudnnGetConvolutionBackwardFilterWorkspaceSize(m_cudnn->get_handle(),
368  m_input_desc,
369  m_output_desc,
370  m_convolution_desc,
371  m_filter_desc,
372  m_backward_filter_algo,
373  &required_work_space));
374  max_work_space = Max(max_work_space, required_work_space);
375  checkCUDNN(cudnnGetConvolutionBackwardDataWorkspaceSize(m_cudnn->get_handle(),
376  m_filter_desc,
377  m_output_desc,
378  m_convolution_desc,
379  m_input_desc,
380  m_backward_data_algo,
381  &required_work_space));
382  max_work_space = Max(max_work_space, required_work_space);
383  for(Int i=0; i<m_cudnn->get_num_gpus(); ++i) {
384  if(max_work_space > m_cudnn->get_work_space_size(i))
385  m_cudnn->set_work_space_size(i, max_work_space);
386  }
387 
388  // Set activation descriptor
389  if(m_activation_type != activation_type::ID) {
390  checkCUDNN(cudnnSetActivationDescriptor(m_activation_desc,
391  get_cudnn_activation_mode(m_activation_type),
392  CUDNN_PROPAGATE_NAN,
393  0.0));
394  }
395 
396  // Allocate GPU memory
397  m_cudnn->allocate_on_gpus(m_weights_d,
398  m_filter_size+m_num_output_channels,
399  1);
400  m_cudnn->allocate_on_gpus(m_weights_gradient_d,
401  m_filter_size+m_num_output_channels,
402  1);
403  m_cudnn->allocate_on_gpus(m_weighted_sum_d,
404  NumNeurons,
405  m_mini_batch_size_per_gpu);
406  m_cudnn->allocate_on_gpus(m_activations_d,
407  NumNeurons,
408  m_mini_batch_size_per_gpu);
409  m_cudnn->allocate_on_gpus(m_error_signal_d,
410  m_num_prev_neurons,
411  m_mini_batch_size_per_gpu);
412  if(!m_prev_layer_using_gpus) {
413  m_cudnn->allocate_on_gpus(m_prev_activations_d,
414  m_num_prev_neurons,
415  m_mini_batch_size_per_gpu);
416  }
417  if(!m_next_layer_using_gpus) {
418  m_cudnn->allocate_on_gpus(m_prev_error_signal_d,
419  NumNeurons,
420  m_mini_batch_size_per_gpu);
421  }
422 
423 #endif // #ifdef __LIB_CUDNN

662 void lbann::convolutional_layer::fp_linearity_cpu() {
663 
664  // Get local matrices
665  const Mat& prev_activations_local = m_prev_activations_v->LockedMatrix();
666  const Mat& weights_local = m_weights->LockedMatrix();
667  Mat& weighted_sum_local = m_weighted_sum_v->Matrix();
668  Mat& activations_local = m_activations_v->Matrix();
669 
670  // Get filter and bias
671  const Mat filter_local = LockedView(weights_local, IR(0,m_filter_size), ALL);
672  const Mat bias_local = LockedView(weights_local, IR(m_filter_size,END), ALL);
673 
674  // Input, output, and filter entries are divided amongst channels
675  const Int num_per_output_channel = NumNeurons / m_num_output_channels;
676  const Int num_per_input_channel = m_num_prev_neurons / m_num_input_channels;
677  const Int current_filter_size = m_filter_size / m_num_output_channels;
678  const Int current_filter_size_per_input_channel = current_filter_size / m_num_input_channels;
679 
680  // Apply bias
681  for(Int i=0; i<m_num_output_channels; ++i) {
682  Mat weighted_sum_channel
683  = View(weighted_sum_local,
684  IR(i*num_per_output_channel, (i+1)*num_per_output_channel),
685  ALL);
686  Fill(weighted_sum_channel, bias_local.Get(i,0));
687  }
688 
690  // Apply convolution
691  // Note: We are implicitly applying a convolution matrix where
692  // filter_entry is the (output_index, input_index) entry.
694 
695  // Iterate through samples, output channels, and input channels
696 #pragma omp parallel for
697  for(Int output_channel = 0;
698  output_channel < m_num_output_channels;
699  ++output_channel) {
700  std::vector<Int> filter_offsets(m_num_dims);
701  std::vector<Int> filter_pos(m_num_dims);
702  for(Int input_channel = 0;
703  input_channel < m_num_input_channels;
704  ++input_channel) {
705  for(Int sample = 0; sample < prev_activations_local.Width(); ++sample) {
706 
707  // Iterate through output entries in current output channel
708  // Note: each output entry corresponds to a different offset
709  // of the convolutional kernel
710  for(Int d = 0; d < m_num_dims; ++d) {
711  filter_offsets[d] = -m_conv_pads[d];
712  }
713  const Int start_output_index = output_channel*num_per_output_channel;
714  const Int end_output_index = (output_channel+1)*num_per_output_channel;
715  for(Int output_index = start_output_index;
716  output_index < end_output_index;
717  ++output_index) {
718  DataType& output_entry = weighted_sum_local(output_index, sample);
719 
720  // Iterate through filter entries for current input and output channel
721  for(Int d = 0; d < m_num_dims; ++d) {
722  filter_pos[d] = 0;
723  }
724  const Int start_filter_index
725  = output_channel*current_filter_size + input_channel*current_filter_size_per_input_channel;
726  const Int end_filter_index
727  = output_channel*current_filter_size + (input_channel+1)*current_filter_size_per_input_channel;
728  for(Int filter_index = start_filter_index;
729  filter_index < end_filter_index;
730  ++filter_index) {
731  const DataType filter_entry = filter_local(filter_index,0);
732 
733  // Get input entry corresponding to filter entry
734  Int input_index = 0;
735  bool valid_input_entry = true;
736  for(Int d = 0; d < m_num_dims; ++d) {
737  if(filter_offsets[d] + filter_pos[d] < 0
738  || filter_offsets[d] + filter_pos[d] >= m_input_dims[d]) {
739  valid_input_entry = false;
740  break;
741  }
742  input_index *= m_input_dims[d];
743  input_index += filter_offsets[d] + filter_pos[d];
744  }
745  input_index += input_channel*num_per_input_channel;
746 
747  // Update output entry
748  if(valid_input_entry) {
749  const DataType input_entry = prev_activations_local(input_index, sample);
750  output_entry += filter_entry*input_entry;
751  }
752 
753  // Move to next filter entry
754  ++filter_pos[m_num_dims-1];
755  for(Int d = m_num_dims - 1; d > 0; --d) {
756  if(filter_pos[d] >= m_filter_dims[d]) {
757  filter_pos[d] = 0;
758  ++filter_pos[d-1];
759  }
760  }
761 
762  }
763 
764  // Move to next filter offset and output entry
765  filter_offsets[m_num_dims-1] += m_conv_strides[m_num_dims-1];
766  for(Int d = m_num_dims - 1; d > 0; --d) {
767  if(filter_offsets[d] + m_filter_dims[d] > m_input_dims[d] + m_conv_pads[d]) {
768  filter_offsets[d] = -m_conv_pads[d];
769  filter_offsets[d-1] += m_conv_strides[d-1];
770  }
771  }
772 
773  }
774 
775  }
776  }
777  }
778 
779  // weighted_sum and output are identical after fp linearity step
780  Copy(weighted_sum_local, activations_local);
