121 void lbann::partitioned_minibatch::calculate_num_iterations_per_epoch(generic_data_reader *data_reader) {
122  int max_mini_batch_size = data_reader->getm_batch_max();
123  int num_parallel_readers_per_model = max(1, (data_reader->get_batch_stride() / m_comm->get_num_models()) / max_mini_batch_size);
124  int min_stride_across_models = max_mini_batch_size * m_comm->get_num_models();
125 
126  data_reader->set_last_mini_batch_size(max_mini_batch_size);
127 
128  int num_whole_mini_batches_per_model = floor(data_reader->getNumData() / min_stride_across_models);
129  int num_whole_mini_batches_per_reader = floor(num_whole_mini_batches_per_model / num_parallel_readers_per_model);
130  // int parallel_readers_with_extra_mini_batch = num_whole_mini_batches_per_model % num_parallel_readers_per_model;
131  int per_model_partial_mini_batch_size = (data_reader->getNumData() - (num_whole_mini_batches_per_model * min_stride_across_models))/(m_comm->get_num_models());
132  int world_master_remainder_data = 0;
133 
134  // Compute how many full "parallel" mini-batches are available
135  data_reader->set_last_mini_batch_threshold(num_whole_mini_batches_per_model * min_stride_across_models);
136 
137  data_reader->set_num_mini_batches_per_reader(num_whole_mini_batches_per_reader);
138 
139  int world_master_remainder_adjustment = data_reader->getNumData()
140  - (num_whole_mini_batches_per_model * min_stride_across_models)
141  - (per_model_partial_mini_batch_size * m_comm->get_num_models());
142  if(m_comm->am_world_master()) {
143  world_master_remainder_data = world_master_remainder_adjustment;
144  world_master_remainder_adjustment = 0;
145  }
146  per_model_partial_mini_batch_size += world_master_remainder_data;
147 
148  if(per_model_partial_mini_batch_size > 0 || world_master_remainder_adjustment > 0) {
149  data_reader->set_num_mini_batches_per_reader(data_reader->get_num_mini_batches_per_reader()+1);
150  data_reader->set_last_mini_batch_size(per_model_partial_mini_batch_size);
151  }
152 
153  data_reader->set_num_iterations_per_epoch(data_reader->get_num_mini_batches_per_reader());
154 
155  if(data_reader->get_last_mini_batch_size() > max_mini_batch_size) {
156  throw new lbann_exception("Error in calculating the partial mini-batch size, exceeds the max mini-batch size");
157  }
158 
161 
163  data_reader->set_last_mini_batch_stride(data_reader->get_batch_stride());
164 
165  int last_mini_batch_offset = max(0, num_whole_mini_batches_per_reader - 1) * data_reader->get_batch_stride();
166 
168  if(per_model_partial_mini_batch_size > 0 || world_master_remainder_adjustment > 0) {
169  data_reader->set_last_mini_batch_stride((data_reader->get_last_mini_batch_threshold() - data_reader->get_base_offset() - data_reader->get_model_offset() - last_mini_batch_offset) + m_comm->get_model_rank() * per_model_partial_mini_batch_size + m_comm->get_rank_in_model());
170  }
