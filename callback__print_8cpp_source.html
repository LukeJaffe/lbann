102 void lbann_callback_print::on_epoch_end(model *m) {
103  lbann_comm *comm = m->get_comm();
104  if (comm->am_model_master()) {
106  for (auto&& metric : m->get_metrics()) {
107  double train_score = metric->report_metric(execution_mode::training);
108  double validate_score = metric->report_metric(execution_mode::validation);
109  if (comm->am_world_master()) {
110  std::vector<double> train_scores(comm->get_num_models());
111  std::vector<double> validate_scores(comm->get_num_models());
112  comm->intermodel_gather(train_score, train_scores);
113  comm->intermodel_gather(validate_score, validate_scores);
114 
115  for (size_t i = 0; i < train_scores.size(); ++i) {
116  std::cout << "Model " << i;
117  std::cout << " @" << m->get_cur_step() << " steps";
118  std::cout << " Training " << metric->name() << ": " <<
119  train_scores[i] << metric->display_unit();
120  std::cout << " @" << m->get_cur_validation_step() <<
121  " validation steps Validation " << metric->name() << ": " <<
122  validate_scores[i] << metric->display_unit();
123  std::cout << std::endl;
124  }
125  } else {
126  comm->intermodel_gather(train_score, comm->get_intermodel_master());
127  comm->intermodel_gather(validate_score, comm->get_intermodel_master());
128  }
129  }
130  for (Layer *layer : m->get_layers()) {
131  layer->epoch_print();
132  }
133  }
134 }

136 void lbann_callback_print::on_test_end(model *m) {
137  lbann_comm *comm = m->get_comm();
138  if (comm->am_model_master()) {
140  for (auto&& metric : m->get_metrics()) {
141  double test_score = metric->report_metric(execution_mode::testing);
142  if (comm->am_world_master()) {
143  std::vector<double> test_scores(comm->get_num_models());
144  comm->intermodel_gather(test_score, test_scores);
145  for (size_t i = 0; i < test_scores.size(); ++i) {
146  std::cout << "Model " << i << " @" << m->get_cur_testing_step() <<
147  " testing steps external validation " << metric->name() << ": ";
148  std::cout << test_scores[i] << metric->display_unit() << std::endl;
149  }
150  } else {
151  comm->intermodel_gather(test_score, comm->get_intermodel_master());
152  }
