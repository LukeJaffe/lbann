62 int lbann::data_reader_nci_regression::fetch_data(Mat& X) {
63  if(!generic_data_reader::position_valid()) {
64  return 0;
65  }
66 
67  int current_batch_size = getm_batch_size();
68  ifstream ifs(m_infile.c_str());
69  if (!ifs) {
70  std::cout << "\n In load: can't open file : " << m_infile;
71  exit(1);
72  }
73 
74  string line;
75  int n = 0;
76  for (n = m_current_pos; n < m_current_pos + current_batch_size; ++n) {
77  if (n >= (int)m_shuffled_indices.size()) {
78  break;
79  }
80 
81  int k = n - m_current_pos;
82  int index = m_shuffled_indices[n];
83 
84  std::getline(ifs.seekg(m_index_map[index]),line);
85  istringstream lstream(line);
86  string field;
87  int col = 0, f=0;
88 
89  while(getline(lstream, field, ' ')) {
90  col++;
91  if (col == 3) {
92  if (field.empty()) {
93  break;
94  } else {
95  m_responses[index] = static_cast<DataType>(stof(field));
96  }
97  } else if (col > 5) {
98  if(field.empty()) {
99  X.Set(f, k, static_cast<DataType>(0));
100  } else {
101  X.Set(f,k,stof(field));
102  }
103  f++;
104  }//end if col > 5
105  }// end while loop
106  } // end for loop (batch)
107  ifs.close();
108  return (n - m_current_pos);
109 }

139 void lbann::data_reader_nci_regression::load() {
140  string infile = get_data_filename();
141  ifstream ifs(infile.c_str());
142  if (!ifs) {
143  stringstream err;
144  err << __FILE__ << " " << __LINE__
145  << "\n In load: can't open file : " << infile;
146  throw lbann_exception(err.str());
147  }
148  m_infile = infile;
149  string line;
150  int i;
151 
152  m_index_map.push_back(ifs.tellg());
153 #if 1
154  std::getline (ifs, line); // to skip the header
155  m_index_map.pop_back();
156  m_index_map.push_back(ifs.tellg());
157 #endif
158 
159  while(std::getline (ifs, line) ) {
160 #if 0 // enable to skip empty or comment lines
161  static const std::string whitespaces(" \t\f\v\n\r");
162  std::size_t pos = line.find_first_not_of(whitespaces);
163  if ((pos == std::string::npos) || (line[pos] == '#')) {
164  m_index_map.pop_back();
165  m_index_map.push_back(ifs.tellg());
166  continue;
167  }
168 #endif
169 
170  m_index_map.push_back(ifs.tellg()); // The beginning of the next data to read
171  m_num_samples++;
172 
173  string field;
174  istringstream lstream(line);
175  i=0;
176  m_num_features = 0;
177  while(getline(lstream, field, ' ')) {
178  i++;
179  if (i > 5) {
180  m_num_features++;
181  }
182  }
183  }
184  ifs.close();
185  m_index_map.pop_back();
186  m_index_map.shrink_to_fit();
187  m_responses.resize(m_num_samples);
188  // reset indices
189  m_shuffled_indices.clear();
190  m_shuffled_indices.resize(m_num_samples);
191  for (size_t n = 0; n < m_shuffled_indices.size(); ++n) {
192  m_shuffled_indices[n] = n;
193  }
194 
195  select_subset_of_data();
196 }
