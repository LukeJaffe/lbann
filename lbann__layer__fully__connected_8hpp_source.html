161  Zeros(*m_weights, NumNeurons, numPrevNeurons+1);
162  Zeros(*m_weights_gradient, NumNeurons, numPrevNeurons + 1);
163  Zeros(*m_prev_activations, numPrevNeurons, m_mini_batch_size);
164  Zeros(*m_weighted_sum, NumNeurons, m_mini_batch_size);
165  Zeros(*m_activations, NumNeurons, m_mini_batch_size);
166  Zeros(*m_prev_error_signal, NumNeurons, m_mini_batch_size);

170  View(*m_activation_weights_v, *m_weights, ALL, IR(0, numPrevNeurons));

200  void fp_linearity() {
201  // Apply forward prop linearity
202 
203  // Apply bias
204  Copy(*m_bias_weights_v, *m_bias_weights_repl);
205  const Mat& local_bias_weights = m_bias_weights_repl->Matrix();
206  IndexDependentFill(m_weighted_sum_v->Matrix(), (std::function<DataType(El::Int,El::Int)>)
207  ([&local_bias_weights](El::Int r, El::Int c)->DataType {
208  return local_bias_weights.Get(r);
209  }));
210  Scale(m_bias_term, *m_weighted_sum_v);
211 
212  // Apply weight matrix
213  switch(m_data_layout) {
214  case data_layout::MODEL_PARALLEL:
215  Gemm(NORMAL, NORMAL, DataType(1),
216  *m_activation_weights_v,
217  *m_prev_activations_v,
218  DataType(1),
219  *m_weighted_sum_v);
220  break;
221  case data_layout::DATA_PARALLEL:
222  Gemm(NORMAL, NORMAL, DataType(1),
223  m_activation_weights_v->LockedMatrix(),
224  m_prev_activations_v->LockedMatrix(),
225  DataType(1),
226  m_weighted_sum_v->Matrix());
227  break;
228  }
229 
230  // Copy result to output matrix
231  Copy(*m_weighted_sum_v, *m_activations_v);
232 
233  }

235  void bp_linearity() {
236 
237  switch(m_data_layout) {
238  case data_layout::MODEL_PARALLEL:
239  // Compute the partial delta update for the next lower layer
240  Gemm(TRANSPOSE, NORMAL, DataType(1),
241  *m_activation_weights_v,
242  *m_prev_error_signal_v,
243  DataType(0),
244  *m_error_signal_v);
245  // Compute update for activation weights
246  Gemm(NORMAL, TRANSPOSE, DataType(1)/get_effective_minibatch_size(),
247  *m_prev_error_signal_v,
248  *m_prev_activations_v,
249  DataType(0),
250  *m_activation_weights_gradient_v);
251  // Compute update for bias terms
252  Gemv(NORMAL, DataType(1)/get_effective_minibatch_size(),
253  *m_prev_error_signal_v,
254  *m_bias_bp_t_v,
255  DataType(0),
256  *m_bias_weights_gradient_v);
257  break;
258  case data_layout::DATA_PARALLEL:
259  // Compute the partial delta update for the next lower layer
260  Gemm(TRANSPOSE, NORMAL, DataType(1),
261  m_activation_weights_v->LockedMatrix(),
262  m_prev_error_signal_v->LockedMatrix(),
263  DataType(0),
264  m_error_signal_v->Matrix());
265  // Compute update for activation weights
266  Gemm(NORMAL, TRANSPOSE, DataType(1)/get_effective_minibatch_size(),
267  m_prev_error_signal_v->LockedMatrix(),
268  m_prev_activations_v->LockedMatrix(),
269  DataType(0),
270  m_activation_weights_gradient_v->Matrix());
271  // Compute update for bias terms
272  Gemv(NORMAL, DataType(1)/get_effective_minibatch_size(),
273  m_prev_error_signal_v->LockedMatrix(),
274  m_bias_bp_t_v->LockedMatrix(),
275  DataType(0),
276  m_bias_weights_gradient_v->Matrix());
277  // Add gradients from all processes
278  AllReduce(*m_weights_gradient,
279  m_weights_gradient->RedundantComm());
280  break;
281  }

285  DataType computeCost(DistMat& deltas) {
286  DataType avg_error = 0.0, total_error = 0.0;
287  // Compute the L2 norm on the deltas (activation - y)
288  ColSumMat norms;
289  ColumnTwoNorms(deltas, norms);
290  int c = 0;
291  // Sum the local, total error
292  for(int r = 0; r < norms.LocalHeight(); r++) {
293  total_error += norms.GetLocal(r,c);
294  }
295  mpi::AllReduce(total_error, norms.DistComm());
296  avg_error = total_error / norms.Height();
297  return avg_error;
298  }

322  DataType checkGradient(Layer& PrevLayer, const DataType Epsilon) {
323  DistMat WB_E1(m_weights->Grid());
324  DistMat WB_E2(m_weights->Grid());
325  DistMat Zs_E1(m_weighted_sum->Grid());
326  DistMat Zs_E2(m_weighted_sum->Grid());
327  DistMat Acts_E1(m_activations->Grid());
328  DistMat Acts_E2(m_activations->Grid());
329  DataType grad_diff = 0;
330  DataType grad_sum = 0;
331 
332  Zeros(WB_E1, m_weights->Height(), m_weights->Width());
333  Zeros(WB_E2, m_weights->Height(), m_weights->Width());
334  Zeros(Zs_E1, m_weighted_sum->Height(), m_weighted_sum->Width());
335  Zeros(Zs_E2, m_weighted_sum->Height(), m_weighted_sum->Width());
336  Zeros(Acts_E1, m_activations->Height(), m_activations->Width());
337  Zeros(Acts_E2, m_activations->Height(), m_activations->Width());
338 
339  Copy(*m_weights, WB_E1);
340  Copy(*m_weights, WB_E2);
341 
342  DataType sum_error = 0;
343  Int prow = 0;
344  Int pcol = 0;
345 
346  if(WB_E1.IsLocal(prow, pcol)) {
347  Int _prow = WB_E1.LocalRow(prow);
348  Int _pcol = WB_E1.LocalCol(pcol);
349  WB_E1.SetLocal(_prow, _pcol, WB_E1.GetLocal(_prow, _pcol) + Epsilon);
350  }
351  if(WB_E2.IsLocal(prow, pcol)) {
352  Int _prow = WB_E2.LocalRow(prow);
353  Int _pcol = WB_E2.LocalCol(pcol);
354  WB_E2.SetLocal(_prow, _pcol, WB_E2.GetLocal(_prow, _pcol) - Epsilon);
355  }
356  for (Int row = 0; row < m_weights->Height(); row++) {
357  for (Int col = 0; col < m_weights->Width(); col++) {
358  // printf("Updating %d: %d x %d\n", Index, row, col);
359  if(WB_E1.IsLocal(prow, pcol)) {
360  Int _prow = WB_E1.LocalRow(prow);
361  Int _pcol = WB_E1.LocalCol(pcol);
362  WB_E1.SetLocal(_prow, _pcol, WB_E1.GetLocal(_prow, _pcol) - Epsilon);
363  }
364  if(WB_E2.IsLocal(prow, pcol)) {
365  Int _prow = WB_E2.LocalRow(prow);
366  Int _pcol = WB_E2.LocalCol(pcol);
367  WB_E2.SetLocal(_prow, _pcol, WB_E2.GetLocal(_prow, _pcol) + Epsilon);
368  }
369  if(WB_E1.IsLocal(row, col)) {
370  Int _row = WB_E1.LocalRow(row);
371  Int _col = WB_E1.LocalCol(col);
372  WB_E1.SetLocal(_row, _col, WB_E1.GetLocal(_row, _col) + Epsilon);
373  }
374  if(WB_E2.IsLocal(row, col)) {
375  Int _row = WB_E2.LocalRow(row);
376  Int _col = WB_E2.LocalCol(col);
377  WB_E2.SetLocal(_row, _col, WB_E2.GetLocal(_row, _col) - Epsilon);
378  }
379 
380  // J(theta)
381  // this->fp_linearity(*WB, *(PrevLayer.Acts), *Zs, *Acts);
382  //this->fp_nonlinearity(*Acts);
383 
384  // J(thetaPlus(i))
385  // this->fp_linearity(WB_E1, *(PrevLayer.Acts), Zs_E1, Acts_E1);
386  //this->fp_nonlinearity(Acts_E1);
387 
388  // J(thetaMinus(i))
389  // this->fp_linearity(WB_E2, *(PrevLayer.Acts), Zs_E2, Acts_E2);
390  //this->fp_nonlinearity(Acts_E2);
391 
392  // this->getCost();
393 
394  // gradApprox(i) = J(thetaPlus(i)) - J(thetaMinus(i)) / (2*Epsilon)
395  Axpy(-1.0, *m_activations, Acts_E1);
396  Axpy(-1.0, *m_activations, Acts_E2);
397 
398  bool bad_E1 = false, bad_E2 = false;
399  for(Int r = 0; r < m_activations->Height(); r++) {
400  for(Int c = 0; c < m_activations->Width(); c++) {
401  if(Acts_E1.IsLocal(r,c)) {
402  Int _r = Acts_E1.LocalRow(r);
403  Int _c = Acts_E1.LocalCol(c);
404  if((Acts_E1.GetLocal(_r,_c) > 1e-12 || Acts_E1.GetLocal(_r,_c) < -1e-12) && r != row) {

408  }
409  if(Acts_E2.IsLocal(r,c)) {
410  Int _r = Acts_E2.LocalRow(r);
411  Int _c = Acts_E2.LocalCol(c);
412  if((Acts_E2.GetLocal(_r,_c) > 1e-12 || Acts_E2.GetLocal(_r,_c) < -1e-12) && r != row) {
413  bad_E2 = true;
414  // cout << "Acts_E2=["<< r << "," << c << "]="<<Acts_E2.GetLocal(_r,_c)<<endl;
415  }
416  }
417  }
418  }
419 
420  if(bad_E1) {
421  if(Acts_E1.Grid().Rank() == 0) {
422  printf("BAD ENTRY Acts_E1 %lld x %lld\n", row, col);
423  }
424  cout.precision(20);
425  Print(Acts_E1);
426  if(Acts_E1.Grid().Rank() == 0) {
427  printf("Acts\n");
428  }
429  Print(*m_activations);

432  if(Acts_E2.Grid().Rank() == 0) {
433  printf("BAD ENTRY Acts_E2 %lld x %lld\n", row, col);
434  }
435  Print(Acts_E2);
436  }
437 
438  prow = row;
439  pcol = col;
440  }
441  }
442  DataType grad_error = sqrt(grad_diff / grad_sum);
443  return grad_error;
