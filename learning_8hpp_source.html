50  virtual void l2_regularize_objective_function() {
51  if (m_l2_regularization_factor != DataType(0)) {
52 
53  // Get local weight data
54  const DataType *weights_buffer = m_weights->LockedBuffer();
55  const int weights_ldim = m_weights->LDim();
56  const int local_height = m_weights->LocalHeight();
57  const int local_width = m_weights->LocalWidth();
58 
59  // Compute sum of squares with Kahan summation
60  DataType sum = 0;
61  DataType correction = 0;
62  for (int col = 0; col < local_width; ++col) {
63  for (int row = 0; row < local_height; ++row) {
64  const DataType x = weights_buffer[row + col * weights_ldim];
65  const DataType term = x * x + correction;
66  const double next_sum = sum + term;
67  correction = term - (next_sum - sum);
68  sum = next_sum;
69  }
70  }
71  sum = El::mpi::AllReduce(sum, m_weights->DistComm());
72 
73  // Add regularization term to objective function
74  const DataType regularization_term = m_l2_regularization_factor * sum / 2;
75  this->m_neural_network_model->m_obj_fn->add_to_value(regularization_term);
76 
77  }
78  }
