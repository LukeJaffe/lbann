50  virtual void l2_regularize_objective_function() {
51  if (m_l2_regularization_factor != DataType(0)) {
52 
53  // Get local weight data
54  const DataType *weights_buffer = m_weights->LockedBuffer();
55  const int weights_ldim = m_weights->LDim();
56  const int local_height = m_weights->LocalHeight();
57  const int local_width = m_weights->LocalWidth();
58 
59  // Compute sum of squares
60  DataType sum = 0;
61  const int block_size = std::max((int) (64 / sizeof(DataType)), 1);
62  #pragma omp parallel for reduction(+:sum) collapse(2)
63  for(int col = 0; col < local_width; ++col) {
64  for(int block_start = 0; block_start < local_height; block_start += block_size) {
65  double block_sum = 0;
66  const int block_end = std::min(block_start + block_size, local_height);
67  for(int row = block_start; row < block_end; ++row) {
68  const DataType x = weights_buffer[row + col * weights_ldim];
69  block_sum += x * x;
70  }
71  sum += block_sum;
72  }
73  }
74  sum = El::mpi::AllReduce(sum, m_weights->DistComm());
75 
76  // Add regularization term to objective function
77  const DataType regularization_term = m_l2_regularization_factor * sum / 2;
78  this->m_neural_network_model->m_obj_fn->add_to_value(regularization_term);
79 
80  }
81  }

94  virtual void group_lasso_regularize_objective_function() {
95  if (m_group_lasso_regularization_factor != DataType(0)) {
96 
97  // Get local weight data
98  const DataType *weights_buffer = m_weights->LockedBuffer();
99  const int weights_ldim = m_weights->LDim();
100  const int width = m_weights->Width();
101  const int local_height = m_weights->LocalHeight();
102  const int local_width = m_weights->LocalWidth();
103 
104  // Compute sum of squares with Kahan summation
105  Mat colSumSqrs = Mat(1, width);
106  El::Zero(colSumSqrs);
107  for (int col = 0; col < local_width; ++col) {
108  DataType sum = 0;
109  DataType correction = 0;
110  for (int row = 0; row < local_height; ++row) {
111  const DataType x = weights_buffer[row + col * weights_ldim];
112  const DataType term = x * x + correction;
113  const double next_sum = sum + term;
114  correction = term - (next_sum - sum);
115  sum = next_sum;
116  }
117  colSumSqrs(0, m_weights->GlobalCol(col)) = sum;
118  }
119  El::AllReduce(colSumSqrs, m_weights->DistComm(), El::mpi::SUM);
120 
121  // Add regularization term to objective function
122  DataType sumColL2Norms = 0;
123  for (int i = width - 1; i >= 0; --i) sumColL2Norms += std::sqrt(colSumSqrs(0, i));
124  const DataType regularization_term = m_group_lasso_regularization_factor * sumColL2Norms;
125  this->m_neural_network_model->m_obj_fn->add_to_value(regularization_term);
126  }
127  }

130  virtual void group_lasso_regularize_gradient() {
131  if (m_group_lasso_regularization_factor != DataType(0)) {
132  //Group lasso gradient will be m_weights where each column c of m_weights is normalized by $||c||_{2}$.
133  //So first we compute $L_2$ norm of each column of m_weights as in group_lasso_regularize_objective_function.
134  const DataType *weights_buffer = m_weights->LockedBuffer();
135  const int weights_ldim = m_weights->LDim();
136  const int width = m_weights->Width();
137  const int local_height = m_weights->LocalHeight();
138  const int local_width = m_weights->LocalWidth();
139 
140  // Compute sum of squares with Kahan summation
141  Mat colSumSqrs = Mat(1, width);
142  for (int col = 0; col < local_width; ++col) {
143  DataType sum = 0;
144  DataType correction = 0;
145  for (int row = 0; row < local_height; ++row) {
146  const DataType x = weights_buffer[row + col * weights_ldim];
147  const DataType term = x * x + correction;
148  const double next_sum = sum + term;
149  correction = term - (next_sum - sum);
150  sum = next_sum;
151  }
152  colSumSqrs(0, m_weights->GlobalCol(col)) = sum;
153  }
154  El::AllReduce(colSumSqrs, m_weights->DistComm(), El::mpi::SUM);
155  El::EntrywiseMap(colSumSqrs, std::function<DataType(const DataType&)>(
156  [](const DataType& x) {
157  return std::sqrt(x);
158  }));
159 
160  //update m_weights_graident using colSumSqrs.
161  Mat& m_weights_gradient_local = m_weights_gradient->Matrix();
162  Mat& m_weights_local = m_weights->Matrix();
163  AbsDistMat& boo = *m_weights_gradient;
164 
165  El::IndexDependentMap(m_weights_gradient_local,
166  (std::function<DataType(El::Int,El::Int,const DataType&)>)
167  ([&colSumSqrs, &m_weights_local, &boo](El::Int r, El::Int c, const DataType& x)
168  -> DataType {
169  const DataType colL2Norm = colSumSqrs(0, boo.GlobalCol(c));
170  if (colL2Norm != DataType(0)) {
171  return x + m_weights_local(r, c)/colL2Norm;
172  } else {
173  return x;
174  };
175  }));
176  }
177  }
