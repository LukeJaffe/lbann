169  void fp_compute() {
170  ElMat *input_acts = this->m_prev_activations;
171  Mat& input_acts_local = input_acts->Matrix();
172  Mat& acts_local = this->m_activations->Matrix();
173  Mat& gamma_local = m_gamma->Matrix();
174  Mat& beta_local = m_beta->Matrix();
175  const El::Int local_width = acts_local.Width();
176  const El::Int local_height = acts_local.Height();
177  if (this->get_execution_mode() == execution_mode::training) {
178  // Compute row-wise mean and standard deviation
179  rowwise_mean_and_stdev(*input_acts, *m_mean, *m_stdev);
180  Mat& mean_local = m_mean->Matrix();
181  Mat& stdev_local = m_stdev->Matrix();
182  // Compute transformed activations xhat = (x-mean)/stdev
183 #pragma omp parallel for collapse(2)
184  for (El::Int col = 0; col < local_width; ++col) {
185  for (El::Int row = 0; row < local_height; ++row) {
186  // Normalize.
187  acts_local(row, col) = input_acts_local(row, col) - mean_local(row, 0);
188  acts_local(row, col) /= stdev_local(row, 0) + DataType(1e-7);
189  // Apply scale/shift.
190  acts_local(row, col) *= gamma_local(row, 0);
191  acts_local(row, col) += beta_local(row, 0);
192  }
193  }
194  // Update the running averages.
195  Scale(m_decay, *m_running_mean);
196  Scale(m_decay, *m_running_stdev);
197  Axpy(DataType(1) - m_decay, *m_mean, *m_running_mean);
198  Axpy(DataType(1) - m_decay, *m_stdev, *m_running_stdev);
199  } else if (this->get_execution_mode() == execution_mode::validation ||
200  this->get_execution_mode() == execution_mode::testing) {
201  // Use the running mean/standard deviation to normalize.
202  const Mat& mean_local = m_running_mean->LockedMatrix();
203  const Mat& stdev_local = m_running_stdev->LockedMatrix();
204 #pragma omp parallel for collapse(2)
205  for (El::Int col = 0; col < local_width; ++col) {
206  for (El::Int row = 0; row < local_height; ++row) {
207  // Normalize.
208  acts_local(row, col) = input_acts_local(row, col) - mean_local(row, 0);
209  acts_local(row, col) /= stdev_local(row, 0) + DataType(1e-7);
210  // Apply scale/shift.
211  acts_local(row, col) *= gamma_local(row, 0);
212  acts_local(row, col) += beta_local(row, 0);
213  }
214  }
215  }
216  }

218  void bp_compute() {
219  // No backprop when not training.
220  if (this->get_execution_mode() != execution_mode::training) {
221  return;
222  }
223  const ElMat *input_bpsignal = this->m_prev_error_signal;
224  const ElMat *acts = this->m_prev_activations;
225  const El::Int mbsize = acts->Width();
226  const Mat& input_bp_local = input_bpsignal->LockedMatrix();
227  Mat& bp_local = this->m_error_signal->Matrix();
228  const Mat& acts_local = acts->LockedMatrix();
229  const El::Int local_height = input_bp_local.Height();
230  const El::Int local_width = input_bp_local.Width();
231  const Mat& mean_local = m_mean->LockedMatrix();
232  const Mat& stdev_local = m_stdev->LockedMatrix();
233  const Mat& gamma_local = m_gamma->LockedMatrix();
234  Mat& dgamma_local = m_dgamma->Matrix();
235  Mat& dbeta_local = m_dbeta->Matrix();
236  // Compute the derivatives of gamma and beta.
237 #pragma omp parallel for
238  for (El::Int row = 0; row < local_height; ++row) {
239  dbeta_local(row, 0) = 0.0;
240  dgamma_local(row, 0) = 0.0;
241  for (El::Int col = 0; col < local_width; ++col) {
242  dbeta_local(row, 0) += input_bp_local(row, col);
243  dgamma_local(row, 0) +=
244  ((acts_local(row, col) - mean_local(row, 0)) /
245  (stdev_local(row, 0) + DataType(1e-7))) *
246  input_bp_local(row, col);
247  }
248  }
249  AllReduce(*m_dbeta, m_dbeta->RedundantComm(), mpi::SUM);
250  AllReduce(*m_dgamma, m_dgamma->RedundantComm(), mpi::SUM);
251  // Update the backprop gradient signal.
252 #pragma omp parallel for collapse(2)
253  for (El::Int row = 0; row < local_height; ++row) {
254  for (El::Int col = 0; col < local_width; ++col) {
255  bp_local(row, col) = mbsize * input_bp_local(row, col);
256  bp_local(row, col) -= dbeta_local(row, 0);
257  bp_local(row, col) -=
258  ((acts_local(row, col) - mean_local(row, 0)) /
259  (stdev_local(row, 0) + DataType(1e-7))) *
260  dgamma_local(row, 0);
261  bp_local(row, col) /= DataType(mbsize);
262  bp_local(row, col) *= gamma_local(row, 0);
263  bp_local(row, col) /= stdev_local(row, 0) + DataType(1e-7);
264  bp_local(row, col) += DataType(1e-8); // Avoid very small values.
265  }
266  }
267  }
