119  void fp_compute() {
120  ElMat *input_acts = this->m_prev_activations;
121  Mat& input_acts_local = input_acts->Matrix();
122  Mat& acts_local = this->m_activations->Matrix();
123  Mat& gamma_local = m_gamma->Matrix();
124  Mat& beta_local = m_beta->Matrix();
125  const El::Int local_width = acts_local.Width();
126  const El::Int local_height = acts_local.Height();
127  if (this->get_execution_mode() == execution_mode::training) {
128  // Compute row-wise mean and standard deviation
129  rowwise_mean_and_stdev(*input_acts, *m_mean, *m_stdev);
130  Mat& mean_local = m_mean->Matrix();
131  Mat& stdev_local = m_stdev->Matrix();
132  // Compute transformed activations xhat = (x-mean)/stdev
133 #pragma omp parallel for collapse(2)
134  for (El::Int col = 0; col < local_width; ++col) {
135  for (El::Int row = 0; row < local_height; ++row) {
136  // Normalize.
137  acts_local(row, col) = input_acts_local(row, col) - mean_local(row, 0);
138  acts_local(row, col) /= stdev_local(row, 0) + DataType(1e-7);
139  // Apply scale/shift.
140  acts_local(row, col) *= gamma_local(row, 0);
141  acts_local(row, col) += beta_local(row, 0);
142  }
143  }
144  // Update the running averages.
145  Scale(m_decay, *m_running_mean);
146  Scale(m_decay, *m_running_stdev);
147  Axpy(DataType(1) - m_decay, *m_mean, *m_running_mean);
148  Axpy(DataType(1) - m_decay, *m_stdev, *m_running_stdev);
149  } else if (this->get_execution_mode() == execution_mode::validation ||
150  this->get_execution_mode() == execution_mode::testing) {
151  // Use the running mean/standard deviation to normalize.
152  const Mat& mean_local = m_running_mean->LockedMatrix();
153  const Mat& stdev_local = m_running_stdev->LockedMatrix();
154 #pragma omp parallel for collapse(2)
155  for (El::Int col = 0; col < local_width; ++col) {
156  for (El::Int row = 0; row < local_height; ++row) {
157  // Normalize.
158  acts_local(row, col) = input_acts_local(row, col) - mean_local(row, 0);
159  acts_local(row, col) /= stdev_local(row, 0) + DataType(1e-7);
160  // Apply scale/shift.
161  acts_local(row, col) *= gamma_local(row, 0);
162  acts_local(row, col) += beta_local(row, 0);
163  }
164  }
165  }
166  }

168  void bp_compute() {
169  // No backprop when not training.
170  if (this->get_execution_mode() != execution_mode::training) {
171  return;
172  }
173  const ElMat *input_bpsignal = this->m_prev_error_signal;
174  const ElMat *acts = this->m_prev_activations;
175  const El::Int mbsize = acts->Width();
176  const Mat& input_bp_local = input_bpsignal->LockedMatrix();
177  Mat& bp_local = this->m_error_signal->Matrix();
178  const Mat& acts_local = acts->LockedMatrix();
179  const El::Int local_height = input_bp_local.Height();
180  const El::Int local_width = input_bp_local.Width();
181  const Mat& mean_local = m_mean->LockedMatrix();
182  const Mat& stdev_local = m_stdev->LockedMatrix();
183  const Mat& gamma_local = m_gamma->LockedMatrix();
184  Mat& dgamma_local = m_dgamma->Matrix();
185  Mat& dbeta_local = m_dbeta->Matrix();
186  // Compute the derivatives of gamma and beta.
187 #pragma omp parallel for
188  for (El::Int row = 0; row < local_height; ++row) {
189  dbeta_local(row, 0) = 0.0;
190  dgamma_local(row, 0) = 0.0;
191  for (El::Int col = 0; col < local_width; ++col) {
192  dbeta_local(row, 0) += input_bp_local(row, col);
193  dgamma_local(row, 0) +=
194  ((acts_local(row, col) - mean_local(row, 0)) /
195  (stdev_local(row, 0) + DataType(1e-7))) *
196  input_bp_local(row, col);
197  }
198  }
199  AllReduce(*m_dbeta, m_dbeta->RedundantComm(), mpi::SUM);
200  AllReduce(*m_dgamma, m_dgamma->RedundantComm(), mpi::SUM);
201  // Update the backprop gradient signal.
202 #pragma omp parallel for collapse(2)
203  for (El::Int row = 0; row < local_height; ++row) {
204  for (El::Int col = 0; col < local_width; ++col) {
205  bp_local(row, col) = mbsize * input_bp_local(row, col);
206  bp_local(row, col) -= dbeta_local(row, 0);
207  bp_local(row, col) -=
208  ((acts_local(row, col) - mean_local(row, 0)) /
209  (stdev_local(row, 0) + DataType(1e-7))) *
210  dgamma_local(row, 0);
211  bp_local(row, col) /= DataType(mbsize);
212  bp_local(row, col) *= gamma_local(row, 0);
213  bp_local(row, col) /= stdev_local(row, 0) + DataType(1e-7);
214  bp_local(row, col) += DataType(1e-8); // Avoid very small values.
215  }
216  }
217  }
