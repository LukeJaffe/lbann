216 template<class InputIterator, class OutputIterator>
217  OutputIterator scale (InputIterator first, InputIterator last, OutputIterator result, const double alpha, const double beta)
218 {
219  if ((alpha == 1.0) && (beta == 0.0)) {
220  if ((std::is_same< typename std::remove_cv<InputIterator>::type, typename std::remove_cv<OutputIterator>::type >::value) &&
221  (reinterpret_cast<const void*>(first) == reinterpret_cast<const void*>(result)))
222  {
223  std::advance(result, std::distance(first,last));
224  return result;
225  } else {
226  return std::copy(first, last, result);
227  }
228  }
229 
230  typedef typename std::iterator_traits<OutputIterator>::value_type T;
231  while (first != last) {
232  *result = cv::saturate_cast<T>(alpha * (*first) + beta);
233  ++result; ++first;
234  }
235  return result;
236 }
