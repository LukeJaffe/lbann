95 template<typename T, int NCh>
96 inline bool image_utils::copy_cvMat_to_buf_with_full_info(const cv::Mat& image, std::vector<uint8_t>& buf, const bool vFlip)
97 {
98  if (image.empty()) return false;
99 
100  const int Width = image.cols;
101  const int Height = image.rows;
102  const int sz = Height*Width;
103 
104  typedef cv::Vec<T, NCh> Vec_T;
105 
106  buf.resize(Height*Width*NCh*sizeof(T));
107  T* Pixels = reinterpret_cast<T*>(&(buf[0]));
108 
109  for (int y = 0; y < Height; ++y) {
110  for (int x = 0; x < Width; ++x) {
111  const Vec_T pixel = image.at<Vec_T>(y, x);
112  const int offset = (vFlip) ? ((Height - 1 - y) * Width + x) : (y * Width + x);
113 
114  T* px_ptr = Pixels + offset;
115  for(int ch = 0; ch < NCh; ++ch, px_ptr += sz) {
116  *px_ptr = pixel[ch];
117  }
118  }
119  }
120 
121  return true;
122 }

130 template<typename T>

132 {

150 template<typename T, int NCh>
151 inline cv::Mat image_utils::copy_buf_to_cvMat_with_full_info(const std::vector<uint8_t>& buf, const int Width, const int Height, const bool vFlip)
152 {
153  using namespace lbann::patchworks;
154  typedef cv_image_type<T, NCh> Img_T;
155  typedef cv::Vec<T, NCh> Vec_T;
156 
157  const int sz = Height*Width;
158 
159  if (sz*NCh*sizeof(T) != buf.size()) return cv::Mat();
160 
161  const T* Pixels = reinterpret_cast<const T*>(&(buf[0]));
162 
163  cv::Mat image = cv::Mat(Height, Width, Img_T::T());
164 
165  for (int y = 0; y < Height; y++) {
166  for (int x = 0; x < Width; x++) {
167  Vec_T pixel;
168  const int offset = (vFlip) ? ((Height - 1 - y) * Width + x) : (y * Width + x);
169  const T* px_ptr = Pixels + offset;
170  for(int ch = 0; ch < NCh; ++ch, px_ptr += sz) {
171  pixel[ch] = *px_ptr;
172  }
173  image.at<Vec_T>(y, x) = pixel;
174  }
175  }
176 
177  return image;
178 }

187 template<typename T>
188 inline cv::Mat image_utils::copy_buf_to_cvMat_with_known_type(const std::vector<uint8_t>& buf, const int Width, const int Height, const bool vFlip)
189 {
190  const size_t S = static_cast<size_t>(Width*Height*sizeof(T));
191  const size_t NCh = buf.size()/S;
192 
193  if (S*NCh != buf.size()) return cv::Mat();
194 
195  switch(NCh) {
196  case 1u: return copy_buf_to_cvMat_with_full_info<T,1>(buf, Width, Height, vFlip);
197  case 2u: return copy_buf_to_cvMat_with_full_info<T,2>(buf, Width, Height, vFlip);
198  case 3u: return copy_buf_to_cvMat_with_full_info<T,3>(buf, Width, Height, vFlip);
199  case 4u: return copy_buf_to_cvMat_with_full_info<T,4>(buf, Width, Height, vFlip);
200  }
201  return cv::Mat();
