185  void setup_data() {
186  learning::setup_data();
187  // Initialize matrices
188  // Note: the weights-bias matrix has an extra column so it includes bias term
189  El::Zeros(*this->m_weights, this->m_num_neurons, this->m_num_prev_neurons+1);
190  this->m_weights_gradient->Resize(this->m_num_neurons, this->m_num_prev_neurons + 1);
191 
192  // Initialize the activations part of the weight matrix -- leave the bias term weights zero
193  El::View(*this->m_activation_weights_v, *this->m_weights, El::ALL, El::IR(0, this->m_num_prev_neurons));
194  initialize_matrix(*this->m_activation_weights_v, m_weight_initialization, this->m_num_prev_neurons, this->m_num_neurons);
195 
196  // Initialize optimizer
197  if (this->m_optimizer != NULL) {
198  this->m_optimizer->setup(this->m_weights);
199  }

202  void setup_views() {
203  learning::setup_views();
204  // Setup independent views of the weight matrix for the activations
205  // Note this is duplicated in setup_data for convenience.
206  El::View(*this->m_activation_weights_v, *this->m_weights, El::ALL, El::IR(0, this->m_num_prev_neurons));
207 
208  // Setup independent views of the weights gradient matrix for the activations
209  El::View(*m_activation_weights_gradient_v, *this->m_weights_gradient, El::ALL, El::IR(0, this->m_num_prev_neurons));
210 
211  // Setup independent views of the weights and gradient matrix for the bias terms
212  El::View(*m_bias_weights_v, *this->m_weights, El::ALL, El::IR(this->m_num_prev_neurons));
213  El::View(*m_bias_weights_gradient_v, *this->m_weights_gradient, El::ALL, El::IR(this->m_num_prev_neurons));
214  }

216  void fp_compute() {
217  // Apply weight matrix
218  fp_compute_weights();
219 
220  // Apply bias if needed
221  if(m_bias_scaling_factor != DataType(0)) {
222  El::Copy(*m_bias_weights_v, *m_bias_weights_repl);
223  const Mat& local_bias_weights = m_bias_weights_repl->Matrix();
224  El::IndexDependentMap(this->m_activations_v->Matrix(),
225  (std::function<DataType(El::Int,El::Int,const DataType&)>)
226  ([this,&local_bias_weights](El::Int r, El::Int c,const DataType& z)->DataType {
227  return z + m_bias_scaling_factor * local_bias_weights.Get(r, 0);
228  }));
229  }
230  }
231 
232  void bp_compute() {
233  // Compute the error signal and gradients.
234  bp_compute_weights();
235 
236  // Compute bias update if needed
237  if(m_bias_scaling_factor != DataType(0)) {
238  El::RowSum(*this->m_prev_error_signal_v,
239  *m_bias_weights_gradient_repl);
240  El::Scale(m_bias_scaling_factor /
241  this->m_neural_network_model->get_effective_mini_batch_size(),
242  *m_bias_weights_gradient_repl);
243  El::Copy(*m_bias_weights_gradient_repl, *m_bias_weights_gradient_v);
244  }
245  }
246 
247  DataType computeCost(DistMat& deltas) {
248  DataType avg_error = 0.0, total_error = 0.0;
249  // Compute the L2 norm on the deltas (activation - y)
250  ColSumMat norms;
251  ColumnTwoNorms(deltas, norms);
252  int c = 0;
253  // Sum the local, total error
254  for(int r = 0; r < norms.LocalHeight(); r++) {
255  total_error += norms.GetLocal(r,c);
256  }
257  mpi::AllReduce(total_error, norms.DistComm());
258  avg_error = total_error / norms.Height();
259  return avg_error;
260  }
261 
262  bool update_compute() {
263  if(this->m_execution_mode == execution_mode::training) {
264  this->l2_regularize();
265  this->m_optimizer->update(this->m_weights_gradient);
266  }
267  return true;
268  }
