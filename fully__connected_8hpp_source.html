118  void setup(int numPrevNeurons) {

121  // Initialize matrices

123  El::Zeros(*this->m_weights, this->m_num_neurons, numPrevNeurons+1);

150  void fp_compute(void) {
151 
152  // Apply weight matrix
153  switch(T_layout) {
154  case data_layout::MODEL_PARALLEL:
155  El::Gemm(NORMAL, NORMAL, DataType(1),
156  *this->m_activation_weights_v,
157  *this->m_prev_activations_v,
158  DataType(0),
159  *this->m_activations_v);
160  break;
161  case data_layout::DATA_PARALLEL:
162  El::Gemm(NORMAL, NORMAL, DataType(1),
163  this->m_activation_weights_v->LockedMatrix(),
164  this->m_prev_activations_v->LockedMatrix(),
165  DataType(0),
166  this->m_activations_v->Matrix());
167  break;
168  }
169 
170  // Apply bias if needed
171  if(m_bias_scaling_factor != DataType(0)) {
172  El::Copy(*m_bias_weights_v, *m_bias_weights_repl);
173  const Mat& local_bias_weights = m_bias_weights_repl->Matrix();
174  El::IndexDependentMap(this->m_activations_v->Matrix(),
175  (std::function<DataType(El::Int,El::Int,const DataType&)>)
176  ([this,&local_bias_weights](El::Int r, El::Int c,const DataType& z)->DataType {
177  return z + m_bias_scaling_factor * local_bias_weights.Get(r, 0);
178  }));
179  }
180 
181  }

183  void bp_compute(void) {
184 
185  switch(T_layout) {
186  case data_layout::MODEL_PARALLEL:
187 
188  // Compute the partial delta update for the next lower layer
189  El::Gemm(El::TRANSPOSE, El::NORMAL, DataType(1),
190  *this->m_activation_weights_v,
191  *this->m_prev_error_signal_v,
192  DataType(0),
193  *this->m_error_signal_v);
194 
195  // Compute update for activation weights
196  El::Gemm(El::NORMAL, El::TRANSPOSE, DataType(1)/this->get_effective_minibatch_size(),
197  *this->m_prev_error_signal_v,
198  *this->m_prev_activations_v,
199  DataType(0),
200  *this->m_activation_weights_gradient_v);
201  break;
202 
203  case data_layout::DATA_PARALLEL:
204 
205  // Compute the partial delta update for the next lower layer
206  El::Gemm(El::TRANSPOSE, El::NORMAL, DataType(1),
207  this->m_activation_weights_v->LockedMatrix(),
208  this->m_prev_error_signal_v->LockedMatrix(),
209  DataType(0),
210  this->m_error_signal_v->Matrix());
211 
212  // Compute update for activation weights
213  El::Gemm(El::NORMAL, El::TRANSPOSE, DataType(1)/this->get_effective_minibatch_size(),
214  this->m_prev_error_signal_v->LockedMatrix(),
215  this->m_prev_activations_v->LockedMatrix(),
216  DataType(0),
217  this->m_activation_weights_gradient_v->Matrix());
218  El::AllReduce(*this->m_activation_weights_gradient_v,
219  this->m_activation_weights_gradient_v->RedundantComm());
220  break;
221  }
222 
223  // Compute bias update if needed
224  if(m_bias_scaling_factor != DataType(0)) {
225  El::RowSum(*this->m_prev_error_signal_v,
226  *m_bias_weights_gradient_repl);
227  El::Scale(m_bias_scaling_factor / this->get_effective_minibatch_size(),
228  *m_bias_weights_gradient_v);
229  El::Copy(*m_bias_weights_gradient_repl, *m_bias_weights_gradient_v);
230  }
231 
232  }

234  DataType computeCost(DistMat& deltas) {
235  DataType avg_error = 0.0, total_error = 0.0;
236  // Compute the L2 norm on the deltas (activation - y)
237  ColSumMat norms;
238  ColumnTwoNorms(deltas, norms);
239  int c = 0;
240  // Sum the local, total error
241  for(int r = 0; r < norms.LocalHeight(); r++) {
242  total_error += norms.GetLocal(r,c);
243  }
244  mpi::AllReduce(total_error, norms.DistComm());
245  avg_error = total_error / norms.Height();
246  return avg_error;
247  }
