116 void lbann::SoftmaxLayer::fp_linearity()
117 {
118 
119  // Apply weight matrix
120  switch(m_data_layout) {
121  case data_layout::MODEL_PARALLEL:
122  Gemm(NORMAL, NORMAL, DataType(1),
123  *m_weights,
124  *m_prev_activations_v,
125  DataType(0),
126  *m_weighted_sum_v);
127  break;
128  case data_layout::DATA_PARALLEL:
129  Gemm(NORMAL, NORMAL, DataType(1),
130  m_weights->LockedMatrix(),
131  m_prev_activations_v->LockedMatrix(),
132  DataType(0),
133  m_weighted_sum_v->Matrix());
134  break;
135  }
136 
137  // Copy result to output matrix
138  Copy(*m_weighted_sum_v, *m_activations_v);
139 
140 }

142 void lbann::SoftmaxLayer::fp_nonlinearity()
143 {
144 
145  // Get local matrices and parameters
146  Mat& workspace_local = m_workspace_v->Matrix();
147  Mat& activations_local = m_activations_v->Matrix();
148  const Int local_height = activations_local.Height();
149  const Int local_width = activations_local.Width();
150 
151  // Find maximum entry in each column
152 #pragma omp parallel for
153  for(Int c=0; c<local_width; ++c) {
154  DataType max_entry = -INFINITY;
155  for(Int r=0; r<local_height; ++r) {
156  max_entry = Max(max_entry, activations_local.Get(r,c));
157  }
158  workspace_local.Set(Int(0), c, max_entry);
159  }
160  AllReduce(*m_workspace_v, m_workspace_v->RedundantComm(), mpi::MAX);
161 
162  // Subtract column max and exponentiate activations
163  // Note: Subtracting the column max prevents activations from blowing
164  // up. Large negative values underflow to 0.
165  IndexDependentMap(activations_local,
166  (std::function<DataType(Int,Int,const DataType&)>)
167  ([this,&workspace_local](Int r, Int c, const DataType& z)->DataType {
168  return Exp(z - workspace_local.Get(Int(0), c));
169  }));
170 
171  // Compute column sums
172 #pragma omp parallel for
173  for(Int c=0; c<local_width; ++c) {
174  DataType sum = 0;
175  for(Int r=0; r<local_height; ++r) {
176  sum += activations_local.Get(r,c);
177  }
178  workspace_local.Set(Int(0), c, sum);
179  }
180  AllReduce(*m_workspace_v, m_workspace_v->RedundantComm(), mpi::SUM);
181 
182  // Divide activations by column sums
183  // This truncates small values to 0 to avoid them becoming denormalized later
184  // in the forward/backward stages. Denormalized values can significantly
185  // impact floating point performance.
186  IndexDependentMap(activations_local,
187  (std::function<DataType(Int,Int,const DataType&)>)
188  ([this,&workspace_local](Int r, Int c, const DataType& z)->DataType {
189  const DataType v = z / workspace_local.Get(Int(0), c);
190  return Abs(v) < 1e-8 ? DataType(1e-8) : v;
191  }));
192 
193 }

195 void lbann::SoftmaxLayer::bp_linearity()
196 {
197 
198  switch(m_data_layout) {
199  case data_layout::MODEL_PARALLEL:
200  // Compute the partial delta update for the next lower layer
201  Gemm(TRANSPOSE, NORMAL, DataType(1),
202  *m_weights,
203  *m_prev_error_signal_v,
204  DataType(0),
205  *m_error_signal_v);
206  // Compute update for activation weights
207  Gemm(NORMAL, TRANSPOSE, DataType(1)/get_effective_minibatch_size(),
208  *m_prev_error_signal_v,
209  *m_prev_activations_v,
210  DataType(0),
211  *m_weights_gradient);
212  break;
213  case data_layout::DATA_PARALLEL:
214  // Compute the partial delta update for the next lower layer
215  Gemm(TRANSPOSE, NORMAL, DataType(1),
216  m_weights->LockedMatrix(),
217  m_prev_error_signal_v->LockedMatrix(),
218  DataType(0),
219  m_error_signal_v->Matrix());
220  // Compute update for activation weights
221  Gemm(NORMAL, TRANSPOSE, DataType(1)/get_effective_minibatch_size(),
222  m_prev_error_signal_v->LockedMatrix(),
223  m_prev_activations_v->LockedMatrix(),
224  DataType(0),
225  m_weights_gradient->Matrix());
226  // Add gradients from all processes
227  AllReduce(*m_weights_gradient,
228  m_weights_gradient->RedundantComm());
229  break;
230  }
231 
232 }

234 void lbann::SoftmaxLayer::bp_nonlinearity()
235 {
236 
237  // Stop early if objective function is categorical cross entropy
238  // Note: error signal is already computed in objective function object
239  if(neural_network_model->obj_fn->type == objective_functions::obj_fn_type::categorical_cross_entropy
240  && (m_next_layer_type == layer_type::target_distributed_minibatch
241  || m_next_layer_type == layer_type::target_distributed_minibatch_parallel_io
242  // || m_next_layer_type == layer_type::target_unsupervised
243  )) {
244  return;
245  }
246 
247  // Get local matrices and parameters
248  const Mat& activations_local = m_activations_v->LockedMatrix();
249  Mat& workspace_local = m_workspace_v->Matrix();
250  Mat& prev_error_signal_local = m_prev_error_signal_v->Matrix();
251  const Int local_height = activations_local.Height();
252  const Int local_width = activations_local.Width();
253 
254  // Compute dot products
255  // Note: prev_error_signal^T activations
256  for(Int c=0; c<local_width; ++c) {
257  workspace_local.Set(Int(0), c,
258  Dot(prev_error_signal_local(ALL,IR(c)),
259  activations_local(ALL,IR(c))));
260  }
261  AllReduce(*m_workspace_v, m_workspace_v->RedundantComm(), mpi::SUM);
262 
263  // Update error signal
264  // Note: prev_error_signal := activations * (prev_error_signal - prev_error_signal^T activations)
265  IndexDependentMap(prev_error_signal_local,
266  (std::function<DataType(Int,Int,const DataType&)>)
267  ([this,&activations_local,&workspace_local]
268  (Int r, Int c, const DataType& z)->DataType {
269  const DataType activations_entry = activations_local.Get(r,c);
270  const DataType dot_product_entry = workspace_local.Get(Int(0),c);
271  return activations_entry * (z - dot_product_entry);
272  }));
273 
274 }
