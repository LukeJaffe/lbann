102  void add_child(const Layer *child) {
103 
104  // Check if child layer is null pointer
105  if(child == NULL) {
106  if(m_comm->am_world_master()) {
107  std::cerr << "split_layer: could not add child layer since pointer is null" << "\n";
108  }
109  return;
110  }
111 
112  // Add child layer if it isn't in list of children
113  auto child_pos = std::find(m_children.begin(), m_children.end(), child);
114  if(child_pos == m_children.end()) {
115  m_children.push_back(child);
116  }
117  else {
118  if(m_comm->am_world_master()) {
119  std::cerr << "split_layer: could not add child layer since it is already in list of children" << "\n";
120  }
121  }

125  void remove_child(const Layer *child) {
126 
127  // Check if child layer is null pointer
128  if(child == NULL) {
129  if(m_comm->am_world_master()) {
130  std::cerr << "split_layer: could not remove child layer since pointer is null" << "\n";
131  }
132  return;

134 
135  // Remove child layer if it is in list of children
136  auto child_pos = std::find(m_children.begin(), m_children.end(), child);
137  if(child_pos != m_children.end()) {
138  m_children.erase(child_pos);

202  void bp_compute_cudnn() {
203  #ifndef __LIB_CUDNN
204  throw lbann_exception("split_layer: cuDNN not detected");
205  #else
206 
207  // Useful constant
208  const DataType one = 1;
209 
210  // Copy error signal from first child layer
211  this->m_cudnn->copy_on_gpus(this->m_error_signal_d,
212  this->m_prev_error_signal_d,
213  this->m_num_neurons,
214  this->m_mini_batch_size_per_gpu);
215 
216  // Iterate through child layers
217  const int num_gpus = this->m_cudnn->get_num_gpus();
218  for(size_t child_index=1; child_index<m_children.size(); ++child_index) {
219  const Layer* child = m_children[child_index];
220 
221  // Get child error signal on GPUs
222  if(child->using_gpus()) {
223  child->get_gpu_bp_output(this->m_prev_error_signal_d, this);
224  }
225  else {
226  child->get_bp_output(*this->m_prev_error_signal, this);
227  this->m_cudnn->scatter_to_gpus(this->m_prev_error_signal_d,
228  this->m_prev_error_signal->LockedMatrix(),
229  this->m_mini_batch_size_per_gpu);
230  }
231 
232  // Add child error signal to this layer's error signal
233  for(int i=0; i<num_gpus; ++i) {
234  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
235  CHECK_CUDNN(cudnnSetStream(this->m_cudnn->get_handle(i),
236  this->m_cudnn->get_stream(i)));
237  CHECK_CUDNN(cudnnAddTensor(this->m_cudnn->get_handle(i),
238  &one,
239  this->m_neurons_cudnn_desc,
240  this->m_prev_error_signal_d[i],
241  &one,
242  this->m_prev_neurons_cudnn_desc,
243  this->m_error_signal_d[i]));
244  }
245 
246  }
247 
248  #endif // #ifndef __LIB_CUDNN
249  }

251  void bp_compute_cpu() {
