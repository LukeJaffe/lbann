91  void fp_compute() {
92 
93  // Get local matrices and parameters
94  Mat& workspace_local = m_workspace_v->Matrix();
95  Mat& prev_activations_local = this->m_prev_activations_v->Matrix();
96  Mat& activations_local = this->m_activations_v->Matrix();
97  const Int local_height = activations_local.Height();
98  const Int local_width = activations_local.Width();
99 
100  // Find maximum entry in each column
101  #pragma omp parallel for
102  for(El::Int col = 0; col < local_width; ++col) {
103  DataType max_entry = prev_activations_local(El::Int(0), col);
104  for(El::Int row = 1; row < local_height; ++row) {
105  max_entry = Max(max_entry, prev_activations_local(row,col));
106  }
107  workspace_local(El::Int(0), col) = max_entry;
108  }
109  AllReduce(*m_workspace_v, m_workspace_v->RedundantComm(), El::mpi::MAX);
110 
111  // Exponentiate activations and compute column sums
112  // Note: Subtracting by the column max prevents activations from
113  // blowing up. Large negative values underflow to 0.
114  #pragma omp parallel for
115  for (El::Int col = 0; col < local_width; ++col) {
116  const DataType shift = workspace_local(0, col);
117  DataType sum = 0;
118  for (El::Int row = 0; row < local_height; ++row) {
119  const DataType prev_activations_entry = prev_activations_local(row, col);
120  const DataType activations_entry = El::Exp(prev_activations_entry - shift);
121  activations_local(row, col) = activations_entry;
122  sum += activations_entry;
123  }
124  workspace_local(El::Int(0), col) = sum;
125  }
126  AllReduce(*m_workspace_v, m_workspace_v->RedundantComm(), El::mpi::SUM);
127 
128  // Divide activations by column sums
129  // This truncates small values to 0 to avoid them becoming denormalized later
130  // in the forward/backward stages. Denormalized values can significantly
131  // impact floating point performance.
132  IndexDependentMap(activations_local,
133  (std::function<DataType(El::Int,El::Int,const DataType&)>)
134  ([this,&workspace_local](El::Int r, El::Int c, const DataType& z)->DataType {
135  const DataType v = z / workspace_local(Int(0), c);
136  return Abs(v) < DataType(1e-8) ? DataType(1e-8) : v;
137  }));
138  }

140  void bp_compute(void) {
141 
142  // Stop early if objective function is categorical cross entropy
143  // Note: error signal is already computed in objective function object
144  if(this->m_neural_network_model->m_obj_fn->type == objective_functions::obj_fn_type::categorical_cross_entropy
145  && (this->m_next_layer_type == layer_type::target_distributed_minibatch
146  || this->m_next_layer_type == layer_type::target_distributed_minibatch_parallel_io
147  || this->m_next_layer_type == layer_type::target_partitioned_minibatch_parallel_io
148  // || m_next_layer_type == layer_type::target_unsupervised
149  )) {
150  View(*this->m_error_signal, *this->m_prev_error_signal);
151  View(*this->m_error_signal_v, *this->m_error_signal,
152  ALL, IR(0,this->m_error_signal->Width()));
153  return;
154  }
155 
156  // Get local matrices and parameters
157  const Mat& activations_local = this->m_activations_v->LockedMatrix();
158  Mat& workspace_local = m_workspace_v->Matrix();
159  Mat& prev_error_signal_local = this->m_prev_error_signal_v->Matrix();
160  Mat& error_signal_local = this->m_error_signal_v->Matrix();
161  const Int local_width = activations_local.Width();
162 
163  // Compute dot products
164  // Note: prev_error_signal^T activations
165  for(El::Int c=0; c<local_width; ++c) {
166  workspace_local(El::Int(0), c) = Dot(prev_error_signal_local(ALL,IR(c)),
167  activations_local(ALL,IR(c)));
168  }
169  AllReduce(*m_workspace_v, m_workspace_v->RedundantComm(), El::mpi::SUM);
170 
171  // Update error signal
172  // Note: error_signal := activations * (prev_error_signal - prev_error_signal^T activations)
173  IndexDependentMap(error_signal_local,
174  (std::function<DataType(El::Int,El::Int,const DataType&)>)
175  ([this,&activations_local,&workspace_local]
176  (El::Int r, El::Int c, const DataType& z)->DataType {
177  const DataType activations_entry = activations_local(r,c);
178  const DataType dot_product_entry = workspace_local(Int(0),c);
179  return activations_entry * (z - dot_product_entry);
180  }));
181 
182  }
