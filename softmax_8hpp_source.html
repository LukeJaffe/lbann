116  void fp_compute() {
117 
118  // Get local matrices and parameters
119  Mat& workspace_local = m_workspace_v->Matrix();
120  const Mat& prev_activations_local = this->m_prev_activations->LockedMatrix();
121  Mat& activations_local = this->m_activations_v->Matrix();
122  const Int local_height = activations_local.Height();
123  const Int local_width = activations_local.Width();
124 
125  // Find maximum entry in each column
126  #pragma omp parallel for
127  for(El::Int col = 0; col < local_width; ++col) {
128  DataType max_entry = prev_activations_local(0, col);
129  for(El::Int row = 1; row < local_height; ++row) {
130  max_entry = std::max(max_entry, prev_activations_local(row,col));
131  }
132  workspace_local(0, col) = max_entry;
133  }
134  El::AllReduce(*m_workspace_v, m_workspace_v->RedundantComm(), El::mpi::MAX);
135 
136  // Exponentiate activations and compute column sums
137  // Note: Subtracting by the column max prevents activations from
138  // blowing up. Large negative values underflow to 0.
139  #pragma omp parallel for
140  for (El::Int col = 0; col < local_width; ++col) {
141  const DataType shift = workspace_local(0, col);
142  DataType sum = 0;
143  for (El::Int row = 0; row < local_height; ++row) {
144  const DataType prev_activations_entry = prev_activations_local(row, col);
145  const DataType activations_entry = std::exp(prev_activations_entry - shift);
146  activations_local(row, col) = activations_entry;
147  sum += activations_entry;
148  }
149  workspace_local(0, col) = sum;
150  }
151  El::AllReduce(*m_workspace_v, m_workspace_v->RedundantComm(), El::mpi::SUM);
152 
153  // Divide activations by column sums
154  // Note: Small values are rounded to minimum output value to avoid
155  // denormalized floats.
156  El::IndexDependentMap(activations_local,
157  (std::function<DataType(El::Int,El::Int,const DataType&)>)
158  ([this,&workspace_local](El::Int r, El::Int c, const DataType& x)
159  ->DataType {
160  const DataType sum = workspace_local(0, c);
161  const DataType y = x / sum;
162  return y > m_min_output ? y : m_min_output;
163  }));
164 
165  }

167  void bp_compute() {
168 
169  // Get local matrices and parameters
170  const Mat& activations_local = this->m_activations_v->LockedMatrix();
171  const Mat& prev_error_signal_local = this->m_prev_error_signal->Matrix();
172  Mat& error_signal_local = this->m_error_signal_v->Matrix();
173  Mat& workspace_local = m_workspace_v->Matrix();
174  const El::Int local_width = activations_local.Width();
175 
176  // Apply softmax-cross-entropy shortcut if activated
177  objective_functions::cross_entropy* obj
178  = dynamic_cast<objective_functions::cross_entropy*>(this->m_neural_network_model->m_obj_fn);
179  if(obj != nullptr && obj->get_shortcut_softmax_layer() == this) {
180  El::IndexDependentFill(error_signal_local,
181  (std::function<DataType(El::Int,El::Int)>)
182  ([this,&activations_local,&prev_error_signal_local]
183  (El::Int r, El::Int c)->DataType {
184  const DataType activations_entry = activations_local(r,c);
185  const DataType prev_error_signal_entry = prev_error_signal_local(r,c);
186  if(activations_entry > m_min_output) {
187  return activations_entry - prev_error_signal_entry;
188  }
189  else {
190  return DataType(0);
191  }
192  }));
193  return;
194  }
195 
196  // Compute dot products
197  // Note: prev_error_signal^T activations
198  for(El::Int c=0; c<local_width; ++c) {
199  workspace_local(0, c) = El::Dot(prev_error_signal_local(El::ALL,El::IR(c)),
200  activations_local(El::ALL,El::IR(c)));
201  }
202  El::AllReduce(*m_workspace_v, m_workspace_v->RedundantComm(), El::mpi::SUM);
203 
204  // Update error signal
205  // Note: error_signal := activations * (prev_error_signal - prev_error_signal^T activations)
206  El::IndexDependentFill(error_signal_local,
207  (std::function<DataType(El::Int,El::Int)>)
208  ([this,&activations_local,&prev_error_signal_local,&workspace_local]
209  (El::Int r, El::Int c)->DataType {
210  const DataType activations_entry = activations_local(r,c);
211  const DataType prev_error_signal_entry = prev_error_signal_local(r,c);
212  const DataType dot_product_entry = workspace_local(Int(0),c);
213  if(activations_entry > m_min_output) {
214  return activations_entry * (prev_error_signal_entry
215  - dot_product_entry);
216  }
217  else {
218  return DataType(0);
219  }
220  }));
221 
222  }
