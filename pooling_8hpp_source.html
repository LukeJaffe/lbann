341  void fp_compute_im2col() {
342 
343  // Throw exception if pooling mode is not max or average pooling
344  if(m_pool_mode != pool_mode::max
345  && m_pool_mode != pool_mode::average) {
346  throw lbann_exception("pooling_layer: CPU pooling layer only implements max and average pooling");
347  }
348 
349  // Local matrices
350  const auto& local_input = get_local_prev_activations();
351  auto& local_output = get_local_activations();
352 
353  // Pool parameters
354  const int local_width = local_input.Width();
355  const int num_channels = this->m_prev_neuron_dims[0];
356  const int num_per_output_channel = this->m_num_neurons / num_channels;
357 
358  // Initialize max pool indices if needed
359  if(m_pool_mode == pool_mode::max) {
360  m_max_pool_indices.assign(this->m_num_neurons * local_width, 0);
361  }
362 
363  // Initialize matrices
364  Mat im2col_mat(m_pool_size * num_channels, num_per_output_channel);
365  Mat input_mat;
366 
367  // Iterate through data samples
368  for(int sample = 0; sample < local_width; ++sample) {
369 
370  // Construct im2col matrix from input
371  El::LockedView(input_mat, local_input,
372  El::ALL, El::IR(sample));
373  im2col(input_mat,
374  im2col_mat,
375  num_channels,
376  this->m_num_prev_neuron_dims - 1,
377  &this->m_prev_neuron_dims[1],
378  m_pads.data(),
379  m_pool_dims.data(),
380  m_strides.data());
381 
382  if(m_pool_mode == pool_mode::max) {
383  // Apply max pooling
384  DataType *output_buffer = local_output.Buffer(0, sample);
385  int *indices_buffer = &m_max_pool_indices[sample * this->m_num_neurons];
386  #pragma omp parallel for
387  for(int channel = 0; channel < num_channels; ++channel) {
388  for(int j = 0; j < num_per_output_channel; ++j) {
389  DataType *im2col_buffer = im2col_mat.Buffer(channel*m_pool_size, j);
390  DataType max_entry = im2col_buffer[0];
391  int max_index = 0;
392  for(int i = 1; i < m_pool_size; ++i) {
393  const DataType current_entry = im2col_buffer[i];
394  if(current_entry > max_entry) {
395  max_entry = current_entry;
396  max_index = i;
397  }
398  }
399  const int output_index = j + channel * num_per_output_channel;
400  output_buffer[output_index] = max_entry;
401  indices_buffer[output_index] = max_index;
402  }
403  }
404  }
405 
406  if(m_pool_mode == pool_mode::average) {
407  // Apply average pooling
408  DataType *output_buffer = local_output.Buffer(0, sample);
409  #pragma omp parallel for
410  for(int channel = 0; channel < num_channels; ++channel) {
411  for(int j = 0; j < num_per_output_channel; ++j) {
412  const DataType *im2col_buffer
413  = im2col_mat.LockedBuffer(channel*m_pool_size, j);
414  DataType output_entry = 0;
415  for(int i = 0; i < m_pool_size; ++i) {
416  output_entry += im2col_buffer[i];
417  }
418  output_entry /= m_pool_size;
419  const int output_index = j + channel * num_per_output_channel;
420  output_buffer[output_index] = output_entry;
421  }
422  }
423  }
424 
425  }
426

430  void bp_compute_im2col() {
431 
432  // Throw exception if pooling mode is not max or average pooling
433  if(m_pool_mode != pool_mode::max
434  && m_pool_mode != pool_mode::average) {
435  throw lbann_exception("pooling_layer: CPU pooling layer only implements max and average pooling");
436  }
437 
438  // Local matrices
439  const auto& local_gradient_wrt_output = get_local_prev_error_signals();
440  auto& local_gradient_wrt_input = get_local_error_signals();
441 
442  // Pool parameters
443  const int input_size = local_gradient_wrt_input.Height();
444  const int local_width = local_gradient_wrt_output.Width();
445  const int num_channels = this->m_prev_neuron_dims[0];
446  const int num_per_input_channel = this->m_num_neurons / num_channels;
447 
448  // Initialize matrices
449  Mat im2col_mat(m_pool_size * num_channels, num_per_input_channel);
450  Mat gradient_wrt_input_col(input_size, 1);
451 
452  // Iterate through data samples
453  for(int sample = 0; sample < local_width; ++sample) {
454 
455  // Compute gradient w.r.t. im2col matrix for max pooling
456  if(m_pool_mode == pool_mode::max) {
457 
458  // Clear im2col matrix
459  El::Zero(im2col_mat);
460 
461  // Copy previous error signal to im2col matrix entries
462  // corresponding to max
463  const DataType *gradient_wrt_output_buffer
464  = local_gradient_wrt_output.LockedBuffer(0, sample);
465  const int *indices_buffer
466  = &m_max_pool_indices[sample * this->m_num_neurons];
467  #pragma omp parallel for
468  for(int channel = 0; channel < num_channels; ++channel) {
469  for(int j = 0; j < num_per_input_channel; ++j) {
470  const int input_index = j + channel * num_per_input_channel;
471  const int max_index = indices_buffer[input_index];
472  DataType *im2col_buffer = im2col_mat.Buffer(channel*m_pool_size, j);
473  im2col_buffer[max_index]
474  = gradient_wrt_output_buffer[input_index];
475  }
476  }
477 
478  }
479 
480  // Compute gradient w.r.t. im2col matrix for average pooling
481  if(m_pool_mode == pool_mode::average) {
482  const DataType *gradient_wrt_output_buffer
483  = local_gradient_wrt_output.LockedBuffer(0, sample);
484  #pragma omp parallel for
485  for(int channel = 0; channel < num_channels; ++channel) {
486  for(int j = 0; j < num_per_input_channel; ++j) {
487  DataType *im2col_buffer = im2col_mat.Buffer(channel*m_pool_size, j);
488  const int input_index = j + channel * num_per_input_channel;
489  const DataType output_entry
490  = gradient_wrt_output_buffer[input_index] / m_pool_size;
491  for(int i = 0; i < m_pool_size; ++i) {
492  im2col_buffer[i] = output_entry;
493  }
494  }
495  }
496 
497  }
498 
499  // Compute error signal (i.e. gradient w.r.t. input)
500  col2im(im2col_mat,
501  gradient_wrt_input_col,
502  num_channels,
503  this->m_num_prev_neuron_dims - 1,
504  &this->m_prev_neuron_dims[1],
505  m_pads.data(),
506  m_pool_dims.data(),
507  m_strides.data());
508  local_gradient_wrt_input(El::ALL, El::IR(sample)) += gradient_wrt_input_col;
509 
510  }
511 
512  }

580  bool using_distconv() const override {
581  if (!(m_pads[0] == 0 && m_pads[1] == 0 &&
582  m_pool_dims[0] % 2 != 0 && m_pool_dims[1] % 2 != 0)) {
583  MPIPrintStreamDebug() << "pooling: unsupported \n";
584  return false;
585  }
586 
587  int stencil_h = (m_pool_dims[0] - 1) / 2;
588  int stencil_w = (m_pool_dims[1] - 1) / 2;
589 
590  if (!((m_strides[0] == 1 && m_strides[1] == 1) ||
591  (m_strides[0] == stencil_h + 1 &&
592  m_strides[1] == stencil_w + 1))) {
593  MPIPrintStreamDebug() << "pooling: unsupported \n";
594  return false;
595  }
596 
597  int input_tensor_w = m_prev_neuron_dims[2];
598  int input_tensor_h = m_prev_neuron_dims[1];
599 
600  // shape dim must be divisible by strides
601  if (!(input_tensor_h % m_strides[0] == 0 &&
602  input_tensor_w % m_strides[1] == 0)) {
603  MPIPrintStreamDebug() << "pooling: Not divisible by strides: "
604  << input_tensor_h << "x" << input_tensor_w
605  << "\n";
606  return false;
607  }
608  return true;
609  }
