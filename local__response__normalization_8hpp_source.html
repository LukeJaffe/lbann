325  void fp_compute_cpu() {
326 
327  // Get local matrices
328  const Mat& prev_activations_local = this->m_prev_activations_v->LockedMatrix();
329  Mat& activations_local = this->m_activations_v->Matrix();
330 
331  // Input and output entries are divided amongst channels
332  const Int num_per_channel = this->m_num_neurons / m_num_channels;
333 
335  // activations(i) = prev_activations(i) / scale_factor(i) ^ beta
336  // scale_factor(i)
337  // = k + alpha / window_width * sum( prev_activations(j) ^ 2 )
338  // Note: The sum is over entries in the normalization window.
340 
341  // Iterate through data samples in mini-batch
342  #pragma omp parallel for collapse(2)
343  for(Int sample = 0; sample < prev_activations_local.Width(); ++sample) {
344  // Iterate through positions in sample
345  for(Int pos = 0; pos < num_per_channel; ++pos) {
346 
347  // Initialize normalization window
348  Int window_start = - m_window_width / 2;
349  Int window_end = m_window_width / 2;
350  DataType window_sum = 0;
351  for(Int c = Max(window_start, 0);
352  c <= Min(window_end, m_num_channels-1);
353  ++c) {
354  const DataType x
355  = prev_activations_local.Get(pos + num_per_channel*c, sample);
356  window_sum += x * x;
357  }
358 
359  // Iterate through channels at current position
360  for(Int channel = 0; channel < m_num_channels; ++channel) {
361  const Int index = pos + num_per_channel * channel;
362 
363  // Apply local response normalization to current entry
364  const DataType input_entry = prev_activations_local.Get(index, sample);
365  const DataType scale_factor = m_lrn_k + m_lrn_alpha / m_window_width * window_sum;
366  const DataType output_entry = input_entry * Pow(scale_factor, -m_lrn_beta);
367  activations_local.Set(index, sample, output_entry);
368 
369  // Shift normalization window by one entry
370  if(window_start >= 0) {
371  const Int i = pos + num_per_channel*window_start;
372  const DataType x = prev_activations_local.Get(i, sample);
373  window_sum -= x * x;
374  }
375  ++window_start;
376  ++window_end;
377  if(window_end < m_num_channels) {
378  const Int i = pos + num_per_channel*window_end;
379  const DataType x = prev_activations_local.Get(i, sample);
380  window_sum += x * x;
381  }
382 
383  }
384 
385  }
386 
387  }

392  void bp_compute_cpu() {
393 
394  // Get local matrices
395  const Mat& prev_activations_local = this->m_prev_activations_v->LockedMatrix();
396  const Mat& activations_local = this->m_activations_v->LockedMatrix();
397  const Mat& prev_error_signal_local = this->m_prev_error_signal_v->LockedMatrix();
398  Mat& error_signal_local = this->m_error_signal_v->Matrix();
399 
400  // Initialize error signal to zero
401  Zero(error_signal_local);
402 
403  // Input and output entries are divided amongst channels
404  const Int num_per_channel = this->m_num_neurons / m_num_channels;
405 
407  // error_signal(i)
408  // = prev_error_signal(i) / scale_factor(i) ^ beta
409  // - 2 * alpha * beta / window_width * prev_activations(i)
410  // * sum( prev_error_signal(j) * activations(j)
411  // / scale_factor(j) )
412  // Note: See comments in fp_linearity_cpu for a definition of
413  // scale_factor. The sum is over entries in the normalization
414  // window.
416 
417  // Iterate through data samples in mini-batch
418  #pragma omp parallel for collapse(2)
419  for(Int sample = 0; sample < prev_activations_local.Width(); ++sample) {
420  // Iterate through positions in sample
421  for(Int pos = 0; pos < num_per_channel; ++pos) {
422 
423  // Initialize normalization window
424  Int window_start = - m_window_width / 2;
425  Int window_end = m_window_width / 2;
426  DataType window_sum = 0;
427  for(Int c = Max(window_start, 0);
428  c <= Min(window_end, m_num_channels-1);
429  ++c) {
430  const DataType x
431  = prev_activations_local.Get(pos + num_per_channel*c, sample);
432  window_sum += x * x;
433  }
434 
435  // Iterate through channels at current position
436  DataType error_signal_update;
437  for(Int channel = 0; channel < m_num_channels; ++channel) {
438  const Int index = pos + num_per_channel * channel;
439 
440  // Get data for current entry
441  const DataType activations_entry = activations_local.Get(index, sample);
442  const DataType prev_error_signal_entry = prev_error_signal_local.Get(index, sample);
443  const DataType scale_factor = m_lrn_k + m_lrn_alpha / m_window_width * window_sum;
444 
445  // Update current error signal entry
446  error_signal_update = prev_error_signal_entry * Pow(scale_factor, -m_lrn_beta);
447  error_signal_local.Update(index, sample, error_signal_update);
448 
449  // Update error signal entries in normalization window
450  for(Int c = Max(window_start, 0);
451  c <= Min(window_end, m_num_channels-1);
452  ++c) {
453  const Int i = pos + num_per_channel * c;
454  const DataType prev_activations_entry = prev_activations_local.Get(i, sample);
455  error_signal_update
456  = (-2 * m_lrn_alpha * m_lrn_beta / m_window_width * prev_activations_entry
457  * prev_error_signal_entry * activations_entry / scale_factor);
458  error_signal_local.Update(i, sample, error_signal_update);
459  }
460 
461  // Shift normalization window by one entry
462  if(window_start >= 0) {
463  const Int i = pos + num_per_channel*window_start;
464  const DataType x = prev_activations_local.Get(i, sample);
465  window_sum -= x * x;
466  }
467  ++window_start;
468  ++window_end;
469  if(window_end < m_num_channels) {
470  const Int i = pos + num_per_channel*window_end;
471  const DataType x = prev_activations_local.Get(i, sample);
472  window_sum += x * x;
473  }
474 
475  }
476 
477  }
478 
479  }
