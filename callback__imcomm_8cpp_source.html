71 void lbann_callback_imcomm::setup(model *m) {
72  std::vector<Layer *>& layers = m->get_layers();
73  m_layer_params.resize(layers.size());
74  for (size_t layer = 0; layer < layers.size(); ++layer) {
75  imcomm_params& params = m_layer_params[layer];
76  learning *learning_layer = dynamic_cast<learning*>(layers[layer]);
77  if (m_param_choices.find(layer) != m_param_choices.end()) {
78  params = m_param_choices[layer];
79  } else if (learning_layer) {
80  // Only do communication for learning layers unless explicitly told.
81  params.ct = m_default_ct;
82  } else {
83  params.ct = NONE;
84  }
85  if (params.ct != NONE) {
86  if (!learning_layer) {
87  throw lbann_exception("imcomm: trying to do inter-model gradient "
88  "communication on layer " + std::to_string(layer)
89  + " without gradients");
90  }
91  if (ct_needs_reshape(params.ct)) {
92  // Currently, no layers need reshaping.
93  //const std::type_info& layer_type = typeid(*(layers[layer]));
94  }
95  if (ct_does_quantization(params.ct)) {
96  const AbsDistMat& gradients = learning_layer->get_weights_gradient();
97  if (params.reshape_height > 0) {
98  El::Zeros(params.error, params.reshape_height, params.reshape_width);
99  } else {
100  El::Zeros(params.error, gradients.LocalHeight(), gradients.LocalWidth());
101  }
102  }
103  }
104  }

107 void lbann_callback_imcomm::on_epoch_end(model *m) {
108  lbann_comm *comm = m->get_comm();
109  if (comm->get_num_models() == 1 ||
110  m->get_execution_mode() != execution_mode::training) {
111  return; // No point with only one model.
112  }
113  std::vector<Layer *>& layers = m->get_layers();
114  for (size_t layer = 0; layer < layers.size(); ++layer) {
115  imcomm_params& params = m_layer_params[layer];
116  if (ct_does_quantization(params.ct)) {
117  comm->intermodel_sum_matrix(params.error);
118  Mat *local_gradients = nullptr;
119  Mat reshaped;
120  learning *learning_layer =
121  dynamic_cast<learning*>(layers[layer]);
122  if (params.reshape_height > 0) {
123  reshape_mat(learning_layer->get_weights_gradient().Matrix(),
124  reshaped, params.reshape_height, params.reshape_width);
125  local_gradients = &reshaped;
126  } else {
127  local_gradients = &(learning_layer->get_weights_gradient().Matrix());
128  }
129  *local_gradients = params.error;
130  // Apply optimizer update with accumulated gradient error.
131  layers[layer]->update();
132  Zero(params.error);
133  }
134  }
135 }

137 void lbann_callback_imcomm::on_backward_prop_end(model *m) {
138  lbann_comm *comm = m->get_comm();
139  if (comm->get_num_models() == 1 ||
140  m->get_execution_mode() != execution_mode::training) {
141  return; // No point with only one model.
142  }
143  std::vector<Layer *>& layers = m->get_layers();
144  for (size_t layer = 0; layer < layers.size(); ++layer) {
145  double start_time = get_time();
146  imcomm_params& params = m_layer_params[layer];
147  if (params.ct == NONE) {
148  continue;
149  }
150  Mat* local_gradients = nullptr;
151  Mat reshaped;
152  learning *learning_layer =
153  dynamic_cast<learning*>(layers[layer]);
154  if (params.reshape_height > 0) {
155  reshape_mat(learning_layer->get_weights_gradient().Matrix(),
156  reshaped, params.reshape_height, params.reshape_width);
157  local_gradients = &reshaped;
158  } else {
159  local_gradients = &(learning_layer->get_weights_gradient().Matrix());
160  }
161  switch (params.ct) {
162  case NORMAL:
163  comm->intermodel_sum_matrix(*local_gradients);
164  break;
165  case ONEBIT_QUANTIZATION:
166  m_quantizer.intermodel_sum_onebit_quantized(
167  comm, *local_gradients, params.error);
168  break;
169  case THRESH_QUANTIZATION:
170  m_quantizer.intermodel_sum_threshold_quantized(
171  comm, *local_gradients, params.error, params.pos_thresh, params.neg_thresh);
172  break;
173  case ADAPTIVE_QUANTIZATION:
174  m_quantizer.intermodel_sum_adaptive_quantized(
175  comm, *local_gradients, params.error, params.proportion);
176  break;
177  default:
178  throw lbann_exception("imcomm: unknown comm type");
179  }
180  double im_time = get_time() - start_time;
181  do_summary(m, learning_layer, im_time);
182  }

185 void lbann_callback_imcomm::do_summary(model *m, learning *layer,
186  double im_time) {
187  if (m_summarizer == nullptr) {
188  return;
189  }
190  uint idx = layer->get_index();
191  lbann_comm *comm = m->get_comm();
192  std::string prefix = "layer" + std::to_string(
193  static_cast<long long>(idx)) + "/imcomm_";
194  m_summarizer->reduce_scalar(prefix + "time",
195  im_time, m->get_cur_step());
196  size_t bytes_sent = 0;
197  size_t bytes_received = 0;
198  if (ct_does_quantization(m_layer_params[idx].ct)) {
199  bytes_sent = comm->get_ar_bytes_sent();
200  bytes_received = comm->get_ar_bytes_received();
201  } else {
202  // Use the same approximation the comm layer does.
203  const Mat& local_gradients =
204  layer->get_weights_gradient().LockedMatrix();
205  bytes_sent =
206  sizeof(DataType) * local_gradients.Height() * local_gradients.Width();
207  bytes_received =
208  sizeof(DataType) * local_gradients.Height() * local_gradients.Width();
209  }
210  m_summarizer->reduce_scalar(prefix + "bytes_sent",
211  bytes_sent, m->get_cur_step());
212  m_summarizer->reduce_scalar(prefix + "bytes_received",
213  bytes_received, m->get_cur_step());
214  if (ct_does_quantization(m_layer_params[idx].ct)) {
215  m_summarizer->reduce_scalar(prefix + "rs_bytes_sent",
216  comm->get_ar_rs_bytes_sent(),
217  m->get_cur_step());
218  m_summarizer->reduce_scalar(prefix + "ag_bytes_sent",
219  comm->get_ar_ag_bytes_sent(),
220  m->get_cur_step());
221  m_summarizer->reduce_scalar(prefix + "rs_bytes_received",
222  comm->get_ar_rs_bytes_received(),
223  m->get_cur_step());
224  m_summarizer->reduce_scalar(prefix + "ag_bytes_received",
225  comm->get_ar_ag_bytes_received(),
226  m->get_cur_step());
227  m_summarizer->reduce_scalar(prefix + "ar_send_trans_time",
228  comm->get_ar_send_transform_time(),
229  m->get_cur_step());
230  m_summarizer->reduce_scalar(prefix + "ar_recv_trans_time",
231  comm->get_ar_recv_transform_time(),
232  m->get_cur_step());
233  m_summarizer->reduce_scalar(prefix + "ar_recv_apply_trans_time",
234  comm->get_ar_recv_apply_transform_time(),
235  m->get_cur_step());
236  if (m_layer_params[idx].ct == ADAPTIVE_QUANTIZATION) {
237  m_summarizer->reduce_scalar(prefix + "quantized_count",
238  m_quantizer.get_quantized_count(),
239  m->get_cur_step());
240  }
241  m_quantizer.reset_counters();
242  comm->reset_stats_counters();
243  }
244 }
