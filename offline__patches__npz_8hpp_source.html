151 template<typename T>
152 inline T& offline_patches_npz::data(
153  const cnpy::NpyArray& na, const std::vector<size_t> indices) const {
154  if ((sizeof(T) != na.word_size) && (sizeof(T) != 1u)) {
155  throw lbann_exception("The data type is not consistent with the word size of the array.");
156  }
157  const size_t offset = compute_cnpy_array_offset(na, indices)
158  * ((sizeof(T) == 1u)? na.word_size : 1u);
159  return *(reinterpret_cast<T*>(&(* na.data_holder)[0]) + offset);
160 }
