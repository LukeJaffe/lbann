73 void data_reader_merge_samples::load() {
74  // Load each subsidiary data reader.
75  for (auto&& reader : m_data_readers) {
76  reader->load();
77  }
78  // Compute the total number of samples and do some sanity checks.
79  int num_samples = 0;
80  int num_labels = m_data_readers[0]->get_num_labels();
81  int data_size = m_data_readers[0]->get_linearized_data_size();
82  int label_size = m_data_readers[0]->get_linearized_label_size();
83  const std::vector<int> data_dims = m_data_readers[0]->get_data_dims();
84  // Prepend a 0 to make things easier.
85  m_num_samples_psum.push_back(0);
86  for (auto&& reader : m_data_readers) {
87  m_num_samples_psum.push_back(reader->get_num_data());
88  num_samples += reader->get_num_data();
89  // TODO: It would be good to relax this.
90  // Large data sets may not have every label in every file.
91  if (num_labels != reader->get_num_labels()) {
92  throw lbann_exception(
93  "data_reader_merge_samples: data readers do not have the same number of labels");
94  }
95  if (data_size != reader->get_linearized_data_size()) {
96  throw lbann_exception(
97  "data_reader_merge_samples: data readers do not have the same data size");
98  }
99  if (label_size != reader->get_linearized_label_size()) {
100  throw lbann_exception(
101  "data_reader_merge_samples: data readers do not have the same label size");
102  }
103  if (data_dims != reader->get_data_dims()) {
104  throw lbann_exception(
105  "data_reader_merge_samples: data readers do not have the same data dims");
106  }
107  }
108  std::partial_sum(m_num_samples_psum.begin(), m_num_samples_psum.end(),
109  m_num_samples_psum.begin());
110  // Set up our indices.
111  // Note each subsidiary data reader presumably shuffled its indices as well.
112  // That's not strictly necessary, but does not impact anything.
113  m_shuffled_indices.resize(num_samples);
114  std::iota(m_shuffled_indices.begin(), m_shuffled_indices.end(), 0);
115  select_subset_of_data();
116 }

118 bool data_reader_merge_samples::fetch_datum(Mat& X, int data_id, int mb_idx,
119  int tid) {
120  // Find the right data reader to delegate to.
121  for (size_t i = 0; i < m_data_readers.size(); ++i) {
122  if (data_id < m_num_samples_psum[i + 1]) {
123  data_id -= m_num_samples_psum[i];
124  return m_data_readers[i]->fetch_datum(X, data_id, mb_idx, tid);
125  }
126  }
127  throw lbann_exception(
128  "data_reader_merge_samples: do not have data ID " +
129  std::to_string(data_id));
130 }
