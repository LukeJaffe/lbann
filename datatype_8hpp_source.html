609 inline unsigned long __seedmix(unsigned long a, unsigned long b, unsigned long c) {
610  a=a-b;
611  a=a-c;
612  a=a^(c >> 13);
613  b=b-c;
614  b=b-a;
615  b=b^(a << 8);
616  c=c-a;
617  c=c-b;
618  c=c^(b >> 13);
619  a=a-b;
620  a=a-c;
621  a=a^(c >> 12);
622  b=b-c;
623  b=b-a;
624  b=b^(a << 16);
625  c=c-a;
626  c=c-b;
627  c=c^(b >> 5);
628  a=a-b;
629  a=a-c;
630  a=a^(c >> 3);
631  b=b-c;
632  b=b-a;
633  b=b^(a << 10);
634  c=c-a;
