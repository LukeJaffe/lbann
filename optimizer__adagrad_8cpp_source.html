87 void adagrad::update(const AbsDistMat *gradient) {
88  // Get local matrix data
89  const int local_height = m_parameters->LocalHeight();
90  const int local_width = m_parameters->LocalWidth();
91  DataType *parameters_buffer = m_parameters->Buffer();
92  const int parameters_ldim = m_parameters->LDim();
93  const DataType *gradient_buffer = gradient->LockedBuffer();
94  const int gradient_ldim = gradient->LDim();
95  DataType *cache_buffer = m_cache->Buffer();
96  const int cache_ldim = m_cache->LDim();
97 
98  // Check if matrix data is contiguous
99  if(parameters_ldim != local_height
100  || gradient_ldim != local_height
101  || cache_ldim != local_height) {
102  // Update with non-contiguous data
103  #pragma omp parallel for collapse(2)
104  for(int j=0; j<local_width; ++j) {
105  for(int i=0; i<local_height; ++i) {
106  DataType& x = parameters_buffer[i+j*parameters_ldim];
107  const DataType g = gradient_buffer[i+j*gradient_ldim];
108  DataType& c = cache_buffer[i+j*cache_ldim];
109  c += g * g;
110  x -= m_learning_rate * g / (std::sqrt(c) + m_eps);
111  }
112  }
113  } else {
114  // Update with contiguous data
115  #pragma omp parallel for
116  for(int i=0; i<local_height*local_width; ++i) {
117  DataType& x = parameters_buffer[i];
118  const DataType g = gradient_buffer[i];
119  DataType& c = cache_buffer[i];
120  c += g * g;
121  x -= m_learning_rate * g / (std::sqrt(c) + m_eps);
122  }
123  }
124 }
