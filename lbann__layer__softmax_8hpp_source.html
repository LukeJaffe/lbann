130  void fp_linearity() {
131 
132  // Apply weight matrix
133  switch(m_data_layout) {
134  case data_layout::MODEL_PARALLEL:
135  Gemm(NORMAL, NORMAL, DataType(1),
136  *m_weights,
137  *m_prev_activations_v,
138  DataType(0),
139  *m_weighted_sum_v);
140  break;
141  case data_layout::DATA_PARALLEL:
142  Gemm(NORMAL, NORMAL, DataType(1),
143  m_weights->LockedMatrix(),
144  m_prev_activations_v->LockedMatrix(),
145  DataType(0),
146  m_weighted_sum_v->Matrix());
147  break;
148  }
149 
150  // Copy result to output matrix
151  Copy(*m_weighted_sum_v, *m_activations_v);
152 
153  }

155  void fp_nonlinearity() {
156 
157  // Get local matrices and parameters
158  Mat& workspace_local = m_workspace_v->Matrix();
159  Mat& activations_local = m_activations_v->Matrix();
160  const Int local_height = activations_local.Height();
161  const Int local_width = activations_local.Width();
162 
163  // Find maximum entry in each column
164 #pragma omp parallel for
165  for(Int c=0; c<local_width; ++c) {
166  DataType max_entry = -INFINITY;
167  for(Int r=0; r<local_height; ++r) {
168  max_entry = Max(max_entry, activations_local.Get(r,c));
169  }
170  workspace_local.Set(Int(0), c, max_entry);
171  }
172  AllReduce(*m_workspace_v, m_workspace_v->RedundantComm(), mpi::MAX);
173 
174  // Subtract column max and exponentiate activations
175  // Note: Subtracting the column max prevents activations from blowing
176  // up. Large negative values underflow to 0.
177  IndexDependentMap(activations_local,
178  (std::function<DataType(Int,Int,const DataType&)>)
179  ([this,&workspace_local](Int r, Int c, const DataType& z)->DataType {
180  return Exp(z - workspace_local.Get(Int(0), c));
181  }));
182 
183  // Compute column sums
184 #pragma omp parallel for
185  for(Int c=0; c<local_width; ++c) {
186  DataType sum = 0;
187  for(Int r=0; r<local_height; ++r) {
188  sum += activations_local.Get(r,c);
189  }
190  workspace_local.Set(Int(0), c, sum);
191  }
192  AllReduce(*m_workspace_v, m_workspace_v->RedundantComm(), mpi::SUM);
193 
194  // Divide activations by column sums
195  // This truncates small values to 0 to avoid them becoming denormalized later
196  // in the forward/backward stages. Denormalized values can significantly
197  // impact floating point performance.
198  IndexDependentMap(activations_local,
199  (std::function<DataType(Int,Int,const DataType&)>)
200  ([this,&workspace_local](Int r, Int c, const DataType& z)->DataType {
201  const DataType v = z / workspace_local.Get(Int(0), c);
202  return Abs(v) < 1e-8 ? DataType(1e-8) : v;
203  }));
204 
205  }

207  void bp_linearity() {
208 
209  switch(m_data_layout) {
210  case data_layout::MODEL_PARALLEL:
211  // Compute the partial delta update for the next lower layer
212  Gemm(TRANSPOSE, NORMAL, DataType(1),
213  *m_weights,
214  *m_prev_error_signal_v,
215  DataType(0),
216  *m_error_signal_v);
217  // Compute update for activation weights
218  Gemm(NORMAL, TRANSPOSE, DataType(1)/get_effective_minibatch_size(),
219  *m_prev_error_signal_v,
220  *m_prev_activations_v,
221  DataType(0),
222  *m_weights_gradient);
223  break;
224  case data_layout::DATA_PARALLEL:
225  // Compute the partial delta update for the next lower layer
226  Gemm(TRANSPOSE, NORMAL, DataType(1),
227  m_weights->LockedMatrix(),
228  m_prev_error_signal_v->LockedMatrix(),
229  DataType(0),
230  m_error_signal_v->Matrix());
231  // Compute update for activation weights
232  Gemm(NORMAL, TRANSPOSE, DataType(1)/get_effective_minibatch_size(),
233  m_prev_error_signal_v->LockedMatrix(),
234  m_prev_activations_v->LockedMatrix(),
235  DataType(0),
236  m_weights_gradient->Matrix());
237  // Add gradients from all processes
238  AllReduce(*m_weights_gradient,
239  m_weights_gradient->RedundantComm());
240  break;
241  }

245  void bp_nonlinearity() {
246 
247  // Stop early if objective function is categorical cross entropy
248  // Note: error signal is already computed in objective function object
249  if(neural_network_model->obj_fn->type == objective_functions::obj_fn_type::categorical_cross_entropy
250  && (m_next_layer_type == layer_type::target_distributed_minibatch
251  || m_next_layer_type == layer_type::target_distributed_minibatch_parallel_io
252  || m_next_layer_type == layer_type::target_partitioned_minibatch_parallel_io
253  // || m_next_layer_type == layer_type::target_unsupervised
254  )) {
255  return;
256  }
257 
258  // Get local matrices and parameters
259  const Mat& activations_local = m_activations_v->LockedMatrix();
260  Mat& workspace_local = m_workspace_v->Matrix();
261  Mat& prev_error_signal_local = m_prev_error_signal_v->Matrix();
262  const Int local_height = activations_local.Height();
263  const Int local_width = activations_local.Width();
264 
265  // Compute dot products
266  // Note: prev_error_signal^T activations
267  for(Int c=0; c<local_width; ++c) {
268  workspace_local.Set(Int(0), c,
269  Dot(prev_error_signal_local(ALL,IR(c)),
270  activations_local(ALL,IR(c))));
271  }
272  AllReduce(*m_workspace_v, m_workspace_v->RedundantComm(), mpi::SUM);
273 
274  // Update error signal
275  // Note: prev_error_signal := activations * (prev_error_signal - prev_error_signal^T activations)
276  IndexDependentMap(prev_error_signal_local,
277  (std::function<DataType(Int,Int,const DataType&)>)
278  ([this,&activations_local,&workspace_local]
279  (Int r, Int c, const DataType& z)->DataType {
280  const DataType activations_entry = activations_local.Get(r,c);
281  const DataType dot_product_entry = workspace_local.Get(Int(0),c);
282  return activations_entry * (z - dot_product_entry);
283  }));
284 
285  }
