65 cv::Mat correct_chromatic_aberration(const cv::Mat& _img)
66 {
67  if (_img.channels() != 3) return _img.clone();
68 
69  const int img_depth = _img.depth();
70 
71  std::pair<double, double> range_org = check_min_max(_img);
72 
73  cv::Mat img; // float matrix
74 
75  _img.convertTo(img, _PW_CV_FP_);
76 
77  static const pw_fp_t a[3] = {-1.0 _f, 2.0 _f, -1.0 _f}; // BGR order
78  static const pw_fp_t aa = a[0]*a[0] + a[1]*a[1]+ a[2]*a[2];
79  // A = a'*a/(a*a')
80  //static const pw_fp_t A[3][3] = {{a[0]*a[0]/aa, a[0]*a[1]/aa, a[0]*a[2]/aa},
81  // {a[1]*a[0]/aa, a[1]*a[1]/aa, a[1]*a[2]/aa},
82  // {a[2]*a[0]/aa, a[2]*a[1]/aa, a[2]*a[2]/aa}};
83  // B = (I - A)'
84  static const pw_fp_t B[3][3] = {{1.0 _f-a[0]*a[0]/aa, a[0]*a[1]/aa, a[0]*a[2]/aa},
85  {a[1]*a[0]/aa, 1.0 _f-a[1]*a[1]/aa, a[1]*a[2]/aa},
86  {a[2]*a[0]/aa, a[2]*a[1]/aa, 1.0 _f-a[2]*a[2]/aa}};
87 
88  cv::MatIterator_<pw_cv_vec3> it = img.begin<pw_cv_vec3>();
89  cv::MatIterator_<pw_cv_vec3> itend = img.end<pw_cv_vec3>();
90 
91  for ( ; it != itend; ++it) {
92  const pw_fp_t b0 = static_cast<pw_fp_t>((*it)[0]);
93  const pw_fp_t g0 = static_cast<pw_fp_t>((*it)[1]);
94  const pw_fp_t r0 = static_cast<pw_fp_t>((*it)[2]);
95 
96  pw_fp_t b = b0 * B[0][0] + g0 * B[1][0] + r0 * B[2][0];
97  pw_fp_t g = b0 * B[0][1] + g0 * B[1][1] + r0 * B[2][1];
98  pw_fp_t r = b0 * B[0][2] + g0 * B[1][2] + r0 * B[2][2];
99 
100  //std::cout << r0 << ' ' << g0 << ' ' << b0 << " " << r << ' ' << g << ' ' << b << std::endl;
101  (*it) = pw_cv_vec3(b,g,r);
102  }
103 
104  std::pair<double, double> range_new = check_min_max(img);
105  cv::Mat img_final;
106  //(x-range_new.first)*(range_org.second-range_org.first)/(range_new.second-range_new.first) + range_org.first;
107  const double alpha = (range_org.second-range_org.first)/(range_new.second-range_new.first);
108  const double beta = range_org.first - range_new.first * alpha;
109  img.convertTo(img_final, img_depth, alpha, beta);
110 
111  //std::pair<double, double> range_final = check_min_max(img_final);
112 
113  return img_final;
114 }

116 cv::Mat drop_2channels(const cv::Mat& _img)
117 {
118  if (_img.channels() != 3) return _img.clone();
119 
120  const int img_depth = _img.depth();
121 
122  cv::Mat img; // pw_fp_t matrix
123  _img.convertTo(img, _PW_CV_FP_);
124 
125  // compute channel to remain
126  pw_fp_t m[3] = {0.0 _f, 0.0 _f, 0.0 _f};
127  const int chosenCh = std::rand() % 3;
128  m[chosenCh] = 1.0 _f;
129 
130  // compute white noise
131  std::vector<image_stats> stats;
132  get_channel_stats(_img, stats);
133 
134  const pw_fp_t avg = static_cast<pw_fp_t>(stats[chosenCh].avg);
135  const pw_fp_t dev = static_cast<pw_fp_t>(stats[chosenCh].stdev/100.0);
136  pw_fp_t avgs[3] = {avg, avg, avg};
137  pw_fp_t devs[3] = {dev, dev, dev};
138 
139  rand_patch<> rg;
140  rg.init_normal(0, avgs[0], devs[0]);
141  rg.init_normal(1, avgs[1], devs[1]);
142  rg.init_normal(2, avgs[2], devs[2]);
143  cv::MatIterator_<pw_cv_vec3> it = img.begin<pw_cv_vec3>();
144  cv::MatIterator_<pw_cv_vec3> itend = img.end<pw_cv_vec3>();
145 
146  for ( ; it != itend; ++it) {
147  const pw_fp_t b0 = static_cast<pw_fp_t>((*it)[0]);
148  const pw_fp_t g0 = static_cast<pw_fp_t>((*it)[1]);
149  const pw_fp_t r0 = static_cast<pw_fp_t>((*it)[2]);
150 
151  #if 1
152  pw_fp_t b = b0*m[0] + (1.0-m[0])*rg.gen_normal(0);
153  pw_fp_t g = g0*m[1] + (1.0-m[1])*rg.gen_normal(1);
154  pw_fp_t r = r0*m[2] + (1.0-m[2])*rg.gen_normal(2);
155  #else
156  pw_fp_t b = b0*m[0];
157  pw_fp_t g = g0*m[1];
158  pw_fp_t r = r0*m[2];
159  #endif
160 
161  //std::cout << r0 << ' ' << g0 << ' ' << b0 << " " << r << ' ' << g << ' ' << b << std::endl;
162  (*it) = pw_cv_vec3(b,g,r);
163  }
164 
165  cv::Mat img_final;
166  img.convertTo(img_final, img_depth);
167 
168  return img_final;
169 }

171 bool take_patch(const cv::Mat& img, const patch_descriptor& pi,
172  const ROI& roi, std::vector<cv::Mat>& patches)
173 {
174  cv::Mat roiImg = img(cv::Rect(cv::Point(roi.left(), roi.top()),

176 
177  if (pi.is_to_correct_chromatic_aberration_at_pixel()) {

183  patches.push_back(roiImg);
184 
185  return true;
186 }

188 bool extract_patches(const cv::Mat& img, patch_descriptor& pi, std::vector<cv::Mat>& patches)
189 {
190  if (img.data == NULL) return false;
191 
192  ROI roi_center;
193  bool ok = pi.get_first_patch(roi_center);
194  if (!ok) return false;
195 
196  take_patch(img, pi, roi_center, patches);
197  //std::cout << "center patch: " << roi_center << std::endl;
198 
199  ROI roi;
200  unsigned int i = 1u;
201 
202  while (pi.get_next_patch(roi)) {
203  take_patch(img, pi, roi, patches);
204  i++;
205  }
206 
207  if (i == 1u) return false;
208  return true;
209 }
