154  void push_back_child(const Layer *child, int slice_point) {
155  std::stringstream err;
156 
157  // Check if child layer is null pointer
158  if(child == NULL) {
159  if(m_comm->am_world_master()) {
160  err << __FILE__ << " " << __LINE__ << " :: slice_layer: could not add child layer since pointer is null";
161  throw lbann_exception(err.str());
162  }
163  return;
164  }
165 
166  // Add first child
167  if(m_children.empty()) {
168  if(m_comm->am_world_master()) {
169  if(slice_point > 0) {
170  err << __FILE__ << " " << __LINE__ << " :: slice_layer: first child should have a slice point of zero";
171  throw lbann_exception(err.str());
172  }
173  }
174  m_children.push_back(child);
175  m_slice_points.push_back(0);
176  }
177 
178  // Add subsequent children
179  else {
180  auto child_pos = std::find(m_children.begin(), m_children.end(), child);
181  if(child_pos != m_children.end()) {
182  err << __FILE__ << " " << __LINE__ << " :: slice_layer: number of slice points should be one less than number of children";
183  throw lbann_exception(err.str());
184  }
185  if(slice_point <= m_slice_points.back()) {
186  err << __FILE__ << " " << __LINE__ << " :: slice_layer: invalid slice point";
187  throw lbann_exception(err.str());
188  }
189  m_children.push_back(child);
190  m_slice_points.push_back(slice_point);
191  }
192 
193  }

195  void pop_back_child() {

199  throw lbann_exception(err.str());
200  }
201  m_children.pop_back();
202  m_slice_points.pop_back();

317  void bp_compute_gpu() {
318  #ifndef __LIB_CUDNN
319  std::stringstream err;
320  err << __FILE__ << " " << __LINE__ << " :: slice_layer: cuDNN not detected";
321  throw lbann_exception(err.str());
322  #else
323 
324  // Split the error signal tensor into slices of width 1 along the
325  // slice axis
326  const int num_slices
327  = std::accumulate(this->m_prev_neuron_dims.begin(),
328  this->m_prev_neuron_dims.begin() + m_slice_axis,
329  1,
330  std::multiplies<int>());
331  const int slice_unit_size
332  = std::accumulate(this->m_prev_neuron_dims.begin() + m_slice_axis + 1,
333  this->m_prev_neuron_dims.end(),
334  1,
335  std::multiplies<int>());
336  const int output_slice_dim = this->m_prev_neuron_dims[m_slice_axis];
337  const int output_slice_size = output_slice_dim * slice_unit_size;
338 
339  // Copy entries in each child to error signal tensor
340  const int num_gpus = this->m_cudnn->get_num_gpus();
341  for(size_t child_index = 0; child_index < m_children.size(); ++child_index) {
342  const Layer* child = m_children[child_index];
343 
344  // Get child error signal on GPUs
345  std::vector<DataType*> input;
346  if(child_index == 0) {
347  input = this->m_prev_error_signal_d;
348  }
349  else {
350  std::vector<void*> work_spaces = this->m_cudnn->get_work_spaces();
351  for(int i=0; i<num_gpus; ++i) {
352  input.push_back((DataType*) work_spaces[i]);
353  }
354  if(child->using_gpus()) {
355  child->get_gpu_bp_output(input, this);
356  }
357  else {
358  child->get_bp_output(*this->m_prev_error_signal, this);
359  this->m_cudnn->scatter_to_gpus(input,
360  this->m_prev_error_signal->LockedMatrix(),
361  this->m_mini_batch_size_per_gpu);
362  }
363  }
364 
365  // Split previous error signal tensor into slices
366  const int input_slice_dim = m_slice_points[child_index+1] - m_slice_points[child_index];
367  const int input_slice_size = input_slice_dim * slice_unit_size;
368  const int input_size = num_slices * input_slice_size;
369  const int slice_offset = m_slice_points[child_index] * slice_unit_size;
370 
371  // Copy slices from previous error signal tensor into error signal tensor
372  for(int slice = 0; slice < num_slices; ++slice) {
373  std::vector<DataType*> input_slice(num_gpus), output_slice(num_gpus);
374  for(int i = 0; i < num_gpus; ++i) {
375  input_slice[i] = input[i] + slice * input_slice_size;
376  output_slice[i] = this->m_error_signal_d[i] + slice * output_slice_size + slice_offset;
377  }
378  this->m_cudnn->copy_on_gpus(output_slice,
379  input_slice,
380  input_slice_size,
381  this->m_mini_batch_size_per_gpu,
382  input_size,
383  this->m_num_prev_neurons);
384  }
385 
386  }
387 
388  #endif // #ifndef __LIB_CUDNN

391  void bp_compute_cpu() {
392 
393  // Split the error signal tensor into slices of width 1 along the
394  // slice axis
395  const int num_slices
396  = std::accumulate(this->m_prev_neuron_dims.begin(),
397  this->m_prev_neuron_dims.begin() + m_slice_axis,
398  1,
399  std::multiplies<int>());
400  const int slice_unit_size
401  = std::accumulate(this->m_prev_neuron_dims.begin() + m_slice_axis + 1,
402  this->m_prev_neuron_dims.end(),
403  1,
404  std::multiplies<int>());
405  const int output_slice_dim = this->m_prev_neuron_dims[m_slice_axis];
406  const int output_slice_size = output_slice_dim * slice_unit_size;
407 
408  // Copy entries in each child to error signal tensor
409  for(size_t i = 0; i < m_children.size(); ++i) {
410 
411  // Split previous error signal tensor into slices
412  m_children[i]->get_bp_output(*this->m_prev_error_signal, this);
413  const int input_slice_dim = m_slice_points[i+1] - m_slice_points[i];
414  const int input_slice_size = input_slice_dim * slice_unit_size;
415  const int slice_offset_start = m_slice_points[i] * slice_unit_size;
416  const int slice_offset_end = m_slice_points[i+1] * slice_unit_size;
417 
418  // Copy slices from previous error signal tensor into error signal tensor
419  for(int slice = 0; slice < num_slices; ++slice) {
420  El::LockedView(*m_input_slice_v,
421  *this->m_prev_error_signal,
422  El::IR(slice * input_slice_size,
423  (slice+1) * input_slice_size),
424  El::ALL);
425  El::View(*m_output_slice_v,
426  *this->m_error_signal_v,
427  El::IR(slice * output_slice_size + slice_offset_start,
428  slice * output_slice_size + slice_offset_end),
429  El::ALL);
430  El::Copy(*m_input_slice_v, *m_output_slice_v);
431  }
432 
433  }

437  void get_fp_output(AbsDistMat& fp_output, const Layer* next_layer) const {
438 
439  // Check if input is in the list of child layers
440  const int child_index = (std::find(m_children.begin(),
441  m_children.end(),
442  next_layer)
443  - m_children.begin());
444  if(child_index >= (int) m_children.size()) {
445  transform::get_fp_output(fp_output, next_layer);
446  }
447 
448  // Split the activations tensor into slices of width 1 along the
449  // slice axis
450  const int num_slices
451  = std::accumulate(this->m_prev_neuron_dims.begin(),
452  this->m_prev_neuron_dims.begin() + m_slice_axis,
453  1,
454  std::multiplies<int>());
455  const int slice_unit_size
456  = std::accumulate(this->m_prev_neuron_dims.begin() + m_slice_axis + 1,
457  this->m_prev_neuron_dims.end(),
458  1,
459  std::multiplies<int>());
460  const int input_slice_dim = this->m_prev_neuron_dims[m_slice_axis];
461  const int output_slice_dim = m_slice_points[child_index+1] - m_slice_points[child_index];
462  const int input_slice_size = input_slice_dim * slice_unit_size;
463  const int output_slice_size = output_slice_dim * slice_unit_size;
464  const int slice_offset_start = m_slice_points[child_index] * slice_unit_size;
465  const int slice_offset_end = m_slice_points[child_index+1] * slice_unit_size;
466 
467  if(num_slices == 1
468  && m_activations_v->DistData() == fp_output.DistData()) {
469  // Return view of activations tensor slice
470  El::LockedView(fp_output,
471  *this->m_activations_v,
472  El::IR(slice_offset_start, slice_offset_end),
473  El::ALL);
474  }
475  else {
476  // Copy slices from activations tensor into output
477  fp_output.Empty(false);
478  fp_output.Resize(output_slice_size, m_activations_v->Width());
479  AbsDistMat* output_slice_v
480  = fp_output.Construct(fp_output.Grid(), fp_output.Root());
481  for(int slice = 0; slice < num_slices; ++slice) {
482  El::LockedView(*m_input_slice_v,
483  *this->m_activations_v,
484  El::IR(slice * input_slice_size + slice_offset_start,
485  slice * input_slice_size + slice_offset_end),
486  El::ALL);
487  El::View(*output_slice_v,
488  fp_output,
489  El::IR(slice * output_slice_size,
490  (slice+1) * output_slice_size),
491  El::ALL);
492  El::Copy(*m_input_slice_v, *output_slice_v);
493  }
494  delete output_slice_v;
495  }
