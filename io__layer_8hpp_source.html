200  long get_linearized_data_size() {
201  long linearized_data_size = -1;
202  dataset& train_ds = get_dataset(execution_mode::training);
203  if (train_ds.m_data_reader) {
204  linearized_data_size = train_ds.m_data_reader->get_linearized_data_size();
205  }
206  dataset& val_ds = get_dataset(execution_mode::validation);
207  if (val_ds.m_data_reader) {
208  long tmp_data_size = val_ds.m_data_reader->get_linearized_data_size();
209  if (linearized_data_size != -1 && linearized_data_size != tmp_data_size) {
210  throw lbann_exception("lbann_io_layer: validation data set size does not "
211  "match the currently established data set size");
212  }
213  }
214  dataset& test_ds = get_dataset(execution_mode::testing);
215  if (test_ds.m_data_reader) {
216  long tmp_data_size = test_ds.m_data_reader->get_linearized_data_size();

218  throw lbann_exception("lbann_io_layer: testing data set size does not "

313  bool loadFromCheckpointShared(persist& p) {
314  // rank 0 reads the file
315  dataset_header header;
316  if (p.get_rank() == 0) {
317  p.read_uint64(persist_type::train, "reader_train_processed", &header.train_proc);
318  p.read_uint64(persist_type::train, "reader_train_total", &header.train_total);
319  p.read_uint64(persist_type::train, "reader_test_processed", &header.test_proc);
320  p.read_uint64(persist_type::train, "reader_test_total", &header.test_total);
321  p.read_uint64(persist_type::train, "reader_validate_processed", &header.validate_proc);
322  p.read_uint64(persist_type::train, "reader_validate_total", &header.validate_total);
323  }
324 
325  // TODO: assumes homogeneous hardware
326  // broadcast data from rank 0
327  MPI_Bcast(&header, sizeof(header), MPI_BYTE, 0, MPI_COMM_WORLD);
328 
329  // set our fields
330  m_training_dataset.m_num_samples_processed = (long) header.train_proc;
331  m_training_dataset.m_total_samples = (long) header.train_total;
332  m_testing_dataset.m_num_samples_processed = (long) header.test_proc;
333  m_testing_dataset.m_total_samples = (long) header.test_total;
334  m_validation_dataset.m_num_samples_processed = (long) header.validate_proc;
335  m_validation_dataset.m_total_samples = (long) header.validate_total;
336 
337  return true;
338  }
