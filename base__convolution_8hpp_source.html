278  virtual void setup_gpu() {
279  learning::setup_gpu();
280  #ifndef __LIB_CUDNN
281  throw lbann_exception("base_convolution_layer: cuDNN not detected");
282  #else
283 
284  // Initialize descriptors
285  CHECK_CUDNN(cudnnCreateTensorDescriptor(&m_bias_cudnn_desc));
286  CHECK_CUDNN(cudnnCreateFilterDescriptor(&m_kernel_cudnn_desc));
287  CHECK_CUDNN(cudnnCreateConvolutionDescriptor(&m_convolution_cudnn_desc));
288 
289  // Set kernel descriptor
290  CHECK_CUDNN(cudnnSetFilterNdDescriptor(m_kernel_cudnn_desc,
291  this->m_cudnn->get_cudnn_data_type(),
292  CUDNN_TENSOR_NCHW,
293  m_kernel_dims.size(),
294  m_kernel_dims.data()));
295 
296  // Set convolution descriptor
297  // Note: upscales are not supported as of cuDNN v5.1
298  std::vector<int> conv_upscales(this->m_num_neuron_dims-1, 1);
299  CHECK_CUDNN(cudnnSetConvolutionNdDescriptor(m_convolution_cudnn_desc,
300  m_conv_pads.size(),
301  m_conv_pads.data(),
302  m_conv_strides.data(),
303  conv_upscales.data(),
304  CUDNN_CONVOLUTION,
305  this->m_cudnn->get_cudnn_data_type()));
306 
307  // Set bias tensor descriptor
308  std::vector<int> bias_dims(this->m_num_neuron_dims+1, 1);
309  bias_dims[1] = this->m_neuron_dims[0];
310  std::vector<int> bias_strides(this->m_num_neuron_dims+1, 1);
311  bias_strides[0] = bias_dims[1];
312  CHECK_CUDNN(cudnnSetTensorNdDescriptor(m_bias_cudnn_desc,
313  this->m_cudnn->get_cudnn_data_type(),
314  bias_dims.size(),
315  bias_dims.data(),
316  bias_strides.data()));
317 
318  // Allocate GPU memory
319  this->m_cudnn->allocate_on_gpus(this->m_kernel_weights_d,
320  this->m_kernel_weights_v->Height(),
321  this->m_kernel_weights_v->Width());
322  this->m_cudnn->allocate_on_gpus(this->m_kernel_weights_gradient_d,
323  this->m_kernel_weights_gradient_v->Height(),
324  this->m_kernel_weights_gradient_v->Width());
325  if(m_bias_scaling_factor != DataType(0)) {
326  this->m_cudnn->allocate_on_gpus(this->m_bias_weights_d,
327  this->m_bias_weights_v->Height(),
328  this->m_bias_weights_v->Width());
329  this->m_cudnn->allocate_on_gpus(this->m_bias_weights_gradient_d,
330  this->m_bias_weights_gradient_v->Height(),
331  this->m_bias_weights_gradient_v->Width());
332  }
333 
334  // Initialize CPU memory for GPU reductions

352  void apply_convolution_cudnn(bool during_forward_prop) {
353  #ifndef __LIB_CUDNN
354  throw lbann_exception("base_convolution_layer: cuDNN not detected");
355  #else
356 
357  // Useful constants
358  const DataType one = 1;
359  const DataType zero = 0;
360 
361  // Initialize input and output
362  cudnnTensorDescriptor_t input_cudnn_desc, output_cudnn_desc;
363  std::vector<DataType*> input_d, output_d;
364  if(during_forward_prop) {
365  input_cudnn_desc = this->m_prev_neurons_cudnn_desc;
366  output_cudnn_desc = this->m_neurons_cudnn_desc;
367  input_d = this->m_prev_activations_d;
368  output_d = this->m_activations_d;
369  }
370  else {
371  input_cudnn_desc = this->m_neurons_cudnn_desc;
372  output_cudnn_desc = this->m_prev_neurons_cudnn_desc;
373  input_d = this->m_prev_error_signal_d;
374  output_d = this->m_error_signal_d;
375  }
376 
377  // Transfer kernels from CPU to GPUs
378  this->m_cudnn->broadcast_to_gpus(m_kernel_weights_d,
379  this->m_kernel_weights_v->LockedMatrix());
380 
381  // Perform convolution on each GPU
382  const int num_gpus = this->m_cudnn->get_num_gpus();
383  for(int i=0; i<num_gpus; ++i) {
384  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
385  CHECK_CUDNN(cudnnSetStream(this->m_cudnn->get_handle(i),
386  this->m_cudnn->get_stream(i)));
387  CHECK_CUDNN(cudnnConvolutionForward(this->m_cudnn->get_handle(i),
388  &one,
389  input_cudnn_desc,
390  input_d[i],
391  m_kernel_cudnn_desc,
392  m_kernel_weights_d[i],
393  m_convolution_cudnn_desc,
394  m_convolution_cudnn_algorithm,
395  this->m_cudnn->get_work_space(i),
396  this->m_cudnn->get_work_space_size(i),
397  &zero,
398  output_cudnn_desc,
399  output_d[i]));
400  }
401 
402  #endif // #ifndef __LIB_CUDNN
403  }

492  void compute_gradients_cudnn(bool using_transposed_convolution) {
493  #ifndef __LIB_CUDNN
494  throw lbann_exception("base_convolution_layer: cuDNN not detected");
495  #else
496 
497  // Useful constant
498  const DataType one = 1;
499  const DataType zero = 0;
500 
501  // Clear unused columns in previous error signal matrix
502  this->m_cudnn->clear_unused_columns_on_gpus(this->m_prev_error_signal_d,
503  this->m_num_neurons,
504  this->m_prev_error_signal->LocalWidth(),
505  this->m_mini_batch_size_per_gpu);
506 
507  // Compute gradients on GPUs
508  const int num_gpus = this->m_cudnn->get_num_gpus();
509  for(int i=0; i<num_gpus; ++i) {
510  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
511  CHECK_CUDNN(cudnnSetStream(this->m_cudnn->get_handle(i),
512  this->m_cudnn->get_stream(i)));
513  if(m_bias_scaling_factor != DataType(0)) {
514  CHECK_CUDNN(cudnnConvolutionBackwardBias(this->m_cudnn->get_handle(i),
515  &m_bias_scaling_factor,
516  this->m_neurons_cudnn_desc,
517  this->m_prev_error_signal_d[i],
518  &zero,
519  m_bias_cudnn_desc,
520  m_bias_weights_gradient_d[i]));
521  }
522  if(using_transposed_convolution) {
523  CHECK_CUDNN(cudnnConvolutionBackwardFilter(this->m_cudnn->get_handle(i),
524  &one,
525  this->m_neurons_cudnn_desc,
526  this->m_prev_error_signal_d[i],
527  this->m_prev_neurons_cudnn_desc,
528  this->m_prev_activations_d[i],
529  m_convolution_cudnn_desc,
530  m_kernel_gradient_cudnn_algorithm,
531  this->m_cudnn->get_work_space(i),
532  this->m_cudnn->get_work_space_size(i),
533  &zero,
534  m_kernel_cudnn_desc,
535  m_kernel_weights_gradient_d[i]));
536  }
537  else {
538  CHECK_CUDNN(cudnnConvolutionBackwardFilter(this->m_cudnn->get_handle(i),
539  &one,
540  this->m_prev_neurons_cudnn_desc,
541  this->m_prev_activations_d[i],
542  this->m_neurons_cudnn_desc,
543  this->m_prev_error_signal_d[i],
544  m_convolution_cudnn_desc,
545  m_kernel_gradient_cudnn_algorithm,
546  this->m_cudnn->get_work_space(i),
547  this->m_cudnn->get_work_space_size(i),
548  &zero,
549  m_kernel_cudnn_desc,
550  m_kernel_weights_gradient_d[i]));
551  }
552  }
553 
554  // Transfer gradients from GPUs to CPU and reduce
555  const int kernel_weights_width = m_kernel_weights_gradient_v->Width();
556  const int bias_weights_width = m_bias_weights_gradient_v->Width();
557  this->m_cudnn->gather_from_gpus(m_kernel_weights_gradient_per_gpu->Matrix(),
558  m_kernel_weights_gradient_d,
559  kernel_weights_width);
560  if(m_bias_scaling_factor != DataType(0)) {
561  this->m_cudnn->gather_from_gpus(m_bias_weights_gradient_per_gpu->Matrix(),
562  m_bias_weights_gradient_d,
563  bias_weights_width);
564  }
565  this->m_cudnn->synchronize();
566  Mat& local_kernel_weights_gradient = m_kernel_weights_gradient_v->Matrix();
567  for(int i=0; i<num_gpus; ++i) {
568  const Mat kernel_weights_on_current_gpu
569  = El::LockedView(m_kernel_weights_gradient_per_gpu->LockedMatrix(),
570  El::ALL,
571  El::IR(i * kernel_weights_width, (i+1) * kernel_weights_width));
572  if(i == 0) {
573  El::Copy(kernel_weights_on_current_gpu, local_kernel_weights_gradient);
574  }
575  else {
576  local_kernel_weights_gradient += kernel_weights_on_current_gpu;
577  }
578  }

723  void apply_bias_cpu() {
724 
725  // Return immediately if there is no bias
726  if(m_bias_scaling_factor == DataType(0)) return;
727 
728  // Get local matrices
729  const Mat& bias_weights_local = this->m_bias_weights_v->LockedMatrix();
730  Mat& activations_local = m_activations_v->Matrix();
731 
732  // Get output parameters
733  const El::Int width_local = activations_local.Width();
734  const El::Int num_output_channels = this->m_neuron_dims[0];
735  const El::Int num_per_output_channel = this->m_num_neurons / num_output_channels;
736 
737  // Apply bias to each output channel
738  #pragma omp parallel for
739  for(El::Int channel = 0; channel < num_output_channels; ++channel) {
740  const El::Int row_start = channel * num_per_output_channel;
741  const El::Int row_end = (channel+1) * num_per_output_channel;
742  const DataType bias_term
743  = m_bias_scaling_factor * bias_weights_local(0, channel);
744  for(El::Int col = 0; col < width_local; ++col) {
745  for(El::Int row = row_start; row < row_end; ++row) {
746  activations_local(row, col) += bias_term;
747  }
748  }

753  void compute_gradients_im2col(bool using_transposed_convolution) {
754 
755  // Get local matrices
756  const Mat& prev_activations_local = this->m_prev_activations->LockedMatrix();
757  const Mat& prev_error_signal_local = this->m_prev_error_signal->LockedMatrix();
758  Mat& kernel_weights_gradient_local = m_kernel_weights_gradient_v->Matrix();
759  Mat& bias_weights_gradient_local = m_bias_weights_gradient_v->Matrix();
760 
761  // Get convolution parameters
762  const El::Int width_local = prev_activations_local.Width();
763  const int num_input_channels = this->m_prev_neuron_dims[0];
764  const int num_output_channels = this->m_neuron_dims[0];
765  const int num_per_output_channel = this->m_num_neurons / num_output_channels;
766 
767  // Initialize weight gradients to zero
768  El::Zero(*this->m_weights_gradient);
769 
770  // Compute bias gradient
771  // Note: Sum is computed with Kahan summation
772  if(m_bias_scaling_factor != DataType(0)) {
773  #pragma omp parallel for
774  for(int channel = 0; channel < num_output_channels; ++channel) {
775  const El::Int row_start = channel * num_per_output_channel;
776  const El::Int row_end = (channel+1) * num_per_output_channel;
777  DataType sum = 0;
778  DataType correction = 0;
779  for(El::Int col = 0; col < width_local; ++col) {
780  for(El::Int row = row_start; row < row_end; ++row) {
781  DataType term = prev_error_signal_local(row, col);
782  term += correction;
783  const DataType next_sum = sum + term;
784  correction = term - (next_sum - sum);
785  sum = next_sum;
786  }
787  }
788  bias_weights_gradient_local(0, channel) = m_bias_scaling_factor * sum;
789  }
790  }
791 
792  // Initialize im2col matrix
793  Mat im2col_matrix;
794  if(using_transposed_convolution) {
795  im2col_matrix.Resize(m_kernel_size / num_input_channels,
796  this->m_num_prev_neurons / num_input_channels);
797  }
798  else {
799  im2col_matrix.Resize(m_kernel_size / num_output_channels,
800  this->m_num_neurons / num_output_channels);
801  }
802 
803  // Compute kernel gradient contributions from each data sample
804  for(El::Int col = 0; col < width_local; ++col) {
805  if(using_transposed_convolution) {
806  const Mat prev_activations_col(this->m_num_prev_neurons / num_input_channels,
807  num_input_channels,
808  prev_activations_local.LockedBuffer(0,col),
809  this->m_num_prev_neurons / num_input_channels);
810  const Mat prev_error_signal_col
811  = El::LockedView(prev_error_signal_local, El::ALL, El::IR(col));
812  im2col(prev_error_signal_col,
813  im2col_matrix,
814  num_output_channels,
815  this->m_num_neuron_dims - 1,
816  &this->m_neuron_dims[1],
817  m_conv_pads.data(),
818  &m_kernel_dims[2],
819  m_conv_strides.data());
820  El::Gemm(NORMAL, NORMAL,
821  DataType(1), im2col_matrix, prev_activations_col,
822  DataType(1), kernel_weights_gradient_local);
823  }
824  else {
825  const Mat prev_activations_col
826  = El::LockedView(prev_activations_local, El::ALL, El::IR(col));
827  const Mat prev_error_signal_col(this->m_num_neurons / num_output_channels,
828  num_output_channels,
829  prev_error_signal_local.LockedBuffer(0,col),
830  this->m_num_neurons / num_output_channels);
831  im2col(prev_activations_col,
832  im2col_matrix,
833  num_input_channels,
834  this->m_num_prev_neuron_dims - 1,
835  &this->m_prev_neuron_dims[1],
836  m_conv_pads.data(),
837  &m_kernel_dims[2],
838  m_conv_strides.data());
839  El::Gemm(NORMAL, NORMAL,
840  DataType(1), im2col_matrix, prev_error_signal_col,
841  DataType(1), kernel_weights_gradient_local);
842  }
843  }
844 
845  // Scale and accumulate gradients
846  *this->m_weights_gradient *=
847  DataType(1) / this->m_neural_network_model->get_effective_mini_batch_size();
848  El::AllReduce(*this->m_weights_gradient,
849  this->m_weights_gradient->RedundantComm());
850 
851  }
