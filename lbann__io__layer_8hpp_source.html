313  bool loadFromCheckpointShared(persist& p) {
314  // rank 0 reads the file
315  dataset_header header;
316  if (p.get_rank() == 0) {
317  p.read_uint64(persist_type::train, "reader_train_processed", &header.train_proc);
318  p.read_uint64(persist_type::train, "reader_train_total", &header.train_total);
319  p.read_uint64(persist_type::train, "reader_test_processed", &header.test_proc);
320  p.read_uint64(persist_type::train, "reader_test_total", &header.test_total);
321  p.read_uint64(persist_type::train, "reader_validate_processed", &header.validate_proc);
322  p.read_uint64(persist_type::train, "reader_validate_total", &header.validate_total);
323  }
324 
325  // TODO: assumes homogeneous hardware
326  // broadcast data from rank 0
327  MPI_Bcast(&header, sizeof(header), MPI_BYTE, 0, MPI_COMM_WORLD);
328 
329  // set our fields
330  m_training_dataset.num_samples_processed = (long) header.train_proc;
331  m_training_dataset.total_samples = (long) header.train_total;
332  m_testing_dataset.num_samples_processed = (long) header.test_proc;
333  m_testing_dataset.total_samples = (long) header.test_total;
334  m_validation_dataset.num_samples_processed = (long) header.validate_proc;
335  m_validation_dataset.total_samples = (long) header.validate_total;
336 
337  return true;
