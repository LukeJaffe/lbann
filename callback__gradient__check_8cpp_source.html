37 void lbann_callback_gradient_check::on_test_begin(model *m) {
38 
39  // Get model members
40  lbann_comm *comm = m->get_comm();
41  std::vector<Layer*>& layers = m->get_layers();
42 
43  // Initialize network for testing
44  m->set_execution_mode(execution_mode::testing);
45  for (size_t l = 0; l < layers.size(); ++l) {
46  layers[l]->set_execution_mode(execution_mode::testing);
47  }
48  layers[0]->forward_prop();
49 
50  // Compute objective function
51  const DataType objective = compute_objective_function(m);
52 
53  // Choose finite difference step
54  // Note: Consider a central difference scheme:
55  // f'(x) ~ ( - f(x+2h) + 8 f(x+h) - 8 f(x-h) + f(x-2h) ) / 12h
56  // By Taylor's theorem, the truncation error is bounded by
57  // E_trunc <= | f'''''(xi) | / 18 * h^4
58  // Assuming f can be computed to a relative accuracy of epsilon,
59  // E_fl <= epsilon * | f(chi) | / h
60  // For simplicity, we assume f(chi) ~ f(x), and | f'''''(xi) | ~ 1.
61  // If step size is not specified, then we choose h so that
62  // E_fl <= sqrt(epsilon)
63  const DataType epsilon = std::pow(std::numeric_limits<DataType>::epsilon(), 0.9);
64  DataType step_size = m_step_size;
65  if (m_step_size <= DataType(0)) {
66  step_size = std::fabs(objective) * std::sqrt(epsilon);
67  }
68  DataType expected_error = (epsilon * objective / step_size
69  + std::pow(step_size, 4) / 18);
70  expected_error = std::pow(expected_error, 0.9);
71 
72  // Compute gradients
73  for (size_t l = layers.size(); l-- > 0u;) {
74  layers[l]->back_prop();
75  }
76 
77  // Print objective function value
78  if (comm->am_world_master()) {
79  std::cout << "--------------------------------------------------------------------------------" << std::endl
80  << "Gradient checking..." << std::endl
81  << " Objective function value = " << objective << std::endl
82  << " Step size = " << step_size << std::endl
83  << " Expected gradient error = " << expected_error << std::endl;
84  }
85 
86  // Iterate through layers
87  for (size_t layer_index = 1; layer_index < layers.size() - 1; ++layer_index) {
88 
89  // Check that current layer is a learning layer
90  learning* layer = dynamic_cast<learning*>(layers[layer_index]);
91  if (layer == nullptr) {
92  continue;
93  }
94 
95  // Get weights and gradients
96  auto& weights = layer->get_weights();
97  auto& weights_gradient = layer->get_weights_gradient();
98 
99  // Iterate through weights in current layer
100  for (El::Int col = 0; col < weights.Width(); ++col) {
101  for (El::Int row = 0; row < weights.Height(); ++row) {
102  const DataType initial_weight = weights.Get(row, col);
103 
104  // Compute objective function values
105  weights.Set(row, col, initial_weight + 2 * step_size);
106  const DataType f_2h = compute_objective_function(m);
107  weights.Set(row, col, initial_weight + step_size);
108  const DataType f_h = compute_objective_function(m);
109  weights.Set(row, col, initial_weight - step_size);
110  const DataType f_nh = compute_objective_function(m);
111  weights.Set(row, col, initial_weight - 2 * step_size);
112  const DataType f_n2h = compute_objective_function(m);
113 
114  // Compute relative error in gradient
115  const DataType analytical_gradient = weights_gradient.Get(row, col);
116  const DataType numerical_gradient
117  = (- f_2h + 8 * f_h - 8 * f_nh + f_n2h) / (12 * step_size);
118  const DataType error = std::fabs(analytical_gradient - numerical_gradient);
119  DataType relative_error = DataType(0);
120  if (error != DataType(0)) {
121  relative_error = error / std::max(std::fabs(analytical_gradient),
122  std::fabs(numerical_gradient));
123  }
124 
125  // Print warning if relative error is large
126  if ((error > expected_error || m_verbose)
127  && comm->am_world_master()) {
128  std::cout << " Gradient error in layer " << layer_index << ", "
129  << "entry (" << row << "," << col << ")" << std::endl;
130  std::cout << " Weight = " << initial_weight << std::endl
131  << " Analytical gradient = " << analytical_gradient << std::endl
132  << " Numerical gradient = " << numerical_gradient << std::endl
133  << " Error = " << error << std::endl
134  << " Relative error = " << relative_error << std::endl;
135  }
136 
137  // Reset weight
138  weights.Set(row, col, initial_weight);
139 
140  }
141  }
142 
143  }
144 
145  if (comm->am_world_master()) {
146  std::cout << "--------------------------------------------------------------------------------" << std::endl;
147  }
148 
149 }

151 DataType lbann_callback_gradient_check::compute_objective_function(model *m) {
152  std::vector<Layer*>& layers = m->get_layers();
153  m->m_obj_fn->reset_statistics();
154  for (size_t l = 1; l < layers.size(); l++) {
155  layers[l]->forward_prop();
156  }
157  const DataType value = m->m_obj_fn->get_value();
158  m->m_obj_fn->reset_statistics();
159  return value;
