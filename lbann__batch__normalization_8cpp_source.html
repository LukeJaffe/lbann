90 void batch_normalization::fp_weights() {
91  // Get output from linearity.
92  ElMat* acts = m_layer->m_activations_v;
93  Copy(*(m_layer->m_weighted_sum_v), *m_weighted_sum_copy);
94  Int mbsize = acts->Width();
95  Mat& acts_local = acts->Matrix();
96  Mat& gamma_local = m_gamma->Matrix();
97  Mat& beta_local = m_beta->Matrix();
98  const Int local_width = acts_local.Width();
99  const Int local_height = acts_local.Height();
100  if (m_layer->m_execution_mode == execution_mode::training) {
101  Mat& mean_local = m_mean->Matrix();
102  Mat& var_local = m_var->Matrix();
103  // Compute the mean and variance of the activations over the minibatch.
104  // Todo: Better access pattern for locality.
105  for (Int row = 0; row < local_height; ++row) {
106  DataType sum = 0.0;
107  DataType sqsum = 0.0;
108  for (Int col = 0; col < local_width; ++col) {
109  sum += acts_local(row, col);
110  sqsum += acts_local(row, col) * acts_local(row, col);
111  }
112  mean_local(row, 0) = sum;
113  var_local(row, 0) = sqsum;
114  }
115  AllReduce(*m_mean, m_mean->RedundantComm(), mpi::SUM);
116  AllReduce(*m_var, m_var->RedundantComm(), mpi::SUM);
117  for (Int row = 0; row < local_height; ++row) {
118  mean_local(row, 0) /= mbsize;
119  var_local(row, 0) = Sqrt(var_local(row, 0) / mbsize -
120  (mean_local(row, 0) * mean_local(row, 0)));
121  }
122  // Compute transformed activations xhat = (x-mean)/sqrt(var)
123  for (Int col = 0; col < local_width; ++col) {
124  for (Int row = 0; row < local_height; ++row) {
125  // Normalize.
126  acts_local(row, col) -= mean_local(row, 0);
127  acts_local(row, col) /= var_local(row, 0) + DataType(1e-7);
128  // Apply scale/shift.
129  acts_local(row, col) *= gamma_local(row, 0);
130  acts_local(row, col) += beta_local(row, 0);
131  }
132  }
133  // Update the running averages.
134  Scale(m_decay, *m_running_mean);
135  Scale(m_decay, *m_running_var);
136  Axpy(DataType(1) - m_decay, *m_mean, *m_running_mean);
137  Axpy(DataType(1) - m_decay, *m_var, *m_running_var);
138  } else if (m_layer->m_execution_mode == execution_mode::validation ||
139  m_layer->m_execution_mode == execution_mode::testing) {
140  // Use the running mean/variance to normalize.
141  const Mat& mean_local = m_running_mean->LockedMatrix();
142  const Mat& var_local = m_running_var->LockedMatrix();
143  for (Int col = 0; col < local_width; ++col) {
144  for (Int row = 0; row < local_height; ++row) {
145  acts_local(row, col) -= mean_local(row, 0);
146  acts_local(row, col) /= var_local(row, 0) + DataType(1e-7);
147  acts_local(row, col) *= gamma_local(row, 0);
148  acts_local(row, col) += beta_local(row, 0);
149  }
150  }
151  }
152 }

154 void batch_normalization::bp_weights() {
155  // No backprop when not training.
156  if (m_layer->m_execution_mode != execution_mode::training) return;
157  ElMat* bpsignal = m_layer->m_prev_error_signal_v;
158  // "activations" here are from *before* the nonlinearity is applied.
159  //const ElMat* acts = m_layer->m_weighted_sum_v;
160  const ElMat* acts = m_weighted_sum_copy;
161  const Int mbsize = acts->Width();
162  Mat& bp_local = bpsignal->Matrix();
163  const Mat& acts_local = acts->LockedMatrix();
164  const Int local_height = bp_local.Height();
165  const Int local_width = bp_local.Width();
166  const Mat& mean_local = m_mean->LockedMatrix();
167  const Mat& var_local = m_var->LockedMatrix();
168  const Mat& gamma_local = m_gamma->LockedMatrix();
169  Mat& dgamma_local = m_dgamma->Matrix();
170  Mat& dbeta_local = m_dbeta->Matrix();
171  // Compute the derivatives of gamma and beta.
172  for (Int row = 0; row < local_height; ++row) {
173  dbeta_local(row, 0) = 0.0;
174  dgamma_local(row, 0) = 0.0;
175  for (Int col = 0; col < local_width; ++col) {
176  dbeta_local(row, 0) += bp_local(row, col);
177  dgamma_local(row, 0) +=
178  ((acts_local(row, col) - mean_local(row, 0)) /
179  (var_local(row, 0) + DataType(1e-7))) *
180  bp_local(row, col);
181  }
182  }
183  AllReduce(*m_dbeta, m_dbeta->RedundantComm(), mpi::SUM);
184  AllReduce(*m_dgamma, m_dgamma->RedundantComm(), mpi::SUM);
185  // Update the backprop gradient signal.
186  for (Int row = 0; row < local_height; ++row) {
187  for (Int col = 0; col < local_width; ++col) {
188  bp_local(row, col) = mbsize * bp_local(row, col);
189  bp_local(row, col) -= dbeta_local(row, 0);
190  bp_local(row, col) -=
191  ((acts_local(row, col) - mean_local(row, 0)) /
192  (var_local(row, 0) + DataType(1e-7))) *
193  dgamma_local(row, 0);
194  bp_local(row, col) /= DataType(mbsize);
195  bp_local(row, col) *= gamma_local(row, 0);
196  bp_local(row, col) /= var_local(row, 0) + DataType(1e-7);
197  }
198  }
199 }

201 void batch_normalization::setup(Layer* l) {
202  regularizer::setup(l);
203  Ones(*m_gamma, l->NumNeurons, 1);
204  Scale(m_gamma_init, *m_gamma);
205  Ones(*m_beta, l->NumNeurons, 1);
206  Scale(m_beta_init, *m_beta);
207  Zeros(*m_dgamma, l->NumNeurons, 1);
208  Zeros(*m_dbeta, l->NumNeurons, 1);
209  Zeros(*m_mean, l->NumNeurons, 1);
210  Zeros(*m_var, l->NumNeurons, 1);
211  Zeros(*m_running_mean, l->NumNeurons, 1);
212  Zeros(*m_running_var, l->NumNeurons, 1);
213  m_gamma_optimizer = l->neural_network_model->create_optimizer();
214  m_beta_optimizer = l->neural_network_model->create_optimizer();
215  m_gamma_optimizer->setup(m_gamma);
216  m_beta_optimizer->setup(m_beta);
217 }
