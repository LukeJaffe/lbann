416  inline int get_adaptive_quantization_threads(El::Int width) {
417  int num_threads = omp_get_max_threads();
418  if (width <= 64) {
419  num_threads = 2;
420  } else if (width <= 128) {
421  num_threads = 8;
422  } else if (width <= 256) {
423  num_threads = 12;
424  } else if (width <= 1024) {
425  num_threads = 24;
426  }
427  return std::min(omp_get_max_threads(), num_threads);
428  }

439  inline int get_adaptive_quantization_copy_threads(El::Int width) {
