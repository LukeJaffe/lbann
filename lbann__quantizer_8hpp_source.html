464  inline int get_adaptive_quantization_threads(El::Int width) {
465  int num_threads = omp_get_max_threads();
466  if (width <= 64) {
467  num_threads = 2;
468  } else if (width <= 128) {
469  num_threads = 8;
470  } else if (width <= 256) {
471  num_threads = 12;
472  } else if (width <= 1024) {
473  num_threads = 24;
474  }
475  return std::min(omp_get_max_threads(), num_threads);
476  }

487  inline int get_adaptive_quantization_copy_threads(El::Int width) {
