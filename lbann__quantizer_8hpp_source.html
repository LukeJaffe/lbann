327  inline int get_adaptive_quantization_threads(El::Int width) {
328  int num_threads = omp_get_max_threads();
329  if (width <= 64) {
330  num_threads = 2;
331  } else if (width <= 128) {
332  num_threads = 8;
333  } else if (width <= 256) {
334  num_threads = 12;
335  } else if (width <= 1024) {
336  num_threads = 24;
337  }
338  return std::min(omp_get_max_threads(), num_threads);
339  }
340 
350  inline int get_adaptive_quantization_copy_threads(El::Int width) {
351  int num_threads = get_adaptive_quantization_threads(width);
352  if (width >= 16384) {
353  num_threads /= 2;
354  }
355  return num_threads;
356  }
