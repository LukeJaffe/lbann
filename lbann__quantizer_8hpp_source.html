465  inline int get_adaptive_quantization_threads(El::Int width) {
466  int num_threads = omp_get_max_threads();
467  if (width <= 64) {
468  num_threads = 2;
469  } else if (width <= 128) {
470  num_threads = 8;
471  } else if (width <= 256) {
472  num_threads = 12;
473  } else if (width <= 1024) {
474  num_threads = 24;
475  }
476  return std::min(omp_get_max_threads(), num_threads);
477  }

488  inline int get_adaptive_quantization_copy_threads(El::Int width) {
