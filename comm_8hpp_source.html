551  static inline bool is_sendable(const Mat& mat) {
552  // This assumes we do not transmit mat with a datatype smaller than
553  // DataType.
554  // MPI uses "int" as its count type; do calculations with larger ints.
555  size_t count = (size_t) mat.Height() * (size_t) mat.Width();
556  return count <= (size_t) std::numeric_limits<int>::max();
557  }
